15  public class ModelBuilder
16  {
17  	static final String IDBASE = "";
18  	static final BioPAXFactory factory = BioPAXLevel.L3.getDefaultFactory();
19  	static final String REACTION_ARROW = "_->_";
20  	static final String ACTIV_ARROW = "_activates_";
21  	static final String DIR = "../biopax/Level3/examples/";

23  	public void generateModel1() throws Throwable
24  	{
25  		Model model = factory.createModel();
26  
27  		createStateChange("Z", model);
28  		Conversion last = (Conversion) model.getByID(IDBASE + "Z-1" + REACTION_ARROW + "Z-2");
29  
30  		createStateChange("A", model);
31  
32  		createCascadeBetween((Protein) model.getByID(IDBASE + "A-1"), last, "B", 5, model);
33  		createCascadeBetween((Protein) model.getByID(IDBASE + "A-2"), last, "C", 5, model);
34  
35  		createStateChange("D", model);
36  		createStateChange("E", model);
37  		createComplex(model, "Comp1",
38  			(Protein) model.getByID(IDBASE + "D-2"),
39  			(Protein) model.getByID(IDBASE + "E-2"));
40  
41  		createControl((Protein) model.getByID(IDBASE + "D-2_cm"),
42  			(Conversion) model.getByID(IDBASE + "A-1" + REACTION_ARROW + "A-2"),
43  			Catalysis.class, ControlType.INHIBITION, model);
44  
45  		BioPAXIOHandler io = new SimpleIOHandler();
46  		io.convertToOWL(model, new FileOutputStream(DIR + "temp2.owl"));
47  	}

49  	public static void createStateChange(String name, Model model)
50  	{
51  		String stdName = name + "_PR";
52  		ProteinReference ref = model.addNew(ProteinReference.class, IDBASE + stdName);
53  		ref.setStandardName(stdName);
54  
55  		stdName = name + "-1";
56  		Protein p1 = model.addNew(Protein.class, IDBASE + stdName);
57  		p1.setEntityReference(ref);
58  		p1.setStandardName(stdName);
59  		
60  		stdName = name + "-2";
61  		Protein p2 = model.addNew(Protein.class, IDBASE + stdName);
62  		p2.setEntityReference(ref);
63  		p2.setStandardName(stdName);
64  
65  		stdName = p1.getStandardName() + REACTION_ARROW + p2.getStandardName();
66  		BiochemicalReaction reac = model.addNew(BiochemicalReaction.class, 
67  				IDBASE + stdName);
68  		reac.setStandardName(stdName);
69  		reac.addLeft(p1);
70  		reac.addRight(p2);
71  		reac.setConversionDirection(ConversionDirectionType.LEFT_TO_RIGHT);
72  	}
73  
74  	public static void createActivatedStateChange(Protein activator, String name, Model model)
75  	{
76  		createStateChange(name, model);
77  		BiochemicalReaction reac = (BiochemicalReaction)
78  			model.getByID(IDBASE + name + "-1" + REACTION_ARROW + name + "-2");
79  
80  		createControl(activator, reac, Catalysis.class, ControlType.ACTIVATION, model);
81  	}
82  
83  	public static void createControl(Protein effector, Conversion conv,
84  									 Class<? extends Control> cls, ControlType type, Model model)
85  	{
86  		String std = effector.getStandardName() + ACTIV_ARROW + conv.getStandardName();
87  		Control con = factory.create(cls, IDBASE + std);
88  		con.setStandardName(std);
89  		con.addController(effector);
90  		con.addControlled(conv);
91  		con.setControlType(type);
92  		model.add(con);
93  	}
94  
95  	public static void createCascadeBetween(Protein first, Conversion last, String name, int length,
96  											Model model)
97  	{
98  		createActivatedStateChange(first, (name + 1), model);
99  
100 		for (int i = 2; i <= length; i++)
101 		{
102 			Protein act = (Protein) model.getByID(IDBASE + (name + (i - 1)) + "-2");
103 			createActivatedStateChange(act, (name + i), model);
104 		}
105 
106 		Protein act = (Protein) model.getByID(IDBASE + (name + length) + "-2");
107 		createControl(act, last, Catalysis.class, ControlType.ACTIVATION, model);
108 	}
109 
110 	public static void createComplex(Model model, String compName, Protein... prots)
111 	{
112 		Complex comp = model.addNew(Complex.class, IDBASE + compName);
113 		comp.setStandardName(compName);
114 
115 		for (Protein prot : prots)
116 		{
117 			String std = prot.getStandardName() + "_cm";
118 			Protein p = model.addNew(Protein.class, IDBASE + std);
119 			p.setStandardName(std);
120 			comp.addComponent(p);
121 		}
122 
123 		String std = "Assembly_of_" + compName;
124 		ComplexAssembly ca = model.addNew(ComplexAssembly.class, IDBASE + std);
125 		ca.setStandardName(std);
126 		ca.setConversionDirection(ConversionDirectionType.LEFT_TO_RIGHT);
127 		for (Protein prot : prots)
128 		{
129 			ca.addLeft(prot);
130 		}
131 		ca.addRight(comp);
132 	}
