30  	@Test
31  	public final void testSearch() throws IOException {
32  		SimpleIOHandler reader = new SimpleIOHandler();
33  		Model model = reader.convertFromOWL(getClass().getResourceAsStream("/pathwaydata1.owl"));
34  		SearchEngine searchEngine = new SearchEngine(model, indexLocation + "index1");
35  		searchEngine.index();
36  		assertTrue(new File(indexLocation + "index1").exists());
37  		
38  		SearchResult response = searchEngine.search("ATP", 0, null, null, null);
39  		assertNotNull(response);
40  		assertFalse(response.getHits().isEmpty());
41  		assertEquals(7, response.getHits().size());
42  		assertEquals(7, response.getTotalHits());
43  		
44  		response = searchEngine.search("ATP", 0, Interaction.class, null, null);
45  		assertNotNull(response);
46  		assertEquals(2, response.getHits().size());
47  		
48  		response = searchEngine.search("ATP", 0, Pathway.class, null, null);
49  		assertNotNull(response);
50  		assertEquals(1, response.getHits().size());
51  		
52  		BioPAXElement hit = response.getHits().get(0);
53  		assertEquals(4, hit.getAnnotations().get(SearchEngine.HitAnnotation.HIT_SIZE.name()));
54  		assertTrue(hit instanceof Pathway);
55  		assertEquals(5, hit.getAnnotations().keySet().size());
56  		
57  		//test a special implementation for wildcard queries
58  		response = searchEngine.search("*", 0, Pathway.class, null, null);
59  		assertNotNull(response);
60  		assertEquals(1, response.getHits().size());
61  		
62  		//find all objects (this here works with page=0 as long as the 
63  		//total no. objects in the test model < max hits per page)
64  		response = searchEngine.search("*", 0, null, null, null);
65  		assertEquals(50, response.getHits().size());
66  			
67  		response = searchEngine.search("*", 0, PhysicalEntity.class, null, null);
68  		assertEquals(8, response.getHits().size());
69  		
70  		response = searchEngine.search("*", 0, PhysicalEntity.class, null, new String[] {"562"});
71  		assertEquals(2, response.getHits().size());
72  		
73  		response = searchEngine.search("*", 0, PhysicalEntity.class, null, new String[] {"Escherichia"});
74  		assertEquals(2, response.getHits().size());
75  		
76  		response = searchEngine.search("*", 0, PhysicalEntity.class, null, new String[] {"Escherichia coli√º"});
77  		assertEquals(2, response.getHits().size());
78  		
79  		response = searchEngine.search("*", 0, Provenance.class, null, null);
80  		assertEquals(2, response.getHits().size());
81  		
82  		response = searchEngine.search("*", 0, Provenance.class, new String[] {"kegg"}, null);
83  		assertEquals(1, response.getHits().size());
84  		
85  		//now - same but using the long name (with 'of' an 'and') as filter value:
86  		response = searchEngine.search("*", 0, Provenance.class, new String[] {"Kyoto Encyclopedia of Genes and Genomes"}, null);
87  		assertEquals(1, response.getHits().size());
88  		
89  		//filter by data source (Provenance) URI - to tell Provenance objects having same name (but perhaps different organism or version)
90  		// from each other if necessary -
91  		response = searchEngine.search("*", 0, Pathway.class, new String[] {"http://identifiers.org/kegg.pathway/"}, null);
92  		assertEquals(1, response.getHits().size());
93  		
94  		response = searchEngine.search("glycolysis", 0, SmallMoleculeReference.class, null, null);
95  		assertTrue(response.getHits().isEmpty()); //parent pathway names are searched for in the keywords, names, etc. default fields
96  		
97  		response = searchEngine.search("pathway:glycolysis", 0, SmallMoleculeReference.class, null, null);
98  		assertEquals(5, response.getHits().size());
99  		
100 		//test search with pagination
101 		searchEngine.setMaxHitsPerPage(10);
102 		response = searchEngine.search("*", 0, null, null, null);
103 		assertEquals(50, response.getTotalHits());
104 		assertEquals(10, response.getHits().size());
105 		response = searchEngine.search("*", 1, null, null, null);
106 		assertEquals(10, response.getHits().size());
107 		
108 	}

110 	@Test
111 	public final void testHitsOrder() throws IOException {
112 		SimpleIOHandler reader = new SimpleIOHandler();
113 		Model model = reader.convertFromOWL(new GZIPInputStream(
114 				getClass().getResourceAsStream("/three-bmp-pathways.owl.gz")));
115 		
116 		//there are three BMP pathways (one is an empty pathway), and a trivial sub-pathway (not bmp):
117 		//"http://purl.org/pc2/7/Pathway_3f75176b9a6272a62f9257f0540dc63b" ("bmppathway", "BMP receptor signaling")
118 		//"http://purl.org/pc2/7/Pathway_b8fa8401d3053b57a10d4c29a3211258" ("BMP signaling pathway")
119 		//"http://identifiers.org/reactome/REACT_12034.3" ("Signaling by BMP" - the one we want...)
120 		//"http://purl.org/pc2/7/Pathway_aaa00bbf872fc777ce1e5eafc108752b" ("proteasomal ubiquitin-dependent protein catabolic process", a black-box)
121 		
122 		SearchEngine searchEngine = new SearchEngine(model, indexLocation + "index2");
123 		searchEngine.index();
124 		assertTrue(new File(indexLocation + "index2").exists());
125 		
126 		// A wide search in the default fields -
127 		SearchResult response = searchEngine.search("signaling by bmp", 0, Pathway.class, null, null);
128 		assertNotNull(response);
129 		assertFalse(response.getHits().isEmpty());		
130 		int i=0;
131 		for(BioPAXElement bpe : response.getHits()) {
132 			log.debug(String.format("Hit %d: %s; size: %s; excerpt: %s",
133 					++i, bpe.getUri(), bpe.getAnnotations().get(HitAnnotation.HIT_SIZE.name())
134 					, bpe.getAnnotations().get(HitAnnotation.HIT_EXCERPT.name())));
135 		}		
136 		assertEquals(3, response.getHits().size());
137 		assertEquals(3, response.getTotalHits());
138 		assertEquals("http://purl.org/pc2/7/Pathway_3f75176b9a6272a62f9257f0540dc63b", response.getHits().get(0).getUri());
139 		//the order of hits is alright, though we'd love to see REACT_12034.3 on top...
140 		
141 		// more accurate search in all the default fields using quotation marks around -
142 		response = searchEngine.search("\"signaling by bmp\"", 0, Pathway.class, null, null);
143 		assertNotNull(response);
144 		assertFalse(response.getHits().isEmpty());
145 		assertEquals(1, response.getTotalHits());
146 		assertEquals("http://identifiers.org/reactome/REACT_12034.3", response.getHits().get(0).getUri());
147 		
148 		//Next, narrow search in 'name' field only using quoted string
149 		response = searchEngine.search("name:\"Signaling by BMP\"", 0, Pathway.class, null, null);
150 		assertNotNull(response);
151 		assertFalse(response.getHits().isEmpty());
152 //		for(BioPAXElement bpe : response.getHits()) {
153 //			log.debug(String.format("Hit: %s; size: %s; excerpt: %s", bpe.getUri(),
154 //					bpe.getAnnotations().get(HitAnnotation.HIT_SIZE.name())
155 //					, bpe.getAnnotations().get(HitAnnotation.HIT_EXCERPT.name())));
156 //		}
157 		//there are three BMP pathways (one is an empty pathway), and a sub-pathway (should not match) -
158 		assertEquals(1, response.getHits().size());
159 		assertEquals(1, response.getTotalHits());
160 		assertEquals("http://identifiers.org/reactome/REACT_12034.3", response.getHits().get(0).getUri());
161 		
162 		
163 		//Next, search in 'pathway' field only, using quoted string
164 		response = searchEngine.search("pathway:\"signaling by bmp\"", 0, Pathway.class, null, null);
165 		assertNotNull(response);
166 		assertFalse(response.getHits().isEmpty());		
167 //		for(BioPAXElement bpe : response.getHits()) {
168 //			log.debug(String.format("Hit: %s; size: %s; excerpt: %s", bpe.getUri(),
169 //					bpe.getAnnotations().get(HitAnnotation.HIT_SIZE.name())
170 //					, bpe.getAnnotations().get(HitAnnotation.HIT_EXCERPT.name())));
171 //		}		
172 		//there is one pathway, and no sub-pathways of it
173 		assertEquals(1, response.getHits().size());
174 		assertEquals(1, response.getTotalHits());
175 		assertEquals("http://identifiers.org/reactome/REACT_12034.3", response.getHits().get(0).getUri());
176 		
177 		response = searchEngine.search("pathway:\"bmp receptor signaling\"", 0, Pathway.class, null, null);
178 		assertNotNull(response);
179 		assertFalse(response.getHits().isEmpty());		
180 //		for(BioPAXElement bpe : response.getHits()) {
181 //			log.debug(String.format("Hit: %s; size: %s; excerpt: %s", bpe.getUri(),
182 //					bpe.getAnnotations().get(HitAnnotation.HIT_SIZE.name())
183 //					, bpe.getAnnotations().get(HitAnnotation.HIT_EXCERPT.name())));
184 //		}	
185 		assertEquals(2, response.getTotalHits());
186 		// - the second pathway is a trivial one, member of Pathway_3f75176b9a6272a62f9257f0540dc63b and of many other parent pathways.
187 		
188 		//check the top pathway is the main one (parent)
189 		assertEquals("http://purl.org/pc2/7/Pathway_3f75176b9a6272a62f9257f0540dc63b", response.getHits().get(0).getUri());
190 	}
