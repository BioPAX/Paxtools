001package org.biopax.paxtools.impl.level3;
002
003import org.biopax.paxtools.model.level3.Entity;
004import org.biopax.paxtools.model.level3.Interaction;
005import org.biopax.paxtools.model.level3.InteractionVocabulary;
006import org.biopax.paxtools.util.BPCollections;
007import org.slf4j.Logger;
008import org.slf4j.LoggerFactory;
009
010import java.util.Set;
011
012public class InteractionImpl extends ProcessImpl implements Interaction
013{
014// ------------------------------ FIELDS ------------------------------
015
016        Set<Entity> participant;
017        private Set<InteractionVocabulary> interactionType;
018    private final Logger log = LoggerFactory.getLogger(InteractionImpl.class);
019
020// --------------------------- CONSTRUCTORS ---------------------------
021
022        public InteractionImpl()
023        {
024                this.interactionType = BPCollections.I.createSafeSet();
025                this.participant = BPCollections.I.createSafeSet();
026        }
027
028// ------------------------ INTERFACE METHODS ------------------------
029
030
031// --------------------- Interface BioPAXElement ---------------------
032
033        public Class<? extends Interaction> getModelInterface()
034        {
035                return Interaction.class;
036        }
037
038// --------------------- Interface Interaction ---------------------
039
040// --------------------- ACCESORS and MUTATORS---------------------
041
042        public Set<InteractionVocabulary> getInteractionType()
043        {
044           return interactionType;
045        }
046
047        public void addInteractionType(
048                InteractionVocabulary newinteractionType)
049        {
050           if(newinteractionType != null)
051                        this.interactionType.add(newinteractionType);
052        }
053
054        public void removeInteractionType(
055                InteractionVocabulary oldinteractionType)
056        {
057                if(oldinteractionType != null)
058                        this.interactionType.remove(oldinteractionType);
059        }
060
061        public Set<Entity> getParticipant()
062        {
063                return participant;
064        }
065
066        protected void setParticipant(Set<Entity> participant)
067        {
068        this.participant = participant;
069    }
070
071        public void addParticipant(Entity aParticipant)
072        {
073                if (aParticipant != null) {
074                        if (aParticipant != null) {
075                                this.participant.add(aParticipant);
076                                aParticipant.getParticipantOf().add(this);
077                        } else {
078                                if (log.isWarnEnabled())
079                                        log.warn("Null object passed to addParticipant @"
080                                                        + this.getUri());
081                        }
082                }
083    }
084
085        public void removeParticipant(Entity aParticipant)
086        {
087                if (aParticipant != null) {
088                        this.participant.remove(aParticipant);
089                        aParticipant.getParticipantOf().remove(this);
090                }
091        }
092}
