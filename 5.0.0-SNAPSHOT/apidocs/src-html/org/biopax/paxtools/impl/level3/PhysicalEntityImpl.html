122        // --------------------- Interface BioPAXElement ---------------------
123
124
125        @Override
126        protected boolean semanticallyEquivalent(BioPAXElement element)
127        {
128                if (!(element instanceof PhysicalEntity))
129                        return false;
130
131                PhysicalEntity that = (PhysicalEntity) element;
132                return hasEquivalentCellularLocation(that)
133                       && hasEquivalentFeatures(that)
134                       && SetEquivalenceChecker
135                                .isEquivalent(getMemberPhysicalEntity(), that.getMemberPhysicalEntity())
136                       &&
137                       super.semanticallyEquivalent(element); // StackOverflow BUG fixed: was isEquivalent !
138        }
139
140        @Override
141        public int equivalenceCode()
142        {
143                return hashCode();
144        }
145
146        public boolean hasEquivalentCellularLocation(PhysicalEntity that)
147        {
148                boolean equivalency = false;
149                if (that != null)
150                {
151                        equivalency = (cellularLocation != null)
152                                      ? cellularLocation.isEquivalent(that.getCellularLocation())
153                                      : that.getCellularLocation() == null;
154                }
155                return equivalency;
156        }
157
158        public boolean hasEquivalentFeatures(PhysicalEntity that)
159        {
160                boolean equivalency = false;
161                if (that != null)
162                {
163                        equivalency =
164                                        SetEquivalenceChecker.isEquivalent(this.getFeature(), that.getFeature()) &&
165                                        SetEquivalenceChecker.isEquivalent(this.getNotFeature(), that.getNotFeature());
166                }
167                return equivalency;
168        }
169
170        protected int locationAndFeatureCode()
171        {
172                int result = (cellularLocation != null ? cellularLocation.hashCode() : 0);
173                result = 31 * result + (feature != null ? feature.hashCode() : 0);
174                result = 31 * result + (notFeature != null ? notFeature.hashCode() : 0);
175                return result;
176
177        }
178
179        public Set<Control> getControllerOf()
180        {
181                return controllerOf;
182        }
