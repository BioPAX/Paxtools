001package org.biopax.paxtools.pattern.miner;
002
003import org.biopax.paxtools.model.Model;
004import org.biopax.paxtools.model.level3.SmallMoleculeReference;
005import org.biopax.paxtools.pattern.util.Blacklist;
006import org.biopax.paxtools.pattern.util.RelType;
007
008import java.io.FileInputStream;
009import java.io.FileNotFoundException;
010import java.io.InputStream;
011import java.util.HashMap;
012import java.util.Map;
013import java.util.Scanner;
014
015/**
016 * This class generates a blacklist for the given model. It is important that the given model is the
017 * very big integrated corpus. It won't work on tiny little model.
018 *
019 * @author Ozgun Babur
020 */
021public class BlacklistGenerator3
022{
023        /**
024         * Known ubiquitous small molecule names along with their
025         */
026        private Map<String, RelType> knownNames;
027
028        /**
029         * Constructor.
030         *
031         * @param knownNamesIS a special file that lists the known names of ubiquitous unwanted chemicals
032         */
033        public BlacklistGenerator3(InputStream knownNamesIS)
034        {
035                knownNames = new HashMap<String, RelType>();
036                Scanner sc = new Scanner(knownNamesIS);
037                while (sc.hasNextLine())
038                {
039                        String[] token = sc.nextLine().split("\t");
040                        RelType type = token.length < 2 ? null : token[1].equals("I") ?
041                                RelType.INPUT : RelType.OUTPUT;
042                        knownNames.put(token[0], type);
043                }
044        }
045
046        /**
047         * Default Constructor.
048         *
049         * Uses the blacklisted small molecule names file from the classpath.
050         */
051        public BlacklistGenerator3()
052        {
053                this(BlacklistGenerator3.class.getResourceAsStream("blacklist-names.txt"));
054        }
055
056        /**
057         * Generates the blacklist.
058         * @param model model to use
059         * @return the blacklist
060         */
061        public Blacklist generateBlacklist(Model model)
062        {
063                Blacklist blacklist = new Blacklist();
064
065                // populate the blacklist
066
067                for (SmallMoleculeReference smr : model.getObjects(SmallMoleculeReference.class))
068                {
069                        String name = smr.getDisplayName();
070                        if (name == null) continue;
071                        name = name.toLowerCase();
072
073
074                        if (knownNames.containsKey(name))
075                        {
076                                blacklist.addEntry(smr.getUri(), 1,
077                                        knownNames.get(name));
078                        }
079                }
080
081                return blacklist;
082        }
083}
