088        /**
089         * Gets the pattern, constructs if null.
090         * @return pattern
091         */
092        public Pattern getPattern()
093        {
094                final Pattern pattern = constructPattern();
095
096                if (this instanceof SIFMiner && idFetcher != null && idMap != null)
097                {
098                        pattern.add(new HasAnID(idFetcher, idMap), ((SIFMiner) this).getSourceLabel());
099                        pattern.add(new HasAnID(idFetcher, idMap), ((SIFMiner) this).getTargetLabel());
100                }
101
102                pattern.optimizeConstraintOrder();
103
104                return pattern;
105        }

157        /**
158         * Searches for the gene symbol of the given EntityReference.
159         * @param pr to search for a symbol
160         * @return symbol
161         */
162        protected String getGeneSymbol(ProteinReference pr)
163        {
164                for (Xref xr : pr.getXref())
165                {
166                        String db = xr.getDb();
167                        if (db != null)
168                        {
169                                db = db.toLowerCase();
170                                if (db.startsWith("hgnc"))
171                                {
172                                        String id = xr.getId();
173                                        if (id != null)
174                                        {
175                                                String symbol = HGNC.getSymbol(id);
176                                                if (symbol != null && !symbol.isEmpty())
177                                                {
178                                                        return symbol;
179                                                }
180                                        }
181                                }
182                        }
183                }
184
185                return null;
186        }

268        /**
269         * Sorts the modifications and gets them in a String.
270         * @param set modifications
271         * @return a String listing the modifications
272         */
273        public Set<String> toStringSet(Set<ModificationFeature> set)
274        {
275                List<ModificationFeature> list = new ArrayList<ModificationFeature>(set);
276
277                Collections.sort(list, new Comparator<ModificationFeature>()
278                {
279                        @Override
280                        public int compare(ModificationFeature o1, ModificationFeature o2)
281                        {
282                                String t1 = getModificationTerm(o1);
283                                String t2 = getModificationTerm(o2);
284
285                                Integer l1 = getPositionStart(o1);
286                                Integer l2 = getPositionStart(o2);
287
288                                if (t1 == null && t2 == null) return l1.compareTo(l2);
289                                if (t1 == null) return 1;
290                                if (t2 == null) return -1;
291                                if (t1.equals(t2)) return l1.compareTo(l2);
292                                return t1.compareTo(t2);
293                        }
294                });
295
296                return getInString(list);
297        }

299        /**
300         * Gets the modifications is a string that is separated with comma.
301         * @param list modification list
302         * @return String representing the modifications
303         */
304        private Set<String> getInString(List<ModificationFeature> list)
305        {
306                Set<String> text = new HashSet<String>(list.size());
307
308                for (ModificationFeature mf : list)
309                {
310                        String term = getModificationTerm(mf);
311                        String loc = getPositionInString(mf);
312
313                        if (term != null)
314                        {
315                                String s = term + loc;
316                                if (!text.contains(s)) text.add(s);
317                        }
318                }
319                return text;
320        }

334        /**
335         * Gets the first position of the modification feature.
336         * @param mf modification feature
337         * @return first location
338         */
339        public int getPositionStart(ModificationFeature mf)
340        {
341                Set vals = SITE_ACC.getValueFromBean(mf);
342
343                if (!vals.isEmpty())
344                {
345                        return ((Integer) vals.iterator().next());
346                }
347
348                vals = INTERVAL_BEGIN_ACC.getValueFromBean(mf);
349
350                if (!vals.isEmpty())
351                {
352                        return ((Integer) vals.iterator().next());
353                }
354
355                return -1;
356        }

358        /**
359         * Gets the position of the modification feature as a String.
360         * @param mf modification feature
361         * @return location
362         */
363        public String getPositionInString(ModificationFeature mf)
364        {
365                Set vals = SITE_ACC.getValueFromBean(mf);
366
367                if (!vals.isEmpty())
368                {
369                        int x = ((Integer) vals.iterator().next());
370                        if (x > 0) return "@" + x;
371                }
372
373                vals = INTERVAL_BEGIN_ACC.getValueFromBean(mf);
374
375                if (!vals.isEmpty())
376                {
377                        int begin = ((Integer) vals.iterator().next());
378
379                        vals = INTERVAL_END_ACC.getValueFromBean(mf);
380
381                        if (!vals.isEmpty())
382                        {
383                                int end = ((Integer) vals.iterator().next());
384
385                                if (begin > 0 && end > 0 && begin <= end)
386                                {
387                                        if (begin == end) return "@"  + begin;
388                                        else return "@" + "[" + begin + "-" + end + "]";
389                                }
390                        }
391                }
392
393                return "";
394        }

438        /**
439         * Gets delta modifications of the given elements in string sets. The elements has to be two
440         * PhysicalEntity chains. The result array is composed of two strings: gained (0) and lost (1).
441         *
442         * @param m match
443         * @param memLabel1 the member-end of the first PhysicalEntity chain
444         * @param comLabel1 the complex-end of the first PhysicalEntity chain
445         * @param memLabel2 the member-end of the second PhysicalEntity chain
446         * @param comLabel2 the complex-end of the second PhysicalEntity chain
447         * @return delta modifications
448         */
449        protected Set<String>[] getDeltaModifications(Match m, String memLabel1, String comLabel1,
450                String memLabel2, String comLabel2)
451        {
452                PhysicalEntityChain chain1 = getChain(m, memLabel1, comLabel1);
453                PhysicalEntityChain chain2 = getChain(m, memLabel2, comLabel2);
454
455                Set<ModificationFeature> before = chain1.getModifications();
456                Set<ModificationFeature> after = chain2.getModifications();
457
458                Set<String> afterMods = toStringSet(after);
459                Set<String> beforeMods = toStringSet(before);
460                removeCommon(afterMods, beforeMods);
461
462                return new Set[]{afterMods, beforeMods};
463        }

495        protected void removeCommon(Set<String> set1, Set<String> set2)
496        {
497                Set<String> common = new HashSet<String>(set1);
498                common.retainAll(set2);
499                set1.removeAll(common);
500                set2.removeAll(common);
501        }

505         * @param set the set
506         * @param sep separator string
507         * @return concatenated string
508         */
509        protected String concat(Set<String> set, String sep)
510        {
511                String s = "";
512
513                int i = set.size();
514                for (String ss : set)
515                {
516                        s += ss;
517                        if (--i > 0) s += sep;
518                }
519                return s;
520        }

536         * @param m result match
537         * @param ctrlLabel labels for controls
538         * @return sign
539         */
540        protected int sign(Match m, String... ctrlLabel)
541        {
542                int sign = 1;
543
544                for (String lab : ctrlLabel)

546                        Control ctrl = (Control) m.get(lab, getPattern());
547                        sign *= sign(ctrl);
548                }
549                return sign;

568        /**
569         * This method writes the output as pairs of gene symbols of the given two ProteinReference.
570         * Parameters labels have to map to ProteinReference.
571         * @param matches the search result
572         * @param out output stream for text output
573         * @param directed if true, reverse pairs is treated as different pairs
574         * @param label1 label for the first ProteinReference in the result matches
575         * @param label2 label for the second ProteinReference in the result matches
576         * @throws IOException if cannot write to output stream
577         */
578        protected void writeResultAsSIF(Map<BioPAXElement, List<Match>> matches, OutputStream out,
579                boolean directed, String label1, String label2) throws IOException
580        {
581                if (matches.isEmpty()) return;
582                if (this instanceof SIFMiner)
583                {
584                        writeSIFsUsingSIFFramework(matches, out);
585                        return;
586                }
587
588                // Memory for already written pairs.
589                Set<String> mem = new HashSet<String>();
590
591                String mid = getRelationType() == null ? "\t" : "\trelation\t";
592
593                OutputStreamWriter writer = new OutputStreamWriter(out);
594                String header = getHeader();
595                writer.write(header  == null ? label1 + mid + label2 : header);
596
597                for (BioPAXElement ele : matches.keySet())
598                {
599                        for (Match m : matches.get(ele))
600                        {
601                                Set<String> s1 = getIdentifiers(m, label1);
602                                Set<String> s2 = getIdentifiers(m, label2);
603
604                                for (String s1s : s1)
605                                {
606                                        for (String s2s : s2)
607                                        {
608                                                String type = getRelationType();
609                                                String sep = type == null ? "\t" : "\t" + type + "\t";
610
611                                                String relation = s1s + sep + s2s;
612                                                String reverse = s2s + sep + s1s;
613
614                                                if (!mem.contains(relation) && (directed || !mem.contains(reverse)))
615                                                {
616                                                        writer.write("\n" + relation);
617                                                        mem.add(relation);
618                                                        if (!directed) mem.add(reverse);
619                                                }
620                                        }
621                                }
622                        }
623                }
624                writer.flush();
625        }

627        /**
628         * This method writes the output as pairs of gene symbols of the given two ProteinReference.
629         * Parameters labels have to map to ProteinReference.
630         * @param matches the search result
631         * @param out output stream for text output
632         * @throws IOException if cannot write to output stream
633         */
634        protected void writeSIFsUsingSIFFramework(Map<BioPAXElement, List<Match>> matches,
635                OutputStream out) throws IOException
636        {
637                Map<SIFInteraction, SIFInteraction> sifMap = new HashMap<SIFInteraction, SIFInteraction>();
638
639                for (List<Match> matchList : matches.values())
640                {
641                        for (Match match : matchList)
642                        {
643                                for (SIFInteraction inter : this.createSIFInteraction(match, new CommonIDFetcher()))
644                                {
645                                        if (inter.hasIDs())
646                                        {
647                                                if (sifMap.containsKey(inter))
648                                                {
649                                                        sifMap.get(inter).mergeWith(inter);
650                                                }
651                                                else sifMap.put(inter, inter);
652                                        }
653                                }
654                        }
655                }
656                OutputStreamWriter writer = new OutputStreamWriter(out);
657
658                boolean first = true;
659                for (SIFInteraction inter : sifMap.keySet())
660                {
661                        if (first) first = false;
662                        else writer.write("\n");
663
664                        writer.write(inter.toString());
665                }
666                writer.flush();
667        }

696        /**
697         * Writes the result as a tab delimited format, where the column values are customized.
698         * @param matches result matches
699         * @param out output stream
700         * @param columns number of columns in the result
701         * @throws IOException if cannot write to the stream
702         */
703        protected void writeResultDetailed(Map<BioPAXElement, List<Match>> matches, OutputStream out,
704                int columns) throws IOException
705        {
706                OutputStreamWriter writer = new OutputStreamWriter(out);
707
708                // write the header
709
710                String header = getHeader();
711                if (header != null)
712                {
713                        writer.write(header);
714                }
715                else
716                {
717                        for (int i = 0; i < columns; i++)
718                        {
719                                writer.write("col-" + (i+1));
720                                if (i < columns - 1) writer.write("\t");
721                        }
722                }
723
724                // memory for already written lines
725                Set<String> mem = new HashSet<String>();
726
727                // write values
728
729                for (BioPAXElement ele : matches.keySet())
730                {
731                        for (Match m : matches.get(ele))
732                        {
733                                String line = "";
734                                boolean aborted = false;
735
736                                for (int i = 0; i < columns; i++)
737                                {
738                                        String s = getValue(m, i);
739
740                                        if (s == null)
741                                        {
742                                                aborted = true;
743                                                break;
744                                        }
745                                        else
746                                        {
747                                                line += s + "\t";
748                                        }
749                                }
750
751                                if (aborted) continue;
752
753                                line = line.trim();
754
755                                if (!mem.contains(line))
756                                {
757                                        writer.write("\n" + line);
758                                        mem.add(line);
759                                }
760                        }
761                }
762                writer.flush();
763        }

778        /**
779         * Creates a SIF interaction for the given match.
780         * @param m match to use for SIF creation
781         * @param fetcher ID generator from BioPAX object
782         * @return SIF interaction
783         */
784        public Set<SIFInteraction> createSIFInteraction(Match m, IDFetcher fetcher)
785        {
786                BioPAXElement sourceBpe = m.get(((SIFMiner) this).getSourceLabel(), getPattern());
787                BioPAXElement targetBpe = m.get(((SIFMiner) this).getTargetLabel(), getPattern());
788
789                Set<String> sources = fetchIDs(sourceBpe, fetcher);
790                Set<String> targets = fetchIDs(targetBpe, fetcher);
791
792                SIFType sifType = ((SIFMiner) this).getSIFType();
793
794                Set<SIFInteraction> set = new HashSet<SIFInteraction>();
795
796                for (String source : sources)
797                {
798                        for (String target : targets)
799                        {
800                                if (source.equals(target)) continue;
801                                else if (sifType.isDirected() || source.compareTo(target) < 0)
802                                {
803                                        set.add(new SIFInteraction(source, target, sourceBpe, targetBpe, sifType,
804                                                new HashSet<BioPAXElement>(m.get(getMediatorLabels(), getPattern())),
805                                                new HashSet<BioPAXElement>(m.get(getSourcePELabels(), getPattern())),
806                                                new HashSet<BioPAXElement>(m.get(getTargetPELabels(), getPattern()))));
807                                }
808                                else
809                                {
810                                        set.add(new SIFInteraction(target, source, targetBpe, sourceBpe, sifType,
811                                                new HashSet<BioPAXElement>(m.get(getMediatorLabels(), getPattern())),
812                                                new HashSet<BioPAXElement>(m.get(getTargetPELabels(), getPattern())),
813                                                new HashSet<BioPAXElement>(m.get(getSourcePELabels(), getPattern()))));
814                                }
815                        }
816                }
817                return set;
818        }

821        {
822                if (!idMap.containsKey(ele))
823                {
824                        Set<String> ids = fetcher.fetchID(ele);
825                        idMap.put(ele, ids);
826                }
827                return idMap.get(ele);
828        }

860        /**
861         * Uses uniprot name or gene symbol as identifier.
862         * @param m current match
863         * @param label label of the related EntityReference in the pattern
864         * @return identifier
865         */
866        protected Set<String> getIdentifiers(Match m, String label)
867        {
868                BioPAXElement el = m.get(label, getPattern());
869
870                if (idFetcher != null) return idFetcher.fetchID(el);
871
872                Set<String> set = new HashSet<String>();
873                if (el instanceof ProteinReference)
874                {
875                        set.add(getGeneSymbol((ProteinReference) el));
876                }
877                else if (el instanceof SmallMoleculeReference)
878                {
879                        set.add(getCompoundName((SmallMoleculeReference) el));
880                }
881
882                return set;
883        }
