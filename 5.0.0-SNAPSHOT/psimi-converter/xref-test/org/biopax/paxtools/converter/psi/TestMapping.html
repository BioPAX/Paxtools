86      @Test
87      public void testApi() throws Exception {
88  
89      	// unmarshall the data, close the stream
90      	PsimiXmlReader reader = new PsimiXmlReader();
91      	InputStream is = getClass().getClassLoader().getResourceAsStream(PSI_MI_TEST_FILE);
92      	EntrySet es = reader.read(is);
93      	is.close();
94      	Collection<Entry> entries =  es.getEntries();			
95      	// we should only have 1 entry
96      	assertEquals(1, entries.size());
97      	// get entry
98      	Entry entry = (Entry)entries.iterator().next();
99  
100     	assertTrue(entry.hasExperiments());
101     	assertEquals(4, entry.getExperiments().size());
102     	assertEquals(10, entry.getInteractions().size());
103 
104     	Interaction interaction = null;
105     	for(Interaction it : entry.getInteractions()) {
106     		if(it.getId()==1) {
107     			interaction = it;
108     			break;
109     		}
110     	}
111 
112     	//despite there is <experimentList><experimentRef> instead of <experimentList><experimentDescription> items -
113     	assertFalse(interaction.hasExperimentRefs());
114     	assertEquals(0, interaction.getExperimentRefs().size());
115     	//(- weird, but this is how PSIMI API works!)	
116     	//at the same time the following is set correctly:
117     	assertTrue(interaction.hasExperiments());
118     	assertEquals(1, interaction.getExperiments().size());
119 
120     	assertFalse(interaction.getParticipants().isEmpty());
121     	Participant participant = interaction.getParticipants().iterator().next();
122     	//despite there was interactorRef,
123     	assertFalse(participant.hasInteractorRef());
124     	assertNull(participant.getInteractorRef());
125     	//at the same time, the following is set:
126     	assertTrue(participant.hasInteractor());
127     	assertNotNull(participant.getInteractor());
128 
129     	assertFalse(participant.hasInteractionRef());
130     	assertFalse(participant.hasInteraction());			
131 
132     	
133     	//TODO assert experimentalRole.hasExperiments()==true always (if there're experiments); experimentalRole.hasExperimentRefs()==false
134 	}

194     @Test
195     public void testParticipantsAreNotDuplicated() throws IOException {
196 
197 		Model bpModel = BioPAXLevel.L3.getDefaultFactory().createModel();
198 		// open file
199 		try {
200 			// unmarshall the data, close the stream
201 			PsimiXmlReader reader = new PsimiXmlReader();
202             InputStream is = getClass().getClassLoader().getResourceAsStream(BIND_TEST_FILE);
203 			EntrySet es = reader.read(is);
204 			is.close();
205 			Collection<Entry> entries =  es.getEntries();
206             assertEquals(1, entries.size());
207 			Entry entry = (Entry)entries.iterator().next();
208 			EntryMapper mapper = new EntryMapper(bpModel, false);
209 			mapper.run(entry);
210 		}
211 		catch (Exception e) {
212 			e.printStackTrace();
213 		}
214 			
215 		assertNotNull(bpModel);
216 		assertFalse(bpModel.getObjects().isEmpty());
217 		
218 		assertEquals(3, bpModel.getObjects(MolecularInteraction.class).size());
219 		
220 		save(bpModel, getClass().getClassLoader().getResource("").getPath() 
221 			+ File.separator + "testConvertBindPsimi.owl");
222 		
223 		// there are 6 PSI-MI participants that refer to: 6 interactors and 3 experimental interactors 
224 		// (but these refer to only 5 unique primary xref IDs - 4 of protein and 1 of dna type) 
225 		// 4 "Max"/"GST-Max" and "Myc-associated factor X" interactors that have same ID makes only one PR
226 		// 2 HMBS promoter dnas make one DnaReference
227 		assertEquals(4, bpModel.getObjects(ProteinReference.class).size());
228 		// 2 dna type interactors ans experimental inter. merge into one
229 		assertEquals(1, bpModel.getObjects(DnaReference.class).size());
230 		// 9 original physical entities, some were merged (e.g., two gst-max...)
231 		assertEquals(8, bpModel.getObjects(SimplePhysicalEntity.class).size());
232 		
233 		ProteinReference pr = (ProteinReference) bpModel.getByID("ProteinReference_refseq_NP_002373_identity");
234 		assertNotNull(pr);
235 		assertEquals(2, pr.getName().size());
236 		assertTrue(pr.getName().contains("Max"));
237 		assertTrue(pr.getName().contains("Myc-associated factor X"));
238 		
239 	}
