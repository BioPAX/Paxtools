59  /**
60   * PSIMI 'entry' to BioPAX converter. 
61   * 
62   * PS (rodche, IR):
63   * The PSI-MI api is weird. It has methods/classes that pretend to follow the XML schema (OXM like)
64   * but in fact some of those do not make any practical sense, because they are either cleared or reset by the 
65   * PSI-MI parser (post-processing); e.g., look at (try) all those get*Ref or has*Ref, etc. methods.
66   * PSI-MI specification itself is tricky (e.g., the ability to define/express multiple experiment 
67   * descriptions per interaction, multiple roles per participant, multiple experimentalInteractors 
68   * per participant of an interaction are probably altogether unnecessary) and uses conventions 
69   * and practices not formally defined in the schema (e.g., how/whether to use "id" attribute, 
70   * absence of some xml elements means sometimes ALL such elements from the parent element (see e.g. experimentRef).
71   *
72   * @author Benjamin Gross, rodche (major re-factoring for Level3; fixing, adding genetic interactions, experimental form entities and features, etc.)
73   */

124 	 * Convert a PSIMI entry to BioPAX
125 	 * interactions, participants, etc. objects 
126 	 * and add to the target BioPAX model.
127 	 * 
128 	 * @param entry
129 	 */
130 	public void run(Entry entry) {	
131 		
132 		// get availabilities
133 		final Set<String> avail = new HashSet<String>();
134 		if(entry.hasAvailabilities()) {
135 			for (Availability a : entry.getAvailabilities())
136 				if (a.hasValue()) 
137 					avail.add(a.getValue());	
138 		}
139 		
140 		// get data source
141 		final Provenance pro = createProvenance(entry.getSource());
142 		
143 		//build a skip-set of "interactions" linked by participant.interactionRef element;
144 		//we'll then create Complex type participants from these in-participant interactions.
145 		Set<Interaction> participantInteractions = new HashSet<Interaction>();
146 		for(Interaction interaction : entry.getInteractions()) {
147 			for(Participant participant : interaction.getParticipants()) {
148 				//checking for hasInteraction()==true only is sufficient; 
149 				//i.e., we ignore hasInteractionRef(), getInteractionRefs(), 
150 				//because these in fact get cleared by the PSI-MI parser:
151 				if(participant.hasInteraction()) {
152 					participantInteractions.add(participant.getInteraction());
153 				}
154 			}
155 		}
156 		
157 		// iterate through the "root" psimi interactions only and create biopax interactions or complexes
158 		for (Interaction interaction : entry.getInteractions()) {
159 			if(!participantInteractions.contains(interaction)) {
160 				// TODO future (hard): make a Complex or Interaction based on the interaction type ('direct interaction' or 'physical association' (IntAct) -> complex)
161 				processInteraction(interaction, avail, pro, false);
162 			}
163 		}
164 	}

167 	private Provenance createProvenance(Source source) {
168 		Provenance pro = null;		
169 		String name = null;
170 		PublicationXref px = null;
171 		UnificationXref ux = null;
172 		
173 		if (source.hasNames()) {
174 			name = getName(source.getNames());
175 		} 
176 		
177 		if(source.hasXref()) {
178 			ux = getPrimaryUnificationXref(source.getXref());
179 			if(name==null && ux != null)
180 				name = ux.getDb()+"_"+ux.getId();
181 		} 
182 			
183 		if(source.hasBibref()) {
184 			psidev.psi.mi.xml.model.Xref psiXREF = source.getBibref().getXref();
185 			if (psiXREF != null) {
186 				px = publicationXref(psiXREF.getPrimaryRef().getDb(), psiXREF.getPrimaryRef().getId());
187 			}
188 			
189 			if(name==null && px != null)
190 				name = px.getDb()+"_"+px.getId();
191 		}
192 		
193 		String ver = null;
194 		if(source.hasReleaseDate()) {
195 			ver = source.getRelease();
196 			if(name==null)
197 				name = ver;
198 		}
199 		
200 		String sourceUri = (name!=null) 
201 			? xmlBase + "Provenance_" + encode(name)
202 				: genUri(Provenance.class, bpModel);
203 		
204 		//unless it's already there,
205 		pro = (Provenance) bpModel.getByID(sourceUri);		
206 		if(pro == null) { //generate a new one
207 			pro = bpModel.addNew(Provenance.class, sourceUri);
208 			
209 			if(name != null)
210 				pro.setDisplayName(name);
211 			
212 			if(px != null)
213 				pro.addXref(px);
214 			
215 			if(ux != null)
216 				pro.addXref(ux);
217 			
218 			if(source.hasAttributes())
219 				for(Attribute attr : source.getAttributes())
220 					pro.addComment(attr.toString());
221 			
222 			if(ver != null)
223 				pro.addComment("Release Date: " + ver);
224 		}
225 		
226 		return pro;
227 	}

231 	 * Creates a paxtools object that
232 	 * corresponds to the psi interaction.
233 	 *
234 	 * Note:
235 	 * psi.interactionElementType                 -&gt; biopax Complex, MolecularInteraction, or GeneticInteraction
236 	 * psi.interactionElementType.participantList -&gt; biopax interaction/complex participants/components
237 	 */
238 	private Entity processInteraction(Interaction interaction, Set<String> avail, 
239 			Provenance pro, boolean isComplex) 
240 	{	
241 		Entity bpInteraction = null; //interaction or complex
242 		boolean isGeneticInteraction = false;
243 		
244 		// get interaction name/short name
245 		String name = null;
246 		String shortName = null;
247 		if (interaction.hasNames()) {
248 			Names names = interaction.getNames();
249 			name = (names.hasFullName()) ? names.getFullName() : "";
250 			shortName = (names.hasShortLabel()) ? names.getShortLabel() : "";
251 		}
252 		
253 		final Set<InteractionVocabulary> interactionVocabularies = new HashSet<InteractionVocabulary>();
254 		if (interaction.hasInteractionTypes()) {
255 			for(CvType interactionType : interaction.getInteractionTypes()) {
256 				//generate InteractionVocabulary and set interactionType
257 				InteractionVocabulary cv = findOrCreateControlledVocabulary(interactionType, InteractionVocabulary.class);
258 				if(cv != null)
259 					interactionVocabularies.add(cv);
260 			}
261 		}		
262 		
263 		// using experiment descriptions, create Evidence objects 
264 		// (yet, no experimental forms/roles/entities are created here)
265 		Set<Evidence> bpEvidences = new HashSet<Evidence>();
266 		if (interaction.hasExperiments()) {			
267 			bpEvidences = createBiopaxEvidences(interaction);
268 		}
269 		
270 		//A hack for e.g. IntAct or BIND "gene-protein" interactions (ChIp and EMSA experiments)
271 		// where the interactor type should probably not be 'gene' (but 'dna' or 'rna')
272 		Set<String> participantTypes = new HashSet<String>();
273 		for(Participant p : interaction.getParticipants()) {
274 			if(p.hasInteractor()) {
275 				String type = getName(p.getInteractor().getInteractorType().getNames());
276 				if(type==null) type = "protein"; //default type (if unspecified)
277 				participantTypes.add(type.toLowerCase());
278 			} else if (p.hasInteraction()) {
279 				participantTypes.add("complex"); //hierarchical complex build up
280 			} // else? (impossible!)
281 		}
282 		// If there are both genes and physical entities present, let's 
283 		// replace 'gene' with 'dna' (esp. true for "ch-ip", "emsa" experiments);
284 		// (this won't affect experimental form entities if experimentalInteractor element exists)
285 		if(participantTypes.size() > 1 && participantTypes.contains("gene")) {
286 			//TODO a better criteria to reliably detect whether 'gene' interactor type actually means Dna/DnaRegion or Rna/RnaRegion, or indeed Gene)
287 			LOG.warn("Interaction: " + interaction.getId() + ", name(s): " + shortName + " " + name 
288 					+ "; has both 'gene' and physical entity type participants: " + participantTypes 
289 					+ "; so we'll replace 'gene' with 'dna' (a quick fix)");
290 			for(Participant p : interaction.getParticipants()) {
291 				if(p.hasInteractor() && p.getInteractor().getInteractorType().hasNames()) {
292 					String type = getName(p.getInteractor().getInteractorType().getNames());
293 					if("gene".equalsIgnoreCase(type)) {
294 						p.getInteractor().getInteractorType().getNames().setShortLabel("dna");
295 					}
296 				}
297 			}	
298 		}		
299 		
300 		// interate through the psi-mi participants, create corresp. biopax entities
301 		final Set<Entity> bpParticipants = new HashSet<Entity>();
302 		for (Participant participant : interaction.getParticipants()) {
303 			// get paxtools physical entity participant and add to participant list
304 			// (this also adds experimental evidence and forms)
305 			Entity bpParticipant = createBiopaxEntity(participant, avail, pro); 
306 			if (bpParticipant != null) {
307 				if(!bpParticipants.contains(bpParticipant))
308 					bpParticipants.add(bpParticipant);
309 			}
310 		}
311 				
312 		// Set GeneticInteraction flag.
313 		// As of BioGRID v3.1.72 (at least), genetic interaction code can reside
314 		// as an attribute of the Interaction via "BioGRID Evidence Code" key
315 		if (interaction.hasAttributes()) {
316 			for (Attribute attribute : interaction.getAttributes()) {
317 				if (attribute.getName().equalsIgnoreCase(BIOGRID_EVIDENCE_CODE)) {
318 					String value = (attribute.hasValue()) ? attribute.getValue().toLowerCase() : "";
319 					if (GENETIC_INTERACTIONS.contains(value)) {
320 						isGeneticInteraction = true;
321 					}
322 				}
323 			}
324 		}
325 		// or, if all participants are 'gene' type, make a biopax GeneticInteraction
326 		if(participantTypes.size() == 1 && participantTypes.contains("gene")) {
327 			isGeneticInteraction = true;
328 		}		
329 		//or, check another genetic interaction flag (criteria)
330 		if(!isGeneticInteraction) {
331 			isGeneticInteraction = isGeneticInteraction(bpEvidences);
332 		}
333 		
334 		if ((isComplex || forceInteractionToComplex) && !isGeneticInteraction) {
335 			bpInteraction = createComplex(bpParticipants);
336 		} else if(isGeneticInteraction) {
337 			bpInteraction = createGeneticInteraction(bpParticipants, interactionVocabularies);
338 		} else {
339 			bpInteraction = createMolecularInteraction(bpParticipants, interactionVocabularies);
340 		}
341 			
342 		//add evidences to the interaction/complex bpEntity
343 		for (Evidence evidence : bpEvidences) {
344 			bpInteraction.addEvidence(evidence);
345 		}
346 		
347 		addAvailabilityAndProvenance(bpInteraction, avail, pro);
348 		
349 		if (name != null)
350 			bpInteraction.addName(name);
351 		if (shortName != null) {
352 			if(shortName.length()<51)
353 				bpInteraction.setDisplayName(shortName);
354 			else
355 				bpInteraction.addName(shortName);
356 		}
357 				
358 		// add xrefs		
359 		Set<Xref> bpXrefs = new HashSet<Xref>();		
360 		if (interaction.hasXref()) {
361 			bpXrefs.addAll(getXrefs(interaction.getXref()));
362 		}
363 		
364 		for (Xref bpXref : bpXrefs) {
365 			bpInteraction.addXref(bpXref);
366 		}
367 		
368 		return bpInteraction;
369 	}

372 	private Set<Evidence> createBiopaxEvidences(Interaction interaction) {
373 		Set<Evidence> evidences = new HashSet<Evidence>();
374 		
375 		for (ExperimentDescription experimentDescription : interaction.getExperiments()) {
376 			// build and add evidence
377 			String evUri = genUri(Evidence.class, bpModel) + 
378 					"_i" + interaction.getId() + "_e" + experimentDescription.getId();
379 			
380 			Evidence evidence = bpModel.addNew(Evidence.class, evUri);
381 
382 			if (experimentDescription.hasXref()) {
383 				for(Xref xref : getXrefs(experimentDescription.getXref()))
384 					evidence.addXref(xref);
385 			}
386 			if (experimentDescription.getBibref() != null) {
387 				psidev.psi.mi.xml.model.Xref psiXREF = experimentDescription.getBibref().getXref();
388 				if (psiXREF != null) {
389 					PublicationXref px = publicationXref(psiXREF.getPrimaryRef().getDb(), psiXREF.getPrimaryRef().getId());
390 					if(px != null) 
391 						evidence.addXref(px);
392 				}
393 			}
394 					
395 			// create comments
396 			// from names (there is no biopax Evidence.name property)
397 			if (experimentDescription.hasNames()) {
398 				Names names = experimentDescription.getNames();
399 				if(names.hasFullName())
400 					evidence.addComment(names.getFullName().trim());
401 				if(names.hasShortLabel())
402 					evidence.addComment(names.getShortLabel().trim());
403 			}
404 			
405 			// from attributes
406 			if (experimentDescription.hasAttributes()) {
407 				for(String attr : getAttributes(experimentDescription.getAttributes()))
408 					evidence.addComment(attr.trim());
409 			}
410 
411 			//add hostOrganism info to comments (there is no 'organism' property of Evidence, EF, etc)
412 			if(experimentDescription.hasHostOrganisms()) {
413 				for(Organism organism : experimentDescription.getHostOrganisms())
414 					evidence.addComment("Host " + organism.toString());
415 			}
416 
417 			// confidence list
418 			if (experimentDescription.hasConfidences()) {
419 				for (Confidence psiConfidence : experimentDescription.getConfidences()) {
420 					Score bpScoreOrConfidence = getScoreOrConfidence(psiConfidence);
421 					if (bpScoreOrConfidence != null) 
422 						evidence.addConfidence(bpScoreOrConfidence);
423 				}
424 			}		
425 			
426 			// interaction detection method, participant detection method, feature detection method
427 			Set<EvidenceCodeVocabulary> evidenceCodes = getEvidenceCodes(experimentDescription);			
428 			if(evidenceCodes != null)
429 				for(EvidenceCodeVocabulary ecv : evidenceCodes)
430 					evidence.addEvidenceCode(ecv);
431 			
432 			evidences.add(evidence);
433 		}
434 		
435 		return evidences;
436 	}

451 	 * Converts a PSI-MI interaction's participant 
452 	 * to a BioPAX physical entity or gene
453 	 * and corresponding experimental form entities.
454 	 */
455 	private Entity createBiopaxEntity(Participant participant, Set<String> avail, Provenance pro) 
456 	{	
457 		LOG.debug("createBiopaxEntity, processing participant: " + participant.getId());
458 		
459 		// create a new PE or Gene to be
460 		Entity entity = null;
461 	
462 		// The PSI-MI parser does not set 'interactorRef' 
463 		// (or it rather clears that property and updates 'interactor' prop. instead;  
464 		// see junit tests).
465 		if (participant.hasInteractor()) {
466 			//participant.getInteractor() is not null
467 			entity = createParticipant(participant, participant.getInteractor(), avail, pro, false);
468 		} else if(participant.hasInteraction()) {
469 			//hierarchical buildup of a complex (participant.hasInteraction==true)...
470 			entity = (Complex) processInteraction(participant.getInteraction(), avail, pro, true);
471 		}
472 				
473 		if(entity == null) 
474 			return null;
475 		
476 		//create new Evidence for this participant
477 		Evidence participantEvidence = bpModel.getLevel().getDefaultFactory()
478 			.create(Evidence.class, genUri(Evidence.class, bpModel) + "_p" + participant.getId());
479 		
480 		if(participant.hasExperimentalRoles() 
481 			|| participant.hasExperimentalInteractors() 
482 				|| participant.hasFeatures()) 
483 		{
484 			if(participant.hasExperimentalInteractors()) {
485 				//create and collect experimental entities for current participant:
486 				for(ExperimentalInteractor experimentalInteractor : participant.getExperimentalInteractors()) {
487 					Interactor interactor = experimentalInteractor.getInteractor();
488 					LOG.debug("createBiopaxEntity, participant: " + participant.getId() 
489 							+ ", exp. interactor: " + interactor.getId());
490 					Entity expEntity = createParticipant(participant, interactor, avail, pro, true);
491 					assert expEntity!=null : "expEntity is null";
492 					
493 					//try to reuse existing experimental form entity
494 					expEntity = findEquivalentEntity(expEntity);
495 					if(!bpModel.contains(expEntity))
496 						bpModel.add(expEntity);
497 					
498 					//workaround a PSI-MI parser issue (no exp. or exp.refs means to apply to all interaction's experiments):
499 					if(experimentalInteractor.hasExperiments()) {
500 						for(ExperimentDescription exp : experimentalInteractor.getExperiments()) {
501 							LOG.debug("createBiopaxEntity, making EFs for exp: " 
502 									+ exp.getId() + "; exp.ent: " + expEntity.getUri());
503 							createAddExperimentalForm(participant, participantEvidence, expEntity, exp);
504 							
505 						}
506 					} else {
507 						LOG.debug("createBiopaxEntity, making a EForm - one for all experiments " 
508 								+ "; exp.ent: " + expEntity.getUri());
509 						createAddExperimentalForm(participant, participantEvidence, expEntity, null);
510 					}
511 				}		
512 			} else {
513 				LOG.debug("createBiopaxEntity, participant: " + participant.getId() 
514 						+ " doesn't have any exp. interactors");
515 				createAddExperimentalForm(participant, participantEvidence, null, null);
516 			}
517 			
518 			//save the participant entity evidence if it's not empty (roles, forms were actually generated)
519 			if(!participantEvidence.getExperimentalForm().isEmpty()) {
520 				bpModel.add(participantEvidence);
521 				entity.addEvidence(participantEvidence);
522 				//do not call findEquivalentEntity(entity) for this one having not empty evidence
523 			} else {
524 				//try to merge/reuse existing equivalent entity
525 				entity = findEquivalentEntity(entity);
526 			}
527 		}
528 	
529 		//finally, add, if new, entity to the model and return
530 		if(!bpModel.contains(entity))
531 			bpModel.add(entity);
532 		
533 		return entity;
534 	}

538 	 * Carefully find an existing equivalent physical entity or gene  
539 	 * or return the same one unchanged.
540 	 */
541 	private Entity findEquivalentEntity(final Entity entity) {
542 		Entity toReturn = entity;
543 		
544 		Class<? extends Entity> entityClass = PhysicalEntity.class;
545 		if(entity instanceof Gene)
546 			entityClass = Gene.class;
547 		
548 		for(Entity existingEntity : bpModel.getObjects(entityClass)) {
549 			// replace with an existing equivalent entity iif
550 			if( (//both are not experimental form entities
551 				 !entity.getComment().contains(EXPERIMENTAL_FORM_ENTITY_COMMENT)
552 				 && !existingEntity.getComment().contains(EXPERIMENTAL_FORM_ENTITY_COMMENT)
553 				)
554 				|| 
555 				(//both are experimental form entities
556 				 entity.getComment().contains(EXPERIMENTAL_FORM_ENTITY_COMMENT)
557 				 && existingEntity.getComment().contains(EXPERIMENTAL_FORM_ENTITY_COMMENT)
558 				 //and if disp.names match: names like 'GST-Max' are often used to describe states, instead of using other psi-mi features...
559 				 && entity.getDisplayName().equalsIgnoreCase(existingEntity.getDisplayName())
560 				)
561 			){	
562 				//and if there are no evidences yet, and the two are equivalent, then return existing one (to replace the entity)
563 				if(existingEntity.getEvidence().isEmpty() && entity.getEvidence().isEmpty() 
564 						&& existingEntity.isEquivalent(entity)) {
565 					toReturn = (Entity) existingEntity;
566 					break;
567 				}
568 			}
569 		}
570 		
571 		return toReturn;
572 	}

575 	/*
576 	 * Converts a PSIMI participant to BioPAX physical entity or gene. 
577 	 * It can be then used either as a participant of an interaction 
578 	 * or exp. form entity of a exp. form of a participant's evidence,
579 	 * depending on the caller method and the map provided.
580 	 * The result entity is not added to the model yet (but its entity reference, 
581 	 * if applicable, Xrefs, CVs are added to the model).
582 	 * 
583 	 * @param participant - of a PSI-MI interaction
584 	 * @param interactor - participant's interactor or experimental interactor's interactor
585 	 * @param avail - (biopax) availability
586 	 * @param pro - (biopax) provenance
587 	 * @param isExperimentalForm - a flag that controls entity's key comments (important)  
588 	 * 							 and how entity references are merged (e.g., copy names or not)
589 	 */
590 	private Entity createParticipant(Participant participant, Interactor interactor, 
591 			Set<String> avail, Provenance pro, boolean isExperimentalForm) 
592 	{
593 		if(interactor==null)
594 			throw new AssertionError("createParticipant: participant: " 
595 					+ participant.getId() + " has got null interactor");
596 		
597 		// Find or create the physical entity and entity reference -		
598 		// figure out physical entity type (protein, dna, rna, small molecule, gene)
599 		String entityType = null;
600 		CvType interactorType = interactor.getInteractorType();
601 		if (interactorType != null && interactorType.hasNames()) {
602 			entityType = getName(interactorType.getNames());
603 		}
604 
605 		// get names/synonyms from the psimi interactor (participant does not have them)
606 		String name = null;
607 		String shortName = null;
608 		Set<String> synonyms = new HashSet<String>();
609 		Names psiNames = interactor.getNames();
610 		if (psiNames != null) {
611 			name = (psiNames.hasFullName()) ? psiNames.getFullName() : null;
612 			shortName = (psiNames.hasShortLabel()) ? psiNames.getShortLabel() : null;
613 			if (psiNames.hasAliases()) {
614 				for (Alias alias : psiNames.getAliases()) {
615 					if (alias.hasValue()) {
616 						synonyms.add(alias.getValue());
617 					}
618 				}
619 			}
620 		}
621 		
622 		//get cellular location, if any
623 		CellularLocationVocabulary cellularLocation = null;
624 		if((interactor.hasOrganism() && interactor.getOrganism().hasCompartment())) {
625 			cellularLocation = findOrCreateControlledVocabulary(
626 					interactor.getOrganism().getCompartment(), CellularLocationVocabulary.class);
627 		}
628 		
629 		// make one entity reference (ER) per unique interactor (merge duplicate/equivalent ones)
630 		// make one physical entity (PE) per unique participant... 		
631 		// (a unique state PE is defined by the ER and cell. location, but not by entity features - which are to go with ExperimentalForms)		
632 		Class<? extends Entity> entityClass = Protein.class; //default
633 		Class<? extends EntityReference> entityReferenceClass = ProteinReference.class; //default
634 		if(entityType!=null && !entityType.isEmpty()) {
635 			//entityType should be a term from MI:0313 "interactor type" CV
636 			entityType = entityType.toLowerCase();
637 			if ("small molecule".equals(entityType)
638 					|| "polysaccharide".equals(entityType)
639 					|| "bioactive entity".equals(entityType))
640 			{
641 				entityClass = SmallMolecule.class;
642 				entityReferenceClass = SmallMoleculeReference.class;
643 			} else if ("dna".equals(entityType) 
644 					|| "deoxyribonucleic acid".equals(entityType) 
645 					|| entityType.contains("deoxyribonucleic")
646 					|| entityType.contains("dna")
647 					|| "nucleic acid".equals(entityType)) //when can't tell dna from rna? guess DNA
648 			{
649 				entityClass = Dna.class;
650 				entityReferenceClass = DnaReference.class;
651 			} else if ("rna".equals(entityType)
652 					|| "ribonucleic acid".equals(entityType)
653 					|| "poly adenine".equals(entityType)
654 					|| entityType.contains("rna"))
655 			{
656 				entityClass = Rna.class;
657 				entityReferenceClass = RnaReference.class;
658 			} else if ("gene".equals(entityType))
659 			{ //NOTE: Gene "gene" type is for Genetic Interactions only (not transcription/translation regulation, etc.), but...
660 				entityClass = Gene.class;
661 				entityReferenceClass = null;
662 			} else if ("complex".equals(entityType) || entityType.contains("complex"))
663 			{
664 				entityClass = Complex.class;
665 				entityReferenceClass = null;
666 			} else if ("interaction".equals(entityType))
667 			{
668 //				entityClass = null;
669 //				entityReferenceClass = null;
670 				LOG.warn("EntryMapper.createParticipant(): skip for interactor: " + interactor.getId()
671 						+ " that has type: " + entityType 
672 						+ ", participant:" + participant.getId() + ").");
673 				return null;
674 			}
675 						
676 			//else - peptide, biopolymer, molecule set, unknown, etc. - consider this is a Protein.
677 		}
678 		
679 		//make consistent biopax URI (for the entity ref., if applies, or phys. entity)
680 		String entityUri = "";
681 		String baseUri = "";
682 		
683 		final RelationshipXref x = getInteractorPrimaryRef(interactor.getXref());
684 		if(x != null) {
685 			baseUri += encode(x.getDb() + "_" + x.getId());
686 			if(x.getRelationshipType()!=null)
687 				baseUri += "_" + encode(x.getRelationshipType().getTerm().iterator().next());
688 			entityUri = baseUri + "_" + String.valueOf(counter++);
689 		} else { //when no xrefs present, use a number ending (always increment);
690 			baseUri = String.valueOf(counter++); //new unique part (seldom happens, when no primary xref...)
691 			entityUri = baseUri;
692 		}
693 		
694 		if(cellularLocation != null)
695 			entityUri += "_" + encode(cellularLocation.getTerm().iterator().next());
696 		
697 		//makes URI look like xmlBase+type+unique_suffix...
698 		entityUri = xmlBase + entityClass.getSimpleName() + "_" + entityUri;	
699 		if(entityReferenceClass != null)
700 			baseUri = xmlBase + entityReferenceClass.getSimpleName() + "_" + baseUri;	
701 		
702 		// create a new PE or Gene (don't add to the model yet, for there may exist an equivalent one)
703 		Entity entity = bpModel.getLevel().getDefaultFactory().create(entityClass, entityUri);
704 		
705 		addAvailabilityAndProvenance(entity, avail, pro);
706 		//and names
707 		if (name != null) {
708 			//quite a few PSIMI providers use too long text (like comments) for <fullName> fields...
709 			if(name.length() > 100 && shortName != null) 
710 				entity.addComment(name);
711 			else 
712 				entity.addName(name);
713 		}
714 		if (shortName != null) {
715             entity.setDisplayName(shortName);
716 		}				
717 		// set cellular location if possible
718 		if(cellularLocation != null && entity instanceof PhysicalEntity)
719 			((PhysicalEntity)entity).setCellularLocation(cellularLocation);
720 		
721 		final Set<Xref> bpXrefsOfInteractor = getXrefs(interactor.getXref());
722 		final BioSource bioSource = getBioSource(interactor);
723 		
724 		// when not a Gene, we are to find/generate corresponding EntityRererence;
725 		// but we do not want to gererate apparently duplicate objects...
726 		if(entityReferenceClass != null) {
727 			EntityReference entityReference = null;
728 			
729 			/* check if the entity ref. with this URI already exists 
730 			 * (we'll the use that instead of generating a new one every time;
731 			 * unfortunately, chances still are that we generate same URI 
732 			 * from non-equivalent psi-mi interactors, though this would probably 
733 			 * flag for semantic errors in the psi-mi data, such as using same xrefs,
734 			 * features, etc. for different xml entries unnecessarily).
735 			 * Unfortunately, we cannot simply create a new ER and compare
736 			 * with an existing one using Paxtools (as of paxtools-core 4.x and older, 
737 			 * two ERs are NOT equivalent unless they either have the same URI, or - same organism and sequence).
738 			 */
739 			EntityReference er = (EntityReference) bpModel.getByID(baseUri);			
740 			if( er != null 
741 				&& er.getModelInterface()==entityReferenceClass 
742 				&& (!(er instanceof SequenceEntityReference) 
743 					|| sameNameOrUndefined(((SequenceEntityReference)er).getOrganism(),bioSource)
744 					)
745 			) {
746 				
747 				entityReference = er; // ok to reuse
748 				
749 			} else if(er != null) {
750 				String newUri = baseUri + "_" + (counter++);
751 				LOG.warn("A different " + er.getModelInterface().getSimpleName()
752 					+ ", URI=" + baseUri + ", was found; for interactor:" + interactor.getId()
753 					+ ", a new ("+entityReferenceClass.getSimpleName()+") URI will be used:" + newUri);
754 				baseUri = newUri;
755 			}
756 			
757 			//if not found above, create a new ER
758 			if(entityReference == null) { 
759 				entityReference = bpModel.addNew(entityReferenceClass, baseUri);	
760 				
761 				if (shortName != null) {
762 					entityReference.setDisplayName(shortName);
763 				}
764 				
765 				//set organism if it's not a small molecule
766 				if(entityReference instanceof SequenceEntityReference) {
767 					SequenceEntityReference ser = (SequenceEntityReference)entityReference;
768 					ser.setOrganism(bioSource);
769 					ser.setSequence(interactor.getSequence());
770 				}
771 			}
772 				
773 			//update the displayName if we've got to reuse the existing ER,
774 			//and this is not an experimental interactor, and new name is shorter
775 			if(entityReference == er && shortName != null && !isExperimentalForm) {
776 				if(er.getDisplayName() != null) {
777 					if(shortName.length() < er.getDisplayName().length()) {
778 						entityReference.addName(er.getDisplayName()); //keep both names
779 						entityReference.setDisplayName(shortName);
780 					} else {
781 						entityReference.addName(shortName); //keep both names
782 					}
783 				} else {
784 					entityReference.setDisplayName(shortName);
785 				}
786 			}
787 			
788 			// if we're not reusing existing ER, and this interactor is not experimental,
789 			// copy names and xrefs to the exisiting ER
790 			if( !(isExperimentalForm && entityReference == er) ) {
791 				if (name != null) {
792 					if(name.length() > 100 && //and there are other names
793 							(entityReference.getDisplayName()!=null || (synonyms!=null && !synonyms.isEmpty())))
794 						entityReference.addComment(name); //comment instead of 'standardName'
795 					else		
796 						entityReference.addName(name);
797 				}
798 				if (synonyms != null) {
799 					for (String synonym : synonyms)
800 						entityReference.addName(synonym);
801 				}
802 				if (bpXrefsOfInteractor != null) {
803 					for (Xref xref : bpXrefsOfInteractor)
804 						entityReference.addXref((Xref) xref);
805 				}
806 			}
807 			
808 			// set ER
809 			((SimplePhysicalEntity)entity).setEntityReference(entityReference);		
810 			
811 		} else { //i.e., entity is a Gene or Complex
812 			//add gene's other names, xrefs, organism (for non-genes, these're added to the ER)
813 			if (synonyms != null) {
814 				for (String synonym : synonyms) {
815 					entity.addName(synonym);
816 				}
817 			}		
818 			if (bpXrefsOfInteractor != null) {
819 				for (Xref xref : bpXrefsOfInteractor)
820 					entity.addXref((Xref) xref);
821 			}
822 			if(entity instanceof Gene) //can be Complex here as well
823 				((Gene)entity).setOrganism(bioSource);
824 		}
825 		
826 		//Merge some of equivalent entities to avoid unnecessary duplicates 
827 		// CAREFULLY, e.g.:
828 		// - never merge a participant entity with an experimental form entity,
829 		// - never merge if one entity have evidence, etc.
830 		//(equiv. ERs are being merged regardless this)
831 		//Merging entities (states) can mess up names, features, exp. roles!
832 		if(isExperimentalForm)
833 			entity.addComment(EXPERIMENTAL_FORM_ENTITY_COMMENT);
834 		else
835 			entity.addComment("psi-mi participant");
836 		
837 		return entity;
838 	}

895 	 * Given a psi organism, return a paxtools biosource.
896 	 */
897 	private BioSource getBioSource(Interactor interactor) {
898 		
899 		Organism organism = interactor.getOrganism();
900 		
901 		// trivial check
902 		if (organism == null) 
903 			return null;
904 
905 		// cell type (can be undefined, i.e., null)
906 		CellVocabulary cellType = findOrCreateControlledVocabulary(organism.getCellType(), CellVocabulary.class);
907 		// tissue (can be null)
908 		TissueVocabulary tissue = findOrCreateControlledVocabulary(organism.getTissue(), TissueVocabulary.class);
909 
910 		// set the BioPAXElement URI and taxonomy xref id
911 		String ncbiId = Integer.toString(organism.getNcbiTaxId());
912 		String name = null;
913 		
914 		if (organism.hasNames()) {
915 			name = getName(organism.getNames());
916 			//a hack for BIND data (and perhaps other DBs)
917 			if(name != null && ("homo sapiens".equalsIgnoreCase(name)
918 					|| "human".equalsIgnoreCase(name))
919 					&& !"9606".equals(ncbiId)) {
920 				LOG.error("Taxonomy: " 
921 					+ ncbiId + " of organism: " + organism 
922 					+ ", interactor: " + interactor.getId() 
923 					+ " does not belong to '" 
924 					+ name + "'; the name will be removed. "
925 					+ "Data provider should probably use <experimentalInteractor> in addition to <interactor>");
926 				name = null;
927 			}
928 		}
929 		
930 		String uri = xmlBase + "BioSource_" + 
931 			"taxonomy_" + ncbiId; //tissue and cell type terms can be added below		
932 		if(tissue!=null && !tissue.getTerm().isEmpty()) 
933 			uri += "_" + encode(tissue.getTerm().iterator().next());
934 		if(cellType!=null && !cellType.getTerm().isEmpty()) 
935 			uri += "_" + encode(cellType.getTerm().iterator().next());
936 
937 		//return if element already exists in model
938 		BioSource toReturn = (BioSource) bpModel.getByID(uri);
939 		if (toReturn != null) 
940 			return toReturn;
941 		
942 		toReturn = bpModel.addNew(BioSource.class, uri);
943 		
944 		String taxonXrefUri = xmlBase + "UnificationXref_taxonomy_" + ncbiId;
945 		UnificationXref taxonXref = (UnificationXref) bpModel.getByID(taxonXrefUri);
946 		if(taxonXref == null) {
947 			taxonXref = bpModel.addNew(UnificationXref.class, taxonXrefUri);
948 			taxonXref.setDb("Taxonomy");
949 			taxonXref.setId(ncbiId);
950 		}
951 		toReturn.addXref((Xref)taxonXref);
952 		
953 		if (cellType != null)
954 		{
955 			toReturn.setCellType((CellVocabulary) cellType);
956 		}
957 		if (tissue != null)
958 		{
959 			toReturn.setTissue((TissueVocabulary) tissue);
960 		}
961 		if (name != null) {
962 			toReturn.setStandardName(name);
963 		}
964 				
965 		return toReturn;
966 	}

1019 	private Set<Xref> getXrefs(psidev.psi.mi.xml.model.Xref psiXREF) {
1020 
1021 		// set to return
1022 		Set<Xref> toReturn = new HashSet<Xref>();
1023 
1024 		// check args
1025 		if (psiXREF == null) return toReturn;
1026 		
1027 		// create the list of all other psimi xrefs
1028 		List<DbReference> psiDBRefList = new ArrayList<DbReference>();
1029 
1030 		DbReference primaryRef = psiXREF.getPrimaryRef();
1031 		if(primaryRef != null) { //looks, it should never be null in a valid PSI-MI model
1032             if ("identity".equals(primaryRef.getRefType()) || "identical object".equals(primaryRef.getRefType())) {
1033                 UnificationXref bpXref = unificationXref(primaryRef.getDb(), primaryRef.getId());
1034 				toReturn.add(bpXref);
1035             } else {
1036 				psiDBRefList.add(psiXREF.getPrimaryRef());
1037 			}
1038 		}
1039 
1040 		if (psiXREF.hasSecondaryRef()) {
1041 			psiDBRefList.addAll(psiXREF.getSecondaryRef());
1042 		}
1043 
1044 		for (DbReference psiref : psiDBRefList) {
1045 			if(psiref==null) 
1046 				continue;
1047 			xref(psiref, toReturn);
1048         }
1049 
1050 		return toReturn;
1051 	}

1054 	private void xref(DbReference psiref, Set<Xref> bpXrefs) {
1055 		// process ref type
1056 		String refType = (psiref.hasRefType()) ? psiref.getRefType() : null;
1057 		String refTypeAc = (psiref.hasRefTypeAc()) ? psiref.getRefTypeAc() : null;
1058         String psiDBRefId = psiref.getId();
1059         String psiDBRefDb = psiref.getDb();
1060         // If multiple ids given with comma separated values, then split them.
1061         for (String dbRefId : psiDBRefId.split(",")) {	
1062         	Xref bpXref = null;
1063 			// Let's not make UnificationXrefs. RelationshipXref is more safe.
1064 			// Often, a gene or omim ID (can be another species') is a protein's xref id with 'identity' type...
1065             if(!"pubmed".equalsIgnoreCase(psiDBRefDb)) {
1066 				bpXref = relationshipXref(psiDBRefDb, dbRefId, refType, refTypeAc);
1067             }
1068             else {
1069             	//TODO shall we skip PublicationXref here (IntAct puts the same PSIMI paper pmid everywhere...)?
1070 				bpXref = publicationXref(psiDBRefDb, dbRefId);
1071             }
1072 
1073             if (bpXref != null) 
1074             	bpXrefs.add(bpXref);
1075         }
1076 	}

1095
