177 	{
178 
179 		String ID_COMPARTMENT_1 = "compartment_1";
180 		String ID_PROTEIN_1 = "PROTEIN_1";
181 		String ID_PROTEIN_2 = "PROTEIN_2";
182 		String ID_PROTEIN_3 = "PROTEIN_3";
183 
184 		String ID_PROTEIN_REFERENCE_1 = "PROTEIN_REFERENCE_1";
185 		String ID_PROTEIN_REFERENCE_2 = "PROTEIN_REFERENCE_2";
186 		String ID_PROTEIN_REFERENCE_3 = "PROTEIN_REFERENCE_3";
187 
188 		BioPAXFactory level3Factory = BioPAXLevel.L3.getDefaultFactory();
189 		Model biopaxModel = level3Factory.createModel();
190 
191 		// Create a compartment
192 		CellularLocationVocabulary clv =
193 				biopaxModel.addNew(CellularLocationVocabulary.class, ID_COMPARTMENT_1);
194 		String compartmentName = "golgi";
195 		clv.addTerm(compartmentName);
196 
197 
198 		// Create three proteins in golgi
199 		Protein p1 = biopaxModel.addNew(Protein.class, ID_PROTEIN_1);
200 		p1.setDisplayName("PE-Pro-BACE-1");
201 		Protein p2 = biopaxModel.addNew(Protein.class, ID_PROTEIN_2);
202 		p2.setDisplayName("PE-BACE-1");
203 		Protein p3 = biopaxModel.addNew(Protein.class, ID_PROTEIN_3);
204 		p3.setDisplayName("Furin");
205 		p1.setCellularLocation(clv);
206 		p2.setCellularLocation(clv);
207 		p3.setCellularLocation(clv);
208 
209 		// Create entity references for the proteins
210 		ProteinReference pr1 = biopaxModel.addNew(ProteinReference.class, ID_PROTEIN_REFERENCE_1);
211 		pr1.setStandardName("Pro-BACE-1");
212 		p1.setEntityReference(pr1);
213 		Stoichiometry stoichiometry1 = biopaxModel.addNew(Stoichiometry.class, "ST1");
214 		stoichiometry1.setPhysicalEntity(p1);
215 		stoichiometry1.setStoichiometricCoefficient(1);
216 
217 		ProteinReference pr2 = biopaxModel.addNew(ProteinReference.class, ID_PROTEIN_REFERENCE_2);
218 		pr2.setStandardName("BACE-1");
219 		p2.setEntityReference(pr2);
220 		Stoichiometry stoichiometry2 = biopaxModel.addNew(Stoichiometry.class, "ST2");
221 		stoichiometry2.setPhysicalEntity(p2);
222 		stoichiometry2.setStoichiometricCoefficient(1);
223 
224 		ProteinReference pr3 = biopaxModel.addNew(ProteinReference.class, ID_PROTEIN_REFERENCE_3);
225 		pr3.setStandardName("Furin");
226 		p3.setEntityReference(pr3);
227 		Stoichiometry stoichiometry3 = biopaxModel.addNew(Stoichiometry.class, "ST3");
228 		stoichiometry3.setPhysicalEntity(p3);
229 		stoichiometry3.setStoichiometricCoefficient(1);
230 
231 		// Create a reaction involving the three proteins
232 		BiochemicalReaction r = biopaxModel.addNew(BiochemicalReaction.class, "r1");
233 		r.addLeft(p1);
234 		r.addRight(p2);
235 
236 		Control c = biopaxModel.addNew(Catalysis.class, "cat1");
237 		c.setControlType(ControlType.ACTIVATION);
238 		c.addControlled(r);
239 
240 
241 		// Write out the owl file
242 		try
243 		{
244 			System.out.println("test");
245 			File f = new File(getClass().getClassLoader()
246 					.getResource("").getPath() + File.separator + "test.owl");
247 			FileOutputStream anOutputStream = new FileOutputStream(f);
248 			outputModel(biopaxModel, anOutputStream);
249 		}
250 		catch (Exception e)
251 		{
252 			e.printStackTrace();
253 		}
254

297     public void readWriteReadCheckModel(SimpleIOHandler io) throws IOException
298     {
299         // Read
300 		Model model = getL3Model(io);
301 
302         // Write
303         ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
304         io.convertToOWL(model, outputStream);
305         outputStream.flush();
306 
307         // Read
308         ByteArrayInputStream inputStream = new ByteArrayInputStream(outputStream.toByteArray());
309         outputStream.close();
310         Model newModel = io.convertFromOWL(inputStream);
311         assertTrue(newModel != null);
312         inputStream.close();
313 
314         // Let's do a simple check to see if whether new model is OK
315         for(BioPAXElement bpe: model.getObjects())
316         {
317             assertTrue(newModel.containsID(bpe.getUri()));
318         }
319     }

337         BioPAXIOHandler io = new SimpleIOHandler(); //auto-detects level
338

356         outputStream.close();
