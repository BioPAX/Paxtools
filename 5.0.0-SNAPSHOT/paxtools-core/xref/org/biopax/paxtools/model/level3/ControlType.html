4   /**
5    * Defines the nature of the control relationship between the controller and the controlled entities.
6   */
7   public enum ControlType
8   {
9   	/**
10  	 * General activation. Compounds that activate the specified enzyme activity by an unknown mechanism. The
11  	 * mechanism is defined as unknown, because either the mechanism has yet to be elucidated in the experimental
12  	 * literature, or the paper(s) curated thus far do not define the mechanism, and a full literature search has
13  	 * yet to be performed.
14  	 */
15  	ACTIVATION,
16  	/**
17  	 * General inhibition. Compounds that inhibit the specified enzyme activity by an unknown mechanism. The
18  	 * mechanism is defined as unknown, because either the mechanism has yet to be elucidated in the experimental
19  	 * literature, or the paper(s) curated thus far do not define the mechanism, and a full literature search has
20  	 * yet to be performed.
21  	 */
22  	INHIBITION,
23  
24  	/**
25  	 * @deprecated LEVEL 1 workaround
26  	 */
27  	ACTIVATION_UNKMECH,
28  	/**
29  	 * @deprecated LEVEL 1 workaround
30  	 */
31  	INHIBITION_UNKMECH,
32  	/**
33  	 * Allosteric inhibitors decrease the specified enzyme activity by binding reversibly to the enzyme and
34  	 * inducing a conformational change that decreases the affinity of the enzyme to its substrates without
35  	 * affecting its VMAX. Allosteric inhibitors can be competitive or noncompetitive inhibitors, therefore,
36  	 * those inhibition categories can be used in conjunction with this category.
37  	 */
38  	INHIBITION_ALLOSTERIC,
39  	/**
40  	 * Competitive inhibitors are compounds that competitively inhibit the specified enzyme activity by binding
41  	 * reversibly to the enzyme and preventing the substrate from binding. Binding of the inhibitor and substrate
42  	 * are mutually exclusive because it is assumed that the inhibitor and substrate can both bind only to the free
43  	 * enzyme. A competitive inhibitor can either bind to the active site of the enzyme,
44  	 * directly excluding the substrate from binding there, or it can bind to another site on the enzyme,
45  	 * altering the conformation of the enzyme such that the substrate can not bind to the active site.
46  	 */
47  	INHIBITION_COMPETITIVE,
48  	/**
49  	 * Irreversible inhibitors are compounds that irreversibly inhibit the specified enzyme activity by binding to
50  	 * the enzyme and dissociating so slowly that it is considered irreversible. For example, alkylating agents,
51  	 * such as iodoacetamide, irreversibly inhibit the catalytic activity of some enzymes by modifying cysteine
52  	 * side chains.
53  	 */
54  	INHIBITION_IRREVERSIBLE,
55  	/**
56  	 * Noncompetitive inhibitors are compounds that noncompetitively inhibit the specified enzyme by binding
57  	 * reversibly to both the free enzyme and to the enzyme-substrate complex. The inhibitor and substrate may be
58  	 * bound to the enzyme simultaneously and do not exclude each other. However, only the enzyme-substrate complex
59  	 * (not the enzyme-substrate-inhibitor complex) is catalytically active.
60  	 */
61  	INHIBITION_NONCOMPETITIVE,
62  	/**
63  	 * Compounds that inhibit the specified enzyme activity by a mechanism that has been characterized,
64  	 * but that cannot be clearly classified as irreversible, competitive, noncompetitive, uncompetitive,
65  	 * or allosteric.
66  	 */
67  	INHIBITION_OTHER,
68  	/**
69  	 * Uncompetitive inhibitors are compounds that uncompetitively inhibit the specified enzyme activity by binding
70  	 * reversibly to the enzyme-substrate complex but not to the enzyme alone.
71  	 */
72  	INHIBITION_UNCOMPETITIVE,
73  	/**
74  	 * Nonallosteric activators increase the specified enzyme activity by means other than allosteric.
75  	 */
76  	ACTIVATION_NONALLOSTERIC,
77  	/**
78  	 * Allosteric activators increase the specified enzyme activity by binding reversibly to the enzyme and
79  	 * inducing a conformational change that increases the affinity of the enzyme to its substrates without
80  	 * affecting its VMAX.
81  	 */
82  	ACTIVATION_ALLOSTERIC
83  }
