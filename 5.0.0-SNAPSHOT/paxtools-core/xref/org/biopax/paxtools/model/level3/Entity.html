1   package org.biopax.paxtools.model.level3;
2   
3   import java.util.Set;
4   
5   /**
6    * This class represents a discrete biological unit used when describing
7    * pathways. This is the root class for all biological concepts in the ontology,
8    * which include pathways, interactions and physical entities. As the most
9    * abstract class in the ontology, instances of the entity class should never be
10   * created. Instead, more specific classes should be used. Synonyms: thing,
11   * object, bioentity.
12   */
13  public interface Entity extends Level3Element, Observable, Named
14  {
15  	// --------------------- ACCESORS and MUTATORS---------------------
16  
17  	// Property AVAILABILITY
18  
19  	/**
20  	 * The contents of this set can be modified but semantic consistency is not
21  	 * guaranteed. Using {@link #addAvailability(String)} and {@link
22  	 * #removeAvailability(String)} is recommended.
23  	 * @return a set of  strings describing the availability of this data (e.g. a
24  	 *         copyright statement).
25  	 */
26  	public Set<String> getAvailability();
27  
28  
29  	/**
30  	 * This method adds the given text to the avaialability set.
31  	 * @param availability a string describing the availability of this data
32  	 * (e.g. a copyright statement).
33  	 */
34  	public void addAvailability(String availability);
35  
36  	/**
37  	 * This method removes the given text from the avaialability set.
38  	 * @param availability a string describing the availability of this data
39  	 * (e.g. a copyright statement).
40  	 */
41  	public void removeAvailability(String availability);
42  
43  
44  	// Property DATA-SOURCE
45  
46  	/**
47  	 * This method returns a set of free text descriptions of the source of this
48  	 * data, e.g. a database or person name. This property should be used to
49  	 * describe the source of the data. This is meant to be used by databases that
50  	 * export their data to the BioPAX format or by systems that are integrating
51  	 * data from multiple sources. The granularity of use (specifying the data
52  	 * source in many or few instances) is up to the user. It is intended that
53  	 * this property report the last data source, not all data sources that the
54  	 * data has passed through from creation.
55  	 * 
56  	 * The contents of this set can be modified but semantic consistency is not
57  	 * guaranteed. Using {@link #addDataSource(Provenance)} and {@link #removeDataSource(Provenance)} is
58  	 * recommended.
59  	 * @return a set of free text descriptions of the source of this data, e.g. a
60  	 *         database or person name.
61  	 */
62  	public Set<Provenance> getDataSource();
63  
64  
65  	/**
66  	 * This method adds the given value to the DATA_SOURCE set.
67  	 * @param dataSource a free text description of the source of this data,
68  	 * e.g. a database or person name.
69  	 */
70  	public void addDataSource(Provenance dataSource);
71  
72  	/**
73  	 * This method removes the given value from the DATA_SOURCE set.
74  	 * @param dataSource a free text description of the source of this data,
75  	 * e.g. a database or person name.
76  	 */
77  	public void removeDataSource(Provenance dataSource);
78  
79  
80  
81  
82  	/**
83  	 * This method  returns the interaction that this entity/pep takes part in.
84  	 * Contents of this set should not be modified.
85  	 * Reverse of {@link org.biopax.paxtools.model.level3.Interaction#getParticipant()}
86  	 * @return a set of interactions that
87  	 */
88  	public Set<Interaction> getParticipantOf();
89  
90  }
