3   /**
4    * Tagger interface for protein, dna and rna entities
5    */
6   public interface SequenceEntityReference extends EntityReference
7   {
8   
9   	/**
10  	 * An organism, e.g. 'Homo sapiens'. This is the organism that the entity is found in. Pathways may
11  	 * not have an organism associated with them, for instance, reference pathways from KEGG.
12  	 * Sequence-based entities (DNA, protein, RNA) may contain an xref to a sequence database that
13  	 * contains organism information, in which case the information should be consistent with the value
14  	 * for ORGANISM.
15  	 *
16  	 * @return the organism for this gene.
17  	 */
18  	BioSource getOrganism();
19  
20  	/**
21  	 * An organism, e.g. 'Homo sapiens'. This is the organism that the entity is found in. Pathways may
22  	 * not have an organism associated with them, for instance, reference pathways from KEGG.
23  	 * Sequence-based entities (DNA, protein, RNA) may contain an xref to a sequence database that
24  	 * contains organism information, in which case the information should be consistent with the value
25  	 * for ORGANISM.
26  	 *
27  	 * @param source new organism for this gene
28  	 */
29  	void setOrganism(BioSource source);
30  
31  	/**
32  	 * Polymer sequence in uppercase letters. For DNA, usually A,C,G,T letters representing the
33  	 * nucleosides of adenine, cytosine, guanine and thymine, respectively; for RNA, usually A, C, U,
34  	 * G; for protein, usually the letters corresponding to the 20 letter IUPAC amino acid code.
35  	 * @return sequence - primary structure of a biopolymer.
36  	 */
37  	String getSequence();
38  
39  	/**
40  	 * Polymer sequence in uppercase letters. For DNA, usually A,C,G,T letters representing the
41  	 * nucleosides of adenine, cytosine, guanine and thymine, respectively; for RNA, usually A, C, U,
42  	 * G; for protein, usually the letters corresponding to the 20 letter IUPAC amino acid code.
43  	 * @param sequence Polymer sequence in uppercase letters.
44  	 */
45  	void setSequence(String sequence);
46  
47  
48  }
