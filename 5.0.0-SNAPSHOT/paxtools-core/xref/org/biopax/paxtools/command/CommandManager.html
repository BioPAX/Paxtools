1   package org.biopax.paxtools.command;
2   
3   import org.biopax.paxtools.controller.PropertyEditor;
4   import org.biopax.paxtools.model.BioPAXElement;
5   import org.biopax.paxtools.model.Model;
6   
7   import java.util.Set;
8   import java.util.Stack;
9   
10  /**
11   * This is a class for manipulating a BioPAX model via undoable commands
12   */
13  
14  public class CommandManager
15  {
16  	private Stack<Command> undoStack = new Stack<Command>();
17  	private Stack<Command> redoStack = new Stack<Command>();
18  	private Model model;
19  
20  
21  	public CommandManager(Model model)
22  	{
23  		this.model = model;
24  	}
25  
26  	public void addProperty(BioPAXElement bpe, PropertyEditor editor, Object value)
27  	{
28  		recordAndRun(editor.isMultipleCardinality() ?
29  		             new PropertyAddCommand(bpe, editor, value) :
30  		             new PropertySetCommand(bpe, editor, value));
31  
32  	}
33  
34  
35  	public void removeProperty(BioPAXElement bpe, PropertyEditor editor, Object value)
36  	{
37  		recordAndRun(new PropertyRemoveCommand(bpe, editor, value));
38  	}
39  
40  
41  	public void addObjects(Set<BioPAXElement> bpes)
42  	{
43  		recordAndRun(new AddCommand(model, bpes));
44  	}
45  
46  	public void removeObjects(Set<BioPAXElement> bpes)
47  	{
48  		recordAndRun(new RemoveCommand(model, bpes));
49  	}
50  
51  	private void recordAndRun(Command command)
52  	{
53  		command.redo();
54  		undoStack.add(command);
55  		if (!redoStack.isEmpty())
56  		{
57  			redoStack.clear();
58  		}
59  	}
60  
61  	public void undo()
62  	{
63  		Command command = undoStack.pop();
64  		command.undo();
65  		redoStack.add(command);
66  	}
67  
68  	public void redo()
69  	{
70  		Command command = redoStack.pop();
71  		command.redo();
72  		undoStack.add(command);
73  
74  	}
75  
76  
77  	public boolean canUndo()
78  	{
79  		return !undoStack.isEmpty();
80  	}
81  
82  	public boolean canRedo()
83  	{
84  		return !redoStack.isEmpty();
85  	}
86  }
