3   import java.util.Set;

5   
6   /**
7    * A FilterSet that filters based on the class. For example, it is useful for creating a Set of Proteins from a
8    * Set of BioPAXElements.
9    * @param <E> Superset's type e.g. BioPAXElement
10   * @param <F> Subset's type e.g. Protein. F must extend E
11   */
12  public class ClassFilterSet<E, F extends E> extends AbstractFilterSet<E, F>
13  {
14  // ------------------------------ FIELDS ------------------------------
15  protected Class<F> filterClass;
16  
17  	// --------------------------- CONSTRUCTORS ---------------------------
18  
19  	public ClassFilterSet(Set<? extends E> baseSet,Class<F> filterClass)
20  	{
21  		super(baseSet);
22  		this.filterClass = filterClass;
23  
24  	}
25  
26  	/**
27  	 * This filter method implementation filters based on isInstance function.
28  	 * @param value Object to be tested
29  	 * @return true if value is instance of F.
30  	 */
31  	@Override public boolean filter(E value)
32  	{
33  		return filterClass.isInstance(value);
34  	}
35  
36  }
