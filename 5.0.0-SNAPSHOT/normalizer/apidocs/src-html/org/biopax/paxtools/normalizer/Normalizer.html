054        /**
055         * Normalizes BioPAX OWL data and returns
056         * the result as BioPAX OWL (string).
057         * 
058         * This public method is actually intended to use 
059         * outside the BioPAX Validator framework, with not too large models.
060         * 
061         * @param biopaxOwlData RDF/XML BioPAX content string
062         * @return normalized BioPAX RDF/XML
063         * @deprecated this method will fail if the data exceeds ~1Gb (max UTF8 java String length)
064         */
065        public String normalize(String biopaxOwlData) {
066                
067                if(biopaxOwlData == null || biopaxOwlData.length() == 0) 
068                        throw new IllegalArgumentException("no data. " + description);
069                
070                // quick-fix for older BioPAX L3 version (v0.9x) property 'taxonXref' (range: BioSource)
071                biopaxOwlData = biopaxOwlData.replaceAll("taxonXref","xref");
072                
073                // build the model
074                Model model = null;
075                try {
076                        model = biopaxReader.convertFromOWL(
077                                new ByteArrayInputStream(biopaxOwlData.getBytes("UTF-8")));
078                } catch (UnsupportedEncodingException e) {
079                        throw new IllegalArgumentException("Failed! " + description, e);
080                }
081                
082                if(model == null) {
083                        throw new IllegalArgumentException("Failed to create Model! " 
084                                        + description);
085                }
086                
087                // auto-convert to Level3 model
088                if (model.getLevel() != BioPAXLevel.L3) {
089                        log.info("Converting model to BioPAX Level3...");
090                        model = (new LevelUpgrader()).filter(model);
091                }
092                
093                normalize(model); // L3 only!
094                
095                // return as BioPAX OWL
096                return convertToOWL(model);
097        }

100        /**
101         * Normalizes all xrefs (not unification xrefs only) to help normalizing/merging other objects, 
102         * and also because some of the original xref URIs ("normalized")
103         * are in fact to be used for other biopax types (e.g., CV or ProteinReference); therefore
104         * this method will replace URI also for "bad" xrefs, i.e., those with empty/illegal 'db' or 'id' values.
105         * 
106         * @param model biopax model to update
107         */
108        private void normalizeXrefs(Model model) {
109                
110                final NormalizerMap map = new NormalizerMap(model);
111                final String xmlBase = getXmlBase(model); //current base, the default or model's one, if set.
112                
113                // use a copy of the xrefs set (to avoid concurrent modif. exception)
114                Set<? extends Xref> xrefs = new HashSet<Xref>(model.getObjects(Xref.class));
115                for(Xref ref : xrefs)
116                {
117                        //skip not well-defined ones (incl. PublicationXrefs w/o db/id - won't normalize)
118                        if(ref.getDb() == null || ref.getId() == null)
119                                continue;
120
121                        ref.setDb(ref.getDb().toLowerCase()); //set lowercase
122                        String idPart = ref.getId();
123
124                        if(ref instanceof RelationshipXref) {
125                                // only normalize (replace URI of) RXs that could potentially clash with other RX, CVs, ERs;
126                                // skip, won't bother, for all other RXs...
127                                if(!ref.getUri().startsWith("http://identifiers.org/"))
128                                        continue;
129
130                                //RXs might have the same db, id but different rel. type.
131                                RelationshipTypeVocabulary cv = ((RelationshipXref) ref).getRelationshipType();
132                                if(ref.getIdVersion()!=null) {
133                                        idPart += "_" + ref.getIdVersion();
134                                }
135                                if(cv != null && !cv.getTerm().isEmpty()) {
136                                        idPart += "_" + StringUtils.join(cv.getTerm(), '_').toLowerCase();
137                                }
138                        }
139                        else if(ref instanceof UnificationXref) {
140                                // first, try to normalize the db name (using MIRIAM EBI registry)
141                                try {
142                                        ref.setDb(MiriamLink.getName(ref.getDb()).toLowerCase());
143                                } catch (IllegalArgumentException e) {
144                                        // - unknown/unmatched db name (normalize using defaults)
145                                        if(ref.getIdVersion()!=null) {
146                                                idPart += "_" + ref.getIdVersion();
147                                        }
148                                        map.put(ref, Normalizer.uri(xmlBase, ref.getDb(), idPart, ref.getModelInterface()));
149                                        continue; //shortcut (non-standard db name)
150                                }
151                        
152                                // a hack for uniprot/isoform xrefs
153                                if (ref.getDb().startsWith("uniprot")) {
154                                        //auto-fix (guess) for possibly incorrect db/id (can be 'uniprot isoform' with/no idVersion, etc..)
155                                        if (isValidDbId("uniprot isoform", ref.getId())
156                                                        && ref.getId().contains("-")) //the second condition is important
157                                        {       //then it's certainly an isoform id; so - fix the db name
158                                                ref.setDb("uniprot isoform"); //fix the db
159                                        }
160                                        else {
161                                                //id does not end with "-\\d+", i.e., not a isoform id
162                                                //(idVersion is a different thing, but id db was "uniprot isoform" they probably misused idVersion)
163                                                if(ref.getDb().equals("uniprot isoform"))
164                                                {
165                                                        if(ref.getIdVersion() != null && ref.getIdVersion().matches("^\\d+$"))
166                                                                idPart = ref.getId()+"-"+ref.getIdVersion(); //guess, by idVersion, they actually meant isoform
167                                                        if(isValidDbId(ref.getDb(), idPart)) {
168                                                                ref.setId(idPart); //moved the isoform # to the ID
169                                                                ref.setIdVersion(null);
170                                                        }
171                                                        else if(!isValidDbId(ref.getDb(), ref.getId())) {
172                                                                //certainly not isoform (might not even uniprot, but try...)
173                                                                ref.setDb("uniprot knowledgebase"); //guess, fix
174                                                        }
175                                                        idPart = ref.getId();
176                                                }
177                                        }
178                                }
179                        }
180
181                        // shelve it for URI replace
182                        map.put(ref, Normalizer.uri(xmlBase, ref.getDb(), idPart, ref.getModelInterface()));
183                }
184                
185                // execute replace xrefs
186                map.doSubs();
187        }

197        /**
198         * Consistently generates a new BioPAX element URI 
199         * using given URI namespace (xml:base), BioPAX class, 
200         * and two different identifiers (at least one is required).
201         * Miriam registry is used to get the standard db name and 
202         * identifiers.org URI, if possible, only for relationship type vocabulary, 
203         * publication xref, and entity reference types.
204         * 
205         * @param xmlBase xml:base (common URI prefix for a BioPAX model), case-sensitive
206         * @param dbName a bio data collection name or synonym, case-insensitive
207         * @param idPart optional (can be null), e.g., xref.id, case-sensitive
208         * @param type BioPAX class
209         * @return URI
210         * @throws IllegalArgumentException if either type is null or both 'dbName' and 'idPart' are all nulls.
211         */
212        public static String uri(final String xmlBase, 
213                String dbName, final String idPart, Class<? extends BioPAXElement> type)
214        {
215                if(type == null || (dbName == null && idPart == null))
216                        throw new IllegalArgumentException("'Either type' is null, or both dbName and idPart are nulls.");
217                        
218                // try to find a standard URI, if exists, for a publication xref, 
219                // or at least a standard name:
220                if (dbName != null) {
221                        try {
222                                // try to get the preferred/standard name
223                                // for any type, for consistency
224                                dbName = MiriamLink.getName(dbName);
225                                // a shortcut: a standard and resolvable URI exists for some BioPAX types
226                                if ((type.equals(PublicationXref.class) && "pubmed".equalsIgnoreCase(dbName))
227                                        || type.equals(RelationshipTypeVocabulary.class)
228                                        || ProteinReference.class.isAssignableFrom(type)
229                                        || SmallMoleculeReference.class.isAssignableFrom(type)
230                                        || (type.equals(BioSource.class) && "taxonomy".equalsIgnoreCase(dbName) && idPart.matches("^\\d+$"))
231                                )
232                                {       //get the standard URI and quit (success), or fail and continue making a new URI below...
233                                        return MiriamLink.getIdentifiersOrgURI(dbName, idPart);
234                                }
235                        } catch (IllegalArgumentException e) {
236                                log.info(String.format("uri(for a %s): db:%s, id:%s are not standard; %s)",
237                                                type.getSimpleName(), dbName, idPart, e.getMessage()));
238                        }
239                }
240
241                // If not returned above this point - no standard URI (Identifiers.org) was found -
242                // then let's consistently build a new URI from args, anyway, the other way around:
243                
244                StringBuilder sb = new StringBuilder();         
245                if (dbName != null) //lowercase for consistency
246                        sb.append(dbName.toLowerCase());        
247                
248                if (idPart != null) {
249                        if (dbName != null) sb.append("_");
250                        sb.append(idPart);
251                }
252                
253                String localPart = sb.toString();
254                String strategy = System.getProperty(PROPERTY_NORMALIZER_URI_STRATEGY, VALUE_NORMALIZER_URI_STRATEGY_MD5);
255                if(VALUE_NORMALIZER_URI_STRATEGY_SIMPLE.equals(strategy) || Xref.class.isAssignableFrom(type))
256                //i.e., for xrefs, always use the simple URI strategy (makes them human-readable)
257                {
258                        //simply replace "unsafe" symbols with underscore (some uri clashes might be possible but rare...)
259                        localPart = localPart.replaceAll("[^-\\w]", "_");
260                }
261                else
262                {
263                        //replace the local part with its md5 sum string (32-byte)
264                        localPart = ModelUtils.md5hex(localPart);
265                }
266                
267                // create URI using the xml:base and digest of other values:
268                return ((xmlBase != null) ? xmlBase : "") + type.getSimpleName() + "_" + localPart;
269        }

290        private void fixDisplayName(Model model) {
291                log.info("Trying to auto-fix 'null' displayName...");
292                // where it's null, set to the shortest name if possible
293                for (Named e : model.getObjects(Named.class)) {
294                        if (e.getDisplayName() == null) {
295                                if (e.getStandardName() != null) {
296                                        e.setDisplayName(e.getStandardName());
297                                        log.info(e + " displayName auto-fix: "
298                                                        + e.getDisplayName() + ". " + description);
299                                } else if (!e.getName().isEmpty()) {
300                                        String dsp = e.getName().iterator().next();
301                                        for (String name : e.getName()) {
302                                                if (name.length() < dsp.length())
303                                                        dsp = name;
304                                        }
305                                        e.setDisplayName(dsp);
306                                        log.info(e + " displayName auto-fix: " + dsp
307                                                + ". " + description);
308                                }
309                        }
310                }
311                // if required, set PE name to (already fixed) ER's name...
312                for(EntityReference er : model.getObjects(EntityReference.class)) {
313                        for(SimplePhysicalEntity spe : er.getEntityReferenceOf()) {
314                                if(spe.getDisplayName() == null || spe.getDisplayName().trim().length() == 0) {
315                                        if(er.getDisplayName() != null && er.getDisplayName().trim().length() > 0) {
316                                                spe.setDisplayName(er.getDisplayName());
317                                        }
318                                }
319                        }
320                }
321        }

331        private Collection<UnificationXref> getUnificationXrefsSorted(XReferrable r) {
332
333                Collection<UnificationXref> urefs = new TreeSet<UnificationXref>(
334                                new Comparator<UnificationXref>() {
335                                        public int compare(UnificationXref o1, UnificationXref o2) {
336                                                String s1 = o1.getDb() + o1.getId();
337                                                String s2 = o2.getDb() + o2.getId();
338                                                return s1.compareTo(s2);
339                                        }
340                                }
341                );
342
343                for(UnificationXref ux : new ClassFilterSet<Xref,UnificationXref>(r.getXref(), UnificationXref.class))
344                {
345                        if(ux.getDb() != null && ux.getId() != null) {
346                                urefs.add(ux);
347                        } 
348                }
349                
350                return urefs;
351        }

354        /**
355         * Finds one preferred unification xref, if possible.
356         * Preferred db values are:
357         * "ncbi gene" - for NucleicAcidReference and the sub-classes;
358         * "uniprot" or "refseq" for ProteinReference;
359         * "chebi" - for SmallMoleculeReference;
360         * 
361         * @param bpe BioPAX object that can have xrefs
362         * @return the "best" first unification xref 
363         */
364        private UnificationXref findPreferredUnificationXref(XReferrable bpe)
365        {
366                UnificationXref toReturn = null;
367
368                Collection<UnificationXref> orderedUrefs = getUnificationXrefsSorted(bpe);
369
370                //use preferred db prefix for different type of ER
371                if(bpe instanceof ProteinReference) {
372                        // prefer xref having db starts with "uniprot" (preferred) or "refseq";
373                        toReturn = findSingleUnificationXref(orderedUrefs, "uniprot");
374                        // if null, next try refseq.
375                        if(toReturn==null)
376                                toReturn = findSingleUnificationXref(orderedUrefs, "refseq");
377                } else if(bpe instanceof SmallMoleculeReference) {
378                        // - "chebi", then  "pubchem";
379                        toReturn = findSingleUnificationXref(orderedUrefs, "chebi");
380                        if(toReturn==null)
381                                toReturn = findSingleUnificationXref(orderedUrefs, "pubchem");
382                } else if(bpe instanceof NucleicAcidReference) {
383                        //that includes NucleicAcidRegionReference, etc. sub-classes;
384                        toReturn = findSingleUnificationXref(orderedUrefs, "ncbi gene");
385                        if(toReturn==null)
386                                toReturn = findSingleUnificationXref(orderedUrefs, "entrez");
387                } else {
388                        //for other XReferrable types (BioSource or ControlledVocabulary)
389                        //use if there's only one xref (return null if many)
390                        if(orderedUrefs.size()==1)
391                                toReturn = orderedUrefs.iterator().next();
392                }
393                
394                return toReturn;
395        }

397        private UnificationXref findSingleUnificationXref(Collection<UnificationXref> uXrefs, String dbStartsWith) {
398                UnificationXref ret = null;
399
400                for(Xref x : uXrefs) {
401                        if(x instanceof UnificationXref && x.getId() != null && x.getDb() != null
402                                        && x.getDb().toLowerCase().startsWith(dbStartsWith))
403                        {
404                                if(ret == null)
405                                {
406                                        ret = (UnificationXref)x;
407                                }
408                                else if (ret.getDb().equalsIgnoreCase(x.getDb()) && !ret.getId().equals(x.getId())) {
409                                        //exactly the same kind (same 'db') xref is found;
410                                        //we can't test here if several IDs map to the same thing;
411                                        //so, give up - for safety - and return null.
412                                        ret = null;
413                                        break;
414                                }
415                        }
416                }
417
418                return ret;
419        }

422        /**
423         * BioPAX normalization 
424         * (modifies the original Model)
425         * 
426         * @param model BioPAX model to normalize
427         * @throws NullPointerException if model is null
428         * @throws IllegalArgumentException if model is not Level3 BioPAX
429         */
430        public void normalize(Model model) {
431                
432                if(model.getLevel() != BioPAXLevel.L3)
433                        throw new IllegalArgumentException("Not Level3 model. " +
434                                "Consider converting it first (e.g., with the PaxTools).");
435                
436                //if set, update the xml:base
437                if(xmlBase != null && !xmlBase.isEmpty())
438                        model.setXmlBase(xmlBase);
439
440                // Normalize/merge xrefs, first, and then CVs
441                // (also because some of original xrefs might have "normalized" URIs 
442                // that, in fact, must be used for other biopax types, such as CV or ProteinReference)
443                log.info("Normalizing xrefs..." + description);
444                normalizeXrefs(model);
445                
446                // fix displayName where possible
447                if(fixDisplayName) {
448                        log.info("Normalizing display names..." + description);
449                        fixDisplayName(model);
450                }
451                        
452                log.info("Normalizing CVs..." + description);
453                normalizeCVs(model);
454                
455                //normalize BioSource objects (better, as it is here, go after Xrefs and CVs)
456                log.info("Normalizing organisms..." + description);
457                normalizeBioSources(model);
458
459                // auto-generate missing entity references:
460                for(SimplePhysicalEntity spe : new HashSet<SimplePhysicalEntity>(model.getObjects(SimplePhysicalEntity.class))) {
461                        //it skips if spe has entityReference or memberPE already
462                        ModelUtils.addMissingEntityReference(model, spe);
463                }
464
465                log.info("Normalizing entity references..." + description);
466                normalizeERs(model);
467                
468                // find/add lost (in replace) children
469                log.info("Repairing..." + description);
470                model.repair(); // it does not remove dangling utility class objects (can be done separately, later, if needed)
471                
472                log.info("Optional tasks (reasoning)..." + description);
473        }

476        private void normalizeCVs(Model model) {
477                
478                NormalizerMap map = new NormalizerMap(model);
479                
480                // process ControlledVocabulary objects (all sub-classes)
481                for(ControlledVocabulary cv : model.getObjects(ControlledVocabulary.class))
482                {
483                        //it does not check/fix the CV terms though (but a validation rule can do if run before the normalizer)...
484                        UnificationXref uref = findPreferredUnificationXref(cv); //usually, there's only one such xref
485                        if (uref != null) {
486                                // so let's generate a consistent URI
487                                map.put(cv, uri(xmlBase, uref.getDb(), uref.getId(), cv.getModelInterface()));
488                        } else if(!cv.getTerm().isEmpty()) {
489                                map.put(cv, uri(xmlBase, null, cv.getTerm().iterator().next(), cv.getModelInterface()));
490                        } else log.info("Cannot normalize " + cv.getModelInterface().getSimpleName() 
491                                + " : no unification xrefs nor terms found in " + cv.getUri()
492                                + ". " + description);
493                } 
494                
495                // replace/update elements in the model
496                map.doSubs();
497        }

500        private void normalizeBioSources(Model model) {
501                
502                NormalizerMap map = new NormalizerMap(model);
503                
504                for(BioSource bs : model.getObjects(BioSource.class))
505                {
506                        UnificationXref uref = findPreferredUnificationXref(bs);
507                        //normally, the xref db is 'Taxonomy' (or a valid synonym)
508                        if (uref != null
509                                && (uref.getDb().toLowerCase().contains("taxonomy") || uref.getDb().equalsIgnoreCase("newt")))
510                        {       
511                                String  idPart = uref.getId();
512
513                                //tissue/cellType terms can be added below:
514                                if(bs.getTissue()!=null && !bs.getTissue().getTerm().isEmpty()) 
515                                        idPart += "_" + bs.getTissue().getTerm().iterator().next();
516                                if(bs.getCellType()!=null && !bs.getCellType().getTerm().isEmpty()) 
517                                        idPart += "_" + bs.getCellType().getTerm().iterator().next();
518                                
519                                String uri = (idPart.equals(uref.getId()) //- no tissue or celltype were attached
520                                                && idPart.matches("^\\d+$")) //- is positive integer id
521                                        ? uri(xmlBase, uref.getDb(), idPart, BioSource.class)
522                                                : "http://identifiers.org/taxonomy/" + idPart;
523                                        // the latter is intentionally invalid identifiers.org/taxonomy URI - good (and important) for merging
524                                map.put(bs, uri);
525
526                        } else 
527                                log.debug("Won't normalize BioSource" 
528                                        + " : no taxonomy unification xref found in " + bs.getUri()
529                                        + ". " + description);
530                } 
531                
532                map.doSubs();
533        }

535        private void normalizeERs(Model model) {
536                
537                NormalizerMap map = new NormalizerMap(model);
538                
539                // process the rest of utility classes (selectively though)
540                for (EntityReference bpe : model.getObjects(EntityReference.class)) {
541                        
542                        //skip those with already normalized URIs
543                        if(bpe.getUri().startsWith("http://identifiers.org/")) {
544                                log.info("Skip already normalized: " + bpe.getUri());
545                                continue;
546                        }                       
547                        
548                        UnificationXref uref = findPreferredUnificationXref(bpe);
549                        if (uref != null) {
550                                // Create (with a new URI made from a unif. xref) 
551                                // and save the replacement object, if possible, 
552                                // but do not replace yet (will call doSubs later, for all).
553                                final String db = uref.getDb();
554                                final String id = uref.getId();
555                                // get the standard ID
556                                String uri = null;
557                                try { // make a new ID for the element
558                                        uri = MiriamLink.getIdentifiersOrgURI(db, id);
559                                } catch (Exception e) {
560                                        log.error("Cannot get a Miriam standard ID for " + bpe 
561                                                        + " (" + bpe.getModelInterface().getSimpleName()
562                                                        + ") " + ", using " + db + ":" + id 
563                                                        + ". " + e.getMessage());
564                                        return;
565                                }
566
567                                if(uri != null) {
568                                        map.put(bpe, uri);
569                                }       
570                        } else
571                                log.info("Cannot normalize EntityReference: "
572                                        + "no unification xrefs found in " + bpe.getUri()
573                                        + ". " + description);
574                }
575                
576                // replace/update elements in the model
577                map.doSubs();
578        }

581        /**
582         * Auto-generates standard and other names for the datasource
583         * from either its ID (if URN) or one of its existing names (preferably - standard name)
584         * 
585         * @param pro data source (BioPAX Provenance)
586         */
587        public static void autoName(Provenance pro) {
588                if(!(pro.getUri().startsWith("urn:miriam:") || pro.getUri().startsWith("http://identifiers.org/"))
589                                && pro.getName().isEmpty()) {
590                        log.info("Skipping: cannot normalize Provenance: " + pro.getUri());
591                }
592                else { // i.e., 'name' is not empty or ID is the URN
593                        final SortedSet<String> names = new TreeSet<String>();
594                        
595                        String key = null;
596                        if(pro.getUri().startsWith("urn:miriam:") || pro.getUri().startsWith("http://identifiers.org/")) {
597                                key = pro.getUri();
598                        } else if (pro.getStandardName() != null) {
599                                key = pro.getStandardName();
600                        } else {
601                                key = pro.getDisplayName(); // can be null
602                        }
603                        
604                        if (key != null) {
605                                try {
606                                        names.addAll(Arrays.asList(MiriamLink.getNames(key)));
607                                        pro.setStandardName(MiriamLink.getName(key));
608                                        // get the datasource description
609                                        String description = MiriamLink.getDataTypeDef(pro.getStandardName());
610                                        pro.addComment(description);
611                                } catch (IllegalArgumentException e) {
612                                        // ignore (then, names is still empty...)
613                                }
614                        } 
615                        
616                        // when the above failed (no match in Miriam), or key was null -
617                        if(names.isEmpty()) {
618                                // finally, trying to find all valid names for each existing one
619                                        for (String name : pro.getName()) {
620                                                try {
621                                                        names.addAll(Arrays.asList(MiriamLink.getNames(name)));
622                                                } catch (IllegalArgumentException e) {
623                                                        // ignore
624                                                }
625                                        }
626                                        // pick up the first name, get the standard name
627                                        if(!names.isEmpty())
628                                                pro.setStandardName(MiriamLink
629                                                        .getName(names.iterator().next()));
630                        }
631                        
632                        // and add all the synonyms if any
633                        for(String name : names)
634                                pro.addName(name);
635                        
636                        //set display name if not set (standard name is set already)
637                        if(pro.getDisplayName() == null)
638                                pro.setDisplayName(pro.getStandardName());                      
639                }
640        }

642        /**
643         * Converts BioPAX L1 or L2 RDF/XML string data to BioPAX L3 string.
644         *
645         * WARN: this is not for huge (larger than 1GB) BioPAX RDF/XML data
646         * due to use of (UTF-8) String and Byte Array internally.
647         * This can be and is used by online web apps, such as the
648         * BioPAX Validator.
649         *
650         * @param biopaxData String
651         * @return BioPAX Level3 RDF/XML string
652         */
653        public static String convertToLevel3(final String biopaxData) {
654                String toReturn = "";
655                
656                try {
657                        ByteArrayOutputStream os = new ByteArrayOutputStream();
658                        InputStream is = new ByteArrayInputStream(biopaxData.getBytes());
659                        SimpleIOHandler io = new SimpleIOHandler();
660                        io.mergeDuplicates(true);
661                        Model model = io.convertFromOWL(is);
662                        if (model.getLevel() != BioPAXLevel.L3) {
663                                log.info("Converting to BioPAX Level3... " + model.getXmlBase());
664                                model = (new LevelUpgrader()).filter(model);
665                                if (model != null) {
666                                        io.setFactory(model.getLevel().getDefaultFactory());
667                                        io.convertToOWL(model, os);
668                                        toReturn = os.toString();
669                                }
670                        } else {
671                                toReturn = biopaxData;
672                        }
673                } catch(Exception e) {
674                        throw new RuntimeException(
675                                "Cannot convert to BioPAX Level3", e);
676                }
677
678                return toReturn;
679        }

711        /**
712         * Helper class, to associate original 
713         * and new (either copy or existing) objects 
714         * within some biopax model and then execute 
715         * batch replace.
716         * 
717         * @author rodche
718         */
719        private static class NormalizerMap {
720
721                //a model to modify by replacing biopax objects
722                final Model model;
723                
724                // this is the biopax elements substitution map (old->new)
725                final Map<BioPAXElement,BioPAXElement> subs;
726                
727                //the next map is to make sure the subs.values()  all have different URIs
728                final Map<String,BioPAXElement> uriToSub;
729                
730                final ShallowCopy copier;
731                                
732                NormalizerMap(Model model) {
733                        subs = BPCollections.I.createMap();
734                        uriToSub = BPCollections.I.createMap();
735                        this.model = model;
736                        copier = new ShallowCopy();
737                }
738
739
740                /**
741                 * Creates (by URI) and saves the replacement object,
742                 * but does not replace yet (call doSubs to replace).
743                 * 
744                 * @param bpe
745                 * @param newUri
746                 */
747                void put(BioPAXElement bpe, String newUri)
748                {
749                        if(model.containsID(newUri)) {
750                                // will use existing original (model) object that has the new Uri
751                                map(bpe, model.getByID(newUri));
752                        } else if(uriToSub.containsKey(newUri)) {
753                                // re-use the new object that's already added to replace another original
754                                map(bpe, uriToSub.get(newUri));
755                        } else {
756                                // will use the object's shallow copy that gets new Uri
757                                BioPAXElement copy = copier.copy(bpe, newUri);
758                                map(bpe, copy);
759                        }
760                }               
761                
762                /**
763                 * Executes the batch replace - migrating  
764                 * to the normalized equivalent objects.
765                 */
766                void doSubs() {
767                        for(BioPAXElement e : subs.keySet()) {
768                                model.remove(e);
769                        }
770                        
771                        try {
772                                ModelUtils.replace(model, subs);
773                        } catch (Exception e) {
774                                log.error("Failed to replace BioPAX elements.", e);
775                                return;
776                        }
777                        
778                        for(BioPAXElement e : subs.values()) {
779                                if(!model.contains(e))
780                                        model.add(e);
781                        }
782                
783                        for(BioPAXElement e : model.getObjects()) {
784                                ModelUtils.fixDanglingInverseProperties(e, model);
785                        }
786                }
787
788                private void map(BioPAXElement bpe, BioPAXElement newBpe) {
789                        subs.put(bpe, newBpe);
790                        uriToSub.put(newBpe.getUri(), newBpe);
791                }
792        }
