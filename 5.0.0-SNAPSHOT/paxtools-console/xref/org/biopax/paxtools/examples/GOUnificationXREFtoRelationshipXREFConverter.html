20  /**
21   * User: Emek Demir Date: Jan 18, 2007 Time: 4:56:53 PM
22   *
23   * In this example we get all the unification xrefs in the model
24   * and check if they point to the Gene Ontology. If this is the case
25   * we convert them to relationship xrefs.
26   *
27   */
28  public class GOUnificationXREFtoRelationshipXREFConverter
29  {
30  	private static Logger log = LoggerFactory.getLogger(GOUnificationXREFtoRelationshipXREFConverter.class);
31     
32  	static BioPAXIOHandler reader = new SimpleIOHandler();
33  
34      //args - a space seperated list of owl files to be processed
35      public static void main(String[] args)
36  		throws IllegalAccessException, InvocationTargetException
37  	{
38          // Process all the args
39          for (String arg : args)
40          {
41              log.info(arg);
42              if (arg.toLowerCase().endsWith("owl"))
43              {
44                  try
45                  {
46                      processXrefs(arg);
47                  }
48                  catch (FileNotFoundException e)
49                  {
50                      e.printStackTrace();
51                  }
52              }
53  
54          }
55  	}
56  
57      /**
58       * Main conversion method. Demonstrates how to read and write a BioPAX
59       * model and accessing its objects.
60       * @param arg file name to be processed
61       */
62      private static void processXrefs(String arg) throws
63  		FileNotFoundException,
64  		IllegalAccessException,
65  		InvocationTargetException
66  	{
67  		//Read in the model
68          FileInputStream in =
69  			new FileInputStream(new File(arg));
70  		Model level2 =
71  			reader.convertFromOWL(in);
72  
73          //Get all unification xrefs.
74          Set<unificationXref> unis =
75  			level2.getObjects(unificationXref.class);
76  		//Create another set for avoiding concurrent modifications
77          Set<unificationXref> gos = new HashSet<unificationXref>();
78  
79          //Process all uni. refs
80          for (unificationXref uni : unis)
81  		{
82  			log.trace(uni.getDB());
83  			//Assuming DB is represented as "GO"
84              if (uni.getDB().equalsIgnoreCase("GO"))
85  			{
86  				//this it to avoid concurrent modification.
87  				log.info("scheduling " + uni.getUri());
88  				gos.add(uni);
89  
90  			}
91  		}
92          //Now we have a list of xrefs to be converted. Let's do it.
93          for (unificationXref go : gos)
94  		{
95  			convert(go, level2);
96  		}
97  		//And finally write out the file. We are done !
98          reader.convertToOWL(level2, new FileOutputStream(
99  			arg.substring(0, arg.lastIndexOf('.')) +
100 				"-converted.owl"));
101 	}
102 
103     /**
104      * This method converts the given unification xref to a relationship xref
105      * @param uni xref to be converted
106      * @param level2 model containing the xref
107      */
108     private static void convert(unificationXref uni, Model level2)
109 	{
110 		//We can not simply convert a class, so we need to remove the
111         //uni and insert a new relationship xref
112 
113         //First get all the objects that refers to this uni
114         Set<XReferrable> referrables =
115 			new HashSet<XReferrable>(uni.isXREFof());
116 
117         //Create the new relationship xref in the model.
118         relationshipXref relationshipXref =
119         	level2.addNew(relationshipXref.class, uni.getUri());
120 
121         //Copy the fields from uni
122 		relationshipXref.setCOMMENT(uni.getCOMMENT());
123 		relationshipXref.setDB(uni.getDB());
124 		relationshipXref.setDB_VERSION(uni.getDB_VERSION());
125 		relationshipXref.setID(uni.getID());
126 		relationshipXref.setID_VERSION(uni.getID_VERSION());
127 		relationshipXref.setRELATIONSHIP_TYPE(
128 			"http://www.biopax.org/paxtools/convertedGOUnificationXREF");
129 
130         //Create a link to the new xref from all the owners.
131         for (XReferrable referrable : referrables)
132 		{
133 			referrable.addXREF(relationshipXref);
134 		}
135 
136         //Remove the references to the old uni
137         for (XReferrable referrable : referrables)
138         {
139             referrable.removeXREF(uni);
140         }
141         //Now remove it from the model.
142         level2.remove(uni);
143 
144         //We are done!
145     }
146 }
