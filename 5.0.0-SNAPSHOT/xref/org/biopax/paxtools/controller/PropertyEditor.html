1   package org.biopax.paxtools.controller;
2   
3   import org.biopax.paxtools.model.BioPAXElement;
4   
5   import java.lang.reflect.Method;
6   import java.util.Set;
7   
8   /**
9   
10   */
11  public interface PropertyEditor<D extends BioPAXElement, R> extends PropertyAccessor<D,R>
12  {
13  	@Override String toString();
14  
15  	/**
16  	 * @return the add method.
17  	 */
18  	Method getAddMethod();
19  
20  	/**
21  	 * @return the get method
22  	 */
23  	Method getGetMethod();
24  
25  	/**
26  	 * @return the proterty name as a string
27  	 */
28  	String getProperty();
29  
30  	/**
31  	 * @return the remove method
32  	 */
33  	Method getRemoveMethod();
34  
35  	/**
36  	 * @return the set method
37  	 */
38  	Method getSetMethod();
39  
40  	/**
41  	 * Sets a maximum cardinality for a domain.
42  	 * @param domain domain on which restriction will be set
43  	 * @param max cardinality
44  	 * @see #isMultipleCardinality()
45  	 */
46  	void addMaxCardinalityRestriction(Class<? extends D> domain, int max);
47  
48  	/**
49  	 * Return the maximum cardinality that is defined for the property to which editor is belong.
50  	 * @param restrictedDomain domain to be checked for the cardinality
51  	 * @return an integer indicating the maximum cardinality
52  	 */
53  	Integer getMaxCardinality(Class<? extends D> restrictedDomain);
54  
55  	/**
56  	 * Gets the unknown <em>value</em>. In an object property or enumeration
57  	 * context a <em>value</em> is regarded to be unknown if it is null (unset);
58  	 * in a primitive property context it depends (can be e.g.,
59  	 * {@link org.biopax.paxtools.model.BioPAXElement#UNKNOWN_FLOAT})
60  	 * @return null or what it means that the property value is unknown
61  	 */
62  	R getUnknown();
63  
64  	/**
65  	 * Removes the <em>value</em> from the <em>bean</em> using the default removeMethod,
66  	 * if such method is defined (i.e., it's a multiple cardinality property),
67  	 * otherwise sets <em>unknown</em> value using {@link #setValueToBean(Object, org.biopax.paxtools.model.BioPAXElement)}
68  	 * (but only if )
69  	 *
70  	 * @param value to be removed from the bean
71  	 * @param bean bean from which the value is going to be removed
72  	 */
73  	void removeValueFromBean(R value, D bean);
74  
75  	/**
76  	 * Removes the <em>values</em> from the <em>bean</em>
77  	 * using the {@link #removeValueFromBean(Object, org.biopax.paxtools.model.BioPAXElement)}
78  	 * for each value in the set.
79  	 *
80  	 * @param values to be removed from the bean
81  	 * @param bean bean from which the value is going to be removed
82  	 */
83  	void removeValueFromBean(Set<R> values, D bean);
84  
85  	/**
86  	 * Sets the <em>value</em> to the <em>bean</em> using the default setMethod if
87  	 * <em>value</em> is not null.
88  	 * @param value to be set to the <em>bean</em>
89  	 * @param bean to which the <em>value</em> is to be set
90  	 */
91  	void setValueToBean(R value, D bean);
92  
93  	void setValueToBean(Set<R> values, D bean);
94  
95  	/**
96  	 * Returns the primary set method of the editor. It is the setMethod for a property of
97  	 * single cardinality, and the addMethod method for a property of multiple cardinality.
98  	 * @return the method to be primarily used for setting a value to an object.
99  	 */
100 	Method getPrimarySetMethod();
101 }
