196 	private static String getPrintable(long time)
197 	{
198 		int div = 1000 * 60 * 60;
199 		int hours = (int) (time / div);
200 		time %= div;
201 		div /= 60;
202 		int minutes = (int) (time / div);
203 		time %= div;
204 		div /= 60;
205 		int seconds = (int) (time / div);
206 
207 		String s = hours > 0 ? hours + "h, " : "";
208 		s += minutes > 0 ? minutes + "m, " : "";
209 		s += seconds + "s";
210 		return s;
211 	}

213 	private void confirmPresenceOfUbiques(Model model, Blacklist blacklist)
214 	{
215 		int present = 0;
216 		int absent = 0;
217 		for (String ubique : blacklist.getListed())
218 		{
219 			if (model.getByID(ubique) != null) present++;
220 			else
221 			{
222 				absent++;
223 			}
224 		}
225 		System.out.println("absent ubique  = " + absent);
226 		System.out.println("present ubique = " + present);
227 	}

239 	public static void generate(String inputModelFile, String ubiqueIDFile, String outputFile)
240 		throws IOException
241 	{
242 		SimpleIOHandler h = new SimpleIOHandler();
243 		Model model = h.convertFromOWL(new FileInputStream(inputModelFile));
244 
245 		List<SIFInteraction> sifs = new ArrayList<SIFInteraction>(generate(model,
246 			new Blacklist(ubiqueIDFile)));
247 
248 		Collections.sort(sifs);
249 
250 		BufferedWriter writer = new BufferedWriter(new FileWriter(outputFile));
251 
252 		for (SIFInteraction sif : sifs)
253 		{
254 			writer.write(sif + "\n");
255 		}
256 
257 		writer.close();
258 	}
