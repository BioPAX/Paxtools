104 	private Model excise(Model model, Set<BioPAXElement> result)
105 	{
106 		Completer c = new Completer(SimpleEditorMap.L3);
107 
108 		result = c.complete(result, model);
109 
110 		Cloner cln = new Cloner(SimpleEditorMap.L3, BioPAXLevel.L3.getDefaultFactory());
111 
112 		return cln.clone(model, result);
113 	}
114 
115 	protected static Set<BioPAXElement> findElements(Model model, String... ids)
116 	{
117 		Set<BioPAXElement> set = new HashSet<BioPAXElement>();
118 
119 		for (String id : ids)
120 		{
121 			BioPAXElement bpe = model.getByID(id);
122 
123 			if (bpe != null)
124 			{
125 				set.add(bpe);
126 			}
127 		}
128 		return set;
129 	}

165 	@Test
166 	public void testFilters()
167 	{
168 		Model model = handler.convertFromOWL(this.getClass().getResourceAsStream(
169 			"raf_map_kinase_cascade_reactome.owl"));
170 
171 		Set<BioPAXElement> source = findElements(model,
172 			"HTTP://WWW.REACTOME.ORG/BIOPAX/48887#PROTEIN2360_1_9606"); //MEK2
173 		Set<BioPAXElement> target = findElements(model,
174 			"HTTP://WWW.REACTOME.ORG/BIOPAX/48887#PROTEIN1631_1_9606"); //ERK1
175 
176 		// test organism filter
177 
178 		Filter f = new OrganismFilter(new String[]{"Homo sapiens"});
179 		Set<BioPAXElement> result = QueryExecuter.runPathsFromTo(
180 			source, target, model, LimitType.NORMAL, 2, f);
181 		assertTrue(!result.isEmpty());
182 
183 		f = new OrganismFilter(new String[]{"Non-existing organism"});
184 		result = QueryExecuter.runPathsFromTo(source, target, model, LimitType.NORMAL, 2, f);
185 		assertTrue(result.isEmpty());
186 
187 		
188 		f = new OrganismFilter(new String[]{"9606"});
189 		result = QueryExecuter.runPathsFromTo(
190 			source, target, model, LimitType.NORMAL, 2, f);
191 		assertTrue(!result.isEmpty());
192 		
193 		// test data source filter
194 
195 		f = new DataSourceFilter(new String[]{"Reactome"});
196 		result = QueryExecuter.runPathsFromTo(source, target, model, LimitType.NORMAL, 2, f);
197 		assertTrue(!result.isEmpty());
198 
199 		f = new DataSourceFilter(new String[]{"Some DB"});
200 		result = QueryExecuter.runPathsFromTo(source, target, model, LimitType.NORMAL, 2, f);
201 		assertTrue(result.isEmpty());
202 
203 		// test both
204 
205 		result = QueryExecuter.runPathsFromTo(source, target, model, LimitType.NORMAL, 2,
206 			new OrganismFilter(new String[]{"Homo sapiens"}), 
207 			new DataSourceFilter(new String[]{"Reactome"}));
208 		assertTrue(!result.isEmpty());
209 
210 		// test ubique filter
211 
212 		source = findElements(model,
213 			"HTTP://WWW.REACTOME.ORG/BIOPAX/48887#SMALLMOLECULE5_1_9606"); //ATP
214 		target = findElements(model,
215 			"HTTP://WWW.REACTOME.ORG/BIOPAX/48887#SMALLMOLECULE6_1_9606"); //ADP
216 
217 		Set<String> ubiqueIDs = new HashSet<String>(Arrays.asList("Some ID", "Another ID"));
218 
219 		result = QueryExecuter.runPathsFromTo(source, target, model, LimitType.NORMAL, 1,
220 			new UbiqueFilter(ubiqueIDs));
221 
222 		assertTrue(!result.isEmpty());
223 
224 		ubiqueIDs = new HashSet<String>(Arrays.asList(
225 			"HTTP://WWW.REACTOME.ORG/BIOPAX/48887#SMALLMOLECULE5_1_9606"));
226 
227 		result = QueryExecuter.runPathsFromTo(source, target, model, LimitType.NORMAL, 1,
228 			new UbiqueFilter(ubiqueIDs));
229 
230 		assertTrue(result.isEmpty());
231 	}
