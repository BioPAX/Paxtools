1   package org.biopax.paxtools.controller;

3   import org.biopax.paxtools.model.BioPAXElement;
4   import org.biopax.paxtools.util.ClassFilterSet;
5   import org.biopax.paxtools.util.IllegalBioPAXArgumentException;
6   
7   import java.util.Set;
8   
9   /**
10   * This class is a decorating property accessor that filters values with a given class. It is used, for example, to
11   * implement restrictions for pattern editors.
12   */
13  public class FilteredPropertyAccessor<D extends BioPAXElement, R> extends DecoratingPropertyAccessor<D,R>
14  {
15  	private Class filter;
16  
17  	private FilteredPropertyAccessor(PropertyAccessor<D, R> impl, Class filter)
18  	{
19  		super(impl);
20  		this.filter=filter;
21  	}
22  
23  	@Override public Set<? extends R> getValueFromBean(D bean) throws IllegalBioPAXArgumentException
24  	{
25  		return new ClassFilterSet(impl.getValueFromBean(bean), filter);
26  	}
27  
28  	/**
29  	 * FactoryMethod that creates a filtered property accessor by decorating a given accessor with a class filter.
30  	 * @param pa to be decorated
31  	 * @param filter Class to be filtered, must extend from R.
32  	 * @param <D> Domain of the original accessor
33  	 * @param <R> Range of the original accessor
34  	 * @return A filtered accessor.
35  	 */
36  	public static <D extends BioPAXElement, R> PropertyAccessor<D,R> create(PropertyAccessor<D,R> pa,
37  	                                                                                     Class filter)
38  	{
39  		 return new FilteredPropertyAccessor<D, R>(pa, filter);
40  	}
41  }
