43  	@Test
44  	public void testWriteL2GSEA() throws Exception {
45  
46  		// grab all owl in test resource dir L2
47  		File testDir = new File(getClass().getResource("/L2").getFile());
48  		FilenameFilter filter = new FilenameFilter() {
49  			public boolean accept(File dir, String name) {
50  				return (name.endsWith("owl"));
51  			}
52  		};
53  
54  		// write the output
55  		out.println("testWriteL2GSEA:");
56  		for (String testFile : testDir.list(filter)) {
57  			InputStream in = getClass().getResourceAsStream("/L2/" + testFile); // this is classpath - no need to use a "separator"
58  			Model level2 = handler.convertFromOWL(in);
59  			GSEAConverter gseaConverter = new GSEAConverter("GENE_SYMBOL", true);
60  			Collection<GSEAEntry> entries = gseaConverter.convert(level2);
61  			// assert some things
62  			assertEquals(1, entries.size());
63  			GSEAEntry entry = entries.iterator().next();
64  			assertEquals("mTOR signaling pathway", entry.name());
65  			assertTrue(entry.description().contains("nci-nature"));
66  			assertEquals("9606", entry.taxID());
67  			assertEquals("GENE_SYMBOL", entry.idType());
68  			assertEquals(27, entry.getIdentifiers().size());
69  			
70  			// dump the output
71  			(new GSEAConverter("GENE_SYMBOL", true)).writeToGSEA(level2, out);
72  			
73  			// test a small hack that allows using sometimes non-standard DB names, like NP, GO, if they are part of xref.id ;)
74  			gseaConverter = new GSEAConverter("NP", true);
75  			entries.clear();
76  			entries = gseaConverter.convert(level2);
77  			// assert some things
78  			assertEquals(1, entries.size());
79  			entry = entries.iterator().next();
80  			assertEquals("mTOR signaling pathway", entry.name());
81  			assertTrue(entry.description().contains("nci-nature"));
82  			assertEquals("9606", entry.taxID());
83  			assertEquals("NP", entry.idType());
84  			assertEquals(33, entry.getIdentifiers().size());
85  		
86  			(new GSEAConverter("NP", true)).writeToGSEA(level2, out);
87  			
88  			in.close();
89  		}
90  	}
