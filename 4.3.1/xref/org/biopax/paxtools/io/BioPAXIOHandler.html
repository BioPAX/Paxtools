1   package org.biopax.paxtools.io;
2   
3   import org.biopax.paxtools.controller.EditorMap;
4   import org.biopax.paxtools.model.BioPAXFactory;
5   import org.biopax.paxtools.model.BioPAXLevel;
6   import org.biopax.paxtools.model.Model;
7   
8   import java.io.FileNotFoundException;
9   import java.io.InputStream;
10  import java.io.OutputStream;
11  
12  /**
13   * This interface defines IO related operations that can be performed on
14   * BioPAX models.
15   */
16  public interface BioPAXIOHandler
17  {
18  
19   /**
20    * This option is only applicable two level 2 models.
21    * When enabled it will replicate illegally reused
22    * pysicalEntityParticipants in Level2 files.
23    * @param fixReusedPEPs true or false
24    */
25   void fixReusedPEPs(boolean fixReusedPEPs);
26  
27   /**
28    * This flag will fix a common legacy bug from BioCyc where the "NIL" string
29    * was used for representing unknown values.
30    * @param treatNILasNull true or false
31    */
32   void treatNilAsNull(boolean treatNILasNull);
33  
34   /**
35    * This method will read the OWL document given by the input stream
36    * and will convert it into an in memory BioPAX model.
37    * @param in a BioPAX data input stream (RDF/XML format)
38    * @return new BioPAX object Model
39    */
40   Model convertFromOWL(InputStream in);
41  
42   /**
43    * This method will read multiple OWL document
44    * and will merge them into an in memory BioPAX model.
45    * @param files input BioPAX files (RDF/XML)
46    * @return new BioPAX object model
47    * @throws FileNotFoundException when a file does not exist.
48    */
49   Model convertFromMultipleOwlFiles(String... files)
50     throws FileNotFoundException;
51  
52   /**
53    * This method will write the model to the output stream. Default encoding
54    * is RDF/XML.
55    * @param model a BioPAX model
56    * @param outputStream output stream
57    */
58   void convertToOWL(Model model, OutputStream outputStream);
59  
60   /**
61    * This flag will allow reader to automatically convert level1 classes to
62    * corresponding level 2 classes.
63    * @param convertingFromLevel1ToLevel2 true/false, whether actual BioPAX data is of Level1
64    * @deprecated BioPAX Level 1 exports are obsolete (anyway, L1 input is always converted to L2 automatically).
65    */
66   void setConvertingFromLevel1ToLevel2(boolean convertingFromLevel1ToLevel2);
67  
68   /**
69    * This flag will fix a common legacy bug from BioCyc where the "NIL" string
70    * was used for representing unknown values.
71    * @return true if this option is enabled.
72    */
73   boolean isTreatNilAsNull();
74  
75   /**
76    * This flag will allow reader to automatically convert level1 classes to
77    * corresponding level 2 classes.
78    * @return true if this option is enabled (e.g. auto-enabled if Level1 data were detected).
79    */
80   boolean isConvertingFromLevel1ToLevel2();
81  
82   /**
83    * This option is only applicable two level 2 models.
84    * When enabled it will replicate illegally reused
85    * pysicalEntityParticipants in Level2 files.
86    * @return true if this option is enabled.
87    */
88   boolean isFixReusedPEPs();
89  
90   /**
91    * @return the factory that is used to create new BioPAX POJOs during a BioPAXIOHandler operation.
92    */
93   BioPAXFactory getFactory();
94  
95   /**
96    * @param factory used for creating objects
97    */
98   void setFactory(BioPAXFactory factory);
99  
100  /**
101   * @return EditorMap used for this handler.
102   */
103  EditorMap getEditorMap();
104 
105  /**
106   * @param editorMap used for this handler.
107   */
108  void setEditorMap(EditorMap editorMap);
109 
110  /**
111   * @return The level of the model that is being read.
112   */
113  BioPAXLevel getLevel();
114 
115  /**
116   * This method will "excise" a new model from the given model that contains
117   * the objects with given ids and their dependents.
118   * @param model BioPAX object model to be exported to the output stream as RDF/XML
119   * @param outputStream the stream
120   * @param ids optional list of absolute URIs of BioPAX objects - roots/seeds for extracting a sub-model to be exported
121   */
122  void convertToOWL(Model model, OutputStream outputStream, String... ids);
123 }
