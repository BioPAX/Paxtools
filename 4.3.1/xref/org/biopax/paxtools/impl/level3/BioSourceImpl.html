52  	protected boolean semanticallyEquivalent(BioPAXElement element)
53  	{
54  		if(!(element instanceof BioSource))
55  			return false;
56  		
57  		final BioSource bioSource = (BioSource) element;
58  
59  		return
60  			(celltype != null ?
61  				celltype.isEquivalent(bioSource.getCellType()) :
62  				bioSource.getCellType() == null)
63  			&&
64  				(tissue != null ?
65  					tissue.isEquivalent(bioSource.getTissue()) :
66  					bioSource.getTissue() == null)
67  			&& hasEquivalentIntersection(
68  				new ClassFilterSet<Xref, UnificationXref>(getXref(), UnificationXref.class),
69  				new ClassFilterSet<Xref, UnificationXref>(bioSource.getXref(), UnificationXref.class));
70      }
71  
72  	public int equivalenceCode()
73  	{
74  		int result = 29 *super.equivalenceCode() + (celltype != null ? celltype.hashCode() : 0);
75  		result = 29 * result + (tissue != null ? tissue.hashCode() : 0);
76  		return result;
77  	}
