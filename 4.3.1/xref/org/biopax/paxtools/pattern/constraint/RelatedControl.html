15  /**
16   * After traversing a PhysicalEntity and the Conversion it participates, this constraint takes us to
17   * the Control. The given relation between PhysicalEntity and Conversion is used to filter out
18   * unrelated controls.
19   *
20   * Var0 - PhysicalEntity (participant)
21   * Var1 - Conversion
22   * Var2 - related Control
23   *
24   * @author Ozgun Babur
25   */
26  public class RelatedControl extends ConstraintAdapter
27  {
28  	/**
29  	 * Relation type between PhysicalEntity and Conversion.
30  	 */
31  	RelType peType;
32  
33  	/**
34  	 * Accessor to controller of Control recursively.
35  	 */
36  	PathAccessor controlledOf = new PathAccessor("Conversion/controlledOf*:Control");
37  
38  	/**
39  	 * Constructor with the relation type between PhysicalEntity and Conversion.
40  	 * @param peType relation type between PhysicalEntity and Conversion
41  	 */
42  	public RelatedControl(RelType peType)
43  	{
44  		this(peType, null);
45  	}
46  
47  	/**
48  	 * Constructor with the relation type between PhysicalEntity and Conversion, and the blacklist.
49  	 * @param peType relation type between PhysicalEntity and Conversion
50  	 * @param blacklist to detect ubiquitous small molecules
51  	 */
52  	public RelatedControl(RelType peType, Blacklist blacklist)
53  	{
54  		super(3, blacklist);
55  		this.peType = peType;
56  	}
57  
58  	/**
59  	 * This is a generative constraint.
60  	 * @return true
61  	 */
62  	@Override
63  	public boolean canGenerate()
64  	{
65  		return true;
66  	}
67  
68  	/**
69  	 * According to the relation between PhysicalEntity and the Conversion, checks of the Control is
70  	 * relevant. If relevant it is retrieved.
71  	 * @param match current pattern match
72  	 * @param ind mapped indices
73  	 * @return related controls
74  	 */
75  	@Override
76  	public Collection<BioPAXElement> generate(Match match, int... ind)
77  	{
78  		PhysicalEntity pe = (PhysicalEntity) match.get(ind[0]);
79  		Conversion conv = (Conversion) match.get(ind[1]);
80  
81  		if (!((peType == RelType.INPUT && getConvParticipants(conv, RelType.INPUT).contains(pe)) ||
82  			(peType == RelType.OUTPUT && getConvParticipants(conv, RelType.OUTPUT).contains(pe))))
83  			throw new IllegalArgumentException("peType = " + peType +
84  				", and related participant set does not contain this PE. Conv dir = " +
85  				getDirection(conv) + " conv.id=" + conv.getRDFId() + " pe.id=" +pe.getRDFId());
86  
87  		boolean rightContains = conv.getRight().contains(pe);
88  		boolean leftContains = conv.getLeft().contains(pe);
89  		
90  		assert rightContains || leftContains : "PE is not a participant.";
91  
92  		Set<BioPAXElement> result = new HashSet<BioPAXElement>();
93  
94  		ConversionDirectionType avoidDir = (leftContains && rightContains) ? null : peType == RelType.OUTPUT ?
95  			(leftContains ? ConversionDirectionType.LEFT_TO_RIGHT : ConversionDirectionType.RIGHT_TO_LEFT) :
96  			(rightContains ? ConversionDirectionType.LEFT_TO_RIGHT : ConversionDirectionType.RIGHT_TO_LEFT);
97  
98  		for (Object o : controlledOf.getValueFromBean(conv))
99  		{
100 			Control ctrl = (Control) o;
101 			ConversionDirectionType dir = getDirection(conv, ctrl);
102 
103 			if (avoidDir != null && dir == avoidDir) continue;
104 
105 			// don't collect this if the pe is ubique in the context of this control
106 			if (blacklist != null && blacklist.isUbique(pe, conv, dir, peType)) continue;
107 
108 			result.add(ctrl);
109 		}
110 		return result;
111 	}
112 
113 }
