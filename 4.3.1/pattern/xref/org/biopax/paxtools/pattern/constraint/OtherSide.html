8   import java.util.Collection;
9   import java.util.Collections;
10  import java.util.HashSet;
11  
12  /**
13   * Given Conversion and its one of the participants (at the left or right), traverses to the other
14   * participants that are at the other side.
15   *
16   * var0 is a PE (PE1)
17   * var1 is a Conv
18   * var2 is a PE (PE2)
19   *
20   * Prerequisite: PE1 is either at left or right of Conv
21   *
22   * @deprecated
23   * @see org.biopax.paxtools.pattern.constraint.ConversionSide
24   *
25   * @author Ozgun Babur
26   */
27  public class OtherSide extends ConstraintAdapter
28  {
29  	/**
30  	 * Constructor.
31  	 */
32  	public OtherSide()
33  	{
34  		super(3);
35  	}
36  
37  	/**
38  	 * This is a generative constraint.
39  	 * @return true
40  	 */
41  	@Override
42  	public boolean canGenerate()
43  	{
44  		return true;
45  	}
46  
47  	/**
48  	 * Checks which side is the first PhysicalEntity, and gathers participants on the other side.
49  	 * @param match current pattern match
50  	 * @param ind mapped indices
51  	 * @return participants at the other side
52  	 */
53  	@Override
54  	public Collection<BioPAXElement> generate(Match match, int... ind)
55  	{
56  		assertIndLength(ind);
57  
58  		PhysicalEntity pe1 = (PhysicalEntity) match.get(ind[0]);
59  		Conversion conv = (Conversion) match.get(ind[1]);
60  
61  		if (conv.getLeft().contains(pe1))
62  		{
63  			return new HashSet<BioPAXElement>(conv.getRight());
64  		}
65  		else if (conv.getRight().contains(pe1))
66  		{
67  			return new HashSet<BioPAXElement>(conv.getLeft());
68  		}
69  		return Collections.emptySet();
70  	}
71  }
