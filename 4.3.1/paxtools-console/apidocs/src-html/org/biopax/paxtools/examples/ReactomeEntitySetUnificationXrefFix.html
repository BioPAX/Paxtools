001package org.biopax.paxtools.examples;
002
003import org.biopax.paxtools.io.BioPAXIOHandler;
004import org.biopax.paxtools.io.SimpleIOHandler;
005import org.biopax.paxtools.model.BioPAXElement;
006import org.biopax.paxtools.model.Model;
007import org.biopax.paxtools.model.level2.physicalEntity;
008import org.biopax.paxtools.model.level2.relationshipXref;
009import org.biopax.paxtools.model.level2.unificationXref;
010import org.biopax.paxtools.model.level2.xref;
011import org.biopax.paxtools.util.ClassFilterSet;
012
013import java.io.*;
014import java.util.HashSet;
015import java.util.Set;
016
017/**
018 * This is a class I wrote for fixing the unification xref problem in Reactome entity sets. It
019 * reads in the model, goes over physical entities and looks for the converted from
020 * reactome entity set comment. For each such generic entities it converts all of its unification
021 * xrefs to relationship xrefs.
022 *
023 * NOTE: This method is now outdated as it is easier to do this now with the new PathAccessors. I did not remove this
024 * example, however, as it demonstrates many "low-level" operations of Paxtools.
025 */
026public class ReactomeEntitySetUnificationXrefFix
027{
028        public static void main(String[] args) throws IOException
029        {
030                if (args.length == 2)
031                {
032                        fixReactome(new FileInputStream(new File(args[0])),
033                                        new FileOutputStream(new File(args[1])));
034                }
035                else
036                fixReactome(System.in, System.out);
037        }
038
039        public static void fixReactome(InputStream in, OutputStream out) throws IOException
040        {
041                BioPAXIOHandler io = new SimpleIOHandler();
042                Model level2 = io.convertFromOWL(in);
043                Set<physicalEntity> physicalEntitySet = new HashSet<physicalEntity>();
044                physicalEntitySet.addAll(level2.getObjects(physicalEntity.class));
045                for (physicalEntity pe : physicalEntitySet)
046                {
047                        boolean entitySet = false;
048                        for (String comment : pe.getCOMMENT())
049                        {
050                                if (comment.contains("Converted from EntitySet in Reactome"))
051                                {
052                                        entitySet = true;
053                                        break;
054
055                                }
056
057                        }
058                        if (entitySet)
059                        {
060                                Set<unificationXref> unis = new HashSet<unificationXref>();
061                                unis.addAll(new ClassFilterSet<xref,unificationXref>(pe.getXREF(), unificationXref.class));
062                                
063                                for (unificationXref uni : unis)
064                                {
065                                        relationshipXref rel;
066                                        String rid = uni.getRDFId() + "-r";
067                                        BioPAXElement exists = level2.getByID(rid);
068                                        if (exists != null)
069                                        {
070                                                rel = (relationshipXref) exists;
071                                        }
072                                        else
073                                        {
074                                                rel = level2.addNew(relationshipXref.class, rid);
075                                                rel.setDB(uni.getDB());
076                                                rel.setID(uni.getID());
077                                        }
078                                        pe.removeXREF(uni);
079                                        if (uni.isXREFof().isEmpty())
080                                        {
081                                                level2.remove(uni);
082                                        }
083                                        pe.addXREF(rel);
084
085                                }
086                        }
087
088                }
089
090                io.convertToOWL(level2, out);
091                out.close();
092        }
093}
