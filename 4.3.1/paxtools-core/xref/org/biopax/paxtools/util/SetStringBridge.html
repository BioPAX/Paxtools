4   import org.hibernate.search.bridge.StringBridge;

6   import java.util.Set;
7   
8   /**
9    * Hibernate - Lucene bridge between a Set&lt;String&gt; and String.
10   * @deprecated Hibernate ORM/Search will be removed in v5
11   */
12  public class SetStringBridge implements StringBridge {
13  
14  	public String objectToString(Object object) {
15  
16  		// string to return
17  		StringBuilder toReturn = new StringBuilder();
18  
19  		// no way to do instanceof on generics.
20  		// trust this bridge is used properly
21  		Set<String> items = (Set<String>)object;
22  
23  		// interate over all strings and append (' ' delimit) to return string
24  		for (String item : items) {
25  			toReturn.append(item).append(" ");
26  		}
27  
28  		return toReturn.toString();
29  	}
30  }
