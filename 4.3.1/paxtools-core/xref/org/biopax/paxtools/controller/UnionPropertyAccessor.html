3   import org.apache.commons.collections15.set.CompositeSet;
4   import org.biopax.paxtools.model.BioPAXElement;
5   import org.biopax.paxtools.util.IllegalBioPAXArgumentException;
6   
7   import java.util.Set;
8   
9   /**
10   *  In OWL a single property can have multiple domains. This is not supported in OO. This class amends this by
11   *  using a Facade pattern. A Union property will act like a OWL property with a union domain but in fact is composed
12   *  of multiple single domain properties.
13   */
14  public class UnionPropertyAccessor<D extends BioPAXElement,R> extends PropertyAccessorAdapter<D,R>
15  {
16  
17  	Set<PropertyAccessor<? extends D, ? extends R>> union;
18  
19  		public UnionPropertyAccessor(Set<PropertyAccessor<? extends D, ? extends R>> union, Class<D> domain)
20  	{
21  		super(domain, (Class<R>) union.iterator().next().getRange(), true);
22  		if(union == null || union.isEmpty())
23  		{
24  			throw new IllegalBioPAXArgumentException("Empty set of editors. Can't create a union");
25  		}
26  		this.union = union;
27  		this.multipleCardinality = union.iterator().next().isMultipleCardinality();
28  	}
29  
30  	@Override public Set getValueFromBean(D bean) throws IllegalBioPAXArgumentException
31  	{
32  		CompositeSet valueFromBean =new CompositeSet();
33  
34  		for (PropertyAccessor atomicAccessor : union)
35  		{
36  			if(atomicAccessor.getDomain().isAssignableFrom(bean.getModelInterface()))
37  			{
38  				valueFromBean.addComposited(atomicAccessor.getValueFromBean(bean));
39  			}
40  		}
41  		return valueFromBean;
42  	}
43  
44  	@Override public boolean isUnknown(Object value)
45  	{
46  		return union.iterator().next().isUnknown(value);
47  	}
48  }
