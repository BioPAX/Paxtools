055        /**
056         * Initialization
057         */
058        static
059        {
060                try
061            {
062            JAXBContext jc = JAXBContext.newInstance(BINDING);
063            Unmarshaller unmarshaller = jc.createUnmarshaller();
064            Miriam mir = null;
065                InputStream is = MiriamLink.class.getResourceAsStream("/Miriam.xml");
066                if(is != null) {
067                        mir = (Miriam) unmarshaller.unmarshal(is);
068                } else {
069                        throw new RuntimeException("Miriam.xml db is missing or broken. " +
070                                                "Please download the XML and schema from http://www.ebi.ac.uk/miriam/main/export/ " +
071                                                        "and put at the classpath's root");
072                }
073            miriam = mir;
074            log.info("MIRIAM XML imported, version: " + miriam.getDataVersion() + ", datatypes: "
075                                        + miriam.getDatatype().size());
076            }
077            catch (JAXBException e) {
078                throw new RuntimeException(e);
079            }
080            
081            // build the name-datatype static hash (once!)
082                for(Datatype dt : miriam.getDatatype()) {
083                        // index by name
084                        datatypesHash.put(dt.getName().toUpperCase(), dt);
085                        
086                        // by identifier
087                        datatypesHash.put(dt.getId().toUpperCase(), dt);
088                        
089                        // index by each synonym
090                        // (Miriam must guarantee: different datatypes cannot have the same synonym!)
091                        Synonyms synonyms = dt.getSynonyms();
092                        if (synonyms != null) {
093                                for (String syn : synonyms.getSynonym()) {
094                                        datatypesHash.put(syn.toUpperCase(), dt);
095                                }
096                        }
097                        
098                        // index by each URI
099                        for(Uris uris : dt.getUris()) {
100                                for(Uri uri : uris.getUri()) {
101                                        datatypesHash.put(uri.getValue(), dt);
102                                }
103                        }
104                }
105        }

183    /**
184     * Retrieves the unique MIRIAM URI of a specific entity (example: "urn:miriam:obo.go:GO%3A0045202").
185     * 
186     * @param name - name, URI/URL, or ID of a data type (examples: "ChEBI", "MIR:00000005")
187     * @param id identifier of an entity within the data type (examples: "GO:0045202", "P62158")
188     * @return unique standard MIRIAM URI of a given entity
189     * 
190     * @throws IllegalArgumentException when datatype not found
191     */
192    public static String getURI(String name, String id)
193    {
194        Datatype datatype = getDatatype(name);
195        String db = datatype.getName();
196        if(checkRegExp(id, db)) {
197                try {
198                                return getOfficialDataTypeURI(datatype) + ":" + URLEncoder.encode(id, "UTF-8");
199                        } catch (UnsupportedEncodingException e) {
200                                throw new RuntimeException("UTF-8 encoding error of id=" + id, e);
201                        }
202        } else 
203                throw new IllegalArgumentException(
204                                "ID pattern mismatch. db=" + db + ", id=" + id
205                                + ", regexp: " + datatype.getPattern());
206    }

224    /**
225     * Retrieves the physical locationS (URLs) of web pageS providing knowledge about an entity.
226     * 
227     * @param datatypeKey name (can be a synonym), ID, or URI of a data type (examples: "Gene Ontology", "UniProt")
228     * @param entityId identifier of an entity within the given data type (examples: "GO:0045202", "P62158")
229     * @return physical locationS (URL templates) of web pageS providing knowledge about the given entity
230     * 
231     * @throws IllegalArgumentException when datatype not found
232     */
233    public static String[] getLocations(String datatypeKey, String entityId)
234    {
235        Set<String> locations = new HashSet<String>();
236                Datatype datatype = getDatatype(datatypeKey);
237                for (Resource resource : getResources(datatype)) {
238                        String link = resource.getDataEntry();
239                        try {
240                                link = link.replaceFirst("\\$id", URLEncoder.encode(entityId, "UTF-8"));
241                        } catch (UnsupportedEncodingException e) {
242                                throw new RuntimeException(e);
243                        }
244                        locations.add(link);
245                }
246        return locations.toArray(ARRAY_OF_STRINGS);
247    }

380     * Checks if the identifier given follows the regular expression 
381     * of its data type (also provided).
382     * 
383     * @param identifier internal identifier used by the data type
384     * @param datatype name, synonym or URI of a data type
385     * @return "true" if the identifier follows the regular expression, "false" otherwise
386     * 
387     * @throws IllegalArgumentException when datatype not found
388     */
389    public static boolean checkRegExp(String identifier, String datatype)
390    {
391        Datatype dt = getDatatype(datatype);
392        return Pattern.compile(dt.getPattern()).matcher(identifier).find();

397     * Retrieves the unique (official) URI of a data type
398         * (example: "http://identifiers.org/uniprot").
399     * 
400     * @param datatype net.biomodels.miriam.Miriam.Datatype
401     * @return URI
402     */
403    public static String getOfficialDataTypeURI(Datatype datatype) {
404        for(Uris uris : datatype.getUris()) {
405                for(Uri uri : uris.getUri()) {
406                        if(!isDeprecated(uri) && uri.getType() == UriType.URN) {
407                                return uri.getValue();
408                        }
409                }
410        }
411        
412        return null;
413    }

425        /**
426     * Gets Miriam Datatype by its ID, Name, Synonym, or URI (URN/URL) 
427     * 
428     * @param datatypeKey - a datatype ID, name, synonym, or URI
429     * @return MIRIAM Datatype bean
430     * 
431         * @throws IllegalArgumentException when not found
432     */
433        public static Datatype getDatatype(String datatypeKey) 
434        {       
435                Datatype dt = null;
436                if(containsIdOrName(datatypeKey))
437                        dt = datatypesHash.get(datatypeKey.toUpperCase());
438                else if(containsUri(datatypeKey)) 
439                        dt = datatypesHash.get(datatypeKey);
440                else
441                        throw new IllegalArgumentException("Datatype not found : " + datatypeKey);
442                
443                if(!useObsoleteDatatypes 
444                                && dt.isObsolete() != null 
445                                && dt.isObsolete().booleanValue() == true)
446                        throw new IllegalArgumentException("Datatype " +
447                                datatypeKey + "(" + dt.getName() + ") is obsolete" +
448                                        " (and useObsoleteDatatypes=false)");
449                
450                // return 
451                return dt;
452        }

516    /**
517     * Gets the list of Miriam {@link Resource} from 
518     * the datatype, also taking the 'obsolete' flag/state
519     * into account.
520     * 
521     * @param datatype a MIRIAM Datatype
522     * @return the list of MIRIAM Resources in that datatype
523     */
524    private static List<Resource> getResources(Datatype datatype) {
525        List<Resource> toReturn = new ArrayList<Resource>();
526        
527        Resources resources = datatype.getResources();
528                if (resources != null) {
529                        for (Resource resource : resources.getResource()) {
530                                if(useObsoleteResources // ok to collect (- do not care about obsolete)
531                                        || resource.isObsolete() == null // undefined - here means non-obsolete (ok)
532                                                || !resource.isObsolete().booleanValue()) {
533                                        toReturn.add(resource);
534                                }
535                        }
536                }
537                
538                return toReturn;
539    }

542    /**
543     * Converts a MIRIAM URN into its equivalent Identifiers.org URL.
544     * 
545     * @see #getURI(String, String) - use this to get the URN
546     * @see #getIdentifiersOrgURI(String, String) - prefered URI
547     * 
548     * @param urn - an existing Miriam URN, e.g., "urn:miriam:obo.go:GO%3A0045202"
549     * @return the Identifiers.org URL corresponding to the data URN, e.g., "http://identifiers.org/obo.go/GO:0045202"
550     * 
551     * @deprecated this method applies {@link URLDecoder#decode(String)} to the last part of the URN, which may not always work as expected (test yours!) 
552     */
553    public static String convertUrn(String urn) {
554        String[] tokens = urn.split(":");
555        return "http://identifiers.org/" + tokens[tokens.length-2] 
556                + "/" + URLDecoder.decode(tokens[tokens.length-1]);
557    }

560    /**
561     * Gets the Identifiers.org URI/URL of the entity (example: "http://identifiers.org/obo.go/GO:0045202").
562     * 
563     * @param name - name, URI/URL, or ID of a data type (examples: "ChEBI", "MIR:00000005")
564     * @param id identifier of an entity within the data type (examples: "GO:0045202", "P62158")
565     * @return Identifiers.org URL for the id
566     * 
567     * @throws IllegalArgumentException when datatype not found
568     */
569    public static String getIdentifiersOrgURI(String name, String id)
570    {
571        String url = null;
572        Datatype datatype = getDatatype(name);
573        String db = datatype.getName();
574        if(checkRegExp(id, db)) {
575                        uris: 
576                        for(Uris uris : datatype.getUris()) {
577                                for(Uri uri : uris.getUri()) {
578                                        if(uri.getValue().startsWith("http://identifiers.org/")) {
579                                                url = uri.getValue() + id;
580                                                break uris;
581                                        }
582                                }
583                        }
584        } else 
585                throw new IllegalArgumentException(
586                                "ID pattern mismatch. db=" + db + ", id=" + id
587                                + ", regexp: " + datatype.getPattern());
588        
589        return url;
590    }
