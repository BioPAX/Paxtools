001package org.biopax.paxtools.model;
002
003import org.biopax.paxtools.impl.BioPAXElementImpl;
004import org.biopax.paxtools.impl.ModelImpl;
005import org.slf4j.Logger;
006import org.slf4j.LoggerFactory;
007
008import java.lang.reflect.Constructor;
009import java.lang.reflect.InvocationTargetException;
010import java.lang.reflect.Method;
011import java.lang.reflect.Modifier;
012
013/**
014 * Abstract factory class for instantiating BioPAX classes. Different implementations of BioPAX model should also
015 * implement their own factory
016 */
017
018public abstract class BioPAXFactory
019{
020    private static final Logger log = LoggerFactory.getLogger(BioPAXFactory.class);
021        
022    private final Method setUriMethod;
023    
024    /**
025     * Protected Constructor without parameters.
026     */
027    protected BioPAXFactory() {
028        try {
029                        setUriMethod = BioPAXElementImpl.class.getDeclaredMethod("setUri", String.class);
030                        setUriMethod.setAccessible(true);
031                } catch (Throwable e) {
032                        throw new RuntimeException("BioPAXFactory Constructor failed", e);
033                } 
034        }
035
036    /**
037     * Gets the level.
038     * 
039     * @return the biopax level
040     */
041    public abstract BioPAXLevel getLevel();
042
043    
044    public BioPAXElement create(String localName, String uri)
045    {
046        Class<? extends BioPAXElement> type = getLevel().getInterfaceForName(localName);
047        return create(type, uri);
048    }
049
050    
051    /**
052     * Universal method that creates a new BioPAX object.
053     * (works with non-public, other package, implementations)
054         *
055         * @param <T> type
056         * @param aClass the class that corresponds to the BioPAX type
057         * @param uri absolute URI of the new BioPAX object
058         * @return new BioPAX object
059     */
060        public <T extends BioPAXElement> T create(Class<T> aClass, String uri)
061        {
062//              //could use the following shortcut, but it might hurt performance of creating/reading large valid models
063//              if(!canInstantiate(aClass)) {
064//                      log.error("Non-instantiable: cannot create BioPAX object, uri:"+uri+", with abstract type: " + aClass);
065//                      return null;
066//              }
067
068                T bpe = null;
069
070                // create a new instance of the BioPAX type
071                try {
072                        Class<T> t = getImplClass(aClass);
073                        if(t != null) {
074                                Constructor<T> c = t.getDeclaredConstructor();
075                                c.setAccessible(true);
076                                bpe = (T) c.newInstance();
077                                setUriMethod.invoke(bpe, uri);
078                        } else {
079                                log.error("Could not find a class implementing " + aClass);
080                        }
081                } catch (InvocationTargetException e) { //this happened, weird (might due to lack of memory or busy...)
082                        log.error("Failed creating BioPAX object: " + aClass + ", URI: " + uri + " - " + e, e.getCause());
083                } catch (Exception e) {
084                        log.error("Failed creating BioPAX object: " + aClass + ", URI: " + uri, e);
085                } 
086
087                return bpe;
088        }
089
090    
091        /**
092         * Maps a BioPAX type (model interface) to the
093         * full-qualified class name of an implementing class.
094         * BioPAX factories have to implement this method.
095         * 
096         * @param aClass BioPAX type (model interface)
097         * @return full class name
098         */
099    public abstract String mapClassName(Class<? extends BioPAXElement> aClass);
100
101    
102    /**
103     * Checks whether objects of a BioPAX model interface
104     * can be created (some types are not official BioPAX 
105     * types, abstract classes).
106     * 
107     * @param aClass BioPAX interface class
108     * @return whether this factory can create an instance of the type
109     */
110    public boolean canInstantiate(Class<? extends BioPAXElement> aClass) 
111    {
112        try {
113            String cname = mapClassName(aClass);
114                return !Modifier.isAbstract(Class.forName(cname).getModifiers());
115        } catch (ClassNotFoundException e)
116        {
117            return false;
118        } catch (Exception ex) {
119                log.error("Error in canInstantiate(" + aClass + ")", ex);
120                        return false;
121                }
122    }
123
124
125        /**
126         * Creates a new BioPAX model.
127         * @return BioPAX object model implementation
128         */
129        public Model createModel() {
130        return new ModelImpl(this);
131    }
132
133    
134    /**
135     * Get a concrete or abstract BioPAX type (not interface), 
136     * from org.biopax.paxtools.impl..*, i.e., one that has 
137     * persistence/search annotations, etc. This may be required for
138     * some DAO and web service controllers; it also returns such 
139     * abstract BioPAX "adapters" as XReferrableImpl, ProcessImpl, etc.
140     * 
141     * @param <T> BioPAX type/interface
142     * @param aModelInterfaceClass interface class for the BioPAX type
143     * @return concrete class that implements the BioPAX interface and can be created with this factory
144     */
145        public <T extends BioPAXElement> Class<T> getImplClass(Class<T> aModelInterfaceClass)
146        {
147                Class<T> implClass = null;
148
149                if (aModelInterfaceClass.isInterface()) {
150                        String name = mapClassName(aModelInterfaceClass);
151                        try {
152                                implClass = (Class<T>) Class.forName(name);
153                        } catch (ClassNotFoundException e) {
154                                log.debug(String.format("getImplClass(%s), %s" , aModelInterfaceClass, e));
155                        }
156                }
157
158                return implClass;
159        }
160
161}
