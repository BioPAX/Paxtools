001package org.biopax.paxtools.model;
002
003import java.io.Serializable;
004import java.util.Map;
005
006/**
007 * This class represents a general BioPAXElement, regardless of Level.
008 */
009public interface BioPAXElement extends Serializable, Cloneable
010{
011// ------------------------------ FIELDS ------------------------------
012
013    /**
014     * Constant for representing unknown doubles. This is required
015     * as by default java would assign 0.
016     */
017    public static final Double UNKNOWN_DOUBLE = Double.MIN_VALUE;
018
019    /**
020     * Constant for representing unknown floats. This is required
021     * as by default java would assign 0.
022     */
023        public static final Float UNKNOWN_FLOAT = Float.MIN_VALUE;
024
025    /**
026     * Constant for representing unknown integers. This is required
027     * as by default java would assign 0.
028     */
029        public static final Integer UNKNOWN_INT = Integer.MIN_VALUE;
030        
031        
032    /**
033     * This method returns the actual model interface that a class implements.
034     * @return an interface from {@link org.biopax.paxtools.model} package
035     * corresponding to a BioPAX class.
036     */
037    Class<? extends BioPAXElement> getModelInterface();
038
039    /**
040     * This method returns the absolute URI of the element.
041     * BioPAX data providers are responsible 
042     * for generating globally unique and standard URIs
043     * for their BioPAX elements.
044     * 
045     * @return unique URI for this object.
046     */
047    String getUri();
048
049
050    /**
051     * This method compares the given element for equivalency. This is different
052     * from equals(), as BioPAX elements resolve equality based on URI.
053     * Equivalent returns true if elements are equal or if
054     *  <ul>
055     *   <li> both elements implement the same model interface AND
056     *   <li> both elements have equivalent key properties
057     *  </ul>
058     *  These key properties vary from class to class.
059     *
060     * @param element to be compared for equivalency
061     * @return true if the element equals to this, or has equivalent critical
062     * properties.
063     */
064    boolean isEquivalent(BioPAXElement element);
065
066    /**
067     * If two elements are equivalent, then their equivalence code should be the
068     * same.
069     * @return an integer that is same across all equivalent entities.
070     */
071    int equivalenceCode();
072
073
074    /**
075     * A general-purpose map to optionally 
076     * store additional application-specific information 
077     * about the BioPAX element, such as statistics,
078     * inferred fields, etc.
079     * 
080     * @return additional (not BioPAX standard) annotations
081     */
082    public Map<String, Object> getAnnotations();
083        
084}
