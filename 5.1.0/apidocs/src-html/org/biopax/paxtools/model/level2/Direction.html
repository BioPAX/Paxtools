001package org.biopax.paxtools.model.level2;
002
003/**
004 * Catalysis direction controlled vocabulary
005 */
006public enum Direction
007{
008        REVERSIBLE,
009        PHYSIOL_LEFT_TO_RIGHT,
010        PHYSIOL_RIGHT_TO_LEFT,
011        IRREVERSIBLE_LEFT_TO_RIGHT,
012        IRREVERSIBLE_RIGHT_TO_LEFT;
013
014        public String toString()
015        {
016                if (this == REVERSIBLE) return "REVERSIBLE";
017                else if (this == PHYSIOL_LEFT_TO_RIGHT) return "PHYSIOL_LEFT_TO_RIGHT";
018                else if (this == PHYSIOL_RIGHT_TO_LEFT) return "PHYSIOL_RIGHT_TO_LEFT";
019                else if (this == IRREVERSIBLE_LEFT_TO_RIGHT) return "IRREVERSIBLE_LEFT_TO_RIGHT";
020                else if (this == IRREVERSIBLE_RIGHT_TO_LEFT) return "IRREVERSIBLE_RIGHT_TO_LEFT";
021                else return null;
022        }
023}
