001package org.biopax.paxtools.model;
002
003import org.biopax.paxtools.util.IllegalBioPAXArgumentException;
004
005import java.io.InputStream;
006
007/**
008 * Enumeration type for BioPAX levels.
009 */
010public enum BioPAXLevel
011{
012        // define enum constants using default BioPAX factories
013        // (L1 is not supported anymore)
014        L2("biopax-level2.owl", new Level2FactoryImpl(), "org.biopax.paxtools.model.level2"),
015        L3("biopax-level3.owl", new Level3FactoryImpl(), "org.biopax.paxtools.model.level3");
016
017        // ------------------------------ FIELDS ------------------------------
018
019        private final String filename;
020
021        private BioPAXFactory factory;
022
023        private final String packageName;
024        
025        // default L2 factory implementation
026        private static class Level2FactoryImpl extends BioPAXFactory {
027            @Override
028            public BioPAXLevel getLevel() {
029                return BioPAXLevel.L2;
030            }
031            
032            public String mapClassName(Class<? extends BioPAXElement> aClass) 
033            {
034                String name = "org.biopax.paxtools.impl.level2."
035                        + aClass.getSimpleName()
036                        + "Impl";
037                return name;
038            }  
039        }
040        
041        // default L3 factory implementation
042        private static class Level3FactoryImpl extends BioPAXFactory {
043            @Override
044            public BioPAXLevel getLevel() {
045                return BioPAXLevel.L3;
046            }
047            
048            public String mapClassName(Class<? extends BioPAXElement> aClass) 
049            {
050                String name = "org.biopax.paxtools.impl.level3."
051                        + aClass.getSimpleName()
052                        + "Impl";
053                return name;
054            }  
055        }
056        
057
058        /**
059         * This is the prefix used for all biopax releases.
060         */
061        public static final String BP_PREFIX = "http://www.biopax.org/release/";
062
063// --------------------------- CONSTRUCTORS ---------------------------
064
065        /**
066         * Default constructor
067         * @param filename File name of the owl file.
068         * @param factory BioPAX factory implementation (default)
069         * @param pm package name of the model implementation
070         */
071        BioPAXLevel(String filename, BioPAXFactory factory, String pm)
072        {
073                this.filename = filename;
074                this.packageName = pm;
075                this.factory = factory;
076        }
077
078// --------------------- GETTER / SETTER METHODS ---------------------
079
080        /**
081         * This method returns the filename of the owl file
082         * @return the filename of the owl file
083         */
084        public String getFilename()
085        {
086                return filename;
087        }
088
089        /**
090         * This method returns the default factory for this level
091         * @return he default factory for this level
092         */
093        public BioPAXFactory getDefaultFactory()
094        {
095                return factory;
096        }
097
098// -------------------------- OTHER METHODS --------------------------
099
100        /**
101         * This method loads the level file as resource and returns it as
102         * an input stream
103         * @return an input stream from the owl file.
104         */
105        public InputStream getLevelFileAsStream()
106        {
107                return this.getClass().getResourceAsStream(filename);
108        }
109
110// --------------------- ACCESORS and MUTATORS---------------------
111
112        /**
113         * This method returns the namespace defined for this level.
114         * @return namespace defined for this level.
115         */
116        public String getNameSpace()
117        {
118                return BP_PREFIX + filename + "#";
119        }
120
121        /**
122         * This method returns true if the given string starts with the
123         * BP_PREFIX
124         * @param nameSpace to be checked
125         * @return rue if the given string starts with the BP_PREFIX
126         */
127        public static boolean isInBioPAXNameSpace(String nameSpace)
128        {
129                return nameSpace != null && nameSpace.startsWith(BP_PREFIX);
130        }
131
132        public static BioPAXLevel getLevelFromNameSpace(String namespace)
133        {
134                if (isInBioPAXNameSpace(namespace))
135                {
136                        {
137                                for (BioPAXLevel level : BioPAXLevel.values())
138                                {
139                                        if (namespace.equalsIgnoreCase(level.getNameSpace()))
140                                        {
141                                                return level;
142                                        }
143                                }
144                        }
145                }
146                return null;
147        }
148
149        public String getPackageName()
150        {
151                return packageName;
152        }
153
154        public boolean hasElement(BioPAXElement element)
155        {
156                return element.getModelInterface().getPackage().getName().equals(this.packageName);
157        }
158
159
160        /**
161         * Gets the BioPAX type (java interface) by name.
162         * @param localName a BioPAX type name
163         * @return the BioPAX interface class
164         * @throws IllegalBioPAXArgumentException when there is no such type / class not found.
165         */
166        public Class<? extends BioPAXElement> getInterfaceForName(String localName)
167        {
168        try
169        {
170            Class modelInterface = Class.forName(this.packageName + "." + localName);
171
172            if (BioPAXElement.class.isAssignableFrom(modelInterface))
173            {
174                return modelInterface;
175            } else
176            {
177                throw new IllegalBioPAXArgumentException(
178                        "BioPAXElement is not assignable from class:" + modelInterface.getSimpleName());
179            }
180        }
181        catch (ClassNotFoundException e)
182        {
183            throw new IllegalBioPAXArgumentException("Could not locate interface for:" + localName);
184        }
185    }
186
