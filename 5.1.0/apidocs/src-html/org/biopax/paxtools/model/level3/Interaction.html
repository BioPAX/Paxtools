001package org.biopax.paxtools.model.level3;
002
003import java.util.Set;
004
005
006/**
007 * Definition: A biological relationship between two or more entities.
008 * 
009 * Rationale: In BioPAX, interactions are atomic from a database modeling perspective,
010 * i.e. interactions can not be decomposed into sub-interactions. When representing non-atomic continuants with
011 * explicit subevents the pathway class should be used instead. Interactions are not necessarily  temporally atomic,
012 * for example genetic interactions cover a large span of time. Interactions as a formal concept is a continuant,
013 * it retains its identitiy regardless of time, or any differences in specific states or properties.
014 * 
015 * Usage: Interaction is a highly abstract class and in almost all cases it is more appropriate to use one of the
016 * subclasses of interaction.
017 * It is partially possible to define generic reactions by using generic participants. A more comprehensive method is
018 * planned for BioPAX L4 for covering all generic cases like oxidization of a generic alcohol.
019 * 
020 * Synonyms: Process, relationship, event.
021 * 
022 * Examples: protein-protein interaction, biochemical reaction, enzyme catalysis
023 */
024public interface Interaction extends Process
025{
026
027        /**
028         * Controlled vocabulary term annotating the interaction type for example, "phosphorylation reaction". This
029         * annotation is meant to be human readable and may not be suitable for computing tasks, like reasoning,
030         * that require formal vocabulary systems. For instance, this information would be useful for display on a web
031         * page or for querying a database. The PSI-MI interaction type controlled vocabulary should be used. This is
032         * browsable at:
033         * <a href =
034         * "http://www.ebi.ac.uk/ontology-lookup/browse.do?ontName=MI&termId=MI%3A0190&termName=interaction%20type">
035         * OLS: PSI-MI</a>
036         * 
037         * Contents of this set should not be modified. Use add/remove instead.
038         * @return Controlled vocabulary term annotating the interaction type.
039         */
040        Set<InteractionVocabulary> getInteractionType();
041
042        /**
043         * Controlled vocabulary term annotating the interaction type for example, "phosphorylation reaction". This
044         * annotation is meant to be human readable and may not be suitable for computing tasks, like reasoning,
045         * that require formal vocabulary systems. For instance, this information would be useful for display on a web
046         * page or for querying a database. The PSI-MI interaction type controlled vocabulary should be used. This is
047         * browsable at:
048         * <a href =
049         * "http://www.ebi.ac.uk/ontology-lookup/browse.do?ontName=MI&termId=MI%3A0190&termName=interaction%20type">
050         * OLS: PSI-MI</a>
051         * 
052         * Contents of this set should not be modified. Use add/remove instead.
053         * @param newinteractionType Controlled vocabulary term annotating the interaction type.
054         */
055        void addInteractionType(InteractionVocabulary newinteractionType);
056
057        /**
058         * Controlled vocabulary term annotating the interaction type for example, "phosphorylation reaction". This
059         * annotation is meant to be human readable and may not be suitable for computing tasks, like reasoning,
060         * that require formal vocabulary systems. For instance, this information would be useful for display on a web
061         * page or for querying a database. The PSI-MI interaction type controlled vocabulary should be used. This is
062         * browsable at:
063         * <a href =
064         * "http://www.ebi.ac.uk/ontology-lookup/browse.do?ontName=MI&termId=MI%3A0190&termName=interaction%20type">
065         * OLS: PSI-MI</a>
066         * 
067         * Contents of this set should not be modified. Use add/remove instead.
068         * @param oldinteractionType Controlled vocabulary term annotating the interaction type.
069         */
070        void removeInteractionType(InteractionVocabulary oldinteractionType);
071
072
073        /**
074         * The entities that participate in this interaction. For example, in a biochemical reaction,
075         * the participants are the union of the reactants and the products of the reaction. Multiple sub-properties of
076         * participant are defined, such as left and right used in the BiochemicalReaction class and controller and
077         * controlled, used in the Control class. Any value of the sub-properties is automatically values of the
078         * participant property.
079         * Contents of this set should not be modified. Use add/remove instead.
080         * @return The entities that participate in this interaction.
081         */
082        Set<Entity> getParticipant();
083
084
085        /**
086         * The entities that participate in this interaction. For example, in a biochemical reaction,
087         * the participants are the union of the reactants and the products of the reaction. Multiple sub-properties of
088         * participant are defined, such as left and right used in the BiochemicalReaction class and controller and
089         * controlled, used in the Control class. Any value of the sub-properties is automatically values of the
090         * participant property.
091         * Contents of this set should not be modified. Use add/remove instead.
092         * @param participant The entities that participate in this interaction.
093         */
094        void addParticipant(Entity participant);
095
096
097        /**
098         * The entities that participate in this interaction. For example, in a biochemical reaction,
099         * the participants are the union of the reactants and the products of the reaction. Multiple sub-properties of
100         * participant are defined, such as left and right used in the BiochemicalReaction class and controller and
101         * controlled, used in the Control class. Any value of the sub-properties is automatically values of the
102         * participant property.
103         * Contents of this set should not be modified. Use add/remove instead.
104         * @param participant The entities that participate in this interaction.
105         */
106        void removeParticipant(Entity participant);
107        
108}
