023        /**
024         * Pattern for a EntityReference has a member PhysicalEntity that is controlling a state change
025         * reaction of another EntityReference.
026         * @return the pattern
027         */
028        public static Pattern controlsStateChange()
029        {
030                Pattern p = new Pattern(SequenceEntityReference.class, "controller ER");
031                p.add(linkedER(true), "controller ER", "generic controller ER");
032                p.add(erToPE(), "generic controller ER", "controller simple PE");
033                p.add(linkToComplex(), "controller simple PE", "controller PE");
034                p.add(peToControl(), "controller PE", "Control");
035                p.add(controlToConv(), "Control", "Conversion");
036                p.add(new NOT(participantER()), "Conversion", "controller ER");
037                p.add(new Participant(RelType.INPUT, true), "Control", "Conversion", "input PE");
038
039                p.add(new NOT(new ConversionSide(ConversionSide.Type.OTHER_SIDE)), "input PE", "Conversion", "input PE");
040
041                p.add(linkToSpecific(), "input PE", "input simple PE");
042                p.add(new Type(SequenceEntity.class), "input simple PE");
043                p.add(peToER(), "input simple PE", "changed generic ER");
044                p.add(new ConversionSide(ConversionSide.Type.OTHER_SIDE), "input PE", "Conversion", "output PE");
045
046                p.add(new NOT(new ConversionSide(ConversionSide.Type.OTHER_SIDE)), "output PE", "Conversion", "output PE");
047
048                p.add(equal(false), "input PE", "output PE");
049                p.add(linkToSpecific(), "output PE", "output simple PE");
050                p.add(peToER(), "output simple PE", "changed generic ER");
051                p.add(linkedER(false), "changed generic ER", "changed ER");
052
053                return p;
054        }

107        /**
108         * Pattern for a EntityReference has a member PhysicalEntity that is controlling a state change
109         * reaction of another EntityReference. In this case the controller is also an input to the
110         * reaction. The affected protein is the one that is represented with different non-generic
111         * physical entities at left and right of the reaction.
112         * @return the pattern
113         */
114        public static Pattern controlsStateChangeBothControlAndPart()
115        {
116                Pattern p = new Pattern(SequenceEntityReference.class, "controller ER");
117                p.add(linkedER(true), "controller ER", "controller generic ER");
118                p.add(erToPE(), "controller generic ER", "controller simple PE");
119                p.add(linkToComplex(), "controller simple PE", "controller PE");
120                p.add(peToControl(), "controller PE", "Control");
121                p.add(controlToConv(), "Control", "Conversion");
122
123                // the controller PE is also an input
124                p.add(new ParticipatesInConv(RelType.INPUT), "controller PE", "Conversion");
125
126                // same controller simple PE is also an output
127                p.add(linkToComplex(), "controller simple PE", "special output PE");
128                p.add(equal(false), "special output PE", "controller PE");
129                p.add(new ParticipatesInConv(RelType.OUTPUT), "special output PE", "Conversion");
130
131                stateChange(p, "Control");
132
133                // non-generic input and outputs are only associated with one side
134                p.add(equal(false), "input simple PE", "output simple PE");
135                p.add(new NOT(simplePEToConv(RelType.OUTPUT)), "input simple PE", "Conversion");
136                p.add(new NOT(simplePEToConv(RelType.INPUT)), "output simple PE", "Conversion");
137
138                p.add(equal(false), "controller ER", "changed ER");
139                p.add(type(SequenceEntityReference.class), "changed ER");
140
141                return p;
142        }

172        /**
173         * Pattern for a Conversion has an input PhysicalEntity and another output PhysicalEntity that
174         * belongs to the same EntityReference.
175         *
176         * @param p pattern to update
177         * @param ctrlLabel label
178         * @return the pattern
179         */
180        public static Pattern stateChange(Pattern p, String ctrlLabel)
181        {
182                if (p == null) p = new Pattern(Conversion.class, "Conversion");
183
184                if (ctrlLabel == null) p.add(new Participant(RelType.INPUT), "Conversion", "input PE");
185                else p.add(new Participant(RelType.INPUT, true), ctrlLabel, "Conversion", "input PE");
186
187                p.add(linkToSpecific(), "input PE", "input simple PE");
188                p.add(peToER(), "input simple PE", "changed generic ER");
189                p.add(new ConversionSide(ConversionSide.Type.OTHER_SIDE), "input PE", "Conversion", "output PE");
190                p.add(equal(false), "input PE", "output PE");
191                p.add(linkToSpecific(), "output PE", "output simple PE");
192                p.add(peToER(), "output simple PE", "changed generic ER");
193                p.add(linkedER(false), "changed generic ER", "changed ER");
194                return p;
195        }

345        /**
346         * Pattern for detecting two EntityReferences are controlling consecutive reactions, where
347         * output of one reaction is input to the other.
348         *
349         * @param blacklist to detect ubiquitous small molecules
350         * @return the pattern
351         */
352        public static Pattern catalysisPrecedes(Blacklist blacklist)
353        {
354                Pattern p = new Pattern(SequenceEntityReference.class, "first ER");
355                p.add(linkedER(true), "first ER", "first generic ER");
356                p.add(erToPE(), "first generic ER", "first simple controller PE");
357                p.add(linkToComplex(), "first simple controller PE", "first controller PE");
358                p.add(peToControl(), "first controller PE", "first Control");
359                p.add(controlToConv(), "first Control", "first Conversion");
360                p.add(new Participant(RelType.OUTPUT, blacklist, true), "first Control", "first Conversion", "linker PE");
361                p.add(new NOT(new ConstraintChain(new ConversionSide(ConversionSide.Type.OTHER_SIDE), linkToSpecific())), "linker PE", "first Conversion", "linker PE");
362                p.add(type(SmallMolecule.class), "linker PE");
363                p.add(new ParticipatesInConv(RelType.INPUT, blacklist), "linker PE", "second Conversion");
364                p.add(new NOT(new ConstraintChain(new ConversionSide(ConversionSide.Type.OTHER_SIDE), linkToSpecific())), "linker PE", "second Conversion", "linker PE");
365                p.add(equal(false), "first Conversion", "second Conversion");
366
367                // make sure that conversions are not replicates or reverse of each other
368                // and also outward facing sides of reactions contain at least one non ubique
369                p.add(new ConstraintAdapter(3, blacklist)
370                {
371                        @Override
372                        public boolean satisfies(Match match, int... ind)
373                        {
374                                Conversion cnv1 = (Conversion) match.get(ind[0]);
375                                Conversion cnv2 = (Conversion) match.get(ind[1]);
376                                SmallMolecule linker = (SmallMolecule) match.get(ind[2]);
377
378                                Set<PhysicalEntity> input1 = cnv1.getLeft().contains(linker) ? cnv1.getRight() : cnv1.getLeft();
379                                Set<PhysicalEntity> input2 = cnv2.getLeft().contains(linker) ? cnv2.getLeft() : cnv2.getRight();
380                                Set<PhysicalEntity> output1 = cnv1.getLeft().contains(linker) ? cnv1.getLeft() : cnv1.getRight();
381                                Set<PhysicalEntity> output2 = cnv2.getLeft().contains(linker) ? cnv2.getRight() : cnv2.getLeft();
382
383                                if (input1.equals(input2) && output1.equals(output2)) return false;
384                                if (input1.equals(output2) && output1.equals(input2)) return false;
385
386                                if (blacklist != null)
387                                {
388                                        Set<PhysicalEntity> set = new HashSet<PhysicalEntity>(input1);
389                                        set = blacklist.getNonUbiques(set, RelType.INPUT);
390                                        set.removeAll(output2);
391                                        if (set.isEmpty())
392                                        {
393                                                set.addAll(output2);
394                                                set = blacklist.getNonUbiques(set, RelType.OUTPUT);
395                                                set.removeAll(input1);
396
397                                                if (set.isEmpty()) return false;
398                                        }
399                                }
400                                return true;
401                        }
402                }, "first Conversion", "second Conversion", "linker PE");
403
404                p.add(new RelatedControl(RelType.INPUT, blacklist), "linker PE", "second Conversion", "second Control");
405                p.add(controllerPE(), "second Control", "second controller PE");
406                p.add(new NOT(compToER()), "second controller PE", "first ER");
407                p.add(linkToSpecific(), "second controller PE", "second simple controller PE");
408                p.add(type(SequenceEntity.class), "second simple controller PE");
409                p.add(peToER(), "second simple controller PE", "second generic ER");
410                p.add(linkedER(false), "second generic ER", "second ER");
411                p.add(equal(false), "first ER", "second ER");
412                return p;
413        }

436        /**
437         * Finds the cases where transcription relation is shown using a Conversion instead of a
438         * TemplateReaction.
439         * @return the pattern
440         */
441        public static Pattern controlsExpressionWithConversion()
442        {
443                Pattern p = new Pattern(SequenceEntityReference.class, "TF ER");
444                p.add(linkedER(true), "TF ER", "TF generic ER");
445                p.add(erToPE(), "TF generic ER", "TF SPE");
446                p.add(linkToComplex(), "TF SPE", "TF PE");
447                p.add(peToControl(), "TF PE", "Control");
448                p.add(controlToConv(), "Control", "Conversion");
449                p.add(new Size(right(), 1, Size.Type.EQUAL), "Conversion");
450                p.add(new OR(new MappedConst(new Empty(left()), 0), new MappedConst(new ConstraintAdapter(1)
451                {
452                        @Override
453                        public boolean satisfies(Match match, int... ind)
454                        {
455                                Conversion cnv = (Conversion) match.get(ind[0]);
456                                Set<PhysicalEntity> left = cnv.getLeft();
457                                if (left.size() > 1) return false;
458                                if (left.isEmpty()) return true;
459                                PhysicalEntity pe = left.iterator().next();
460                                if (pe instanceof NucleicAcid)
461                                {
462                                        PhysicalEntity rPE = cnv.getRight().iterator().next();
463                                        return rPE instanceof Protein;
464                                }
465                                return false;
466                        }
467                }, 0)), "Conversion");
468                p.add(right(), "Conversion", "right PE");
469                p.add(linkToSpecific(), "right PE", "right SPE");
470                p.add(new Type(SequenceEntity.class), "right SPE");
471                p.add(peToER(), "right SPE", "product generic ER");
472                p.add(linkedER(false), "product generic ER", "product ER");
473                p.add(equal(false), "TF ER", "product ER");
474                return p;
475        }

673        /**
674         * Finds ProteinsReference related to an interaction. If specific types of interactions are
675         * desired, they should be sent as parameter, otherwise leave the parameter empty.
676         *
677         * @param seedType specific BioPAX interaction sub-types (interface classes)
678         * @return pattern
679         */
680        public static Pattern relatedProteinRefOfInter(Class<? extends Interaction>... seedType)
681        {
682                Pattern p = new Pattern(Interaction.class, "Interaction");
683
684                if (seedType.length == 1)
685                {
686                        p.add(new Type(seedType[0]), "Interaction");
687                }
688                else if (seedType.length > 1)
689                {
690                        MappedConst[] mc = new MappedConst[seedType.length];
691                        for (int i = 0; i < mc.length; i++)
692                        {
693                                mc[i] = new MappedConst(new Type(seedType[i]), 0);
694                        }
695
696                        p.add(new OR(mc), "Interaction");
697                }
698
699                p.add(new OR(new MappedConst(participant(), 0, 1),
700                        new MappedConst(new PathConstraint(
701                                "Interaction/controlledOf*/controller:PhysicalEntity"), 0, 1)),
702                        "Interaction", "PE");
703
704                p.add(linkToSpecific(), "PE", "SPE");
705                p.add(peToER(), "SPE", "generic PR");
706                p.add(new Type(ProteinReference.class), "generic PR");
707                p.add(linkedER(false), "generic PR", "PR");
708                return p;
709        }
