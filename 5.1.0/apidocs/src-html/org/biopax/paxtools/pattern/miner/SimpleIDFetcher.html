001package org.biopax.paxtools.pattern.miner;
002
003import org.biopax.paxtools.model.BioPAXElement;
004
005import java.util.Collections;

008/**
009 * Simple gets the URI (of an entity reference);
010 */
011public class SimpleIDFetcher implements IDFetcher
012{
013        @Override
014        public Set<String> fetchID(BioPAXElement ele)
015        {
016                return Collections.singleton(ele.getUri());
017        }
018
019}
