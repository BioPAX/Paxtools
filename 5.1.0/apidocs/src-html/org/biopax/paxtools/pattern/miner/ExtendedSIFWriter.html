001package org.biopax.paxtools.pattern.miner;
002
003import org.biopax.paxtools.controller.PathAccessor;
004import org.biopax.paxtools.model.BioPAXElement;
005import org.biopax.paxtools.model.level3.RelationshipXref;
006
007import java.io.IOException;
008import java.io.OutputStream;
009import java.io.OutputStreamWriter;
010import java.io.Writer;
011import java.util.*;
012
013/**
014 * This class is used for writing the binary interactions to a text output stream in the
015 * so called Pathway Commons EXTENDED_BINARY_SIF format.
016 *
017 * @author Ozgun Babur
018 */
019public class ExtendedSIFWriter
020{
021        /**
022         * Path accessors for participant properties.
023         */
024        private static final PathAccessor[] ACCS = new PathAccessor[]{
025                new PathAccessor("Named/displayName"),
026                new PathAccessor("XReferrable/xref:UnificationXref"),
027                new PathAccessor("XReferrable/xref:RelationshipXref")};
028
029        /**
030         * Writes down the given inferred binary interactions into the output stream
031         * using the Pathway Commons one-file EXTENDED_BINARY_SIF format,
032         * where edges (interactions) are written first, followed by a single
033         * blank line, followed by the nodes section (participant details and annotations).
034         * Closes the output stream at the end.
035         * @param inters binary interactions
036         * @param out stream to write
037         * @return true if any output produced successfully
038         */
039        public static boolean write(Set<SIFInteraction> inters, OutputStream out)
040        {
041                SIFToText stt = new CustomFormat(
042                        OutputColumn.Type.RESOURCE.name(),
043                        OutputColumn.Type.PUBMED.name(),
044                        OutputColumn.Type.PATHWAY.name(),
045                        OutputColumn.Type.MEDIATOR.name());
046
047                if (!inters.isEmpty())
048                {
049                        List<SIFInteraction> interList = new ArrayList<SIFInteraction>(inters);
050                        Collections.sort(interList);
051                        try {
052                                OutputStreamWriter writer = new OutputStreamWriter(out);
053                                writer.write("PARTICIPANT_A\tINTERACTION_TYPE\tPARTICIPANT_B\t" +
054                                                                "INTERACTION_DATA_SOURCE\tINTERACTION_PUBMED_ID\tPATHWAY_NAMES\tMEDIATOR_IDS");
055                                for (SIFInteraction inter : interList) {
056                                        writer.write("\n" + stt.convert(inter));
057                                }
058                                writer.write("\n\n");//last line's EOL, + one blank line
059                                writeSourceAndTargetDetails(inters, writer);
060                                writer.close();
061                                return true;
062                        }
063                        catch (IOException e) {
064                                e.printStackTrace();
065                        }
066                }
067
068                return false;
069        }
070
071        /**
072         * Writes down the given interactions into the given "edges" output stream.
073         * Closes the stream at the end.
074         *
075         * @param inters binary interactions
076         * @param out stream to write
077         * @return true if any output produced successfully
078         */
079        public static boolean writeInteractions(Set<SIFInteraction> inters, OutputStream out)
080        {
081                SIFToText stt = new CustomFormat(
082                                OutputColumn.Type.RESOURCE.name(),
083                                OutputColumn.Type.PUBMED.name(),
084                                OutputColumn.Type.PATHWAY.name(),
085                                OutputColumn.Type.MEDIATOR.name());
086
087                if (!inters.isEmpty())
088                {
089                        List<SIFInteraction> interList = new ArrayList<SIFInteraction>(inters);
090                        Collections.sort(interList);
091                        try
092                        {
093                                OutputStreamWriter writer = new OutputStreamWriter(out);
094                                writer.write("PARTICIPANT_A\tINTERACTION_TYPE\tPARTICIPANT_B\t" +
095                                                "INTERACTION_DATA_SOURCE\tINTERACTION_PUBMED_ID\tPATHWAY_NAMES\tMEDIATOR_IDS");
096                                for (SIFInteraction inter : interList)
097                                {
098                                        writer.write("\n" + stt.convert(inter));
099                                }
100                                writer.close();
101                                return true;
102                        }
103                        catch (IOException e)
104                        {
105                                e.printStackTrace();
106                        }
107                }
108                return false;
109        }
110
111        /**
112         * Writes down the interaction participants' (nodes) details
113         * to the given output stream. Closes the stream at the end.
114         *
115         * @param inters binary interactions
116         * @param out stream to write
117         * @return true if any output produced successfully
118         */
119        public static boolean writeParticipants(Set<SIFInteraction> inters, OutputStream out)
120        {
121                if (!inters.isEmpty())
122                {
123                        try
124                        {
125                                OutputStreamWriter writer = new OutputStreamWriter(out);
126                                writeSourceAndTargetDetails(inters, writer);
127                                writer.close();
128                                return true;
129                        }
130                        catch (IOException e)
131                        {
132                                e.printStackTrace();
133                        }
134                }
135                return false;
136        }
137
138        private static void writeSourceAndTargetDetails(Collection<SIFInteraction> inters, Writer writer)
139                throws IOException
140        {
141                Map<String, Set<BioPAXElement>> map = collectEntityRefs(inters);
142                //write the column headers row first
143                writer.write("PARTICIPANT\tPARTICIPANT_TYPE\tPARTICIPANT_NAME\tUNIFICATION_XREF\tRELATIONSHIP_XREF");
144                for (String id : map.keySet())
145                {
146                        writer.write("\n" + id + "\t" + getParticipantTypes(map.get(id)));
147                        for (PathAccessor acc : ACCS)
148                        {
149                                writer.write("\t" + getValue(map.get(id), acc));
150                        }
151                }
152        }
153
154        /**
155         * Collects the sources and targets in given interactions.
156         * @param inters binary interactions
157         * @return map from the primary id to the set of related source and target elements.
158         */
159        private static Map<String, Set<BioPAXElement>> collectEntityRefs(Collection<SIFInteraction> inters)
160        {
161                Map<String, Set<BioPAXElement>> map = new HashMap<String, Set<BioPAXElement>>();
162
163                for (SIFInteraction inter : inters)
164                {
165                        if (!map.containsKey(inter.sourceID))
166                                map.put(inter.sourceID, new HashSet<BioPAXElement>());
167                        if (!map.containsKey(inter.targetID))
168                                map.put(inter.targetID, new HashSet<BioPAXElement>());
169
170                        map.get(inter.sourceID).addAll(inter.sourceERs);
171                        map.get(inter.targetID).addAll(inter.targetERs);
172                }
173                return map;
174        }
175
176        private static String getParticipantTypes(Set<BioPAXElement> elements)
177        {
178                Set<String> set = new HashSet<String>();
179                for (BioPAXElement ele : elements)
180                {
181                        String name = ele.getModelInterface().getName();
182                        set.add(name.substring(name.lastIndexOf(".") + 1));
183                }
184                List<String> list = new ArrayList<String>(set);
185                Collections.sort(list);
186                return concat(list);
187        }
188
189        private static String getValue(Set<BioPAXElement> elements, PathAccessor pa)
190        {
191                Set<String> set = new HashSet<String>();
192                for (Object o : pa.getValueFromBeans(elements))
193                {
194                        if(o instanceof RelationshipXref)
195                        {
196                                RelationshipXref rx = (RelationshipXref) o;
197                                if(rx.getRelationshipType() != null
198                                                && rx.getRelationshipType().toString().toLowerCase().contains("identity")) {
199                                        set.add(rx.toString());
200                                }
201                        }
202                        else {
203                                set.add(o.toString());
204                        }
205                }
206                List<String> list = new ArrayList<String>(set);
207                Collections.sort(list);
208                return concat(list);
209        }
210
211        /**
212         * Concatenates the given collection of strings into a single string where values are separated
213         * with a semicolon.
214         * @param col string collection
215         * @return concatenated string
216         */
217        private static String concat(Collection<String> col)
218        {
219                StringBuilder b = new StringBuilder();
220                boolean first = true;
221                for (String s : col)
222                {
223                        if (first)
224                                first = false;
225                        else
226                                b.append(";");
227
228                        b.append(s);
229                }
230                return b.toString();
231        }
