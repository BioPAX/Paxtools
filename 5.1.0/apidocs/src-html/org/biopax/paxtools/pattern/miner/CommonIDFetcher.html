001package org.biopax.paxtools.pattern.miner;
002
003import org.biopax.paxtools.model.BioPAXElement;
004import org.biopax.paxtools.model.level3.*;
005import org.biopax.paxtools.pattern.util.HGNC;
006import org.biopax.paxtools.util.IllegalBioPAXArgumentException;
007
008import java.util.Comparator;
009import java.util.HashSet;
010import java.util.Set;
011import java.util.TreeSet;
012
013/**
014 * Tries to get Gene Symbols or UniProt IDs for genes
015 * and - display names for small molecules;
016 *
017 * This id-fetcher is mainly to use with the BioPAX pathway data
018 * from Pathway Commons (PC2) db (normalized and enriched with xrefs).
019 *
020 * @author Ozgun Babur et al.
021 */
022public class CommonIDFetcher implements IDFetcher
023{
024        boolean useUniprotIDs = false;
025
026        public Set<String> fetchID(BioPAXElement ele)
027        {
028                Set<String> set = new HashSet<String>();
029
030                if (ele instanceof SmallMoleculeReference || ele instanceof SmallMolecule)
031                {
032                        Named e = (Named) ele;
033                        //avoid shortened/incomplete names -
034                        if (e.getDisplayName() != null && !e.getDisplayName().contains("..."))
035                                set.add(e.getDisplayName());
036                        else if (e.getStandardName() != null && !e.getStandardName().contains("..."))
037                                set.add(e.getStandardName());
038                        else if (!e.getName().isEmpty()) {
039                                Set<String> names = new TreeSet<String>();
040                                for(String name : e.getName()) {
041                                        if(!name.contains("..."))
042                                                names.add(name);
043                                }
044                                set.add(names.toString());
045                        }
046                }
047                else if (useUniprotIDs && ele.getUri().startsWith("http://identifiers.org/uniprot/"))
048                {
049                        set.add(ele.getUri().substring(ele.getUri().lastIndexOf("/") + 1));
050                }
051                else if (ele instanceof XReferrable)
052                {
053                        for (Xref xr : ((XReferrable) ele).getXref())
054                        {
055                                if(xr instanceof PublicationXref)
056                                        continue;
057
058                                String db = xr.getDb();
059                                String id = xr.getId();
060                                if (db != null && id != null && !id.isEmpty())
061                                {
062                                        db = db.toLowerCase();
063                                        if (!useUniprotIDs && db.startsWith("hgnc"))
064                                        {
065                                                //valid id is either a HGNC:1234 ID or Symbol (gene name)
066                                                String symbol = HGNC.getSymbol(id);
067                                                if (symbol != null && !symbol.isEmpty())
068                                                {
069                                                        set.add(symbol);
070                                                }
071                                        }
072                                        else if (useUniprotIDs && db.startsWith("uniprot"))
073                                        {
074                                                set.add(id);
075                                        }
076                                }
077                        }
078                }
079
080                if (set.isEmpty()) {
081                        // a second try - getting mirtarbase ids for a nucleic acid...
082                        if (ele instanceof NucleicAcidReference || ele instanceof NucleicAcid)
083                        {
084                                for (Xref xr : ((XReferrable) ele).getXref()) {
085                                        if(xr instanceof PublicationXref) continue;
086
087                                        String db = xr.getDb();
088                                        String id = xr.getId();
089                                        if (db != null && id != null && !id.isEmpty()) {
090                                                db = db.toLowerCase();
091                                                if (db.equals("mirbase sequence")) {
092                                                        set.add(id);
093                                                }
094                                        }
095                                }
096                        }
097                }
098
099                return set;
100        }
101
102        public void setUseUniprotIDs(boolean useUniprotIDs)
103        {
104                this.useUniprotIDs = useUniprotIDs;
105        }
106}
