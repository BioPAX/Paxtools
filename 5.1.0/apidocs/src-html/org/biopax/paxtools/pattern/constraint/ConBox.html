069         * @return generative constraint to get the Interaction that the PhysicalEntity is a controller
070         */

072        {
073                return new OR(
074                        new MappedConst(peToControlledInter(), 0, 1),
075                        new MappedConst(participatesInInter(), 0, 1));
076        }

614        /**
615         * Makes sure that the PhysicalEntity is controlling more reactions than it participates
616         * (excluding complex assembly).
617         * @return non-generative constraint
618         */
619        public static Constraint moreControllerThanParticipant()
620        {
621                return new ConstraintAdapter(1)
622                {
623                        PathAccessor partConv = new PathAccessor("PhysicalEntity/participantOf:Conversion");
624                        PathAccessor partCompAss = new PathAccessor("PhysicalEntity/participantOf:ComplexAssembly");
625                        PathAccessor effects = new PathAccessor("PhysicalEntity/controllerOf/controlled*:Conversion");
626
627                        @Override
628                        public boolean satisfies(Match match, int... ind)
629                        {
630                                PhysicalEntity pe = (PhysicalEntity) match.get(ind[0]);
631
632                                int partCnvCnt = partConv.getValueFromBean(pe).size();
633                                int partCACnt = partCompAss.getValueFromBean(pe).size();
634                                int effCnt = effects.getValueFromBean(pe).size();
635
636                                return (partCnvCnt - partCACnt) <= effCnt;
637                        }
638                };
639        }
