001package org.biopax.paxtools.pattern.util;
002
003import org.biopax.paxtools.model.level3.*;
004
005import java.util.Collections;
006import java.util.HashSet;
007import java.util.Set;
008
009/**
010 * THis class is used for comparing modification features of two PhysicalEntity objects of same
011 * EntityReference.
012 *
013 * @author Ozgun Babur
014 */
015public class DifferentialModificationUtil
016{
017        /**
018         * Gets the differential features.
019         * @param before first entity
020         * @param after second entity
021         * @return array of differential features. index 0: gained features, index 1: lost features
022         */
023        public static Set<ModificationFeature>[] getChangedModifications(PhysicalEntity before,
024                PhysicalEntity after)
025        {
026                Set<Modification> set1 = collectFeatures(before);
027                Set<Modification> set2 = collectFeatures(after);
028
029                Set<Modification> temp = new HashSet<Modification>(set1);
030                set1.removeAll(set2);
031                set2.removeAll(temp);
032
033                // Remove common features that can be deemed semantically equivalent
034
035                Set<Modification> furtherRemove = new HashSet<Modification>();
036
037                for (Modification m1 : set1)
038                {
039                        for (Modification m2 : set2)
040                        {
041                                if (furtherRemove.contains(m2)) continue;
042
043                                if (m1.getKey().equals(m2.getKey()))
044                                {
045                                        furtherRemove.add(m1);
046                                        furtherRemove.add(m2);
047                                        break;
048                                }
049                        }
050                }
051
052                set1.removeAll(furtherRemove);
053                set2.removeAll(furtherRemove);
054
055                return new Set[]{collectFeatures(set2), collectFeatures(set1)};
056        }
057
058        public static Set<String> collectChangedPhosphorylationSites(PhysicalEntity before,
059                PhysicalEntity after, boolean gained)
060        {
061                Set<ModificationFeature> set = getChangedModifications(before, after)[gained ? 0 : 1];
062                Set<String> sites = new HashSet<String>();
063                for (ModificationFeature mf : set)
064                {
065                        String lett = getPhosphoSiteLetter(mf);
066                        if (lett != null)
067                        {
068                                int site = getPhosphoSite(mf);
069                                if (site > 0)
070                                {
071                                        sites.add(lett + site);
072                                }
073                        }
074                }
075                return sites;
076        }
077
078        private static Set<Modification> collectFeatures(PhysicalEntity pe)
079        {
080                Set<Modification> set = new HashSet<Modification>();
081
082                for (EntityFeature f : pe.getFeature())
083                {
084                        if (f instanceof ModificationFeature)
085                                set.add(new Modification((ModificationFeature) f));
086                }
087                return set;
088        }
089
090        private static Set<ModificationFeature> collectFeatures(Set<Modification> wrappers)
091        {
092                if (wrappers.isEmpty()) return Collections.emptySet();
093
094                Set<ModificationFeature> set = new HashSet<ModificationFeature>();
095
096                for (Modification wrapper : wrappers)
097                {
098                        set.add(wrapper.mf);
099                }
100                return set;
101        }
102
103        /**
104         * Wrapper for ModificationFeature.
105         */
106        private static class Modification
107        {
108                ModificationFeature mf;
109
110                Modification(ModificationFeature mf)
111                {
112                        this.mf = mf;
113                }
114
115                @Override
116                public int hashCode()
117                {
118                        int code = 0;
119
120                        SequenceModificationVocabulary type = mf.getModificationType();
121                        if (type != null)
122                        {
123                                for (String term : type.getTerm())
124                                {
125                                        code += term.hashCode();
126                                }
127                        }
128
129                        SequenceLocation loc = mf.getFeatureLocation();
130                        if (loc != null)
131                        {
132                                if (loc instanceof SequenceInterval)
133                                {
134                                        SequenceSite begin = ((SequenceInterval) loc).getSequenceIntervalBegin();
135                                        SequenceSite end = ((SequenceInterval) loc).getSequenceIntervalEnd();
136
137                                        if (begin != null) code += begin.getSequencePosition();
138                                        if (end != null) code += end.getSequencePosition();
139                                }
140                                else if (loc instanceof SequenceSite)
141                                {
142                                        code += ((SequenceSite) loc).getSequencePosition();
143                                }
144                        }
145                        return code;
146                }
147
148                @Override
149                public boolean equals(Object obj)
150                {
151                        if (!(obj instanceof Modification)) return false;
152
153                        Modification m = (Modification) obj;
154
155                        if (mf == m.mf) return true;
156
157                        if (mf.getModificationType() == null || m.mf.getModificationType() == null)
158                                return false;
159
160                        if (!mf.getModificationType().getTerm().containsAll(
161                                m.mf.getModificationType().getTerm())) return false;
162
163                        if (mf.getFeatureLocation() == null && m.mf.getFeatureLocation() == null) return true;
164
165                        if (mf.getFeatureLocation() == null || m.mf.getFeatureLocation() == null) return false;
166
167                        return mf.getFeatureLocation().equivalenceCode() ==
168                                m.mf.getFeatureLocation().equivalenceCode();
169                }
170
171                public String getKey()
172                {
173                        String k = "";
174                        if (mf.getModificationType() != null && !mf.getModificationType().getTerm().isEmpty())
175                        {
176                                k = mf.getModificationType().getTerm().iterator().next();
177
178                                if (mf.getFeatureLocation() instanceof SequenceSite)
179                                {
180                                        SequenceSite ss = (SequenceSite) mf.getFeatureLocation();
181                                        int site = ss.getSequencePosition();
182                                        // 100,000 is a relaxed upper bound for possible size of a human protein
183                                        if (site > 0 && site < 1E5)
184                                        {
185                                                k += "@" + site;
186                                        }
187                                }
188                                else if (mf.getFeatureLocation() instanceof SequenceInterval)
189                                {
190                                        SequenceInterval si = (SequenceInterval) mf.getFeatureLocation();
191
192                                        if (si.getSequenceIntervalBegin() != null && si.getSequenceIntervalEnd() != null)
193                                        {
194                                                int begin = si.getSequenceIntervalBegin().getSequencePosition();
195                                                int end = si.getSequenceIntervalEnd().getSequencePosition();
196
197                                                if (begin > 0 && begin < end && end < 1E5)
198                                                {
199                                                        k += "@[" + begin + "-" + end + "]";
200                                                }
201                                        }
202                                }
203                        }
204                        return k;
205                }
206        }
207
208        private static String getPhosphoSiteLetter(ModificationFeature mf)
209        {
210                if (mf.getModificationType() != null)
211                {
212                        for (String term : mf.getModificationType().getTerm())
213                        {
214                                term = term.toLowerCase();
215                                if (term.contains("phospho"))
216                                {
217                                        if (term.contains("serine")) return "S";
218                                        if (term.contains("threonine")) return "T";
219                                        if (term.contains("tyrosine")) return "Y";
220                                }
221                        }
222                }
223                return null;
224        }
225
226        private static int getPhosphoSite(ModificationFeature mf)
227        {
228                SequenceLocation loc = mf.getFeatureLocation();
229                if (loc != null && loc instanceof SequenceSite)
230                {
231                        return ((SequenceSite) loc).getSequencePosition();
232                }
233                return -1;
234        }
