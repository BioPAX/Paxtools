001package org.biopax.paxtools.controller;
002
003import org.biopax.paxtools.model.BioPAXElement;
004import org.biopax.paxtools.model.Model;
005import org.biopax.paxtools.model.level3.BiochemicalPathwayStep;
006import org.biopax.paxtools.model.level3.Conversion;
007import org.biopax.paxtools.util.Filter;
008import org.slf4j.Logger;
009import org.slf4j.LoggerFactory;
010
011import java.util.Collection;
012import java.util.Collections;
013import java.util.HashSet;
014import java.util.Set;
015
016/**
017 * A "simple" BioPAX merger, a utility class to merge
018 * 'source' BioPAX models or a set of elements into the target model,
019 * using (URI) identity only. Merging into a normalized,
020 * self-integral model normally makes sense and gives better results
021 * (but it depends on the application though).
022 * 
023 * One can also "merge" a model to itself, i.e.: merge(target,target),
024 * or to an empty one, which adds all implicit child elements 
025 * to the model and makes it self-integral.
026 * 
027 * Note, "URI identity" means that it does not copy
028 * a source element to the target model if the target already has an element 
029 * with the same URI. However, it will update (re-wire) all the object 
030 * properties of new elements to make sure they do not refer to any objects 
031 * outside the updated target model.
032 * 
033 * We do not guarantee the integrity of the source models after the merge is done
034 * (some object properties will refer to target elements).
035 * 
036 * Finally, although called Simple Merger, it is in fact an advanced BioPAX utility,
037 * which should be used wisely. Otherwise, it can actually waste resources.
038 * So, consider using model.add(..), model.addNew(..) approach first (or instead),
039 * especially, when you're adding "new" things (ID not present in the target model),
040 * or/and target model does not contain any references to the source or another one, etc.
041 */
042public class SimpleMerger
043{
044        private static final Logger LOG = LoggerFactory.getLogger(SimpleMerger.class);
045
046        private final EditorMap map;
047        
048        private Filter<BioPAXElement> mergePropOf;
049
050        /**
051         * @param map a class to editor map for the elements to be modified.
052         */
053        public SimpleMerger(EditorMap map)
054        {
055                this.map = map;
056        }
057
058        /** 
059         * @param map a class to editor map for the elements to be modified.
060         * @param mergePropOf when not null, all multiple-cardinality properties
061         *                                              of a source biopax object that passes this filter are updated
062         *                                              and also copied to the corresponding (same URI) target object,
063         *                      unless the source and target are the same thing 
064         *                      (in which case, we simply migrate object properties 
065         *                      to target model objects). 
066         */
067        public SimpleMerger(EditorMap map, Filter<BioPAXElement> mergePropOf)
068        {
069                this(map);
070                this.mergePropOf = mergePropOf;
071        }
072
073        /**
074         * Merges the <em>source</em> models into <em>target</em> model,
075         * one after another (in the order they are listed).
076         * 
077         * If the target model is self-integral (complete) or empty, then
078         * the result of the merge will be also a complete model (contain 
079         * unique objects with unique URIs, all objects referenced from 
080         * any other object in the model).
081         * 
082         * Source models do not necessarily have to be complete and may even
083         * indirectly contain different objects of the same type with the same 
084         * URI. Though, in most cases, one probably wants target model be complete
085         * or empty for the best possible results. So, if your target is incomplete, 
086         * or you are not quite sure, then do simply merge it as the first source 
087         * to a new empty model or itself (or call {@link Model#repair()} first).
088         *       
089         * @param target model into which merging process will be done
090         * @param sources models to be merged/updated to <em>target</em>; order can be important
091         */
092        public void merge(Model target, Model... sources)
093        {
094                for (Model source : sources)
095                        if (source != null)
096                                merge(target, source.getObjects());
097        }
098
099
100        /**
101         * Merges the <em>elements</em> and all their child biopax objects
102         * into the <em>target</em> model.
103         * 
104         * @see #merge(Model, Model...) for details about the target model.
105         * 
106         * @param target model into which merging will be done
107         * @param elements elements that are going to be merged/updated to <em>target</em>
108         */
109        public void merge(Model target, Collection<? extends BioPAXElement> elements)
110        {
111                @SuppressWarnings("unchecked")
112                final Fetcher fetcher = new Fetcher(map);
113                
114                // Auto-complete source 'elements' by discovering all the implicit elements there
115                // copy all elements, as the collection can be immutable or unsafe to add elements to
116                final Set<BioPAXElement> sources = new HashSet<BioPAXElement>(elements);
117                for(BioPAXElement se : elements) {
118                        sources.addAll(fetcher.fetch(se));
119                }
120                                
121                // Next, we only copy elements having new URIs -
122                for (BioPAXElement bpe : sources)
123                {
124                        /* if there exists target element with the same id, 
125                         * do not copy this one! (this 'source' element will 
126                         * be soon replaced with the target's, same id, one 
127                         * in all parent objects)
128                         */
129                        final String uri = bpe.getUri();
130                        if (!target.containsID(uri))
131                        {
132                                /*
133                                 * Warning: other than the default (ModelImpl) target Model 
134                                 * implementations may add child elements recursively (e.g., 
135                                 * using jpa cascades/recursion); it might also override target's
136                                 * properties with the corresponding ones from the source, even
137                                 * though SimpleMerger is not supposed to do this; also, is such cases,
138                                 * the number of times this loop body is called can be less that
139                                 * the number of elements in sourceElements set that weren't
140                                 * originally present in the target model, or - even equals to
141                                 * one)
142                                 */
143                                target.add(bpe);
144                        }
145                        else if(bpe.getModelInterface() != target.getByID(uri).getModelInterface()) {
146                                // source object/model(s) have to be fixed (URI conflicts resolved) before merging by URI
147                                throw new RuntimeException(String.format(
148                                        "URI:%s of %s (to merge) is also URI of %s in target model (different class)",
149                                                uri, bpe.getModelInterface().getSimpleName(),
150                                                        target.getByID(uri).getModelInterface().getSimpleName()));
151                        }
152                }
153
154                // Finally, update object references
155                for (BioPAXElement bpe : sources) {
156                        updateObjectFields(bpe, target);
157                }
158
159        }
160
161
162        /**
163         * Merges the <em>source</em> element (and its "downstream" dependents)
164         * into <em>target</em> model.
165         * 
166         * @see #merge(Model, Collection)
167         * 
168         * @param target the BioPAX model to merge into
169         * @param source object to add or merge
170         */
171        public void merge(Model target, BioPAXElement source)
172        {
173                merge(target, Collections.singleton(source));
174        }
175
176
177        /**
178         * Updates each value of <em>existing</em> element, using the value(s) of <em>update</em>.
179         * @param source BioPAX element of which values are used for update
180         * @param target the BioPAX model
181         */
182        private void updateObjectFields(BioPAXElement source, Model target)
183        {
184                //Skip if target model had a different object with the same URI as the source's,
185                //and no Filter was set (mergePropOf is null); i.e., when
186                //we simply want the source to be replaced with the object
187                //having the same type and URI and that's already present in the target model.
188                BioPAXElement keep = target.getByID(source.getUri());
189                if(keep != source && mergePropOf ==null)
190                {
191                        return; //nothing to do
192                }
193                
194                Set<PropertyEditor> editors = map.getEditorsOf(source);
195                for (PropertyEditor editor : editors)
196                {
197                        if (editor instanceof ObjectPropertyEditor)
198                        {
199                                //copy prop. values (to avoid concurrent modification exception)
200                                Set<BioPAXElement> values = new HashSet<BioPAXElement>((Set<BioPAXElement>) editor.getValueFromBean(source));
201                                if(keep == source) //i.e., it has been just added to the target, - simply update properties
202                                {
203                                        for (BioPAXElement value : values) {
204                                                migrateToTarget(source, target, editor, value);
205                                        }
206                                } else //source is normally to be entirely replaced, but if it passes the filter,
207                                        if(mergePropOf !=null && mergePropOf.filter(source)
208                                                && editor.isMultipleCardinality()) // and the prop. is multi-cardinality,
209                                {
210                                        // - then we want to copy some values
211                                        for (BioPAXElement value : values) {
212                                                mergeToTarget(keep, target, editor, value);
213                                        }
214                                }
215
216                        }
217                        else // - primitive, enum, or string property editor (e.g., comment, name), and -
218                                if (mergePropOf != null && mergePropOf.filter(source) && editor.isMultipleCardinality())
219                        {
220                                Set<Object> values = new HashSet<Object>((Set<Object>) editor.getValueFromBean(source));
221                                for (Object value : values) {
222                                        mergeToTarget(keep, target, editor, value);
223                                }
224                        }
225                }
226        }
227        
228
229        private void migrateToTarget(BioPAXElement source, Model target, PropertyEditor editor, BioPAXElement value)
230        {
231                if (value != null)
232                {
233                        BioPAXElement newValue = target.getByID(value.getUri());
234                        //not null at this point, because every source element was found 
235                        //and either added to the target model, or target had an object with the same URI (to replace this value).
236                        assert newValue != null : "'newValue' is null (a design flaw in the 'merge' method)";           
237                        if (newValue != value) { //not using 'equals' intentionally
238                                editor.removeValueFromBean(value, source);
239                                editor.setValueToBean(newValue, source);
240                        } 
241                }
242        }
243        
244        private void mergeToTarget(BioPAXElement targetElement, Model target, PropertyEditor editor, Object value)
245        {
246                if (value != null) {
247                        Object newValue = (value instanceof BioPAXElement) 
248                                        ? target.getByID(((BioPAXElement)value).getUri()) : value;
249
250                        //a "dirty" fix (avoid warnings in the log that can be ignored)
251                        if(targetElement instanceof BiochemicalPathwayStep
252                                        && editor.getProperty().equalsIgnoreCase("stepProcess")
253                                        && value instanceof Conversion) {
254                                // In this case, the Conversion should be set via stepConversion property editor, which is
255                                // sub-property of stepProcess (getStepProcess can include a step conversion as well, but
256                                // Conversion is not supposed to be set explicitly via stepProcess)
257                                if(LOG.isDebugEnabled())
258                                        LOG.debug("mergeToTarget, ignored an attempt to set a Conversion via " +
259                                                "BiochemicalPathwayStep.stepProcess property editor; target URI:" + targetElement.getUri());
260                                return;
261                        }
262
263                        editor.setValueToBean(newValue, targetElement);
264                }
265        }
