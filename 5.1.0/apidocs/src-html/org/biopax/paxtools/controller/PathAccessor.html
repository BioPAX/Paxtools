012/**
013 * This class is a composite property accessor that allows users to chain multiple
014 * property accessors to define paths in the BioPAX object graph.
015 * 
016 * The path can be defined by either explicitly providing a set of property accessors or
017 * by an xPath like syntax.
018 * 
019 * The path can contain transitive and or restricted sub paths:
020 * A transitive sub-path is traversed recursively as many times as possible. e.g. Complex/component* will
021 * return all components of the complex and the components of the components if those components
022 * are complex recursively.
023 * 
024 * A restricted subpath is restricted with a class and only follows through the domains that are
025 * assignable from the the restriction class. e.g. Control/controlled:Pathway/name will only return the
026 * names of the pathways that are being controlled, but not interactions.
027 */
028public class PathAccessor extends PropertyAccessorAdapter<BioPAXElement, Object>
029{
030
031        List<PropertyAccessor<? extends BioPAXElement, ?>> accessors;
032
033        Class<? extends BioPAXElement> domain;
034
035        List<Class<? extends BioPAXElement>> domainOrder = new ArrayList<Class<? extends BioPAXElement>>();
036
037        public static final Logger log = LoggerFactory.getLogger(PathAccessor.class);
038
039        BioPAXLevel level;
040
041        /**
042         * Constructor for defining the access path with a list of accessors. All of the accessors must be of the same
043         * level. In the case one accessor's range can not be assigned by the domain of the next( broken path),
044         * this accessor will return an empty set.
045         * @param objectAccessors A list of object accessors.
046         * @param level biopax level
047         */
048        public PathAccessor(List<PropertyAccessor<? extends BioPAXElement, ?>> objectAccessors, BioPAXLevel level)
049        {
050                super(BioPAXElement.class, Object.class, true);
051                this.accessors = objectAccessors;
052                this.domain = objectAccessors.get(0).getDomain();
053                this.level = level;
054        }
055
056
057        /**
058         * Constructor for defining the access path via a XPath like string.
059         * @param path The string defining the path. The following operators can be used:
060         * <ul>
061         * <li>The first token is the class name of the starting domain.</li>
062         * <li>The next tokens are names of the properties, separated by "/"</li>
063         * <li>A property step can be restricted to a domain by defining the restriction class with ":" after the
064         * property </li>
065         * <li>A property step can be declared transitive by adding a "*" at the end. </li>
066         * <li>Two accessors can be joined by "|" (Currently not implemented)</li>
067         * <li>A subpath can be defined by a substring enclosed in "(" ")"</li>
068         * <li>Subpaths can also be restricted or made transitive</li>
069         * </ul>
070         * @param level BioPAX level that this path is defined in.
071         */
072        public PathAccessor(String path, BioPAXLevel level)
073        {
074                super(BioPAXElement.class, Object.class, true);
075                this.level = level;
076                parsePath(path, 0);
077        }
078
079        private PathAccessor(StringTokenizer tk, BioPAXLevel level, Class<? extends BioPAXElement> domain, int depth)
080        {
081                super(BioPAXElement.class, Object.class, true);
082                this.domain = domain;
083                this.level = level;
084                if (iterateTheRemainingPath(depth, tk) != depth)
085                        throw new IllegalBioPAXArgumentException("Unexpected element. Parentheses do not match!");
086        }
087
088        private int parsePath(String path, int depth)
089        {
090                StringTokenizer st = new StringTokenizer(path, "/:*|()", true);
091                domain = getClass(st.nextToken());
092                this.accessors = new ArrayList<PropertyAccessor<? extends BioPAXElement, ?>>();
093                if (st.nextToken().equals("/"))
094                {
095                        accessors.add(getStepAccessor(level, st, domain));
096                        domainOrder.add(domain);
097                } else throw new IllegalArgumentException();
098                return iterateTheRemainingPath(depth, st);
099        }
100
101        private int iterateTheRemainingPath(int depth, StringTokenizer st)
102        {
103                Class<? extends BioPAXElement> intermediate = getNextDomain();
104                while (st.hasMoreTokens())
105                {
106                        String s = st.nextToken();
107                        if (s.equals("("))
108                        {
109                                PathAccessor subPath = new PathAccessor(st, level, intermediate, depth++);
110                                accessors.add(subPath);
111                                domainOrder.add(intermediate);
112                                intermediate = getNextDomain();
113                        } else if (s.equals(")"))
114                        {
115                                return depth;
116                        } else if (s.equals("/"))
117                        {
118                                accessors.add(getStepAccessor(level, st, intermediate));
119                                domainOrder.add(intermediate);
120                                intermediate = getNextDomain();
121
122                        } else if (s.equals("*"))
123                        {
124                                PropertyAccessor lastAccessor =
125                                                accessors.remove(accessors.size() - 1);
126                                accessors.add(TransitivePropertyAccessor.create(lastAccessor));
127
128                        } else if (s.equals(":"))
129                        {
130                                Class<? extends BioPAXElement> restricted = getClass(st.nextToken());
131                                if (restricted != null)
132                                {
133                                        PropertyAccessor<? extends BioPAXElement, ?> lastAccessor = accessors.remove(accessors.size() - 1);
134                                        accessors.add(FilteredPropertyAccessor.create(lastAccessor, restricted));
135                                }
136                                intermediate = restricted;
137                        } else if (s.equals("|"))
138                        {
139                                throw new UnsupportedOperationException("Not implemented yet");
140                        } else throw new IllegalArgumentException();
141
142                }
143                return depth;
144        }
145
146        private Class<? extends BioPAXElement> getNextDomain()
147        {
148                return accessors.isEmpty() ? domain : getLastAccessor();
149        }
150
151        private Class<? extends BioPAXElement> getLastAccessor()
152        {
153                return (Class<? extends BioPAXElement>) accessors.get(accessors.size() - 1).getRange();
154        }
155
156        /**
157         * This constructor defaults to BioPAX Level 3.
158         * @param path The string defining the path. The following operators can be used:
159         * <ul>
160         * <li>The first token is the class name of the starting domain.</li>
161         * <li>The next tokens are names of the properties, separated by "/"</li>
162         * <li>A property step can be restricted to a domain by defining the restriction class with ":" after the
163         * property </li>
164         * <li>A property step can be declared transitive by adding a "*" at the end. </li>
165         * <li>Two accessors can be joined by "|" (Currently not implemented)</li>
166         * <li>A subpath can be defined by a substring enclosed in "(" ")"</li>
167         * <li>Subpaths can also be restricted or made transitive</li>
168         * </ul>
169         */
170        public PathAccessor(String path)
171        {
172                this(path, BioPAXLevel.L3);
173        }
174
175        private Class<? extends BioPAXElement> getClass(String domainstr)
176        {
177                Class<? extends BioPAXElement> domain = level.getInterfaceForName(domainstr);
178
179                if (domain == null) throw new IllegalBioPAXArgumentException(
180                                "Could not parse path." + domainstr + " did not resolve to any" + "BioPAX classes in level " + level +
181                                ".");
182
183                return domain;
184        }
185
186        public Set getValueFromBean(BioPAXElement bean) throws IllegalBioPAXArgumentException
187        {
188                Set<BioPAXElement> bpes = new HashSet<BioPAXElement>();
189                bpes.add(bean);
190                return getValueFromBeans(bpes);
191        }
192
193        @Override
194        public Set getValueFromBeans(Collection<? extends BioPAXElement> beans) throws IllegalBioPAXArgumentException
195        {
196                Collection<? extends BioPAXElement> bpes = beans;
197                for (int i = 0; i < accessors.size() - 1; i++)
198                {
199                        PropertyAccessor accessor = accessors.get(i);
200                        if (log.isTraceEnabled()) log.trace(String.valueOf(accessor));
201                        HashSet<BioPAXElement> nextBpes = new HashSet<BioPAXElement>();
202                        for (BioPAXElement bpe : bpes)
203                        {
204                                if (log.isTraceEnabled()) log.trace("\t" + bpe);
205                                Set valueFromBean = accessor.getValueFromBean(bpe);
206                                if (valueFromBean != null || valueFromBean.isEmpty())
207                                {
208                                        if (log.isTraceEnabled()) log.trace("\t\tv:" + valueFromBean);
209                                        nextBpes.addAll(valueFromBean);
210                                        if (log.isTraceEnabled()) log.trace("\t\tn:" + nextBpes);
211                                }
212                        }
213                        bpes = nextBpes;
214                }
215                HashSet values = new HashSet();
216                PropertyAccessor lastStep = accessors.get(accessors.size() - 1);
217                Class<? extends BioPAXElement> lastDomain = domainOrder.get(domainOrder.size() - 1);
218                if(log.isTraceEnabled()) log.trace(String.valueOf(lastStep));
219                for (BioPAXElement bpe : bpes)
220                {
221                        if (!lastDomain.isInstance(bpe)) continue;
222
223                        log.trace("\t" + bpe);
224                        Set valueFromBean = lastStep.getValueFromBean(bpe);
225                        if (valueFromBean != null || valueFromBean.isEmpty())
226                        {
227                                values.addAll(lastStep.getValueFromBean(bpe));
228                                log.trace("\t" + values);
229                        }
230                }
231                return values;
232        }
233
234        /**
235         * This method runs the path query on all the elements within the model.
236         * @param model to be queried
237         * @return a merged set of all values that is reachable by the paths starting from all applicable objects in
238         * the model. For example running ProteinReference/xref:UnificationXref on the model will get all the
239         * unification xrefs of all ProteinReferences.
240         */
241        public Set getValueFromModel(Model model)
242        {
243                Set<? extends BioPAXElement> domains = new HashSet<BioPAXElement>(model.getObjects(this.getDomain()));
244                return getValueFromBeans(domains);
245        }
246
247        private <D extends BioPAXElement> PropertyAccessor getStepAccessor(BioPAXLevel level, StringTokenizer ct,
248                        Class<D> domain)
249        {
250                String property = ct.nextToken();
251                PropertyAccessor simple = null;
252                if (property.endsWith("Of"))
253                {
254                        String forwardName = property.substring(0, property.length() - 2);
255                        Set<ObjectPropertyEditor> iEds = SimpleEditorMap.get(level).getInverseEditorsOf(domain);
256                        if (iEds == null)
257                        {
258                                throw new IllegalBioPAXArgumentException("No inverse editors defined for " + domain);
259                        }
260                        for (ObjectPropertyEditor ope : iEds)
261                        {
262                                if (ope.property.equals(forwardName))
263                                {
264                                        if (simple == null) simple = ope.getInverseAccessor();//TODO why not simply break, after assignment, instead using 'if'?
265                                }
266
267                        }
268                } else
269                {
270                        simple = SimpleEditorMap.get(level).getEditorForProperty(property, domain);
271                        if (simple == null)
272                        {
273                                Set<PropertyEditor<? extends D, ?>> subclassEditorsForProperty =
274                                                SimpleEditorMap.get(level).getSubclassEditorsForProperty(property, domain);
275                                simple = new UnionPropertyAccessor(subclassEditorsForProperty, domain);
276
277                        }
278                }
279                return simple;
280        }

282        @Override public boolean isUnknown(Object value)
283        {
284                if (value instanceof Set)
285                {
286                        for (Object o : (Set) value)
287                        {
288                                if (!isSingleUnknown(o))
289                                {
290                                        return false; // found a "known" value
291                                }
292                        }
293                        // empty set or all unknown
294                        return true;
295                } else
296                {
297                        return isSingleUnknown(value);
298                }
299        }
300
301        private boolean isSingleUnknown(Object value)
302        {
303                return value == null || BioPAXElement.UNKNOWN_DOUBLE.equals(value) ||
304                       BioPAXElement.UNKNOWN_FLOAT.equals(value) || BioPAXElement.UNKNOWN_INT.equals(value);
305        }
306
307        /**
308         * @param bpe BioPAXElement to be checked.
309         * @return true if bpe is an instance of the domain of this accessor.
310         */
311        public boolean applies(BioPAXElement bpe)
312        {
313                Class domain = accessors.iterator().next().getDomain();
314                return domain.isInstance(bpe);
315        }
