001package org.biopax.paxtools.controller;
002
003import org.biopax.paxtools.model.BioPAXElement;
004import org.biopax.paxtools.model.Model;
005import org.biopax.paxtools.model.level2.pathway;
006import org.biopax.paxtools.model.level3.Pathway;
007
008import java.util.Collection;
009import java.util.HashSet;
010import java.util.Set;
011
012/**
013 * This class is used for getting a valid subgraph from a set of BioPAX elements. For instance, if a
014 * BioPAX graph contains a complex with some members, but the subgraph contains the same complex
015 * without the members, then it is not valid. Similarly, an interaction is only valid with its
016 * participants; however, including its controls is optional.
017 *
018 * @author Ozgun Babur
019 */
020public class Completer implements Visitor
021{
022        protected TraverserBilinked traverser;
023
024        private final Set<BioPAXElement> completed;
025
026        private boolean skipSubPathways;
027
028        public Completer(EditorMap map)
029        {
030                this.traverser = new TraverserBilinked(map, this, new PropertyFilterBilinked()
031                {
032                        public boolean filter(PropertyEditor editor)
033                        {
034                                return editor instanceof ObjectPropertyEditor &&
035                                        ((ObjectPropertyEditor) editor).isCompleteForward();
036                        }
037
038                        public boolean filterInverse(PropertyEditor editor)
039                        {
040                                return editor instanceof ObjectPropertyEditor &&
041                                        ((ObjectPropertyEditor) editor).isCompleteBackward();
042                        }
043                });
044
045                completed = new HashSet<BioPAXElement>();
046                skipSubPathways = true; //makes default - skip sub-pathways
047        }
048
049        /**
050         * @deprecated use {@link #complete(Collection)} instead (- model is never explicitly used there)
051         * @param elements
052         * @param model
053         * @return
054         */
055        public Set<BioPAXElement> complete(Collection<BioPAXElement> elements, Model model) {
056                return complete(elements);
057        }
058
059        public Set<BioPAXElement> complete(Collection<BioPAXElement> elements)
060        {
061                completed.clear();
062
063                for (BioPAXElement element : elements)
064                {
065                        if (completed.add(element))
066                        {
067                                traverser.traverse(element, null); //model is not required here because of 'visit' impl. below
068                        }
069                }
070
071                return completed;
072        }
073
074        public void visit(BioPAXElement domain, Object range, Model model, PropertyEditor editor)
075        {
076                if (range instanceof BioPAXElement) //!=null works too (because of prop. filters)
077                {
078                        BioPAXElement element = (BioPAXElement) range;
079                        if (!completed.contains(element))
080                        {
081                                // Ignore sub-pathways if skipSubPathways is true
082                                //(the effect must be the same as if @AutoComplete(forward=false) annotation
083                                // would be set on properties pathwayComponent and controlled, which unfortunately we cannot do
084                                // for values of type Pathway only via the property getter annotations,
085                                // because these properties' range is Process - Interaction or Pathway...)
086                                completed.add(element);
087                                //traverse unless we are to ignore sub-pathways (in fact - sub-pathway's member objects)
088                                if( !(skipSubPathways && (element instanceof Pathway || element instanceof pathway)) ) {
089                                        traverser.traverse(element, null);
090                                }
091                        }
092                }
093        }
094
095        /**
096         * Use this property to optionally
097         * skip (if true) traversing into sub-pathways;
098         * i.e., when the value of BioPAX property 'pathwayComponent' or 'controlled' is a pathway.
099         *
100         * @param skipSubPathways true/false (default is 'true' - do skip)
101         */
102        public void setSkipSubPathways(boolean skipSubPathways) {
103                this.skipSubPathways = skipSubPathways;
104        }
105
106        public boolean isSkipSubPathways() {
107                return skipSubPathways;
108        }
109}
