001package org.biopax.paxtools.controller;
002
003import org.biopax.paxtools.model.BioPAXElement;
004import org.biopax.paxtools.model.level3.*;
005import org.biopax.paxtools.util.ClassFilterSet;
006
007import java.util.*;
008
009/**
010 * Tries to get preferred type IDs of an entity reference.
011 *
012 * Could be used, e.g., as part of a BioPAX to plain text (or SIF, GMT) converter.
013 */
014public class IdFetcher
015{
016        private final List<String> seqDbStartsWithOrEquals;
017        private final List<String> chemDbStartsWithOrEquals;
018        private boolean useNameWhenNoDbMatch;
019
020        /**
021         * Constructor.
022         */
023        public IdFetcher() {
024                seqDbStartsWithOrEquals = new ArrayList<String>();
025                chemDbStartsWithOrEquals = new ArrayList<String>();
026                useNameWhenNoDbMatch = false;
027        }
028
029        /**
030         * Set to prefer collecting gene/sequence IDs of such Xrefs
031         * where the db starts with or equals given string,
032         * ignoring case. You can chain this method calls like
033         * seqDbStartsWithOrEquals(A).seqDbStartsWithOrEquals(B)... -
034         * it will try to match a xref.db and collect xref.id
035         * in the given order/priority.
036         *
037         * @param dbStartsWithOrEquals the Xref.db value or prefix (case-insensitive)
038         * @return this id-fetcher instance
039         */
040        public IdFetcher seqDbStartsWithOrEquals(String dbStartsWithOrEquals) {
041                this.seqDbStartsWithOrEquals.add(dbStartsWithOrEquals.toLowerCase());
042                return this;
043        }
044
045        public List<String> getSeqDbStartsWithOrEquals() {
046                return Collections.unmodifiableList(seqDbStartsWithOrEquals);
047        }
048
049        /**
050         * Set to prefer collecting chemical IDs of such Xrefs
051         * where the small molecules db starts with or equals given string,
052         * ignoring case. You can chain this method calls like
053         * chemDbStartsWithOrEquals(A).chemDbStartsWithOrEquals(B)... -
054         * it will try to match a xref.db and collect xref.id
055         * in the given order/priority.
056         *
057         * @param dbStartsWithOrEquals the Xref.db value or prefix (case-insensitive)
058         * @return this id-fetcher instance
059         */
060        public IdFetcher chemDbStartsWithOrEquals(String dbStartsWithOrEquals) {
061                this.chemDbStartsWithOrEquals.add(dbStartsWithOrEquals.toLowerCase());
062                return this;
063        }
064
065        public List<String> getChemDbStartsWithOrEquals() {
066                return Collections.unmodifiableList(chemDbStartsWithOrEquals);
067        }
068
069        /**
070         * Set the flag to use the entity reference's names
071         * when no desired ID type can be found (none of xref.db
072         * matched before, or there're no xrefs at all).
073         *
074         * @param useNameWhenNoDbMatch true/false (default is 'true' - when this method's never been called)
075         * @return this id-fetcher instance
076         */
077        public IdFetcher useNameWhenNoDbMatch(boolean useNameWhenNoDbMatch) {
078                this.useNameWhenNoDbMatch = useNameWhenNoDbMatch;
079                return this;
080        }
081
082        public Set<String> fetchID(BioPAXElement ele)
083        {
084                Set<String> set = new HashSet<String>();
085
086                if(ele instanceof XReferrable) {
087                        //Iterate the db priority list, match/filter all xrefs to collect the IDs of given type, until 'set' is not empty.
088                        List<String> dbStartsWithOrEquals =
089                                        (ele instanceof SmallMoleculeReference || ele instanceof SmallMolecule)
090                                                        ? chemDbStartsWithOrEquals : seqDbStartsWithOrEquals;
091
092                        for (String dbStartsWith : dbStartsWithOrEquals) {
093                                //a shortcut for URI like "http://identifiers.org/uniprot/", "http://identifiers.org/chebi/";
094                                //this prevents collecting lots of secondary IDs of the same type
095                                if(ele.getUri().startsWith("http://identifiers.org/"+dbStartsWith)) {
096                                        set.add(ele.getUri().substring(ele.getUri().lastIndexOf("/") + 1));
097                                }
098                                else
099                                {
100                                        for (UnificationXref x : new ClassFilterSet<Xref, UnificationXref>(((XReferrable) ele).getXref(),
101                                                        UnificationXref.class)) {
102                                                collectXrefIdIfDbLike(x, dbStartsWith, set);
103                                        }
104                                        //if none was found in unif. xrefs, try rel, xrefs
105                                        if (set.isEmpty()) {
106                                                for (RelationshipXref x : new ClassFilterSet<Xref, RelationshipXref>(((XReferrable) ele).getXref(),
107                                                                RelationshipXref.class)) {
108                                                        collectXrefIdIfDbLike(x, dbStartsWith, set);
109                                                }
110                                        }
111                                }
112
113                                //once we've found some ID, no need to try another id type
114                                if (!set.isEmpty())
115                                        break;
116                        }
117                }
118
119                if (set.isEmpty() && ele instanceof Named && useNameWhenNoDbMatch)
120                {
121                        Named e = (Named) ele;
122                        //avoid shortened/incomplete names -
123                        if (e.getDisplayName() != null && !e.getDisplayName().contains("..."))
124                                set.add(e.getDisplayName());
125                        else if (e.getStandardName() != null && !e.getStandardName().contains("..."))
126                                set.add(e.getStandardName());
127                        else if (!e.getName().isEmpty()) {
128                                Set<String> names = new TreeSet<String>();
129                                for(String name : e.getName()) {
130                                        if(!name.contains("..."))
131                                                names.add(name);
132                                }
133                                set.add(names.toString());
134                        }
135                }
136
137                return set;
138        }
139
140        private void collectXrefIdIfDbLike(final Xref x, final String dbStartsWith, final Set<String> set) {
141                String db = x.getDb();
142                String id = x.getId();
143                if (db != null && id != null && !id.isEmpty()) {
144                        db = db.toLowerCase();
145                        if (db.startsWith(dbStartsWith)) {
146                                if (id != null)
147                                        set.add(id);
148                        }
149                }
150        }
151
152}
