001package org.biopax.paxtools.controller;
002
003import org.biopax.paxtools.model.BioPAXElement;
004import org.biopax.paxtools.model.Model;
005import org.biopax.paxtools.model.level2.physicalEntityParticipant;
006import org.biopax.paxtools.model.level2.sequenceFeature;
007import org.biopax.paxtools.model.level2.sequenceParticipant;
008import org.slf4j.Logger;
009import org.slf4j.LoggerFactory;
010
011import java.net.URI;
012import java.net.URISyntaxException;
013import java.util.HashMap;
014import java.util.Map;
015import java.util.Set;
016
017/**
018 * This class contains methods for handling reused PEPs - a historically common problem in BioPAX L2 exports.
019 *
020 * Note: This class might eventually be deprecated as this problem is mostly fixed and BioPAX L3 is making L2
021 * obsolete.
022 */
023public class ReusedPEPHelper
024{
025    private static final Logger log = LoggerFactory.getLogger(ReusedPEPHelper.class);
026
027    private final Model model;
028    private final Map<physicalEntityParticipant, physicalEntityParticipant> duplicatedPeps;
029
030    /**
031     * @param model the biopax model
032     */
033    public ReusedPEPHelper(Model model)
034    {
035        this.model = model;
036        duplicatedPeps = new HashMap<physicalEntityParticipant, physicalEntityParticipant>();
037    }
038
039    public Object fixReusedPEP(physicalEntityParticipant pep, BioPAXElement bpe)
040    {
041        if (duplicated(pep, bpe))
042        {
043                log.info(pep.getUri() + " is reused, duplicating it to fix");
044
045                String syntheticID = createSyntheticID(pep, bpe);
046            
047            if (model.containsID(syntheticID))
048            {
049                pep = (physicalEntityParticipant) model.getByID(syntheticID);
050            }
051            else
052            {
053                physicalEntityParticipant duplicated = (physicalEntityParticipant) model
054                        .addNew(pep.getModelInterface(), syntheticID);
055                duplicatedPeps.put(duplicated, pep);
056                pep = duplicated;
057            }
058        }
059        return pep;
060    }
061
062    private boolean duplicated(physicalEntityParticipant pep, BioPAXElement bpe)
063    {
064        boolean result = false;
065
066        if (!pep.isPARTICIPANTSof().isEmpty())
067        {
068            if (pep.isPARTICIPANTSof().iterator().next().equals(bpe))
069            {
070                log.debug("Unexpected multiple participant statements");
071            }
072            else
073            {
074                result = true;
075            }
076        }
077        else if (pep.isCOMPONENTof() != null)
078        {
079            if (pep.isCOMPONENTof().equals(bpe))
080            {
081                log.debug("Unexpected multiple participant statements");
082            }
083            else
084            {
085                result = true;
086            }
087        }
088
089        return result;
090
091    }
092
093    private String createSyntheticID(physicalEntityParticipant pep,
094                                     BioPAXElement bpe)
095    {
096        return "http://patywaycommons.org/synthetic"
097               + createDataStringFromURI(pep.getUri(),bpe.getUri());
098    }
099
100    private String createDataStringFromURI(String... uris)
101    {
102        String ssp = "";
103        String fragment = "";
104
105        for (String uri : uris)
106        {
107            try
108            {
109                URI suri = new URI(uri);
110                ssp += suri.getSchemeSpecificPart() + "_";
111                fragment += suri.getFragment() + "_";
112            }
113            catch (URISyntaxException e)
114            {
115                throw new RuntimeException(e);
116            }
117        }
118        return ssp + "#" + fragment;
119    }
120
121    public void copyPEPFields()
122    {
123        Set<physicalEntityParticipant> physicalEntityParticipants = duplicatedPeps.keySet();
124        for (physicalEntityParticipant dup : physicalEntityParticipants)
125        {
126            copyPEPFields(dup, duplicatedPeps.get(dup));
127        }
128    }
129
130    private void copyPEPFields(physicalEntityParticipant duplicated,
131                               physicalEntityParticipant pep)
132    {
133        duplicated.setCELLULAR_LOCATION(pep.getCELLULAR_LOCATION());
134        duplicated.setCOMMENT(pep.getCOMMENT());
135        duplicated.setSTOICHIOMETRIC_COEFFICIENT(pep.getSTOICHIOMETRIC_COEFFICIENT());
136        duplicated.setPHYSICAL_ENTITY(pep.getPHYSICAL_ENTITY());
137        if (pep instanceof sequenceParticipant)
138        {
139            Set<sequenceFeature> sfSet =
140                    ((sequenceParticipant) pep)
141                            .getSEQUENCE_FEATURE_LIST();
142            for (sequenceFeature sf : sfSet)
143            {
144                ((sequenceParticipant) duplicated)
145                        .addSEQUENCE_FEATURE_LIST(sf);
146            }
147        }
148        }
149
150}
