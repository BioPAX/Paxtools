001package org.biopax.paxtools.examples;
002
003import org.biopax.paxtools.io.BioPAXIOHandler;
004import org.biopax.paxtools.io.SimpleIOHandler;
005import org.biopax.paxtools.model.Model;
006import org.biopax.paxtools.model.level2.XReferrable;
007import org.biopax.paxtools.model.level2.relationshipXref;
008import org.biopax.paxtools.model.level2.unificationXref;
009import org.slf4j.Logger;
010import org.slf4j.LoggerFactory;
011
012import java.io.File;
013import java.io.FileInputStream;
014import java.io.FileNotFoundException;
015import java.io.FileOutputStream;
016import java.lang.reflect.InvocationTargetException;
017import java.util.HashSet;
018import java.util.Set;
019
020/**
021 * User: Emek Demir Date: Jan 18, 2007 Time: 4:56:53 PM
022 *
023 * In this example we get all the unification xrefs in the model
024 * and check if they point to the Gene Ontology. If this is the case
025 * we convert them to relationship xrefs.
026 *
027 */
028public class GOUnificationXREFtoRelationshipXREFConverter
029{
030        private static Logger log = LoggerFactory.getLogger(GOUnificationXREFtoRelationshipXREFConverter.class);
031   
032        static BioPAXIOHandler reader = new SimpleIOHandler();
033
034    //args - a space seperated list of owl files to be processed
035    public static void main(String[] args)
036                throws IllegalAccessException, InvocationTargetException
037        {
038        // Process all the args
039        for (String arg : args)
040        {
041            log.info(arg);
042            if (arg.toLowerCase().endsWith("owl"))
043            {
044                try
045                {
046                    processXrefs(arg);
047                }
048                catch (FileNotFoundException e)
049                {
050                    e.printStackTrace();
051                }
052            }
053
054        }
055        }
056
057    /**
058     * Main conversion method. Demonstrates how to read and write a BioPAX
059     * model and accessing its objects.
060     * @param arg file name to be processed
061     */
062    private static void processXrefs(String arg) throws
063                FileNotFoundException,
064                IllegalAccessException,
065                InvocationTargetException
066        {
067                //Read in the model
068        FileInputStream in =
069                        new FileInputStream(new File(arg));
070                Model level2 =
071                        reader.convertFromOWL(in);
072
073        //Get all unification xrefs.
074        Set<unificationXref> unis =
075                        level2.getObjects(unificationXref.class);
076                //Create another set for avoiding concurrent modifications
077        Set<unificationXref> gos = new HashSet<unificationXref>();
078
079        //Process all uni. refs
080        for (unificationXref uni : unis)
081                {
082                        log.trace(uni.getDB());
083                        //Assuming DB is represented as "GO"
084            if (uni.getDB().equalsIgnoreCase("GO"))
085                        {
086                                //this it to avoid concurrent modification.
087                                log.info("scheduling " + uni.getUri());
088                                gos.add(uni);
089
090                        }
091                }
092        //Now we have a list of xrefs to be converted. Let's do it.
093        for (unificationXref go : gos)
094                {
095                        convert(go, level2);
096                }
097                //And finally write out the file. We are done !
098        reader.convertToOWL(level2, new FileOutputStream(
099                        arg.substring(0, arg.lastIndexOf('.')) +
100                                "-converted.owl"));
101        }
102
103    /**
104     * This method converts the given unification xref to a relationship xref
105     * @param uni xref to be converted
106     * @param level2 model containing the xref
107     */
108    private static void convert(unificationXref uni, Model level2)
109        {
110                //We can not simply convert a class, so we need to remove the
111        //uni and insert a new relationship xref
112
113        //First get all the objects that refers to this uni
114        Set<XReferrable> referrables =
115                        new HashSet<XReferrable>(uni.isXREFof());
116
117        //Create the new relationship xref in the model.
118        relationshipXref relationshipXref =
119                level2.addNew(relationshipXref.class, uni.getUri());
120
121        //Copy the fields from uni
122                relationshipXref.setCOMMENT(uni.getCOMMENT());
123                relationshipXref.setDB(uni.getDB());
124                relationshipXref.setDB_VERSION(uni.getDB_VERSION());
125                relationshipXref.setID(uni.getID());
126                relationshipXref.setID_VERSION(uni.getID_VERSION());
127                relationshipXref.setRELATIONSHIP_TYPE(
128                        "http://www.biopax.org/paxtools/convertedGOUnificationXREF");
129
130        //Create a link to the new xref from all the owners.
131        for (XReferrable referrable : referrables)
132                {
133                        referrable.addXREF(relationshipXref);
134                }
135
136        //Remove the references to the old uni
137        for (XReferrable referrable : referrables)
138        {
139            referrable.removeXREF(uni);
140        }
141        //Now remove it from the model.
142        level2.remove(uni);
143
144        //We are done!
145    }
146}
