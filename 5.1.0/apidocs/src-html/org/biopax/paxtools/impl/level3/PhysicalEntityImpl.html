001package org.biopax.paxtools.impl.level3;
002
003import org.biopax.paxtools.util.*;
004import org.biopax.paxtools.model.BioPAXElement;
005import org.biopax.paxtools.model.level3.*;
006import org.biopax.paxtools.util.SetEquivalenceChecker;
007import org.slf4j.Logger;
008import org.slf4j.LoggerFactory;
009
010import java.util.Set;
011
012
013public class PhysicalEntityImpl extends EntityImpl implements PhysicalEntity
014{
015        private CellularLocationVocabulary cellularLocation;
016        private Set<PhysicalEntity> memberPhysicalEntity;
017        private Set<Complex> componentOf;
018        private Set<EntityFeature> feature;
019        private Set<EntityFeature> notFeature;
020        private Set<Control> controllerOf;
021        private static final Logger log = LoggerFactory.getLogger(PhysicalEntityImpl.class);
022        private Set<PhysicalEntity> memberPhysicalEntityOf;
023
024        public PhysicalEntityImpl()
025        {
026                feature = BPCollections.I.createSafeSet();
027                notFeature = BPCollections.I.createSafeSet();
028                controllerOf = BPCollections.I.createSafeSet();
029                componentOf = BPCollections.I.createSafeSet();
030                memberPhysicalEntityOf = BPCollections.I.createSafeSet(); //TODO make generic?
031                memberPhysicalEntity = BPCollections.I.createSafeSet();
032        }
033
034        public Class<? extends PhysicalEntity> getModelInterface()
035        {
036                return PhysicalEntity.class;
037        }
038
039        public Set<Complex> getComponentOf()
040        {
041                return componentOf;
042        }
043
044        public CellularLocationVocabulary getCellularLocation()
045        {
046                return cellularLocation;
047        }
048
049        public void setCellularLocation(CellularLocationVocabulary location)
050        {
051                this.cellularLocation = location;
052        }
053
054        public Set<EntityFeature> getFeature()
055        {
056                return feature;
057        }
058
059        public void addFeature(EntityFeature feature)
060        {
061                if (feature != null) {
062                        this.feature.add(feature);
063                        feature.getFeatureOf().add(this);
064                }
065        }
066
067        public void removeFeature(EntityFeature feature)
068        {
069                if (feature != null) {
070                        this.feature.remove(feature);
071                        feature.getFeatureOf().remove(this);
072                }
073        }
074
075        public Set<EntityFeature> getNotFeature()
076        {
077                return notFeature;
078        }
079
080        public void addNotFeature(EntityFeature feature)
081        {
082                if (feature != null) {
083                        this.notFeature.add(feature);
084                        feature.getNotFeatureOf().add(this);
085                }
086        }
087
088        public void removeNotFeature(EntityFeature feature)
089        {
090                if (feature != null) {
091                        this.notFeature.remove(feature);
092                        feature.getNotFeatureOf().remove(this);
093                }
094        }
095
096        public Set<PhysicalEntity> getMemberPhysicalEntity()
097        {
098                return this.memberPhysicalEntity;
099        }
100
101        public void addMemberPhysicalEntity(PhysicalEntity mpe)
102        {
103                if (mpe != null) {
104                        //Ensure OWL class/property restrictions
105                        if(this.getModelInterface().isInstance(mpe)) {
106                                this.memberPhysicalEntity.add(mpe);
107                                mpe.getMemberPhysicalEntityOf().add(this);
108                        } else {
109                                String c = this.getModelInterface().getSimpleName();
110                                throw new IllegalBioPAXArgumentException(String.format(
111                                        "Value:%s violates class and property restriction: %s.memberPhysycalEntity only %s",
112                                                mpe.getUri(), c, c));
113                        }
114                }
115        }
116
117        public void removeMemberPhysicalEntity(PhysicalEntity oldMember)
118        {
119                if (oldMember != null) {
120                        this.memberPhysicalEntity.remove(oldMember);
121                        oldMember.getMemberPhysicalEntityOf().remove(this);
122                }
123        }
124
125        public Set<PhysicalEntity> getMemberPhysicalEntityOf()
126        {
127                return memberPhysicalEntityOf;
128        }
129
130        // --------------------- Interface BioPAXElement ---------------------
131
132
133        @Override
134        protected boolean semanticallyEquivalent(BioPAXElement element)
135        {
136                if (!(element instanceof PhysicalEntity))
137                        return false;
138
139                PhysicalEntity that = (PhysicalEntity) element;
140                return hasEquivalentCellularLocation(that)
141                       && hasEquivalentFeatures(that)
142                       && SetEquivalenceChecker
143                                .isEquivalent(getMemberPhysicalEntity(), that.getMemberPhysicalEntity())
144                       &&
145                       super.semanticallyEquivalent(element); // StackOverflow BUG fixed: was isEquivalent !
146        }
147
148        @Override
149        public int equivalenceCode()
150        {
151                return hashCode();
152        }
153
154        public boolean hasEquivalentCellularLocation(PhysicalEntity that)
155        {
156                boolean equivalency = false;
157                if (that != null)
158                {
159                        equivalency = (cellularLocation != null)
160                                      ? cellularLocation.isEquivalent(that.getCellularLocation())
161                                      : that.getCellularLocation() == null;
162                }
163                return equivalency;
164        }
165
166        public boolean hasEquivalentFeatures(PhysicalEntity that)
167        {
168                boolean equivalency = false;
169                if (that != null)
170                {
171                        equivalency =
172                                        SetEquivalenceChecker.isEquivalent(this.getFeature(), that.getFeature()) &&
173                                        SetEquivalenceChecker.isEquivalent(this.getNotFeature(), that.getNotFeature());
174                }
175                return equivalency;
176        }
177
178        protected int locationAndFeatureCode()
179        {
180                int result = (cellularLocation != null ? cellularLocation.hashCode() : 0);
181                result = 31 * result + (feature != null ? feature.hashCode() : 0);
182                result = 31 * result + (notFeature != null ? notFeature.hashCode() : 0);
183                return result;
184
185        }
186
187        public Set<Control> getControllerOf()
188        {
189                return controllerOf;
190        }
191
192}
