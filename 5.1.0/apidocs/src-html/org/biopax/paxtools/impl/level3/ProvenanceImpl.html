001package org.biopax.paxtools.impl.level3;
002
003import org.biopax.paxtools.model.level3.Provenance;
004import org.biopax.paxtools.model.level3.Xref;
005import org.slf4j.Logger;
006import org.slf4j.LoggerFactory;
007
008import java.util.Comparator;
009import java.util.TreeSet;
010
011
012public class ProvenanceImpl extends NamedImpl implements Provenance
013{
014        private final static Logger log = LoggerFactory.getLogger(ProvenanceImpl.class);
015        
016        public ProvenanceImpl() {
017        }
018
019    public Class<? extends Provenance> getModelInterface()
020        {
021                return Provenance.class;
022        }
023
024        /*
025         * (non-Javadoc)
026         * @see org.biopax.paxtools.impl.BioPAXElementImpl#toString()
027         * 
028         * TODO this probably makes inconsistent strings (on different systems); fix, if it matters....
029         * 
030         */
031        @Override public String toString()
032        {
033                try {
034                        StringBuilder s = new StringBuilder();
035
036                        for (String name : new TreeSet<String>(this.getName())) 
037                                s.append(name).append(";");
038                        
039                        if (!getXref().isEmpty()) 
040                        {
041                                TreeSet<Xref> xrefs = new TreeSet<Xref>(new Comparator<Xref>() {
042                                        @Override
043                                        public int compare(Xref o1, Xref o2) {
044                                                return o1.toString().compareTo(o2.toString());
045                                        }                                       
046                                });
047                                xrefs.addAll(getXref()); //this also removes duplicate equivalent xrefs (having same id/db)
048                                
049                                s.append(" (");
050                                for (Xref anXref : xrefs)
051                                        s.append(anXref).append(";");
052                                s.append(")");
053                        }
054                        
055                        return s.toString();
056
057                } catch (Exception e) {
058                        // possible issues - when in a persistent context (e.g., lazy
059                        // collections init...)
060                        log.warn("toString: ", e);
061                        return getUri();
062                }
063        }
064}
