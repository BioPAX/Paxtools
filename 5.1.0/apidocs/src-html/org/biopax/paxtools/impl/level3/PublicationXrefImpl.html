001package org.biopax.paxtools.impl.level3;
002
003import org.biopax.paxtools.model.BioPAXElement;
004import org.biopax.paxtools.model.level3.PublicationXref;
005import org.biopax.paxtools.util.BPCollections;
006
007import java.util.Set;
008
009
010public class PublicationXrefImpl extends XrefImpl implements PublicationXref
011{
012        private String title;
013        private Set<String> url;
014        private Set<String> source;
015        private Set<String> author;
016        private int year = UNKNOWN_INT;
017
018        /**
019         * Constructor.
020         */
021        public PublicationXrefImpl()
022        {
023                this.url = BPCollections.I.createSet();
024                this.source = BPCollections.I.createSet();
025                this.author = BPCollections.I.createSet();
026        }
027
028    public Class<? extends PublicationXref> getModelInterface()
029        {
030                return PublicationXref.class;
031        }
032
033        //
034        // PublicationXref interface implementation
035        //
036        ////////////////////////////////////////////////////////////////////////////
037
038    // Property author
039        public Set<String> getAuthor()
040        {
041                return author;
042        }
043
044        public void addAuthor(String author)
045        {
046                if(author != null && author.length() > 0)
047                        this.author.add(author);
048        }
049
050        public void removeAuthor(String author)
051        {
052                if(author != null)
053                        this.author.remove(author);
054        }
055
056        public Set<String> getSource()
057        {
058                return source;
059        }
060
061        public void addSource(String source)
062        {
063                if(source != null && source.length() > 0)
064                        this.source.add(source);
065        }
066
067        public void removeSource(String source)
068        {
069                if(source != null)
070                        this.source.remove(source);
071        }
072
073        public String getTitle()
074        {
075                return title;
076        }
077
078        public void setTitle(String title)
079        {
080                this.title = title;
081        }
082
083        public Set<String> getUrl()
084        {
085                return url;
086        }
087
088        public void addUrl(String url)
089        {
090                if(url != null && url.length() > 0)
091                        this.url.add(url);
092        }
093
094        public void removeUrl(String url)
095        {
096                if(url != null)
097                        this.url.remove(url);
098        }
099
100    // Property year
101        public int getYear()
102        {
103                return year;
104        }
105
106        public void setYear(int year)
107        {
108                this.year = year;
109        }
110        
111        @Override
112        protected boolean semanticallyEquivalent(BioPAXElement other) {
113                if(!(other instanceof PublicationXref)) return false;
114                
115                PublicationXref that = (PublicationXref) other;
116                boolean eqv = (year == that.getYear()) &&
117                        (title != null ? 
118                                title.equals(that.getTitle()) 
119                                : that.getTitle() == null)
120                        && author.containsAll(that.getAuthor())
121                        && source.containsAll(that.getSource())
122                        && url.containsAll(that.getUrl());
123                
124                return eqv      && super.semanticallyEquivalent(other);
125        }
126}
