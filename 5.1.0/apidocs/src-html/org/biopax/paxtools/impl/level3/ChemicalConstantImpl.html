001package org.biopax.paxtools.impl.level3;
002
003
004import org.biopax.paxtools.model.BioPAXElement;
005import org.biopax.paxtools.model.level3.ChemicalConstant;
006
007import static java.lang.Float.compare;
008
009
010public abstract class ChemicalConstantImpl extends L3ElementImpl implements ChemicalConstant {
011    float ionicStrength;
012    float ph;
013    float pMg;
014    float temperature;
015
016    public ChemicalConstantImpl() {
017        this.ionicStrength = KPrimeImpl.UNKNOWN_FLOAT;
018        this.ph = KPrimeImpl.UNKNOWN_FLOAT;
019        this.pMg = KPrimeImpl.UNKNOWN_FLOAT;
020        this.temperature = KPrimeImpl.UNKNOWN_FLOAT;
021    }
022
023    public float getIonicStrength() {
024        return ionicStrength;
025    }
026
027    public void setIonicStrength(float ionicStrength) {
028        this.ionicStrength = ionicStrength;
029    }
030
031    public float getPh() {
032        return ph;
033    }
034
035    public void setPh(float ph) {
036        this.ph = ph;
037    }
038
039    public float getPMg() {
040        return pMg;
041    }
042
043    public void setPMg(float pMg) {
044        this.pMg = pMg;
045    }
046
047    public float getTemperature() {
048        return temperature;
049    }
050
051    public void setTemperature(float temperature) {
052        this.temperature = temperature;
053    }
054
055    public Class<? extends ChemicalConstant> getModelInterface() {
056        return ChemicalConstant.class;
057    }
058
059    @Override
060    protected boolean semanticallyEquivalent(BioPAXElement element) {
061        if(!(element instanceof ChemicalConstant))
062            return false;
063
064        final ChemicalConstant aKPrime = (ChemicalConstant) element;
065        return
066                compare(aKPrime.getIonicStrength(), getIonicStrength()) == 0 &&
067                        compare(aKPrime.getPh(), getPh()) == 0 &&
068                        compare(aKPrime.getPMg(), pMg) == 0 &&
069                        compare(aKPrime.getTemperature(), getTemperature()) == 0;
070
071    }
072
073    @Override
074    public int equivalenceCode() {
075        return super.equivalenceCode();    //To change body of overridden methods use File | Settings | File Templates.
076    }
077}
