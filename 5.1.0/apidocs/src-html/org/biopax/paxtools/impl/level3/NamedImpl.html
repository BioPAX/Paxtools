001package org.biopax.paxtools.impl.level3;
002
003import org.biopax.paxtools.model.level3.Named;
004import org.biopax.paxtools.util.BPCollections;
005
006import java.util.Set;
007
008
009public abstract class NamedImpl extends XReferrableImpl implements Named
010{
011
012        private String standardName;
013        private String displayName;
014        private Set<String> allNames;
015
016        public NamedImpl()
017        {
018                allNames = BPCollections.I.createSet();
019                displayName = null;
020                standardName = null;
021        }
022
023        public String getStandardName()
024        {
025                return standardName;
026        }
027
028        public void setStandardName(String name)
029        {
030                addName(standardName = name);
031        }
032        
033        public String getDisplayName()
034        {
035                return displayName;
036        }
037
038        public void setDisplayName(String displayName)
039        {
040                addName(this.displayName = displayName);
041        }
042        
043        public Set<String> getName()
044        {
045                return allNames;
046        }
047
048        public void setName(Set<String> names)
049        {
050                allNames = names;
051
052                //also include the display and standard names if they're set
053                if(displayName!=null && !displayName.isEmpty())
054                        allNames.add(displayName);
055                if(standardName!=null && !standardName.isEmpty())
056                        allNames.add(standardName);
057        }
058        
059        public void addName(String name)
060        {
061                if (name != null && name.length() > 0)
062                        allNames.add(name);
063        }
064
065        public void removeName(String name)
066        {
067                if (name != null)
068                {
069                        allNames.remove(name);
070                        if (name.equals(standardName))
071                        {
072                                standardName = null;
073                        }
074                        if (name.equals(displayName))
075                        {
076                                displayName = null;
077                        }
078                }
079        }
080
081}
