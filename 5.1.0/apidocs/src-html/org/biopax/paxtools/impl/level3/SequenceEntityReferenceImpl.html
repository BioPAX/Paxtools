001package org.biopax.paxtools.impl.level3;
002
003import org.biopax.paxtools.model.BioPAXElement;
004import org.biopax.paxtools.model.level3.BioSource;
005import org.biopax.paxtools.model.level3.SequenceEntityReference;
006
007
008public abstract class SequenceEntityReferenceImpl extends EntityReferenceImpl
009        implements SequenceEntityReference
010{
011    private BioSource organism;
012    private String sequence;
013
014    public SequenceEntityReferenceImpl() {
015        }
016
017    //
018    // referenceSequenceEntity interface implementation
019    //
020    ////////////////////////////////////////////////////////////////////////////
021
022    // Property organism
023    public BioSource getOrganism()
024    {
025        return organism;
026    }
027
028    public void setOrganism(BioSource organism)
029    {
030        this.organism = organism;
031    }
032
033    // Property sequence
034
035        public String getSequence()
036    {
037        return sequence;
038    }
039
040    public void setSequence(String sequence)
041    {
042        this.sequence = sequence;
043    }
044    
045    @Override
046    protected boolean semanticallyEquivalent(BioPAXElement element) {
047        if(!(element instanceof SequenceEntityReference)) return false;
048        SequenceEntityReference that = (SequenceEntityReference) element;
049        
050        return  getOrganism() != null
051                && getOrganism().isEquivalent(that.getOrganism())
052                && getSequence() != null
053                && getSequence().equalsIgnoreCase(that.getSequence());
054
055    }
056
057    @Override
058    public int equivalenceCode()
059    {
060        return this.organism==null || this.sequence==null? hashCode():
061                this.organism.equivalenceCode()+17*this.sequence.hashCode();
062    }
063}
