001package org.biopax.paxtools.impl.level3;
002
003import org.biopax.paxtools.model.BioPAXElement;
004import org.biopax.paxtools.model.level3.BindingFeature;
005
006
007public class BindingFeatureImpl extends EntityFeatureImpl
008                implements BindingFeature
009{
010
011        private BindingFeature bindsTo;
012        private Boolean intramolecular;
013
014        //
015        // BioPAXElement interface implementation
016        //
017        ////////////////////////////////////////////////////////////////////////////
018
019        public BindingFeatureImpl() {
020        }
021
022        public Class<? extends BindingFeature> getModelInterface()
023        {
024                return BindingFeature.class;
025        }
026
027
028        //
029        // nonCovalentFeature interface implementation
030        //
031        ////////////////////////////////////////////////////////////////////////////
032
033        // Property BOUND-TO
034
035        public BindingFeature getBindsTo()
036        {
037                return bindsTo;
038        }
039
040        /**
041         * This method will set the paired binding feature that binds to this feature.
042         * This method will preserve the symmetric bidirectional semantics. If not-null old feature's
043         * bindsTo will be set to null and if not null new feature's binds to will set to this
044         * @param bindsTo paired binding feature.
045         */
046        public void setBindsTo(BindingFeature bindsTo)
047        {
048                //Check if we need to update
049                if (this.bindsTo != bindsTo)
050                {
051                        //ok go ahead - first get a pointer to the old binds to as we will swap this soon.
052                        BindingFeature old = this.bindsTo;
053                        //swap it
054                        this.bindsTo = bindsTo;
055
056                        //make sure old feature no longer points to this
057                        if (old != null && old.getBindsTo() == this)
058                        {
059                                old.setBindsTo(null);
060                        }
061                        //make sure new feature points to this
062                        if (!(this.bindsTo == null || this.bindsTo.getBindsTo() == this))
063                        {
064                                this.bindsTo.setBindsTo(this);
065                        }
066                }
067        }
068
069        public Boolean getIntraMolecular()
070        {
071                return intramolecular;
072        }
073
074        public void setIntraMolecular(Boolean intramolecular)
075        {
076                this.intramolecular = intramolecular;
077        }
078
079
080        @Override
081        protected boolean semanticallyEquivalent(BioPAXElement element)
082        {
083                if(!(element instanceof BindingFeature))
084                        return false;
085                
086                BindingFeature that = (BindingFeature) element;
087                
088                return  super.semanticallyEquivalent(element) 
089                                && this.getIntraMolecular() == that.getIntraMolecular() 
090                        && ((this.bindsTo == null) 
091                                        ? that.getBindsTo()==null 
092                                : this.bindsTo.equals(that.getBindsTo())
093                                ); //todo check equivalence sem.
094        }
095
096        @Override
097        public int equivalenceCode()
098        {
099                int value = super.equivalenceCode();
100                if(this.intramolecular!=null)
101                {
102                        value +=(this.intramolecular?29:17);
103                }
104                return value;
105        }
106}
