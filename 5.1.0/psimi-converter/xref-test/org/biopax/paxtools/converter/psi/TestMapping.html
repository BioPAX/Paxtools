84      @Test
85      public void testApi() throws Exception {
86  
87      	// unmarshall the data, close the stream
88      	PsimiXmlReader reader = new PsimiXmlReader();
89      	InputStream is = new GZIPInputStream(getClass().getClassLoader().getResourceAsStream(PSI_MI_TEST_FILE));
90      	EntrySet es = reader.read(is);
91      	is.close();
92      	Collection<Entry> entries =  es.getEntries();			
93      	// we should only have 1 entry
94      	assertEquals(1, entries.size());
95      	// get entry
96      	Entry entry = (Entry)entries.iterator().next();
97  
98      	assertTrue(entry.hasExperiments());
99      	assertEquals(4, entry.getExperiments().size());
100     	assertEquals(10, entry.getInteractions().size());
101 
102     	Interaction interaction = null;
103     	for(Interaction it : entry.getInteractions()) {
104     		if(it.getId()==1) {
105     			interaction = it;
106     			break;
107     		}
108     	}
109 
110     	//despite there is <experimentList><experimentRef> instead of <experimentList><experimentDescription> items -
111     	assertFalse(interaction.hasExperimentRefs());
112     	assertEquals(0, interaction.getExperimentRefs().size());
113     	//(- weird, but this is how PSIMI API works!)	
114     	//at the same time the following is set correctly:
115     	assertTrue(interaction.hasExperiments());
116     	assertEquals(1, interaction.getExperiments().size());
117 
118     	assertFalse(interaction.getParticipants().isEmpty());
119     	Participant participant = interaction.getParticipants().iterator().next();
120     	//despite there was interactorRef,
121     	assertFalse(participant.hasInteractorRef());
122     	assertNull(participant.getInteractorRef());
123     	//at the same time, the following is set:
124     	assertTrue(participant.hasInteractor());
125     	assertNotNull(participant.getInteractor());
126 
127     	assertFalse(participant.hasInteractionRef());
128     	assertFalse(participant.hasInteraction());			
129 
130     	
131     	//TODO assert experimentalRole.hasExperiments()==true always (if there're experiments); experimentalRole.hasExperimentRefs()==false
132 	}

192     @Test
193     public void testParticipantsAreNotDuplicated() throws IOException {
194 
195 		Model bpModel = BioPAXLevel.L3.getDefaultFactory().createModel();
196 		// open file
197 		try {
198 			// unmarshall the data, close the stream
199 			PsimiXmlReader reader = new PsimiXmlReader();
200             InputStream is = getClass().getClassLoader().getResourceAsStream(BIND_TEST_FILE);
201 			EntrySet es = reader.read(is);
202 			is.close();
203 			Collection<Entry> entries =  es.getEntries();
204             assertEquals(1, entries.size());
205 			Entry entry = (Entry)entries.iterator().next();
206 			EntryMapper mapper = new EntryMapper(bpModel, false);
207 			mapper.run(entry);
208 		}
209 		catch (Exception e) {
210 			e.printStackTrace();
211 		}
212 			
213 		assertNotNull(bpModel);
214 		assertFalse(bpModel.getObjects().isEmpty());
215 		
216 		assertEquals(3, bpModel.getObjects(MolecularInteraction.class).size());
217 		
218 		save(bpModel, getClass().getClassLoader().getResource("").getPath() 
219 			+ File.separator + "testConvertBindPsimi.owl");
220 		
221 		// there are 6 PSI-MI participants that refer to: 6 interactors and 3 experimental interactors 
222 		// (but these refer to only 5 unique primary xref IDs - 4 of protein and 1 of dna type) 
223 		// 4 "Max"/"GST-Max" and "Myc-associated factor X" interactors that have same ID makes only one PR
224 		// 2 HMBS promoter dnas make one DnaReference
225 		assertEquals(4, bpModel.getObjects(ProteinReference.class).size());
226 		// 2 dna type interactors ans experimental inter. merge into one
227 		assertEquals(1, bpModel.getObjects(DnaReference.class).size());
228 		// 9 original physical entities, some were merged (e.g., two gst-max...)
229 		assertEquals(8, bpModel.getObjects(SimplePhysicalEntity.class).size());
230 		
231 		ProteinReference pr = (ProteinReference) bpModel.getByID("ProteinReference_refseq_NP_002373_identity");
232 		assertNotNull(pr);
233 		assertEquals(2, pr.getName().size());
234 		assertTrue(pr.getName().contains("Max"));
235 		assertTrue(pr.getName().contains("Myc-associated factor X"));
236 		
237 	}

239     @Test
240     public void testMappingIntAct() {
241 		Model bpModel = BioPAXLevel.L3.getDefaultFactory().createModel();   	
242 		try {
243 			PsimiXmlReader reader = new PsimiXmlReader();
244             InputStream is = new GZIPInputStream(getClass().getClassLoader().getResourceAsStream(INTACT_TEST_FILE));
245 			EntrySet es = reader.read(is);
246 			is.close();
247 			Collection<Entry> entries =  es.getEntries();		
248 			// we should only have 1 entry
249             assertEquals(11, entries.size());
250 			Entry entry = (Entry)entries.iterator().next();
251 			EntryMapper mapper = new EntryMapper(bpModel, false);
252 			mapper.run(entry);
253 			save(bpModel,"target/" + INTACT_TEST_FILE + ".owl");
254 		}
255 		catch (Exception e) {
256 			e.printStackTrace();
257 		}			
258 		assertNotNull(bpModel);
259 		assertFalse(bpModel.getObjects().isEmpty());
260 		Set<Evidence> evs = bpModel.getObjects(Evidence.class);
261 		assertFalse(evs.isEmpty());
262 		MolecularInteraction interaction = null;
263 		for(MolecularInteraction mi : bpModel.getObjects(MolecularInteraction.class)) {
264 			if(mi.getUri().contains("1580340")) {
265 				interaction = mi;
266 				break;
267 			}
268 		}
269 		assertNotNull(interaction);
270 		assertTrue(interaction.getComment().contains(EntryMapper.FIGURE_LEGEND_CODE+ ":1C"));
271 	}
