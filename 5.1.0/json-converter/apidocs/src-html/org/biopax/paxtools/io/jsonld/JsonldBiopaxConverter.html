001package org.biopax.paxtools.io.jsonld;
002
003import java.io.File;
004import java.io.FileInputStream;
005import java.io.FileOutputStream;
006import java.io.IOException;
007import java.io.InputStream;
008import java.io.OutputStream;
009import java.text.SimpleDateFormat;
010import java.util.Calendar;
011
012import org.apache.jena.rdf.model.Model;
013import org.apache.jena.rdf.model.ModelFactory;
014import org.apache.jena.riot.Lang;
015import org.apache.jena.riot.RDFDataMgr;
016import org.biopax.paxtools.io.SimpleIOHandler;
017import org.biopax.paxtools.model.BioPAXLevel;
018
019import org.slf4j.Logger;
020import org.slf4j.LoggerFactory;
021
022public class JsonldBiopaxConverter implements JsonldConverter {
023
024        private final static Logger LOG = LoggerFactory.getLogger(JsonldBiopaxConverter.class);
025
026        /*
027         * Convert inputstream in owl/rdf format to outputsream in jsonld format
028         */
029        public void convertToJsonld(InputStream in, OutputStream os)
030                        throws IOException {
031                
032                File inputProcessedFile = preProcessFile(in);
033                LOG.info("OWl File processed successfully ");
034                
035                // print current time
036                SimpleDateFormat sdf = new SimpleDateFormat("HH:mm:ss");
037                LOG.info("Conversion RDF to JSONLD started "
038                                + sdf.format(Calendar.getInstance().getTime()));
039
040                // create an empty model
041                Model modelJena = ModelFactory.createDefaultModel();
042                InputStream internalInputStream = new FileInputStream(inputProcessedFile);
043                // read the RDF/XML file
044                RDFDataMgr.read(modelJena, internalInputStream, Lang.RDFXML);
045                LOG.info("Read into Model finished " + sdf.format(Calendar.getInstance().getTime()));
046
047                try { //close quietly and delete the temp. input file
048                        internalInputStream.close();
049                        inputProcessedFile.delete();
050                } catch(Exception e) {}
051
052                RDFDataMgr.write(os, modelJena, Lang.JSONLD);
053                LOG.info("Conversion RDF to JSONLD finished " + sdf.format(Calendar.getInstance().getTime()));
054                LOG.info(" JSONLD file " + " is written successfully.");
055
056                try { //close, flush quietly
057                        os.close();
058                } catch(Exception e) {}
059        }
060
061        
062        /*
063         * Convert inputstream in jsonld format to outputsream if owl/rdf format
064         */
065        public void convertFromJsonld(InputStream in, OutputStream out) {
066
067                Model modelJena = ModelFactory.createDefaultModel();
068
069                if (in == null) {
070                        throw new IllegalArgumentException("Input File: " + " not found");
071                }
072                if (out == null) {
073                        throw new IllegalArgumentException("Output File: " + " not found");
074                }
075
076                // read the JSONLD file
077                modelJena.read(in, null, "JSONLD");
078
079                RDFDataMgr.write(out, modelJena, Lang.RDFXML);
080                LOG.info(" RDF file " + " is written successfully.");
081
082        }
083
084        /**
085         * Converts the BioPAX data (stream) to an equivalent temporary
086         * BioPAX RDF/XML file that contains absolute instead of (possibly)
087         * relative URIs for all the BioPAX elements out there; and returns that file.
088         *
089         * @param in biopax input stream
090         * @return a temporary file
091         * @throws IOException
092     */
093        public File preProcessFile(InputStream in) throws IOException {
094
095                SimpleDateFormat sdf = new SimpleDateFormat("HH:mm:ss");
096                LOG.info("BIOPAX Conversion started "
097                                + sdf.format(Calendar.getInstance().getTime()));
098
099                if (in == null) {
100                        throw new IllegalArgumentException("Input File: " + " is not found");
101                }
102
103                SimpleIOHandler simpleIO = new SimpleIOHandler(BioPAXLevel.L3);
104
105                // create a Paxtools Model from the BioPAX L3 RDF/XML input file (stream)
106                org.biopax.paxtools.model.Model model = simpleIO.convertFromOWL(in);
107                // - and the input stream 'in' gets closed inside the above method call
108
109                // set for the IO to output full URIs:
110
111                simpleIO.absoluteUris(true);
112
113                File fullUriBiopaxInput = File.createTempFile("paxtools", ".owl");
114
115                fullUriBiopaxInput.deleteOnExit(); // delete on JVM exits
116                FileOutputStream outputStream = new FileOutputStream(fullUriBiopaxInput);
117
118                // write to an output stream (back to RDF/XML)
119
120                simpleIO.convertToOWL((org.biopax.paxtools.model.Model) model,  outputStream); // it closes the stream internally
121
122                LOG.info("BIOPAX Conversion finished " + sdf.format(Calendar.getInstance().getTime()));
123                return fullUriBiopaxInput;
124        }
125
126}
