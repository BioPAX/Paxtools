3   /**
4    * A graph object is the common interface for nodes and edges in a graph.
5    *
6    * @author Ozgun Babur
7    */
8   public interface GraphObject
9   {
10  	/**
11  	 * Every graph object should have an owner graph.
12  	 *
13  	 * @return The owner graph
14  	 */
15  	Graph getGraph();
16  
17  	/**
18  	 * This method is used for storing the object in a Map. The key should be uniquely generated for
19  	 * each edge in a graph.
20  	 *
21  	 * @return The key
22  	 */
23  	String getKey();
24  
25  	/**
26  	 * This method should clear any analysis specific labels on the object.
27  	 */
28  	void clear();
29  }
