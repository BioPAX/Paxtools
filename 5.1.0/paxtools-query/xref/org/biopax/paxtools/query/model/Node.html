1   package org.biopax.paxtools.query.model;
2   
3   import java.util.Collection;
4   
5   /**
6    * Node interface to use in graph algorithms.
7    *
8    * @author Ozgun Babur
9    */
10  public interface Node extends GraphObject
11  {
12  	/**
13  	 * @return Upstream edges
14  	 */
15  	Collection<Edge> getUpstream();
16  
17  	/**
18  	 * @return Downstream edges
19  	 */
20  	Collection<Edge> getDownstream();
21  
22  	/**
23  	 * This method is critical when the algorithm needs to calculate a path length. A graph may
24  	 * contain nodes that will not add to the path length. Such nodes are not breadth node. So the
25  	 * length of the path is the number of breadth nodes - 1.
26  	 *
27  	 * @return Whether this is a breadth node
28  	 */
29  	boolean isBreadthNode();
30  
31  	/**
32  	 * @return Parent equivalent nodes
33  	 */
34  	Collection<Node> getUpperEquivalent();
35  
36  	/**
37  	 * @return Child equivalent nodes
38  	 */
39  	Collection<Node> getLowerEquivalent();
40  
41  	/**
42  	 * Some nodes can have a sign (typically non-breadth nodes).
43  	 * @return Sign of the node
44  	 */
45  	int getSign();
46  
47  	/**
48  	 * In biological graphs, some nodes are used ubiquitously like ATP, H2O, etc. While traversing a
49  	 * graph we do not want these molecules to link two reactions. So they should be labeled.
50  	 * @return Whether this note is a ubiquitous node.
51  	 */
52  	boolean isUbique();
53  
54  	/**
55  	 * A node may be related to a transcription and an algorithm can depend on this information.
56  	 * @return Whether this node is related to a transcription event
57  	 */
58  	boolean isTranscription();
59  
60  	/**
61  	 * Initializes the node.
62  	 */
63  	void init();
64  
65  	/**
66  	 * Positive sign. This is not an Enum because we want to be able calculate sign of a path by
67  	 * multiplying signs of path elements.
68  	 */
69  	public static final int POSITIVE = 1;
70  
71  	/**
72  	 * Negative sign. This is not an Enum because we want to be able calculate sign of a path by
73  	 * multiplying signs of path elements.
74  	 */
75  	public static final int NEGATIVE = -1;
76  
77  	/**
78  	 * Neutral sign. This is not an Enum because we want to be able calculate sign of a path by
79  	 * multiplying signs of path elements.
80  	 */
81  	public static final int NEUTRAL = 0;
82  }
