3   import org.biopax.paxtools.impl.BioPAXElementImpl;
4   import org.biopax.paxtools.model.level3.Level3Element;

6   import java.util.Set;

8   
9   /**
10   * Base BioPAX Level3 element.
11   *
12   */
13  public abstract class L3ElementImpl extends BioPAXElementImpl
14          implements Level3Element
15  {
16      private Set<String> comment;
17      
18      public L3ElementImpl()
19      {
20          this.comment = BPCollections.I.createSet();
21      }
22  
23      public Set<String> getComment()
24      {
25          return this.comment;
26      }
27  
28      public void setComment(Set<String> comment)
29      {
30          this.comment = comment;
31      }
32  
33      public void addComment(String COMMENT)
34      {
35          if(COMMENT != null && COMMENT.length() > 0)
36          	this.comment.add(COMMENT);
37      }
38  
39      public void removeComment(String COMMENT)
40      {
41      	if(COMMENT != null)
42      		this.comment.remove(COMMENT);
43      }
44  }
