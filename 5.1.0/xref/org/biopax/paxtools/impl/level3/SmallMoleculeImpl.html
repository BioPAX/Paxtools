7   public class SmallMoleculeImpl extends SimplePhysicalEntityImpl
8   		implements SmallMolecule
9   {
10  	public SmallMoleculeImpl() {
11  	}
12  	
13  	@Override
14  	public Class<? extends PhysicalEntity> getModelInterface() {
15  		return SmallMolecule.class;
16  	}
17  	
18  	@Override
19  	public void addFeature(EntityFeature feature) {
20  		if(feature==null || feature instanceof BindingFeature)
21  			super.addFeature(feature);
22  		else 
23  			throw new IllegalBioPAXArgumentException(
24  					"Range violation: SmallMolecule.feature can have only BindingFeature");	
25  	}
26  	
27  	@Override
28  	public void addNotFeature(EntityFeature feature) {
29  		if(feature==null || feature instanceof BindingFeature)
30  			super.addNotFeature(feature);
31  		else 
32  			throw new IllegalBioPAXArgumentException(
33  					"Range violation: SmallMolecule.notFeature can have only BindingFeature");	
34  	}
35  
36  
37  	@Override
38  	public void setEntityReference(EntityReference entityReference) {
39  		if(entityReference instanceof SmallMoleculeReference || entityReference == null)
40  			super.setEntityReference(entityReference);
41  		else
42  			throw new IllegalBioPAXArgumentException("setEntityReference failed: "
43  					+ entityReference.getUri() + " is not a SmallMoleculeReference.");
44  	}
45  }
