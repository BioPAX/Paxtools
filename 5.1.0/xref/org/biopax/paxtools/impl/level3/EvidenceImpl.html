136 	/**
137 	 * Answers whether two Evidence objects are semantically equivalent.
138 	 * (Currently, it considers only member UnificationXrefs and EvidenceCodeVocabularies
139 	 * for comparison...)
140 	 * 
141 	 * TODO: review; compare ExperimentalForm and Confidence values; or - simply always return false!
142 	 */
143 	@Override
144 	protected boolean semanticallyEquivalent(BioPAXElement element) {
145 		if(! (element instanceof Evidence) ) return false;
146 		Evidence that = (Evidence) element; // not null (guaranteed by here)
147 		boolean hasAllEquivEvidenceCodes = false;
148 		
149 		if(this.getEvidenceCode().isEmpty()) {
150 			if(that.getEvidenceCode().isEmpty()) {
151 				hasAllEquivEvidenceCodes = true;
152 			}
153 		} else {
154 			if(!that.getEvidenceCode().isEmpty()) {
155 				Set<EvidenceCodeVocabulary> shorter;
156 				Set<EvidenceCodeVocabulary> longer;
157 				if (this.getEvidenceCode().size() < that.getEvidenceCode().size())
158 				{
159 					shorter = this.getEvidenceCode();
160 					longer  = that.getEvidenceCode();
161 				} else {
162 					longer = this.getEvidenceCode();
163 					shorter  = that.getEvidenceCode();
164 				}
165 				
166 				/* each ECV in the 'shorter' set must find its equivalent
167 				 * in the 'longer' set; 
168 				 * otherwise two Evidence objects (this and that) are not equiv.
169 				 */
170 				hasAllEquivEvidenceCodes = true; // initial guess
171 				for(EvidenceCodeVocabulary secv : shorter) {
172 					boolean foundEquiv = false;
173 					for(EvidenceCodeVocabulary lecv : longer) {
174 						if(secv.isEquivalent(lecv)) {
175 							foundEquiv = true;
176 						}
177 					}
178 					if(!foundEquiv) {
179 						hasAllEquivEvidenceCodes = false;
180 						break;
181 					}
182 				}
183 			}
184 		}
185 		
186 		//consider publication xrefs!
187 		boolean hasCommonPublicationXref = hasEquivalentIntersection(
188 				new ClassFilterSet<Xref, PublicationXref>(getXref(), PublicationXref.class),
189 				new ClassFilterSet<Xref, PublicationXref>(that.getXref(), PublicationXref.class));
190 		
191 		return super.semanticallyEquivalent(element) && hasAllEquivEvidenceCodes && hasCommonPublicationXref;
192 	}
