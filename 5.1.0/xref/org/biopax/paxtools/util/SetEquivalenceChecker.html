7   import java.util.HashSet;

9   import java.util.Set;

11  /**
12   * This class performs set operations based on equivalence.
13   */
14  public class SetEquivalenceChecker {
15  
16  	private static final Logger LOG = LoggerFactory.getLogger(SetEquivalenceChecker.class);
17  
18  	/**
19  	 * @param set1 First set to be checked.
20  	 * @param set2 Second set to be checked
21  	 * @param <T> Both sets should be of type that extends from T.
22  	 * @return true iff both sets are of equal size and all objects in set1 has an equivalent object in set2.
23  	 */
24  	public static <T extends BioPAXElement> boolean isEquivalent(Set<? extends T> set1, Set<? extends T> set2)
25  	{
26  		if (set1 != null && !set1.isEmpty() && set2 != null && !set2.isEmpty())
27  		{
28  			int size = set1.size();
29  			if (size == set2.size())
30  			{
31  				EquivalenceGrouper<T> grouper = new EquivalenceGrouper<T>();
32  				grouper.addAll(set1);
33  				if (grouper.getBuckets().size() == size)
34  				{
35  					grouper.addAll(set2);
36  					return (grouper.getBuckets().size() == size);
37  				}
38  			}
39  		} 
40  		
41  		//now, only if both null or both empty - return true 
42  		if( set1 == null && set2 == null //both null
43  			|| // or both not null but empty
44  			set1 != null && set1.isEmpty() && set2 != null && set2.isEmpty()) 
45  		{ 
46  			return true;
47  		}
48  		
49  		return false;	
50  	}
51  
52  	/**
53  	 * @param set Set to test if it contains an element equivalent to query
54  	 * @param query BPE to look for equivalents in set
55  	 * @return true iff there is an element of set that is equivalent to query.
56  	 */
57  	public static boolean containsEquivalent(Set<? extends BioPAXElement> set, BioPAXElement query)
58  	{
59  		if (set != null && query != null)
60  		{
61  			for (BioPAXElement element2 : set)
62  			{
63  				if (query.isEquivalent(element2))
64  				{
65  					return true;
66  				}
67  			}
68  		}
69  		return false;
70  	}
71  
72  	/**
73  	 * @param set1 First set to be checked.
74  	 * @param set2 Second set to be checked
75  	 * @param <T> Both sets should be of type that extends from T.
76  	 * @return elements of set1 that has an equivalent element in set2
77  	 */
78  	public static <T extends BioPAXElement> Set<T> findEquivalentIntersection(Set<? extends T> set1,
79  			Set<? extends T> set2)
80  	{
81  
82  		Set<T> value = new HashSet<T>();
83  
84  		if (set1 != null && !set1.isEmpty() && set2 != null && !set2.isEmpty())
85  		{
86  			EquivalenceGrouper<T> grouper = new EquivalenceGrouper<T>();
87  			grouper.addAll(set1);
88  			if (grouper.getBuckets().size() < set1.size())
89  			{
90  				throw new IllegalArgumentException("There should not be equivalent elements in a set");
91  			}
92  			grouper.addAll(set2);
93  			for (List<T> ts : grouper.getBuckets())
94  			{
95  				if (ts.size() > 1)
96  				{
97  					for (T t : ts)
98  					{
99  						if (set1.contains(t))
100 						{
101 							value.add(t);
102 						}
103 					}
104 				}
105 			}
106 		}
107 		return value;
108 	}
109 
110 	/**
111 	 * @param set1 First set to be checked.
112 	 * @param set2 Second set to be checked
113 	 * @param <T> Both sets should be of type that extends from T.
114 	 * @return true iff there are at least one equivalent element between set1 and set2, or both sets are empty..
115 	 */
116 	public static <T extends BioPAXElement> boolean hasEquivalentIntersection(Set<? extends T> set1,
117 			Set<? extends T> set2)
118 	{
119 		if (!set1.isEmpty() && !set2.isEmpty())
120 		{
121 			EquivalenceGrouper<T> grouper1 = new EquivalenceGrouper<T>(set1);
122 			int size1 = grouper1.getBuckets().size();
123 			if(set1.size() > size1)
124 				LOG.warn("hasEquivalentIntersection: the first set already contains equivalent objects");
125 			EquivalenceGrouper<T> grouper2 = new EquivalenceGrouper<T>(set2);
126 			int size2 = grouper2.getBuckets().size();
127 			if(set2.size() > size2)
128 				LOG.warn("hasEquivalentIntersection: the second set already contains equivalent objects");
129 			//now add both sets into one grouper -
130 			EquivalenceGrouper<T> grouper = new EquivalenceGrouper<T>();
131 			grouper.addAll(set1);
132 			grouper.addAll(set2);
133 			return (grouper.getBuckets().size() < size1 + size2);
134 		} else if(set1.isEmpty() ^ set2.isEmpty()) {
135 			//only one of sets is empty
136 			return false;
137 		} else //both empty
138 			return true;
139 	}
