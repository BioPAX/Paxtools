4   /**
5    * Definition: The chemical structure of a small molecule.
6    *
7    * Usage: Structure information is stored in the property structureData, in one of the three formats:
8    * <a href="www.xml-cml.org">CML</a>, <a href="www.daylight.com/dayhtml/smiles/">SMILES</a> or
9    * <a href="http://www.iupac.org/inchi/">InChI</a>.The structureFormat property specifies which format is used.
10   *
11   * Examples: The following SMILES string describes the structure of glucose-6-phosphate: 'C(OP(=O)(O)O)[CH]1([CH](O)
12   * [CH](O)[CH](O)[CH](O)O1)'.
13   */
14  public interface ChemicalStructure extends UtilityClass
15  {
16  
17  
18  	/**
19  	 * This property holds a string defining chemical structure,in one of the three formats:
20  	 * <a href ="www.xml-cml.org">CML</a>, <a href = "www.daylight.com/dayhtml/smiles/">SMILES</a> or
21  	 * <a href="http://www.iupac.org/inchi/">InChI</a>. If, for example,the CML format is used,
22  	 * then the value of this property is a string containing the XML encoding of the CML data.
23  	 * @return a string defining chemical structure
24  	 */
25  	String getStructureData();
26  
27  	/**
28  	 * This property holds a string of data defining chemical structure,in one of the three formats:
29  	 * <a href ="www.xml-cml.org">CML</a>, <a href = "www.daylight.com/dayhtml/smiles/">SMILES</a> or
30  	 * <a href="http://www.iupac.org/inchi/">InChI</a>. If, for example,the CML format is used,
31  	 * then the value of this property is a string containing the XML encoding of the CML data.
32  	 * @param structureData a string defining chemical structure
33  	 */
34  	 void setStructureData(String structureData);
35  
36  
37  	/**
38  	 * This property specifies which format is used to define chemical structure.
39  	 * @return format used to define chemical structure
40  	 */
41  	StructureFormatType getStructureFormat();
42  
43  	/**
44  	 * This property specifies which format is used to define chemical structure.
45  	 * @param structureFormat format used to define chemical structure
46  	 */
47  	void setStructureFormat(StructureFormatType structureFormat);
48  
49  }
