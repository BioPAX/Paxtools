6   /**
7    * Description: A characteristic of a physical entity that can change while the entity still retains its biological
8    * identity.
9    * 
10   * Rationale: Two phosphorylated forms of a protein are strictly speaking different chemical  molecules. It is,
11   * however, standard in biology to treat them as different states of the same entity,
12   * where the entity is loosely defined based on sequence. Entity Feature class and its subclassses captures these
13   * variable characteristics. A Physical Entity in BioPAX represents a pool of  molecules rather than an individual
14   * molecule. This is a notion imported from chemistry (See {@link PhysicalEntity}). Pools are defined by a set of
15   * Entity Features in the sense that a single molecule must have all of the features in the set in order to be
16   * considered a member of the pool. Since it is impossible to list and experimentally test all potential features for
17   * an  entity, features that are not listed in the selection criteria is neglected Pools can also be defined by the
18   * converse by specifying features  that are known to NOT exist in a specific context. As DNA,
19   * RNA and Proteins can be hierarchically organized into families based on sequence homology so can entity features.
20   * The memberFeature property allows capturing such hierarchical classifications among entity features.
21   * 
22   * Usage: Subclasses of entity feature describe most common biological instances and should be preferred whenever
23   * possible. One common use case for instantiating  entity feature is, for describing active/inactive states of
24   * proteins where more specific feature information is not available.
25   * 
26   * Examples: Open/close conformational state of channel proteins, "active"/"inactive" states,
27   * excited states of photoreactive groups.
28   */
29  public interface EntityFeature extends UtilityClass, Observable
30  {
31  
32  	/**
33  	 * Inverse of {@link EntityReference#getEntityFeature()}
34  	 * @return the EntityReference that this EntityFeature belongs to.
35  	 */
36  	EntityReference getEntityFeatureOf();

38  
39  	/**
40  	 * Inverse of {@link PhysicalEntity#getFeature()}
41  	 * Contents of this set is generated automatically and should not be modified.
42  	 * @return The list of PhysicalEntities that were observed to have this feature.
43  	 */
44  	Set<PhysicalEntity> getFeatureOf();
45  
46  	/**
47  	 * Inverse of {@link PhysicalEntity#getNotFeature()}
48  	 * Contents of this set is generated automatically and should not be modified.
49  	 * @return The list of PhysicalEntities that were observed to NOT have this feature.
50  	 */
51  	Set<PhysicalEntity> getNotFeatureOf();
52  
53  
54  	/**
55  	 * Location of the feature on the sequence of the interactor.
56  	 * 
57  	 * For modification features this is the modified base or residue. For binding features this is the
58  	 * binding site and for fragment features this is the location of the fragment on the "base" sequence.
59  	 * 
60  	 * One feature may have more than one location, used e.g. for features which involve sequence positions close in
61  	 * the folded, three-dimensional state of a protein, but non-continuous along the sequence.
62  	 * 
63  	 * Small Molecules can have binding features but currently it is not possible to define the binding site on the
64  	 * small molecules. In those cases this property should not be specified.
65  	 * @return Location of the feature on the sequence of the interactor.
66  	 */
67  	SequenceLocation getFeatureLocation();
68  
69  	/**
70  	 * Location of the feature on the sequence of the interactor.
71  	 * 
72  	 * For modification features this is the modified base or residue. For binding features this is the
73  	 * binding site and for fragment features this is the location of the fragment on the "base" sequence.
74  	 * 
75  	 * One feature may have more than one location, used e.g. for features which involve sequence positions close in
76  	 * the folded, three-dimensional state of a protein, but non-continuous along the sequence.
77  	 * 
78  	 * Small Molecules can have binding features but currently it is not possible to define the binding site on the
79  	 * small molecules. In those cases this property should not be specified.
80  	 * @param sequenceLocation of the feature
81  	 */
82  	void setFeatureLocation(SequenceLocation sequenceLocation);
83  
84  	/**
85  	 * A controlled vocabulary term describing the type of the sequence location such as C-Terminal or SH2 Domain.
86  	 * Using Sequence Ontology (<a href ="http://www.sequenceontology.org">http://www.sequenceontology.org</a>) is
87  	 * recommended.
88  	 * @return A CV term describing the location of the feature
89  	 */
90  	SequenceRegionVocabulary getFeatureLocationType();
91  
92  	/**
93  	 * A controlled vocabulary term describing the type of the sequence location such as C-Terminal or SH2 Domain.
94  	 * Using Sequence Ontology (<a href ="http://www.sequenceontology.org">http://www.sequenceontology.org</a>) is
95  	 * recommended.
96  	 * @param regionVocabulary A CV term describing the location of the feature
97  	 */
98  	void setFeatureLocationType(SequenceRegionVocabulary regionVocabulary);
99  
100 
101 	/**
102 	 * An entity feature  that belongs to this homology grouping.
103 	 * 
104 	 * Members of this set should be of the same class of this EntityFeature.
105 	 * 
106 	 * Members of this set should be an EntityFeature of an EntityReference which is a memberEntityReference of the
107 	 * EntityReference of this feature.
108 	 * 
109 	 * If this set is not empty than the sequenceLocation of this feature should be null.
110 	 * 
111 	 * Example: a homologous phosphorylation site across a protein family.
112 	 * @return An entity feature  that belongs to this homology grouping.
113 	 */
114 	Set<EntityFeature> getMemberFeature();   //todo generify?

116 	/**
117 	 * An entity feature  that belongs to this homology grouping.
118 	 * 
119 	 * Members of this set should be of the same class of this EntityFeature.
120 	 * 
121 	 * Members of this set should be an EntityFeature of an EntityReference which is a memberEntityReference of the
122 	 * EntityReference of this feature.
123 	 * 
124 	 * If this set is not empty than the sequenceLocation of this feature should be null.
125 	 * 
126 	 * Example: a homologous phosphorylation site across a protein family.
127 	 * @param entityFeature An entity feature  that belongs to this homology grouping.
128 	 */
129 	void addMemberFeature(EntityFeature entityFeature);
130 
131 	/**
132 	 * An entity feature  that belongs to this homology grouping.
133 	 * 
134 	 * Members of this set should be of the same class of this EntityFeature.
135 	 * 
136 	 * Members of this set should be an EntityFeature of an EntityReference which is a memberEntityReference of the
137 	 * EntityReference of this feature.
138 	 * 
139 	 * If this set is not empty than the sequenceLocation of this feature should be null.
140 	 * 
141 	 * Example: a homologous phosphorylation site across a protein family.
142 	 * @param entityFeature An entity feature that belongs to this homology grouping.
143 	 */
144 	void removeMemberFeature(EntityFeature entityFeature);
145 
146 	/**
147 	 * @param that EntityFeature to be compared
148 	 * @return returns true iff the given feature is at the equivalent sequence location with this feature
149 	 */
150 	boolean atEquivalentLocation(EntityFeature that);
151 
152 	/**
153 	 * Reverse of {@link #getMemberFeature()}
154 	 * @return the generic feature(s) that this feature belong to
155 	 */
156 	Set<EntityFeature> getMemberFeatureOf();
