1   package org.biopax.paxtools.query.algorithm;
2   
3   import org.biopax.paxtools.query.model.GraphObject;
4   import org.biopax.paxtools.query.model.Node;
5   
6   import java.util.HashSet;
7   import java.util.Map;
8   import java.util.Set;
9   
10  /**
11   * Searches the neighborhood of given source set of nodes
12   * 
13   * @author Shatlyk Ashyralyev
14   */
15  public class NeighborhoodQuery
16  {
17  	/**
18  	 * Set of source nodes.
19  	 */
20  	private Set<Node> sourceNodes;
21  	
22  	/**
23  	 * Booleans to determine the directin of search.
24  	 */
25  	private Direction direction;
26  
27  	/**
28  	 * Stop distance.
29  	 */
30  	private int limit;
31  
32  	/**
33  	 * Constructor with parameters.
34  	 * @param sourceNodes Seed to the query
35  	 * @param direction Direction of the search
36  	 * @param limit Distance limit
37  	 */
38  	public NeighborhoodQuery(Set<Node> sourceNodes, Direction direction, int limit)
39  	{
40  		if (direction == Direction.UNDIRECTED)
41  			throw new IllegalArgumentException("Direction cannot be undirected, " +
42  				"use BOTHSTREAM instead");
43  
44  		this.sourceNodes = sourceNodes;
45  		this.direction = direction;
46  		this.limit = limit;
47  	}
48  
49  	/**
50  	 * Executes the query.
51  	 * @return Neighborhood
52  	 */
53  	public Set<GraphObject> run()
54  	{
55  		// result set of neighborhood query
56  		Set<GraphObject> queryResult = new HashSet<GraphObject>();
57  
58  		// if upstream is selected
59  		if (direction == Direction.UPSTREAM || direction == Direction.BOTHSTREAM)
60  		{
61  			// run BFS in upstream direction
62  			BFS bfsBackward = new BFS(sourceNodes, null, Direction.UPSTREAM, this.limit);
63  
64  			/*
65  	  Maps to hold forward and backward BFS results
66  	 */
67  			Map<GraphObject, Integer> mapBackward = bfsBackward.run();
68  
69  			// add result of BFS to result Set
70  			queryResult.addAll(mapBackward.keySet());
71  		}
72  
73  		// if downstream is selected
74  		if (direction == Direction.DOWNSTREAM || direction == Direction.BOTHSTREAM)
75  		{
76  			// run BFS in downstream direction
77  			BFS bfsForward = new BFS(sourceNodes, null, Direction.DOWNSTREAM, this.limit);
78  
79  			Map<GraphObject, Integer> mapForward = bfsForward.run();
80  
81  			// add result of BFS to result Set
82  			queryResult.addAll(mapForward.keySet());
83  		}
84  		
85  		// return the result of query
86  		return queryResult;
87  	}
88  }
