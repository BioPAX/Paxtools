3   /**
4    * Direction is used for specifying upstream, downstream or both. Neighborhood and CommonStream
5    * queries use this enum as parameter.
6    *
7    * The difference between BOTHSTREAM and UNDIRECTED: Both-stream means that the linking paths can be
8    * either towards upstream or downstream. A linking path will always be directed towards one
9    * direction, i.e. the directions of its relations will be consistent. Undirected means the
10   * directions are not considered at all, so a linking path can contain relations towards different
11   * directions.
12   *
13   * @author Ozgun Babur
14   */
15  public enum Direction
16  {
17  	UPSTREAM("Search direction backwards in the order of events"),
18  	DOWNSTREAM("Search direction forward in the order of events"),
19  	BOTHSTREAM("Search towards both directions"),
20  	UNDIRECTED("Search without considering directions");
21  
22  	/**
23  	 * Description of the direction.
24  	 */
25  	private final String description;
26  
27  	/**
28  	 * Constructor with description.
29  	 * @param description Description
30  	 */
31  	private Direction(String description)
32  	{
33  		this.description = description;
34  	}
35  
36  	/**
37  	 * Gets the description.
38  	 * @return description
39  	 */
40  	public String getDescription()
41  	{
42  		return description;
43  	}
44  }
