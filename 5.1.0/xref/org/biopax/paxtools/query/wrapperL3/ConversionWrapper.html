1   package org.biopax.paxtools.query.wrapperL3;
2   
3   import org.biopax.paxtools.model.level3.*;
4   
5   /**
6    * Wrapper for the Conversion class.
7    *
8    * @author Ozgun Babur
9    */
10  public class ConversionWrapper extends EventWrapper
11  {
12  	/**
13  	 * Wrapped Conversion.
14  	 */
15  	protected Conversion conv;
16  
17  	/**
18  	 * The direction that the Conversion is wrapped. A separate wrapper is used for each direction
19  	 * if the Conversion is reversible.
20  	 */
21  	protected boolean direction;
22  
23  	/**
24  	 * Link to the wrapper for the same Conversion but for the other direction, if exists.
25  	 */
26  	private ConversionWrapper reverse;
27  
28  	/**
29  	 * Flag to say this Conversion is a transcription.
30  	 */
31  	protected boolean transcription;
32  
33  	/**
34  	 * Constructor with the Conversion to wrap and the owner graph.
35  	 * @param conv Conversion to wrap
36  	 * @param graph Owner graph
37  	 */
38  	protected ConversionWrapper(Conversion conv, GraphL3 graph)
39  	{
40  		super(graph);
41  		this.conv = conv;
42  	}
43  
44  	/**
45  	 * @return Direction of the conversion
46  	 */
47  	public boolean getDirection()
48  	{
49  		return direction;
50  	}
51  
52  	/**
53  	 * @return The reverse Conversion wrapper if exists
54  	 */
55  	public ConversionWrapper getReverse()
56  	{
57  		return reverse;
58  	}
59  
60  	/**
61  	 * Extracts the direction, creates the reverse if necessary.
62  	 */
63  	public void init()
64  	{
65  		if (conv.getConversionDirection() == ConversionDirectionType.REVERSIBLE &&
66  			this.reverse == null)
67  		{
68  			reverse = new ConversionWrapper(conv, (GraphL3) graph);
69  			this.direction = LEFT_TO_RIGHT;
70  			reverse.direction = RIGHT_TO_LEFT;
71  			reverse.reverse = this;
72  		}
73  		else if (conv.getConversionDirection() == ConversionDirectionType.RIGHT_TO_LEFT)
74  		{
75  			this.direction = RIGHT_TO_LEFT;
76  		}
77  		else
78  		{
79  			this.direction = LEFT_TO_RIGHT;
80  		}
81  	}
82  
83  	/**
84  	 * Binds inputs and controllers.
85  	 */
86  	@Override
87  	public void initUpstream()
88  	{
89  		if (direction == LEFT_TO_RIGHT)
90  		{
91  			for (PhysicalEntity pe : conv.getLeft())
92  			{
93  				addToUpstream(pe, getGraph());
94  			}
95  		}
96  		else
97  		{
98  			for (PhysicalEntity pe : conv.getRight())
99  			{
100 				addToUpstream(pe, getGraph());
101 			}
102 		}
103 
104 		for (Control cont : conv.getControlledOf())
105 		{
106 			if (cont instanceof Catalysis)
107 			{
108 				Catalysis cat = (Catalysis) cont;
109 
110 				if ((cat.getCatalysisDirection() == CatalysisDirectionType.LEFT_TO_RIGHT && direction == RIGHT_TO_LEFT) ||
111 					(cat.getCatalysisDirection() == CatalysisDirectionType.RIGHT_TO_LEFT && direction == LEFT_TO_RIGHT))
112 				{
113 					continue;
114 				}
115 			}
116 
117 			addToUpstream(cont, graph);
118 		}
119 	}
120 
121 	/**
122 	 * Binds products.
123 	 */
124 	@Override
125 	public void initDownstream()
126 	{
127 		if (direction == RIGHT_TO_LEFT)
128 		{
129 			for (PhysicalEntity pe : conv.getLeft())
130 			{
131 				addToDownstream(pe, getGraph());
132 			}
133 		}
134 		else
135 		{
136 			for (PhysicalEntity pe : conv.getRight())
137 			{
138 				addToDownstream(pe, getGraph());
139 			}
140 		}
141 	}
142 
143 	/**
144 	 * @return Whether this is a transcription
145 	 */
146 	public boolean isTranscription()
147 	{
148 		return transcription;
149 	}
150 
151 	/**
152 	 * RDF ID of the Conversion and the direction is used for the key.
153 	 * @return Key
154 	 */
155 	public String getKey()
156 	{
157 		return conv.getUri() + "|" + direction;
158 	}
159 
160 	/**
161 	 * @return Wrapped Conversion
162 	 */
163 	public Conversion getConversion()
164 	{
165 		return conv;
166 	}
167 
168 	/**
169 	 * @return Display name with the ID
170 	 */
171 	@Override
172 	public String toString()
173 	{
174 		return conv.getDisplayName() + " -- " + conv.getUri();
175 	}
176 
177 	/**
178 	 * Direction LEFT_TO_RIGHT.
179 	 */
180 	public static final boolean LEFT_TO_RIGHT = true;
181 
182 	/**
183 	 * Direction RIGHT_TO_LEFT.
184 	 */
185 	public static final boolean RIGHT_TO_LEFT = false;
186 }
