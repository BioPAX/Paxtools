11  /**
12   * A Trove4j based java collections (Set, Map) provider for the paxtools-core BioPAX model.
13   *
14   * Enabled by adding the JVM option:
15   * -Dpaxtools.CollectionProvider=org.biopax.paxtools.trove.TProvider
16   * when running Paxtools or another paxtools-based JAR.
17   */
18  public class TProvider implements BPCollections.CollectionProvider
19  {
20  
21  	public <R> Set<R> createSet()
22  	{
23  		return new THashSet<R>();
24  	}
25  
26  	public <D,R > Map<D,R> createMap()
27  	{
28  		return new THashMap<D, R>(4);
29  	}
30  }
