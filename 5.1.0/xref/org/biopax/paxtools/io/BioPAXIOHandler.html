1   package org.biopax.paxtools.io;
2   
3   import org.biopax.paxtools.controller.EditorMap;
4   import org.biopax.paxtools.model.BioPAXFactory;
5   import org.biopax.paxtools.model.BioPAXLevel;
6   import org.biopax.paxtools.model.Model;
7   
8   import java.io.FileNotFoundException;
9   import java.io.InputStream;
10  import java.io.OutputStream;
11  
12  /**
13   * This interface defines IO related operations that can be performed on
14   * BioPAX models.
15   */
16  public interface BioPAXIOHandler
17  {
18  
19   /**
20    * This option is only applicable two level 2 models.
21    * When enabled it will replicate illegally reused
22    * pysicalEntityParticipants in Level2 files.
23    * @param fixReusedPEPs true or false
24    */
25   void fixReusedPEPs(boolean fixReusedPEPs);
26  
27   /**
28    * This method will read the OWL document given by the input stream
29    * and will convert it into an in memory BioPAX model.
30    * @param in a BioPAX data input stream (RDF/XML format)
31    * @return new BioPAX object Model
32    */
33   Model convertFromOWL(InputStream in);
34  
35   /**
36    * This method will write the model to the output stream. Default encoding
37    * is RDF/XML.
38    * @param model a BioPAX model
39    * @param outputStream output stream
40    */
41   void convertToOWL(Model model, OutputStream outputStream);
42  
43   /**
44    * This option is only applicable two level 2 models.
45    * When enabled it will replicate illegally reused
46    * pysicalEntityParticipants in Level2 files.
47    * @return true if this option is enabled.
48    */
49   boolean isFixReusedPEPs();
50  
51   /**
52    * @return the factory that is used to create new BioPAX POJOs during a BioPAXIOHandler operation.
53    */
54   BioPAXFactory getFactory();
55  
56   /**
57    * @param factory used for creating objects
58    */
59   void setFactory(BioPAXFactory factory);
60  
61   /**
62    * @return EditorMap used for this handler.
63    */
64   EditorMap getEditorMap();
65  
66   /**
67    * @param editorMap used for this handler.
68    */
69   void setEditorMap(EditorMap editorMap);
70  
71   /**
72    * @return The level of the model that is being read.
73    */
74   BioPAXLevel getLevel();
75  
76   /**
77    * This method will "excise" a new model from the given model that contains
78    * the objects with given ids and their dependents.
79    * @param model BioPAX object model to be exported to the output stream as RDF/XML
80    * @param outputStream the stream
81    * @param ids optional list of absolute URIs of BioPAX objects - roots/seeds for extracting a sub-model to be exported
82    */
83   void convertToOWL(Model model, OutputStream outputStream, String... ids);
84  
85  }
