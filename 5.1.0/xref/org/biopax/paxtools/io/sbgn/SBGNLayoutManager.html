40      /**
41       * Applies CoSE layout to the given SBGN PD model.
42       *
43       * @param sbgn model where layout is performed and stored
44       * @param doLayout whether to actually run the CoSE or just complete the SBGN model
45       */
46      public void createLayout(final Sbgn sbgn, boolean doLayout)
47      {
48          viewToLayout = new HashMap();
49          glyphToVNode = new HashMap();
50          idToGLyph = new HashMap();
51          idToCompartmentGlyphs = new HashMap();
52          portIDToOwnerGlyph = new HashMap();
53          layoutToView = new HashMap();
54          idToArcs = new HashMap<String, Arc>();
55  
56          this.layout = new SbgnPDLayout();
57          final LGraphManager graphMgr = layout.getGraphManager();
58          graphMgr.addRoot();
59  
60          root = new VCompound(new Glyph());
61  
62          // Detect compartment glyphs and put them in a hashmap;
63          // also set compartment glyphs of members of complexes.
64          for (Glyph g: sbgn.getMap().getGlyph())
65          {
66              if(glyphClazzOneOf(g, GlyphClazz.COMPARTMENT)) {
67                  idToCompartmentGlyphs.put(g.getId(), g);
68              }
69              //Set compartmentRef to all children of a Complex node.
70              Glyph compartment = (Glyph)g.getCompartmentRef();
71              if(compartment != null && glyphClazzOneOf(g, GlyphClazz.COMPLEX)) {
72                  setCompartmentRefForComplexMembers(g, compartment, new HashSet<Glyph>());
73              }
74          }
75  
76          // Nest glyphs inside compartment glyphs according to their compartmentRef.
77          // This list holds the glyphs that will be deleted after corresponding glyph
78          // is added to child glyph of another glyph.
79          if(!idToCompartmentGlyphs.isEmpty()) {
80              List<Glyph> deletedList = new ArrayList<Glyph>();
81              for (Glyph g : sbgn.getMap().getGlyph()) {
82                  Glyph containerCompartment = (Glyph) g.getCompartmentRef();
83                  if (containerCompartment != null) {
84                      idToCompartmentGlyphs.get(containerCompartment.getId()).getGlyph().add(g);
85                      deletedList.add(g);
86                  }
87              }
88              // Delete the duplicate glyphs, after they are moved to corresponding compartment glyph.
89              for (Glyph g : deletedList) {
90                  sbgn.getMap().getGlyph().remove(g);
91              }
92          }
93  
94          // initialize the map for keeping ports and their owner glyphs
95          // with entries like: <portID, ownerGlyph>
96          initPortIdToGlyphMap(sbgn.getMap().getGlyph());
97  
98          //Remove ports from source and target field of ports
99          //replace them with owner glyphs of these ports
100         removePortsFromArcs(sbgn.getMap().getArc());
101 
102         // Assign logical operator and Process nodes to compartment
103         assignProcessAndLogicOpNodesToCompartment(sbgn);
104 
105         // Create Vnodes for ChiLay layout component
106         createVNodes(root, sbgn.getMap().getGlyph());
107 
108         for (VNode vNode: root.children) {
109             createLNode(vNode, null);
110         }
111 
112         // Create LEdges for ChiLay layout component
113         createLEdges(sbgn.getMap().getArc());
114 
115         // Apply layout
116         if(doLayout) {
117             layout.runLayout();
118         }
119 
120         graphMgr.updateBounds();
121 
122         // Here if any SbgnProcessNode node is returned from SBGNPD Layout
123         // this means that we will have two additional port info. We should
124         // add this information to libSbgn objects
125         for (Object lNode : layout.getAllNodes()) {
126             if (lNode instanceof SbgnProcessNode) {
127                 //Set geometry of corresponding node
128                 VNode vNode = layoutToView.get(((SbgnProcessNode) lNode).label);
129                 Bbox tempBbox = vNode.glyph.getBbox();
130                 tempBbox.setX((float) (((SbgnProcessNode) lNode).getLeft()));
131                 tempBbox.setY((float) (((SbgnProcessNode) lNode).getTop()));
132                 vNode.glyph.setBbox(tempBbox);
133 
134                 //Created port objects in layout level
135                 SbgnPDNode inputLPort = ((SbgnProcessNode) lNode).getInputPort();
136                 SbgnPDNode outputLPort = ((SbgnProcessNode) lNode).getOutputPort();
137 
138                 // New port objects
139                 Port inputPort = new Port();
140                 Port outputPort = new Port();
141 
142                 // Set port attributes
143                 inputPort.setX((float) (inputLPort.getCenterX()));
144                 inputPort.setY((float) (inputLPort.getCenterY()));
145                 inputPort.setId(inputLPort.label);
146 
147                 outputPort.setX((float) (outputLPort.getCenterX()));
148                 outputPort.setY((float) (outputLPort.getCenterY()));
149                 outputPort.setId((outputLPort.label));
150 
151                 //Clear existing ports !
152                 vNode.glyph.getPort().clear();
153                 //Connect existing arcs to newly created ports
154                 //These ports are created by ChiLay and SBGNPD Layout
155                 connectArcToPort(inputLPort, inputPort);
156                 connectArcToPort(outputLPort, outputPort);
157 
158                 //Add ports to the corresponding glyph
159                 vNode.glyph.getPort().add(inputPort);
160                 vNode.glyph.getPort().add(outputPort);
161             }
162         }
163 
164         // Update the bounds
165         for (VNode vNode: root.children) {
166             updateCompoundBounds(vNode.glyph, vNode.glyph.getGlyph());
167         }
168 
169         // Clear inside of the compartmentGlyphs
170         for (Glyph compGlyph: idToCompartmentGlyphs.values()) {
171             //Again add the members of compartments
172             for(Glyph memberGlyph: compGlyph.getGlyph() ) {
173                 sbgn.getMap().getGlyph().add(memberGlyph);
174             }
175             compGlyph.getGlyph().clear();
176         }
177     }

179     /**
180      * This method connects the existing arcs to the newly created
181      * ports which are created by ChiLay and SBGNPD Layout.
182      * @param lPort l level port object.
183      * @param vPort v level port object.
184      * */
185     private void connectArcToPort(SbgnPDNode lPort, Port vPort)
186     {
187         //Iterate over the edges of l level port
188         for  (Object e: (lPort.getEdges()))
189         {
190             //Ignore rigid edges
191             if(((LEdge)e).type.equals("rigid edge"))
192                 continue;
193 
194             //Determine the if vPort is source or target
195             Arc arc = idToArcs.get(((LEdge)e).label);
196             if( lPort.label.equals(((LEdge)e).getSource().label ))
197             {
198                 arc.setSource(vPort);
199             }
200             else if ( lPort.label.equals(((LEdge)e).getTarget().label ) )
201             {
202                 arc.setTarget(vPort);
203             }
204         }
205     }

207     /**
208      * This method finds process nodes and logical operator nodes in sbgn map and assigns them to a compartment by using majority rule.
209      * @param sbgn Given Sbgn map.
210      * */
211     private void assignProcessAndLogicOpNodesToCompartment(Sbgn sbgn)
212     {
213         // Create a hashmap for keeping a node( generally logical operators and process nodes ) and its neighbours.
214         // TreeMap value of the hash map keeps track of compartment nodes that includes neighbours of the node by String id and
215         // Integer value holds the number of occurences of that compartment among the neighbours of the node as parent.
216         HashMap <String, HashMap<String,Integer>>  nodetoNeighbours = new HashMap<String, HashMap<String,Integer>>();
217         List<Glyph> glyphList = sbgn.getMap().getGlyph();
218         List<Arc>   arcList   = sbgn.getMap().getArc();
219 
220         // Keeps track of process and logical operator nodes that will be assigned to a compartment.
221         ArrayList<Glyph> targetNodes = new ArrayList<Glyph>();
222 
223         //Iterate over glyphs of sbgn map
224         for(Glyph glyph: glyphList)
225         {
226             // Here logical operator nodes and process nodes are interested !
227             if(glyphClazzOneOf(glyph, GlyphClazz.PROCESS, GlyphClazz.OMITTED_PROCESS,
228                     GlyphClazz.UNCERTAIN_PROCESS, GlyphClazz.PHENOTYPE, GlyphClazz.ASSOCIATION,
229                     GlyphClazz.DISSOCIATION, GlyphClazz.AND, GlyphClazz.OR, GlyphClazz.NOT))
230             {
231                 // Add a new value to hash map and also store the node as target node
232                 String processGlyphID = glyph.getId();
233 //                String rootID = "root"; //was not used anywhere
234                 nodetoNeighbours.put(processGlyphID, new HashMap<String, Integer>());
235                 targetNodes.add(glyph);
236 
237                 // Iterate over arc list
238                 for(Arc arc: arcList)
239                 {
240                     Glyph target = null;
241                     Glyph source = null;
242 
243                     // If source and target of node is port find its owner glyph ! else just assign it.
244                     if(arc.getSource() instanceof Port)
245                         source = portIDToOwnerGlyph.get(((Port)arc.getSource()).getId());
246                     else
247                         source = (Glyph)arc.getSource();
248 
249                     if(arc.getTarget() instanceof Port)
250                         target = portIDToOwnerGlyph.get(((Port)arc.getTarget()).getId());
251                     else
252                         target = (Glyph)arc.getTarget();
253 
254                     // If source of any arc is our node, then target must be neighbour of this node !
255                     if(source.getId().equals(processGlyphID))
256                     {
257                         populateCompartmentOccurencesMap(target, nodetoNeighbours.get(processGlyphID));
258                     }
259                     // same as target part !!
260                     else if(target.getId().equals(processGlyphID))
261                     {
262                         populateCompartmentOccurencesMap(source, nodetoNeighbours.get(processGlyphID));
263                     }
264                 }
265             }
266         }
267 
268         //Finally assign nodes to compartments by majority rule
269         for(Glyph glyph: targetNodes)
270         {
271             String id = glyph.getId();
272             HashMap<String, Integer> compartmentsOfTargetNode = nodetoNeighbours.get(id);
273 
274             // Finally sort the hashmap to obtain the compartment that includes majority of the neighbours of the targetNode 
275             List<Map.Entry<String,Integer>> entries = new LinkedList<Map.Entry<String,Integer>>(compartmentsOfTargetNode.entrySet());
276             Collections.sort(entries, new Comparator<Map.Entry<String,Integer>>() {
277                 @Override
278                 public int compare(Map.Entry<String, Integer> o1, Map.Entry<String, Integer> o2)
279                 {
280                     return -(o1.getValue().compareTo(o2.getValue()));
281                 }
282             });
283 
284             if(entries.size() > 0)
285             {
286                 // if the process belongs to root do not make any changes
287                 if(entries.get(0).getKey().equals("root"))
288                     continue;
289 
290                 Glyph compartment = idToCompartmentGlyphs.get(entries.get(0).getKey());
291                 if(compartment != null) {
292                     //Set compartmentRef of process glyph also!
293                     glyph.setCompartmentRef(compartment);
294                     compartment.getGlyph().add(glyph);
295                     //Remove it from sbgn also
296                     sbgn.getMap().getGlyph().remove(glyph);
297                 }
298             }
299         }
300     }

302     /**
303      * Updates a hashmap by incrementing the number of nodes in the compartment glyph that includes targetGlyph.
304      * It is assumed that given hashmap includes compartment ids' as keys and number of nodes as values.This method 
305      * is an utility method that will be used to populate a hashmap while determining the compartment node of a process
306      * node by majority rule.
307      *
308      * @param targetGlyph glyph whose occurence will be updated in the given hashmap.
309      * @param compartmentIDandOccurenceMap  Map that references number of nodes in a compartment by compartment ids .
310      * */
311     private void populateCompartmentOccurencesMap(Glyph targetGlyph,  HashMap<String, Integer> compartmentIDandOccurenceMap)
312     {
313         String rootID = "root";
314 
315         // if compartment ref of targetGlyph node is not null, increment its occurence by 1
316         if(targetGlyph.getCompartmentRef() != null)
317         {
318             Glyph  containerCompartment = (Glyph)targetGlyph.getCompartmentRef();
319             String compartmentID = containerCompartment.getId();
320             Integer compartmentOccurrenceValue = compartmentIDandOccurenceMap.get(compartmentID);
321 
322             if( compartmentOccurrenceValue != null)
323             {
324                 compartmentIDandOccurenceMap.put(compartmentID, compartmentOccurrenceValue + 1);
325             }
326             else
327                 compartmentIDandOccurenceMap.put(compartmentID, 1);
328         }
329         // else targetGlyph is in root graph so increment root graphs counter value by 1
330         else
331         {
332             Integer compartmentOccurrenceValue = compartmentIDandOccurenceMap.get(rootID);
333 
334             if( compartmentOccurrenceValue != null)
335             {
336                 compartmentIDandOccurenceMap.put(rootID, compartmentOccurrenceValue + 1);
337             }
338             else
339                 compartmentIDandOccurenceMap.put(rootID, 1);
340         }
341     }

343     /**
344      * Updates bounds of a compound node ( i.e. complex glyph ) from its children .
345      * @param parent compound glyph.
346      * @param childGlyphs related children of parent .
347      * */
348     private void updateCompoundBounds(Glyph parent,List<Glyph> childGlyphs)
349     {
350         float PAD = (float) 2.0;
351         float minX = Float.MAX_VALUE; float minY = Float.MAX_VALUE;
352         float maxX = Float.MIN_VALUE; float maxY = Float.MIN_VALUE;
353 
354         for (Glyph tmpGlyph : childGlyphs)
355         {
356             if(!glyphClazzOneOf(tmpGlyph, GlyphClazz.UNIT_OF_INFORMATION, GlyphClazz.STATE_VARIABLE))
357             {
358                 if(tmpGlyph.getGlyph().size() > 0)
359                     updateCompoundBounds(tmpGlyph, tmpGlyph.getGlyph());
360 
361                 float w = tmpGlyph.getBbox().getW();
362                 float h = tmpGlyph.getBbox().getH();
363 
364                 // Verify MIN and MAX x/y again:
365                 minX = Math.min(minX, (tmpGlyph.getBbox().getX()));
366                 minY = Math.min(minY, (tmpGlyph.getBbox().getY()));
367                 maxX = Math.max(maxX, (tmpGlyph.getBbox().getX())+w);
368                 maxY = Math.max(maxY, (tmpGlyph.getBbox().getY())+h);
369 
370                 if (minX == Float.MAX_VALUE) minX = 0;
371                 if (minY == Float.MAX_VALUE) minY = 0;
372                 if (maxX == Float.MIN_VALUE) maxX = 0;
373                 if (maxY == Float.MIN_VALUE) maxY = 0;
374 
375                 parent.getBbox().setX(minX - PAD);
376                 parent.getBbox().setY(minY - PAD);
377                 parent.getBbox().setW(maxX -  parent.getBbox().getX() + PAD);
378                 parent.getBbox().setH(maxY -  parent.getBbox().getY() + PAD);
379             }
380         }
381     }

383     /**
384      * Recursively creates VNodes from Glyphs of Sbgn.
385      *
386      * @param parent Parent of the glyphs that are passed as second arguement.
387      * @param glyphs Glyphs that are child of parent which is passed as first arguement.
388      *
389      * */
390     private void createVNodes(VCompound parent,List<Glyph> glyphs)
391     {
392         for(Glyph glyph: glyphs )
393         {
394             if (!glyphClazzOneOf(glyph, GlyphClazz.UNIT_OF_INFORMATION, GlyphClazz.STATE_VARIABLE))
395             {
396 //                if(glyph.getClazz().equals(GlyphClazz.PROCESS.getClazz()))
397 //                {
398 //                    VCompound v = new VCompound(glyph); //TODO: v is never used; wat's the idea?
399 //                }
400 
401                 if(!isChildless(glyph))
402                 {
403                     VCompound v = new VCompound(glyph);
404                     idToGLyph.put(glyph.getId(), glyph);
405                     glyphToVNode.put(glyph, v);
406                     parent.children.add(v);
407                     createVNodes(v, glyph.getGlyph());
408                 }
409                 else
410                 {
411                     VNode v = new VNode(glyph);
412                     idToGLyph.put(glyph.getId(), glyph);
413                     glyphToVNode.put(glyph, v);
414                     parent.children.add(v);
415                 }
416             }
417         }
418     }

420     /**
421      * Creates LNodes from Arcs of Sbgn and adds it to the passed layout object.
422      *
423      * @param arcs List of arc objects from which the LEdges will be constructed for ChiLay Layout component.
424      *
425      * */
426     private void createLEdges(List<Arc> arcs)
427     {
428         for(Arc arc: arcs )
429         {
430             LEdge lEdge = layout.newEdge(null);
431             lEdge.type = arc.getClazz();
432             lEdge.label = arc.getId();
433             LNode sourceLNode = viewToLayout.get(glyphToVNode.get(arc.getSource()));
434             LNode targetLNode = viewToLayout.get(glyphToVNode.get(arc.getTarget()));
435             idToArcs.put(arc.getId(), arc);
436 
437             // Add edge to the layout
438             layout.getGraphManager().add(lEdge, sourceLNode, targetLNode);
439         }
440     }

442     /**
443      * Helper function for creating LNode objects from VNode objects and adds them to the given layout.
444      *
445      * @param vNode  VNode object from which a corresponding LNode object will be created.
446      * @param parent parent of vNode, if not null vNode will be added to layout as child node.
447      * */
448 
449     private void createLNode(VNode vNode, VNode parent)
450     {
451         LNode lNode = layout.newNode(vNode);
452         lNode.type  = vNode.glyph.getClazz();
453         lNode.label = vNode.glyph.getId();
454         LGraph rootLGraph = layout.getGraphManager().getRoot();
455 
456         //Add corresponding nodes to corresponding maps
457         viewToLayout.put(vNode, lNode);
458         layoutToView.put(lNode.label,vNode);
459 
460         // if the vNode has a parent, add the lNode as a child of the parent l-node.
461         // otherwise, add the node to the root graph.
462         if (parent != null)
463         {
464             LNode parentLNode = viewToLayout.get(parent);
465             parentLNode.getChild().add(lNode);
466         }
467 
468         else
469         {
470             rootLGraph.add(lNode);
471         }
472 
473         lNode.setLocation(vNode.glyph.getBbox().getX(), vNode.glyph.getBbox().getY());
474 
475         if (vNode instanceof VCompound)
476         {
477             VCompound vCompound = (VCompound) vNode;
478             // add new LGraph to the graph manager for the compound node
479             layout.getGraphManager().add(layout.newGraph(null), lNode);
480             // for each VNode in the node set create an LNode
481             for (VNode vChildNode: vCompound.getChildren())
482             {
483                 createLNode(vChildNode, vCompound);
484             }
485         }
486 
487         else
488         {
489             lNode.setWidth(vNode.glyph.getBbox().getW());
490             lNode.setHeight(vNode.glyph.getBbox().getH());
491         }
492     }

494     /**
495      * Recursively sets compartmentRef fields of members of a complex glyph
496      * to the same value as complex's compartment.
497      *
498      * @param glyph target glyph where compartmentRef(compartment parameter) will be set.
499      * @param compartment compartmentRef value that will be set.
500      * @param visited (initially an empty set of) previously processed glyphs, to escape infinite loops
501      * */
502     private void setCompartmentRefForComplexMembers(final Glyph glyph, final Glyph compartment,
503                                                     final Set<Glyph> visited)
504     {
505         if(!visited.add(glyph))
506             return;
507 
508         glyph.setCompartmentRef(compartment);
509         if(glyph.getGlyph().size() > 0) {
510             for(Glyph g: glyph.getGlyph() ) {
511                 setCompartmentRefForComplexMembers(g, compartment, visited);
512             }
513         }
514     }

516     /**
517      * This method replaces ports of arc objects with their owners.
518      * @param arcs Arc list of SBGN model
519      * */
520     private void removePortsFromArcs(List<Arc> arcs)
521     {
522         for(Arc arc: arcs )
523         {
524             // If source is port, first clear port indicators else retrieve it from hashmaps
525             if (arc.getSource() instanceof Port )
526             {
527                 Glyph source = portIDToOwnerGlyph.get(((Port)arc.getSource()).getId());
528                 arc.setSource(source);
529             }
530 
531             // If target is port, first clear port indicators else retrieve it from hashmaps
532             if (arc.getTarget() instanceof Port)
533             {
534                 Glyph target = portIDToOwnerGlyph.get(((Port)arc.getTarget()).getId());
535                 arc.setTarget(target);
536             }
537         }
538     }

541      * This method initializes map for glyphs and their respective ports.
542      * @param glyphs Glyph list of SBGN model
543      * */
544     private void initPortIdToGlyphMap(List<Glyph> glyphs)

546         for(Glyph glyph: glyphs)
547         {
548             for(Port p: glyph.getPort())
549             {
550                 portIDToOwnerGlyph.put(p.getId(), glyph );
551             }
552             if(glyph.getGlyph().size() > 0)
553                 initPortIdToGlyphMap(glyph.getGlyph());
554         }

557     /**
558      * Returns true if a glyph includes child glyphs (state and info glyphs are out of count!)
559      * @param targetGlyph target glyph that will be queried.
560      * @return true/false
561      * */
562     private boolean isChildless(Glyph targetGlyph)
563     {
564         boolean checker = true;
565         for(Glyph glyph: targetGlyph.getGlyph() )
566         {
567             if ( !glyphClazzOneOf(glyph, GlyphClazz.STATE_VARIABLE, GlyphClazz.UNIT_OF_INFORMATION) )
568             {
569                 checker = false;
570                 break;
571             }
572         }
573         return checker;
574     }

576     private boolean glyphClazzOneOf(Glyph g, GlyphClazz... clazzes) {
577         for(GlyphClazz clazz : clazzes) {
578             if(clazz.getClazz().equals(g.getClazz()))
579                 return true;
580         }
581         return false;
582     }
