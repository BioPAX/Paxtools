55  /**
56   * PSIMI 'entry' to BioPAX converter. 
57   * 
58   * PS (rodche, IR):
59   * The PSI-MI api is weird. It has methods/classes that pretend to follow the XML schema (OXM like)
60   * but in fact some of those do not make any practical sense, because they are either cleared or reset by the 
61   * PSI-MI parser (post-processing); e.g., look at (try) all those get*Ref or has*Ref, etc. methods.
62   * PSI-MI specification itself is tricky (e.g., the ability to define/express multiple experiment 
63   * descriptions per interaction, multiple roles per participant, multiple experimentalInteractors 
64   * per participant of an interaction are probably altogether unnecessary) and uses conventions 
65   * and practices not formally defined in the schema (e.g., how/whether to use "id" attribute, 
66   * absence of some xml elements means sometimes ALL such elements from the parent element (see e.g. experimentRef).
67   *
68   * @author Benjamin Gross, rodche (major re-factoring for Level3; fixing, adding genetic interactions, experimental form entities and features, etc.)
69   */

122 	 * Convert a PSIMI entry to BioPAX
123 	 * interactions, participants, etc. objects 
124 	 * and add to the target BioPAX model.
125 	 * 
126 	 * @param entry
127 	 */
128 	public void run(Entry entry) {	
129 		
130 		// get availabilities
131 		final Set<String> avail = new HashSet<String>();
132 		if(entry.hasAvailabilities()) {
133 			for (Availability a : entry.getAvailabilities())
134 				if (a.hasValue()) 
135 					avail.add(a.getValue());	
136 		}
137 		
138 		// get data source
139 		final Provenance pro = createProvenance(entry.getSource());
140 		
141 		//build a skip-set of "interactions" linked by participant.interactionRef element;
142 		//we'll then create Complex type participants from these in-participant interactions.
143 		Set<Interaction> participantInteractions = new HashSet<Interaction>();
144 		for(Interaction interaction : entry.getInteractions()) {
145 			for(Participant participant : interaction.getParticipants()) {
146 				//checking for hasInteraction()==true only is sufficient; 
147 				//i.e., we ignore hasInteractionRef(), getInteractionRefs(), 
148 				//because these in fact get cleared by the PSI-MI parser:
149 				if(participant.hasInteraction()) {
150 					participantInteractions.add(participant.getInteraction());
151 				}
152 			}
153 		}
154 		
155 		// iterate through the "root" psimi interactions only and create biopax interactions or complexes
156 		for (Interaction interaction : entry.getInteractions()) {
157 			if(!participantInteractions.contains(interaction)) {
158 				// TODO future (hard): make a Complex or Interaction based on the interaction type ('direct interaction' or 'physical association' (IntAct) -> complex)
159 				processInteraction(interaction, avail, pro, false);
160 			}
161 		}
162 	}

165 	private Provenance createProvenance(Source source) {
166 		Provenance pro = null;		
167 		String name = null;
168 		PublicationXref px = null;
169 		UnificationXref ux = null;
170 		
171 		if (source.hasNames()) {
172 			name = getName(source.getNames());
173 		} 
174 		
175 		if(source.hasXref()) {
176 			ux = getPrimaryUnificationXref(source.getXref());
177 			if(name==null && ux != null)
178 				name = ux.getDb()+"_"+ux.getId();
179 		} 
180 			
181 		if(source.hasBibref()) {
182 			psidev.psi.mi.xml.model.Xref psiXREF = source.getBibref().getXref();
183 			if (psiXREF != null) {
184 				px = publicationXref(psiXREF.getPrimaryRef().getDb(), psiXREF.getPrimaryRef().getId());
185 			}
186 			
187 			if(name==null && px != null)
188 				name = px.getDb()+"_"+px.getId();
189 		}
190 		
191 		String ver = null;
192 		if(source.hasReleaseDate()) {
193 			ver = source.getRelease();
194 			if(name==null)
195 				name = ver;
196 		}
197 		
198 		String sourceUri = (name!=null) 
199 			? xmlBase + "Provenance_" + encode(name)
200 				: genUri(Provenance.class);
201 		
202 		//unless it's already there,
203 		pro = (Provenance) bpModel.getByID(sourceUri);		
204 		if(pro == null) { //generate a new one
205 			pro = bpModel.addNew(Provenance.class, sourceUri);
206 			
207 			if(name != null)
208 				pro.setDisplayName(name);
209 			
210 			if(px != null)
211 				pro.addXref(px);
212 			
213 			if(ux != null)
214 				pro.addXref(ux);
215 			
216 			if(source.hasAttributes())
217 				for(Attribute attr : source.getAttributes())
218 					pro.addComment(attr.toString());
219 			
220 			if(ver != null)
221 				pro.addComment("Release Date: " + ver);
222 		}
223 		
224 		return pro;
225 	}

229 	 * Creates a paxtools object that
230 	 * corresponds to the psi interaction.
231 	 *
232 	 * Note:
233 	 * psi.interactionElementType                 -&gt; biopax Complex, MolecularInteraction, or GeneticInteraction
234 	 * psi.interactionElementType.participantList -&gt; biopax interaction/complex participants/components
235 	 */
236 	private Entity processInteraction(Interaction interaction, Set<String> avail, 
237 			Provenance pro, boolean isComplex) 
238 	{	
239 		Entity bpInteraction = null; //interaction or complex
240 		boolean isGeneticInteraction = false;
241 		
242 		// get interaction name/short name
243 		String name = null;
244 		String shortName = null;
245 		if (interaction.hasNames()) {
246 			Names names = interaction.getNames();
247 			name = (names.hasFullName()) ? names.getFullName() : "";
248 			shortName = (names.hasShortLabel()) ? names.getShortLabel() : "";
249 		}
250 		
251 		final Set<InteractionVocabulary> interactionVocabularies = new HashSet<InteractionVocabulary>();
252 		if (interaction.hasInteractionTypes()) {
253 			for(CvType interactionType : interaction.getInteractionTypes()) {
254 				//generate InteractionVocabulary and set interactionType
255 				InteractionVocabulary cv = findOrCreateControlledVocabulary(interactionType, InteractionVocabulary.class);
256 				if(cv != null)
257 					interactionVocabularies.add(cv);
258 			}
259 		}		
260 		
261 		// using experiment descriptions, create Evidence objects 
262 		// (yet, no experimental forms/roles/entities are created here)
263 		Set<Evidence> bpEvidences = new HashSet<Evidence>();
264 		if (interaction.hasExperiments()) {			
265 			bpEvidences = createBiopaxEvidences(interaction);
266 		}
267 		
268 		//A hack for e.g. IntAct or BIND "gene-protein" interactions (ChIp and EMSA experiments)
269 		// where the interactor type should probably not be 'gene' (but 'dna' or 'rna')
270 		Set<String> participantTypes = new HashSet<String>();
271 		for(Participant p : interaction.getParticipants()) {
272 			if(p.hasInteractor()) {
273 				String type = getName(p.getInteractor().getInteractorType().getNames());
274 				if(type==null) type = "protein"; //default type (if unspecified)
275 				participantTypes.add(type.toLowerCase());
276 			} else if (p.hasInteraction()) {
277 				participantTypes.add("complex"); //hierarchical complex build up
278 			} // else? (impossible!)
279 		}
280 		// If there are both genes and physical entities present, let's 
281 		// replace 'gene' with 'dna' (esp. true for "ch-ip", "emsa" experiments);
282 		// (this won't affect experimental form entities if experimentalInteractor element exists)
283 		if(participantTypes.size() > 1 && participantTypes.contains("gene")) {
284 			//TODO a better criteria to reliably detect whether 'gene' interactor type actually means Dna/DnaRegion or Rna/RnaRegion, or indeed Gene)
285 			LOG.warn("Interaction: " + interaction.getId() + ", name(s): " + shortName + " " + name 
286 					+ "; has both 'gene' and physical entity type participants: " + participantTypes 
287 					+ "; so we'll replace 'gene' with 'dna' (a quick fix)");
288 			for(Participant p : interaction.getParticipants()) {
289 				if(p.hasInteractor() && p.getInteractor().getInteractorType().hasNames()) {
290 					String type = getName(p.getInteractor().getInteractorType().getNames());
291 					if("gene".equalsIgnoreCase(type)) {
292 						p.getInteractor().getInteractorType().getNames().setShortLabel("dna");
293 					}
294 				}
295 			}	
296 		}		
297 		
298 		// interate through the psi-mi participants, create corresp. biopax entities
299 		final Set<Entity> bpParticipants = new HashSet<Entity>();
300 		for (Participant participant : interaction.getParticipants()) {
301 			// get paxtools physical entity participant and add to participant list
302 			// (this also adds experimental evidence and forms)
303 			Entity bpParticipant = createBiopaxEntity(participant, avail, pro); 
304 			if (bpParticipant != null) {
305 				if(!bpParticipants.contains(bpParticipant))
306 					bpParticipants.add(bpParticipant);
307 			}
308 		}
309 				
310 		// Process interaction attributes.
311 		final Set<String> comments = new HashSet<String>();
312 		// Set GeneticInteraction flag.
313 		// As of BioGRID v3.1.72 (at least), genetic interaction code can reside
314 		// as an attribute of the Interaction via "BioGRID Evidence Code" key
315 		if (interaction.hasAttributes()) {
316 			for (Attribute attribute : interaction.getAttributes()) {
317 				String key = attribute.getName(); //may be reset below
318 				String value = (attribute.hasValue()) ? attribute.getValue() : "";
319 				if(key.equalsIgnoreCase(BIOGRID_EVIDENCE_CODE)
320 						&& GENETIC_INTERACTIONS.contains(value))
321 				{
322 					isGeneticInteraction = true; // important!
323 				}
324 				comments.add(key + ":" + value);
325 			}
326 		}
327 		// or, if all participants are 'gene' type, make a biopax GeneticInteraction
328 		if(participantTypes.size() == 1 && participantTypes.contains("gene")) {
329 			isGeneticInteraction = true;
330 		}		
331 		//or, check another genetic interaction flag (criteria)
332 		if(!isGeneticInteraction) {
333 			isGeneticInteraction = isGeneticInteraction(bpEvidences);
334 		}
335 		
336 		if ((isComplex || forceInteractionToComplex) && !isGeneticInteraction) {
337 			bpInteraction = createComplex(bpParticipants, interaction.getImexId(), interaction.getId());
338 		} else if(isGeneticInteraction) {
339 			bpInteraction = createGeneticInteraction(bpParticipants, interactionVocabularies,
340 					interaction.getImexId(), interaction.getId()
341 			);
342 		} else {
343 			bpInteraction = createMolecularInteraction(bpParticipants, interactionVocabularies,
344 					interaction.getImexId(), interaction.getId());
345 		}
346 
347 		for(String c : comments) {
348 			bpInteraction.addComment(c);
349 		}
350 
351 		//add evidences to the interaction/complex bpEntity
352 		for (Evidence evidence : bpEvidences) {
353 			bpInteraction.addEvidence(evidence);
354 			//TODO: shall we add IntAct "figure legend" comment to the evidences as well?
355 		}
356 		
357 		addAvailabilityAndProvenance(bpInteraction, avail, pro);
358 		
359 		if (name != null)
360 			bpInteraction.addName(name);
361 		if (shortName != null) {
362 			if(shortName.length()<51)
363 				bpInteraction.setDisplayName(shortName);
364 			else
365 				bpInteraction.addName(shortName);
366 		}
367 				
368 		// add xrefs		
369 		Set<Xref> bpXrefs = new HashSet<Xref>();		
370 		if (interaction.hasXref()) {
371 			bpXrefs.addAll(getXrefs(interaction.getXref()));
372 		}
373 
374 		for (Xref bpXref : bpXrefs) {
375 			bpInteraction.addXref(bpXref);
376 		}
377 		
378 		return bpInteraction;
379 	}

382 	private Set<Evidence> createBiopaxEvidences(Interaction interaction) {
383 		Set<Evidence> evidences = new HashSet<Evidence>();
384 		
385 		for (ExperimentDescription experimentDescription : interaction.getExperiments()) {
386 			// build and add evidence
387 			String evUri = genUri(Evidence.class,
388 					interaction.getImexId(), interaction.getId(), experimentDescription.getId());
389 			
390 			Evidence evidence = bpModel.addNew(Evidence.class, evUri);
391 
392 			if (experimentDescription.hasXref()) {
393 				for(Xref xref : getXrefs(experimentDescription.getXref()))
394 					evidence.addXref(xref);
395 			}
396 			if (experimentDescription.getBibref() != null) {
397 				psidev.psi.mi.xml.model.Xref psiXREF = experimentDescription.getBibref().getXref();
398 				if (psiXREF != null) {
399 					PublicationXref px = publicationXref(psiXREF.getPrimaryRef().getDb(), psiXREF.getPrimaryRef().getId());
400 					if(px != null) 
401 						evidence.addXref(px);
402 				}
403 			}
404 					
405 			// create comments
406 			// from names (there is no biopax Evidence.name property)
407 			if (experimentDescription.hasNames()) {
408 				Names names = experimentDescription.getNames();
409 				if(names.hasFullName())
410 					evidence.addComment(names.getFullName().trim());
411 				if(names.hasShortLabel())
412 					evidence.addComment(names.getShortLabel().trim());
413 			}
414 			
415 			// from attributes
416 			if (experimentDescription.hasAttributes()) {
417 				for(String attr : getAttributes(experimentDescription.getAttributes()))
418 					evidence.addComment(attr.trim());
419 			}
420 
421 			//add hostOrganism info to comments (there is no 'organism' property of Evidence, EF, etc)
422 			if(experimentDescription.hasHostOrganisms()) {
423 				for(Organism organism : experimentDescription.getHostOrganisms())
424 					evidence.addComment("Host " + organism.toString());
425 			}
426 
427 			// confidence list
428 			if (experimentDescription.hasConfidences()) {
429 				for (Confidence psiConfidence : experimentDescription.getConfidences()) {
430 					Score bpScoreOrConfidence = getScoreOrConfidence(psiConfidence);
431 					if (bpScoreOrConfidence != null) 
432 						evidence.addConfidence(bpScoreOrConfidence);
433 				}
434 			}		
435 			
436 			// interaction detection method, participant detection method, feature detection method
437 			Set<EvidenceCodeVocabulary> evidenceCodes = getEvidenceCodes(experimentDescription);			
438 			if(evidenceCodes != null)
439 				for(EvidenceCodeVocabulary ecv : evidenceCodes)
440 					evidence.addEvidenceCode(ecv);
441 			
442 			evidences.add(evidence);
443 		}
444 		
445 		return evidences;
446 	}

461 	 * Converts a PSI-MI interaction's participant 
462 	 * to a BioPAX physical entity or gene
463 	 * and corresponding experimental form entities.
464 	 */
465 	private Entity createBiopaxEntity(Participant participant, Set<String> avail, Provenance pro) 
466 	{	
467 		LOG.debug("createBiopaxEntity, processing participant: " + participant.getId());
468 		
469 		// create a new PE or Gene to be
470 		Entity entity = null;
471 	
472 		// The PSI-MI parser does not set 'interactorRef' 
473 		// (or it rather clears that property and updates 'interactor' prop. instead;  
474 		// see junit tests).
475 		if (participant.hasInteractor()) {
476 			//participant.getInteractor() is not null
477 			entity = createParticipant(participant, participant.getInteractor(), avail, pro, false);
478 		} else if(participant.hasInteraction()) {
479 			//hierarchical buildup of a complex (participant.hasInteraction==true)...
480 			entity = (Complex) processInteraction(participant.getInteraction(), avail, pro, true);
481 		}
482 				
483 		if(entity == null) 
484 			return null;
485 		
486 		//create new Evidence for this participant
487 		Evidence participantEvidence = bpModel.getLevel().getDefaultFactory()
488 			.create(Evidence.class, genUri(Evidence.class, "p",participant.getId()));
489 		
490 		if(participant.hasExperimentalRoles() 
491 			|| participant.hasExperimentalInteractors() 
492 				|| participant.hasFeatures()) 
493 		{
494 			if(participant.hasExperimentalInteractors()) {
495 				//create and collect experimental entities for current participant:
496 				for(ExperimentalInteractor experimentalInteractor : participant.getExperimentalInteractors()) {
497 					Interactor interactor = experimentalInteractor.getInteractor();
498 					LOG.debug("createBiopaxEntity, participant: " + participant.getId() 
499 							+ ", exp. interactor: " + interactor.getId());
500 					Entity expEntity = createParticipant(participant, interactor, avail, pro, true);
501 					assert expEntity!=null : "expEntity is null";
502 					
503 					//try to reuse existing experimental form entity
504 					expEntity = findEquivalentEntity(expEntity);
505 					if(!bpModel.contains(expEntity))
506 						bpModel.add(expEntity);
507 					
508 					//workaround a PSI-MI parser issue (no exp. or exp.refs means to apply to all interaction's experiments):
509 					if(experimentalInteractor.hasExperiments()) {
510 						for(ExperimentDescription exp : experimentalInteractor.getExperiments()) {
511 							LOG.debug("createBiopaxEntity, making EFs for exp: " 
512 									+ exp.getId() + "; exp.ent: " + expEntity.getUri());
513 							createAddExperimentalForm(participant, participantEvidence, expEntity, exp);
514 							
515 						}
516 					} else {
517 						LOG.debug("createBiopaxEntity, making a EForm - one for all experiments " 
518 								+ "; exp.ent: " + expEntity.getUri());
519 						createAddExperimentalForm(participant, participantEvidence, expEntity, null);
520 					}
521 				}		
522 			} else {
523 				LOG.debug("createBiopaxEntity, participant: " + participant.getId() 
524 						+ " doesn't have any exp. interactors");
525 				createAddExperimentalForm(participant, participantEvidence, null, null);
526 			}
527 			
528 			//save the participant entity evidence if it's not empty (roles, forms were actually generated)
529 			if(!participantEvidence.getExperimentalForm().isEmpty()) {
530 				bpModel.add(participantEvidence);
531 				entity.addEvidence(participantEvidence);
532 				//do not call findEquivalentEntity(entity) for this one having not empty evidence
533 			} else {
534 				//try to merge/reuse existing equivalent entity
535 				entity = findEquivalentEntity(entity);
536 			}
537 		}
538 	
539 		//finally, add, if new, entity to the model and return
540 		if(!bpModel.contains(entity))
541 			bpModel.add(entity);
542 		
543 		return entity;
544 	}

548 	 * Carefully find an existing equivalent physical entity or gene  
549 	 * or return the same one unchanged.
550 	 */
551 	private Entity findEquivalentEntity(final Entity entity) {
552 		Entity toReturn = entity;
553 		
554 		Class<? extends Entity> entityClass = PhysicalEntity.class;
555 		if(entity instanceof Gene)
556 			entityClass = Gene.class;
557 		
558 		for(Entity existingEntity : bpModel.getObjects(entityClass)) {
559 			// replace with an existing equivalent entity iif
560 			if( (//both are not experimental form entities
561 				 !entity.getComment().contains(EXPERIMENTAL_FORM_ENTITY_COMMENT)
562 				 && !existingEntity.getComment().contains(EXPERIMENTAL_FORM_ENTITY_COMMENT)
563 				)
564 				|| 
565 				(//both are experimental form entities
566 				 entity.getComment().contains(EXPERIMENTAL_FORM_ENTITY_COMMENT)
567 				 && existingEntity.getComment().contains(EXPERIMENTAL_FORM_ENTITY_COMMENT)
568 				 //and if disp.names match: names like 'GST-Max' are often used to describe states, instead of using other psi-mi features...
569 				 && entity.getDisplayName().equalsIgnoreCase(existingEntity.getDisplayName())
570 				)
571 			){	
572 				//and if there are no evidences yet, and the two are equivalent, then return existing one (to replace the entity)
573 				if(existingEntity.getEvidence().isEmpty() && entity.getEvidence().isEmpty() 
574 						&& existingEntity.isEquivalent(entity)) {
575 					toReturn = (Entity) existingEntity;
576 					break;
577 				}
578 			}
579 		}
580 		
581 		return toReturn;
582 	}

585 	/*
586 	 * Converts a PSIMI participant to BioPAX physical entity or gene. 
587 	 * It can be then used either as a participant of an interaction 
588 	 * or exp. form entity of a exp. form of a participant's evidence,
589 	 * depending on the caller method and the map provided.
590 	 * The result entity is not added to the model yet (but its entity reference, 
591 	 * if applicable, Xrefs, CVs are added to the model).
592 	 * 
593 	 * @param participant - of a PSI-MI interaction
594 	 * @param interactor - participant's interactor or experimental interactor's interactor
595 	 * @param avail - (biopax) availability
596 	 * @param pro - (biopax) provenance
597 	 * @param isExperimentalForm - a flag that controls entity's key comments (important)  
598 	 * 							 and how entity references are merged (e.g., copy names or not)
599 	 */
600 	private Entity createParticipant(Participant participant, Interactor interactor, 
601 			Set<String> avail, Provenance pro, boolean isExperimentalForm) 
602 	{
603 		if(interactor==null)
604 			throw new AssertionError("createParticipant: participant: " 
605 					+ participant.getId() + " has got null interactor");
606 		
607 		// Find or create the physical entity and entity reference -		
608 		// figure out physical entity type (protein, dna, rna, small molecule, gene)
609 		String entityType = null;
610 		CvType interactorType = interactor.getInteractorType();
611 		if (interactorType != null && interactorType.hasNames()) {
612 			entityType = getName(interactorType.getNames());
613 		}
614 
615 		// get names/synonyms from the psimi interactor (participant does not have them)
616 		String name = null;
617 		String shortName = null;
618 		Set<String> synonyms = new HashSet<String>();
619 		Names psiNames = interactor.getNames();
620 		if (psiNames != null) {
621 			name = (psiNames.hasFullName()) ? psiNames.getFullName() : null;
622 			shortName = (psiNames.hasShortLabel()) ? psiNames.getShortLabel() : null;
623 			if (psiNames.hasAliases()) {
624 				for (Alias alias : psiNames.getAliases()) {
625 					if (alias.hasValue()) {
626 						synonyms.add(alias.getValue());
627 					}
628 				}
629 			}
630 		}
631 		
632 		//get cellular location, if any
633 		CellularLocationVocabulary cellularLocation = null;
634 		if((interactor.hasOrganism() && interactor.getOrganism().hasCompartment())) {
635 			cellularLocation = findOrCreateControlledVocabulary(
636 					interactor.getOrganism().getCompartment(), CellularLocationVocabulary.class);
637 		}
638 		
639 		// make one entity reference (ER) per unique interactor (merge duplicate/equivalent ones)
640 		// make one physical entity (PE) per unique participant... 		
641 		// (a unique state PE is defined by the ER and cell. location, but not by entity features - which are to go with ExperimentalForms)		
642 		Class<? extends Entity> entityClass = Protein.class; //default
643 		Class<? extends EntityReference> entityReferenceClass = ProteinReference.class; //default
644 		if(entityType!=null && !entityType.isEmpty()) {
645 			//entityType should be a term from MI:0313 "interactor type" CV
646 			entityType = entityType.toLowerCase();
647 			if ("small molecule".equals(entityType)
648 					|| "polysaccharide".equals(entityType)
649 					|| "bioactive entity".equals(entityType))
650 			{
651 				entityClass = SmallMolecule.class;
652 				entityReferenceClass = SmallMoleculeReference.class;
653 			} else if ("dna".equals(entityType) 
654 					|| "deoxyribonucleic acid".equals(entityType) 
655 					|| entityType.contains("deoxyribonucleic")
656 					|| entityType.contains("dna")
657 					|| "nucleic acid".equals(entityType)) //when can't tell dna from rna? guess DNA
658 			{
659 				entityClass = Dna.class;
660 				entityReferenceClass = DnaReference.class;
661 			} else if ("rna".equals(entityType)
662 					|| "ribonucleic acid".equals(entityType)
663 					|| "poly adenine".equals(entityType)
664 					|| entityType.contains("rna"))
665 			{
666 				entityClass = Rna.class;
667 				entityReferenceClass = RnaReference.class;
668 			} else if ("gene".equals(entityType))
669 			{ //NOTE: Gene "gene" type is for Genetic Interactions only (not transcription/translation regulation, etc.), but...
670 				entityClass = Gene.class;
671 				entityReferenceClass = null;
672 			} else if ("complex".equals(entityType) || entityType.contains("complex"))
673 			{
674 				entityClass = Complex.class;
675 				entityReferenceClass = null;
676 			} else if ("interaction".equals(entityType))
677 			{
678 //				entityClass = null;
679 //				entityReferenceClass = null;
680 				LOG.warn("EntryMapper.createParticipant(): skip for interactor: " + interactor.getId()
681 						+ " that has type: " + entityType 
682 						+ ", participant:" + participant.getId() + ").");
683 				return null;
684 			}
685 						
686 			//else - peptide, biopolymer, molecule set, unknown, etc. - consider this is a Protein.
687 		}
688 		
689 		//make consistent biopax URI (for the entity ref., if applies, or phys. entity)
690 		String entityUri = "";
691 		String baseUri = "";
692 		
693 		final RelationshipXref x = getInteractorPrimaryRef(interactor.getXref());
694 		if(x != null) {
695 			baseUri += encode(x.getDb() + "_" + x.getId());
696 			if(x.getRelationshipType()!=null)
697 				baseUri += "_" + encode(x.getRelationshipType().getTerm().iterator().next());
698 			entityUri = baseUri + "_" + String.valueOf(counter++);
699 		} else { //when no xrefs present, use a number ending (always increment);
700 			baseUri = String.valueOf(counter++); //new unique part (seldom happens, when no primary xref...)
701 			entityUri = baseUri;
702 		}
703 		
704 		if(cellularLocation != null)
705 			entityUri += "_" + encode(cellularLocation.getTerm().iterator().next());
706 		
707 		//makes URI look like xmlBase+type+unique_suffix...
708 		entityUri = xmlBase + entityClass.getSimpleName() + "_" + entityUri;	
709 		if(entityReferenceClass != null)
710 			baseUri = xmlBase + entityReferenceClass.getSimpleName() + "_" + baseUri;	
711 		
712 		// create a new PE or Gene (don't add to the model yet, for there may exist an equivalent one)
713 		Entity entity = bpModel.getLevel().getDefaultFactory().create(entityClass, entityUri);
714 		
715 		addAvailabilityAndProvenance(entity, avail, pro);
716 		//and names
717 		if (name != null) {
718 			//quite a few PSIMI providers use too long text (like comments) for <fullName> fields...
719 			if(name.length() > 100 && shortName != null) 
720 				entity.addComment(name);
721 			else 
722 				entity.addName(name);
723 		}
724 		if (shortName != null) {
725             entity.setDisplayName(shortName);
726 		}				
727 		// set cellular location if possible
728 		if(cellularLocation != null && entity instanceof PhysicalEntity)
729 			((PhysicalEntity)entity).setCellularLocation(cellularLocation);
730 		
731 		final Set<Xref> bpXrefsOfInteractor = getXrefs(interactor.getXref());
732 		final BioSource bioSource = getBioSource(interactor);
733 		
734 		// when not a Gene, we are to find/generate corresponding EntityRererence;
735 		// but we do not want to gererate apparently duplicate objects...
736 		if(entityReferenceClass != null) {
737 			EntityReference entityReference = null;
738 			
739 			/* check if the entity ref. with this URI already exists 
740 			 * (we'll the use that instead of generating a new one every time;
741 			 * unfortunately, chances still are that we generate same URI 
742 			 * from non-equivalent psi-mi interactors, though this would probably 
743 			 * flag for semantic errors in the psi-mi data, such as using same xrefs,
744 			 * features, etc. for different xml entries unnecessarily).
745 			 * Unfortunately, we cannot simply create a new ER and compare
746 			 * with an existing one using Paxtools (as of paxtools-core 4.x and older, 
747 			 * two ERs are NOT equivalent unless they either have the same URI, or - same organism and sequence).
748 			 */
749 			EntityReference er = (EntityReference) bpModel.getByID(baseUri);			
750 			if( er != null 
751 				&& er.getModelInterface()==entityReferenceClass 
752 				&& (!(er instanceof SequenceEntityReference) 
753 					|| sameNameOrUndefined(((SequenceEntityReference)er).getOrganism(),bioSource)
754 					)
755 			) {
756 				
757 				entityReference = er; // ok to reuse
758 				
759 			} else if(er != null) {
760 				String newUri = baseUri + "_" + (counter++);
761 				LOG.warn("A different " + er.getModelInterface().getSimpleName()
762 					+ ", URI=" + baseUri + ", was found; for interactor:" + interactor.getId()
763 					+ ", a new ("+entityReferenceClass.getSimpleName()+") URI will be used:" + newUri);
764 				baseUri = newUri;
765 			}
766 			
767 			//if not found above, create a new ER
768 			if(entityReference == null) { 
769 				entityReference = bpModel.addNew(entityReferenceClass, baseUri);	
770 				
771 				if (shortName != null) {
772 					entityReference.setDisplayName(shortName);
773 				}
774 				
775 				//set organism if it's not a small molecule
776 				if(entityReference instanceof SequenceEntityReference) {
777 					SequenceEntityReference ser = (SequenceEntityReference)entityReference;
778 					ser.setOrganism(bioSource);
779 					ser.setSequence(interactor.getSequence());
780 				}
781 			}
782 				
783 			//update the displayName if we've got to reuse the existing ER,
784 			//and this is not an experimental interactor, and new name is shorter
785 			if(entityReference == er && shortName != null && !isExperimentalForm) {
786 				if(er.getDisplayName() != null) {
787 					if(shortName.length() < er.getDisplayName().length()) {
788 						entityReference.addName(er.getDisplayName()); //keep both names
789 						entityReference.setDisplayName(shortName);
790 					} else {
791 						entityReference.addName(shortName); //keep both names
792 					}
793 				} else {
794 					entityReference.setDisplayName(shortName);
795 				}
796 			}
797 			
798 			// if we're not reusing existing ER, and this interactor is not experimental,
799 			// copy names and xrefs to the exisiting ER
800 			if( !(isExperimentalForm && entityReference == er) ) {
801 				if (name != null) {
802 					if(name.length() > 100 && //and there are other names
803 							(entityReference.getDisplayName()!=null || (synonyms!=null && !synonyms.isEmpty())))
804 						entityReference.addComment(name); //comment instead of 'standardName'
805 					else		
806 						entityReference.addName(name);
807 				}
808 				if (synonyms != null) {
809 					for (String synonym : synonyms)
810 						entityReference.addName(synonym);
811 				}
812 				if (bpXrefsOfInteractor != null) {
813 					for (Xref xref : bpXrefsOfInteractor)
814 						entityReference.addXref((Xref) xref);
815 				}
816 			}
817 			
818 			// set ER
819 			((SimplePhysicalEntity)entity).setEntityReference(entityReference);		
820 			
821 		} else { //i.e., entity is a Gene or Complex
822 			//add gene's other names, xrefs, organism (for non-genes, these're added to the ER)
823 			if (synonyms != null) {
824 				for (String synonym : synonyms) {
825 					entity.addName(synonym);
826 				}
827 			}		
828 			if (bpXrefsOfInteractor != null) {
829 				for (Xref xref : bpXrefsOfInteractor)
830 					entity.addXref((Xref) xref);
831 			}
832 			if(entity instanceof Gene) //can be Complex here as well
833 				((Gene)entity).setOrganism(bioSource);
834 		}
835 		
836 		//Merge some of equivalent entities to avoid unnecessary duplicates 
837 		// CAREFULLY, e.g.:
838 		// - never merge a participant entity with an experimental form entity,
839 		// - never merge if one entity have evidence, etc.
840 		//(equiv. ERs are being merged regardless this)
841 		//Merging entities (states) can mess up names, features, exp. roles!
842 		if(isExperimentalForm)
843 			entity.addComment(EXPERIMENTAL_FORM_ENTITY_COMMENT);
844 		else
845 			entity.addComment("psi-mi participant");
846 		
847 		return entity;
848 	}

905 	 * Given a psi organism, return a paxtools biosource.
906 	 */
907 	private BioSource getBioSource(Interactor interactor) {
908 		
909 		Organism organism = interactor.getOrganism();
910 		
911 		// trivial check
912 		if (organism == null) 
913 			return null;
914 
915 		// cell type (can be undefined, i.e., null)
916 		CellVocabulary cellType = findOrCreateControlledVocabulary(organism.getCellType(), CellVocabulary.class);
917 		// tissue (can be null)
918 		TissueVocabulary tissue = findOrCreateControlledVocabulary(organism.getTissue(), TissueVocabulary.class);
919 
920 		// set the BioPAXElement URI and taxonomy xref id
921 		String ncbiId = Integer.toString(organism.getNcbiTaxId());
922 		String name = null;
923 		
924 		if (organism.hasNames()) {
925 			name = getName(organism.getNames());
926 			//a hack for BIND data (and perhaps other DBs)
927 			if(name != null && ("homo sapiens".equalsIgnoreCase(name)
928 					|| "human".equalsIgnoreCase(name))
929 					&& !"9606".equals(ncbiId)) {
930 				LOG.error("Taxonomy: " 
931 					+ ncbiId + " of organism: " + organism 
932 					+ ", interactor: " + interactor.getId() 
933 					+ " does not belong to '" 
934 					+ name + "'; the name will be removed. "
935 					+ "Data provider should probably use <experimentalInteractor> in addition to <interactor>");
936 				name = null;
937 			}
938 		}
939 		
940 		String uri = xmlBase + "BioSource_" + 
941 			"taxonomy_" + ncbiId; //tissue and cell type terms can be added below		
942 		if(tissue!=null && !tissue.getTerm().isEmpty()) 
943 			uri += "_" + encode(tissue.getTerm().iterator().next());
944 		if(cellType!=null && !cellType.getTerm().isEmpty()) 
945 			uri += "_" + encode(cellType.getTerm().iterator().next());
946 
947 		//return if element already exists in model
948 		BioSource toReturn = (BioSource) bpModel.getByID(uri);
949 		if (toReturn != null) 
950 			return toReturn;
951 		
952 		toReturn = bpModel.addNew(BioSource.class, uri);
953 		
954 		String taxonXrefUri = xmlBase + "UnificationXref_taxonomy_" + ncbiId;
955 		UnificationXref taxonXref = (UnificationXref) bpModel.getByID(taxonXrefUri);
956 		if(taxonXref == null) {
957 			taxonXref = bpModel.addNew(UnificationXref.class, taxonXrefUri);
958 			taxonXref.setDb("Taxonomy");
959 			taxonXref.setId(ncbiId);
960 		}
961 		toReturn.addXref((Xref)taxonXref);
962 		
963 		if (cellType != null)
964 		{
965 			toReturn.setCellType((CellVocabulary) cellType);
966 		}
967 		if (tissue != null)
968 		{
969 			toReturn.setTissue((TissueVocabulary) tissue);
970 		}
971 		if (name != null) {
972 			toReturn.setStandardName(name);
973 		}
974 				
975 		return toReturn;
976 	}

1029 	private Set<Xref> getXrefs(psidev.psi.mi.xml.model.Xref psiXREF) {
1030 
1031 		// set to return
1032 		Set<Xref> toReturn = new HashSet<Xref>();
1033 
1034 		// check args
1035 		if (psiXREF == null) return toReturn;
1036 		
1037 		// create the list of all other psimi xrefs
1038 		List<DbReference> psiDBRefList = new ArrayList<DbReference>();
1039 
1040 		DbReference primaryRef = psiXREF.getPrimaryRef();
1041 		if(primaryRef != null) { //looks, it should never be null in a valid PSI-MI model
1042             if ("identity".equals(primaryRef.getRefType()) || "identical object".equals(primaryRef.getRefType())) {
1043                 UnificationXref bpXref = unificationXref(primaryRef.getDb(), primaryRef.getId());
1044 				toReturn.add(bpXref);
1045             } else {
1046 				psiDBRefList.add(psiXREF.getPrimaryRef());
1047 			}
1048 		}
1049 
1050 		if (psiXREF.hasSecondaryRef()) {
1051 			psiDBRefList.addAll(psiXREF.getSecondaryRef());
1052 		}
1053 
1054 		for (DbReference psiref : psiDBRefList) {
1055 			if(psiref==null) 
1056 				continue;
1057 			xref(psiref, toReturn);
1058         }
1059 
1060 		return toReturn;
1061 	}

1064 	private void xref(DbReference psiref, Set<Xref> bpXrefs) {
1065 		// process ref type
1066 		String refType = (psiref.hasRefType()) ? psiref.getRefType() : null;
1067 		String refTypeAc = (psiref.hasRefTypeAc()) ? psiref.getRefTypeAc() : null;
1068         String psiDBRefId = psiref.getId();
1069         String psiDBRefDb = psiref.getDb();
1070         // If multiple ids given with comma separated values, then split them.
1071         for (String dbRefId : psiDBRefId.split(",")) {	
1072         	Xref bpXref = null;
1073 			// Let's not make UnificationXrefs. RelationshipXref is more safe.
1074 			// Often, a gene or omim ID (can be another species') is a protein's xref id with 'identity' type...
1075             if(!"pubmed".equalsIgnoreCase(psiDBRefDb)) {
1076 				bpXref = relationshipXref(psiDBRefDb, dbRefId, refType, refTypeAc);
1077             }
1078             else {
1079             	//TODO shall we skip PublicationXref here (IntAct puts the same PSIMI paper pmid everywhere...)?
1080 				bpXref = publicationXref(psiDBRefDb, dbRefId);
1081             }
1082 
1083             if (bpXref != null) 
1084             	bpXrefs.add(bpXref);
1085         }
1086 	}
