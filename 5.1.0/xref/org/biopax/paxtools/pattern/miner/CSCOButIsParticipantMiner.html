1   package org.biopax.paxtools.pattern.miner;
2   
3   import org.biopax.paxtools.model.level3.ProteinReference;
4   import org.biopax.paxtools.pattern.Pattern;
5   import org.biopax.paxtools.pattern.PatternBox;
6   import org.biopax.paxtools.pattern.constraint.Type;
7   
8   /**
9    * Miner for the controls-state-change pattern.
10   * @author Ozgun Babur
11   */
12  public class CSCOButIsParticipantMiner extends ControlsStateChangeOfMiner
13  {
14  	/**
15  	 * Constructor that sets name and description.
16  	 */
17  	public CSCOButIsParticipantMiner()
18  	{
19  		super("-but-a-participant", "The controller is modeled as a " +
20  			"participant of the Conversion. This is in fact a modeling error, but this pattern " +
21  			"exists in some resources. The controller PhysicalEntity appears at both left and " +
22  			"right of the Conversion.");
23  	}
24  
25  	/**
26  	 * Constructs the pattern.
27  	 * @return pattern
28  	 */
29  	@Override
30  	public Pattern constructPattern()
31  	{
32  		Pattern p = PatternBox.controlsStateChangeButIsParticipant();
33  		p.add(new Type(ProteinReference.class), "controller ER");
34  		p.add(new Type(ProteinReference.class), "changed ER");
35  		return p;
36  	}
37  
38  	@Override
39  	public String[] getMediatorLabels()
40  	{
41  		return new String[]{"Conversion"};
42  	}
43  
44  	@Override
45  	public String getSourceLabel()
46  	{
47  		return "controller ER";
48  	}
49  
50  	@Override
51  	public String getTargetLabel()
52  	{
53  		return "changed ER";
54  	}
55  
56  
57  	@Override
58  	public String[] getSourcePELabels()
59  	{
60  		return new String[]{"controller simple PE", "controller PE"};
61  	}
62  
63  	@Override
64  	public String[] getTargetPELabels()
65  	{
66  		return new String[]{"input PE", "input simple PE", "output PE", "output simple PE"};
67  	}
68  }
