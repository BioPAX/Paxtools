18  /**
19   * Miner for the related genes (participant or controller) of interactions.
20   * @author Ozgun Babur
21   */
22  public class RelatedGenesOfInteractionsMiner extends MinerAdapter
23  {
24  	private static final PathAccessor controlAcc = new PathAccessor("Interaction/controlledOf*");
25  
26  	/**
27  	 * Constructor that sets name and description.
28  	 */
29  	public RelatedGenesOfInteractionsMiner()
30  	{
31  		super("related-genes-of-interactions", "This miner finds any related gene that is a " +
32  			"participant or a controller of an Interaction (Conversion or TemplateReaction). " +
33  			"The output lists the ID of the interaction and the related gene symbols in a row.");
34  	}
35  
36  	/**
37  	 * Constructs the pattern.
38  	 * @return pattern
39  	 */
40  	public Pattern constructPattern()
41  	{
42  		return PatternBox.relatedProteinRefOfInter(
43  			Conversion.class,
44  			TemplateReaction.class);
45  	}
46  
47  	/**
48  	 * Writes the IDs of interaction, then gene symbols of related proteins in a line.
49  	 * @param matches pattern search result
50  	 * @param out output stream
51  	 */
52  	public void writeResult(Map<BioPAXElement, List<Match>> matches, OutputStream out)
53  		throws IOException
54  	{
55  		OutputStreamWriter writer = new OutputStreamWriter(out);
56  
57  		for (BioPAXElement ele : matches.keySet())
58  		{
59  			Set<String> syms = new HashSet<String>();
60  
61  			for (Match m : matches.get(ele))
62  			{
63  				ProteinReference pr = (ProteinReference) m.get("PR", getPattern());
64  
65  				String sym = getGeneSymbol(pr);
66  				if (sym != null) syms.add(sym);
67  			}
68  
69  			if (syms.size() > 1)
70  			{
71  				writer.write("\n" + ele.getUri());
72  
73  				for (Object o : controlAcc.getValueFromBean(ele))
74  				{
75  					Control ctrl = (Control) o;
76  					writer.write(" " + ctrl.getUri());
77  				}
78  
79  				for (String sym : syms)
80  				{
81  					writer.write("\t" + sym);
82  				}
83  			}
84  		}
85  		writer.flush();
86  	}
87  
88  }
