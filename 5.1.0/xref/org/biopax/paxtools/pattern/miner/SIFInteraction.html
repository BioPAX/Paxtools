27  	public SIFInteraction(String sourceID, String targetID, BioPAXElement sourceER,
28  		BioPAXElement targetER, SIFType type, Set<BioPAXElement> mediators,
29  		Set<BioPAXElement> sourcePEs, Set<BioPAXElement> targetPEs)
30  	{
31  		this.sourceID = sourceID;
32  		this.targetID = targetID;
33  		this.sourceERs = new HashSet<BioPAXElement>();
34  		this.targetERs = new HashSet<BioPAXElement>();
35  
36  		this.sourceERs.add(sourceER);
37  		this.targetERs.add(targetER);
38  		this.type = type;
39  
40  		this.mediators = mediators;
41  		this.sourcePEs = sourcePEs;
42  		this.targetPEs = targetPEs;
43  	}
44  
45  	public boolean hasIDs()
46  	{
47  		return sourceID != null && targetID != null;
48  	}
49  
50  	@Override
51  	public int hashCode()
52  	{
53  		return sourceID.hashCode() + targetID.hashCode() + type.hashCode();
54  	}

56  	@Override
57  	public boolean equals(Object obj)
58  	{
59  		if (obj instanceof SIFInteraction)
60  		{
61  			SIFInteraction i = (SIFInteraction) obj;
62  
63  			return i.type.equals(type) && i.sourceID.equals(sourceID) && i.targetID.equals(targetID);
64  		}
65  
66  		return false;
67  	}
68  
69  	@Override
70  	public int compareTo(Object o)
71  	{
72  		if (o instanceof SIFInteraction)
73  		{
74  			SIFInteraction i = (SIFInteraction) o;
75  
76  			return (sourceID + targetID + type.getTag()).compareTo(
77  				i.sourceID + i.targetID + i.type.getTag());
78  		}
79  
80  		return 0;
81  	}

83  	/**
84  	 * Merges publications of the parameter equivalent sif with this one.
85  	 * @param equivalent the equivalent sif interaction to get its publications.
86  	 */
87  	public void mergeWith(SIFInteraction equivalent)
88  	{
89  		if (!this.equals(equivalent))
90  			throw new IllegalArgumentException("SIF interactions are not equivalent.");
91  
92  		sourceERs.addAll(equivalent.sourceERs);
93  		targetERs.addAll(equivalent.targetERs);
94  
95  		sourcePEs.addAll(equivalent.sourcePEs);
96  		targetPEs.addAll(equivalent.targetPEs);
97  
98  		if (mediators == null) mediators = equivalent.mediators;
99  		else if (equivalent.mediators != null)
100 		{
101 			mediators.addAll(equivalent.mediators);
102 		}
103 	}

141 	/**
142 	 * Collects Pubmed or PMC ids
143 	 * @return publication IDs
144 	 */
145 	public List<String> getPublicationIDs(boolean pubmed)
146 	{
147 		if (mediators == null) return Collections.emptyList();
148 
149 		Set<PublicationXref> xrefs = harvestPublicationXrefs(mediators.toArray(new BioPAXElement[mediators.size()]));
150 		Set<String> set = pubmed ? harvestPMIDs(xrefs) : harvestPMCIDs(xrefs);
151 
152 		List<String> list = new ArrayList<String>(set);
153 		Collections.sort(list);
154 
155 		return list;
156 	}

161 	/**
162 	 * Collects publication xrefs of the given elements.
163 	 * @param ele element array
164 	 * @return publication xrefs
165 	 */
166 	private Set<PublicationXref> harvestPublicationXrefs(BioPAXElement... ele)
167 	{
168 		Set<PublicationXref> set = new HashSet<PublicationXref>();
169 
170 		for (Object o : xrefAcc.getValueFromBeans(Arrays.asList(ele)))
171 		{
172 			set.add((PublicationXref) o);
173 		}
174 		for (Object o : evidAcc.getValueFromBeans(Arrays.asList(ele)))
175 		{
176 			set.add((PublicationXref) o);
177 		}
178 		return set;
179 	}

201 	/**
202 	 * Collects things from given xrefs.
203 	 * @param xrefs
204 	 * @return whatever collected
205 	 */
206 	private Set<String> harvestXrefs(Set<PublicationXref> xrefs, String dbID)
207 	{
208 		Set<String> set = new HashSet<String>();
209 
210 		for (PublicationXref xref : xrefs)
211 		{
212 			if (xref.getDb() != null && xref.getDb().equalsIgnoreCase(dbID))
213 				if (xref.getId() != null && !xref.getId().isEmpty())
214 					set.add(xref.getId());
215 		}
216 		return set;
217 	}

356 	/**
357 	 * Collects and sorts sourceID and targetID of the given collection of sif interactions.
358 	 * @param sifInts interactions to consider
359 	 * @param type types of interest, all types accepted if empty
360 	 * @return sorted source and target IDS
361 	 */
362 	public static List<String> getSortedGeneNames(Collection<SIFInteraction> sifInts,
363 		SIFType... type)
364 	{
365 		Set<String> genes = new HashSet<String>();
366 		Set<SIFType> types = new HashSet<SIFType>(Arrays.asList(type));
367 
368 		// collect gene names
369 
370 		for (SIFInteraction sifInt : sifInts)
371 		{
372 			if (!types.isEmpty() && !types.contains(sifInt.type)) continue;
373 
374 			genes.add(sifInt.sourceID);
375 			genes.add(sifInt.targetID);
376 		}
377 
378 		// sort all gene names
379 
380 		List<String> names = new ArrayList<String>(genes);
381 		Collections.sort(names);
382 
383 		return names;
384 	}

386 	/**
387 	 * Converts the given collection of interactions into an adjacency matrix.
388 	 * @param sifInts interactions to consider
389 	 * @param type types of interest, all types accepted if empty
390 	 * @return the sif network as adjacency matrix
391 	 */
392 	public static boolean[][] convertToAdjacencyMatrix(Collection<SIFInteraction> sifInts,
393 		SIFType... type)
394 	{
395 		Set<SIFType> types = new HashSet<SIFType>(Arrays.asList(type));
396 
397 		// collect gene names
398 		List<String> names = getSortedGeneNames(sifInts, type);
399 
400 		// record name indexes for efficient lookup
401 
402 		Map<String, Integer> name2ind = new HashMap<String, Integer>();
403 		int i = 0;
404 		for (String name : names)
405 		{
406 			name2ind.put(name, i++);
407 		}
408 
409 		// generate adjacency matrix
410 
411 		boolean[][] matrix = new boolean[names.size()][names.size()];
412 		for (boolean[] m : matrix) Arrays.fill(m, false);
413 
414 		for (SIFInteraction sifInt : sifInts)
415 		{
416 			if (!types.isEmpty() && !types.contains(sifInt.type)) continue;
417 
418 			matrix[name2ind.get(sifInt.sourceID)][name2ind.get(sifInt.targetID)] = true;
419 
420 			if (!sifInt.type.isDirected())
421 			{
422 				matrix[name2ind.get(sifInt.targetID)][name2ind.get(sifInt.sourceID)] = true;
423 			}
424 		}
425 
426 		return matrix;
427 	}
