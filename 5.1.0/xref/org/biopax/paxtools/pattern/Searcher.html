32  	 * Searches the pattern starting from the given match. The first element of the match should be
33  	 * assigned. Others are optional.
34  	 * @param m match to start from
35  	 * @param pattern pattern to search
36  	 * @return result matches
37  	 */
38  	public static List<Match> search(Match m, Pattern pattern)
39  	{
40  		assert pattern.getStartingClass().isAssignableFrom(m.get(0).getModelInterface());
41  
42  		return searchRecursive(m, pattern.getConstraints(), 0);
43  	}

46  	 * Searches the pattern starting from the given element.
47  	 * @param ele element to start from
48  	 * @param pattern pattern to search
49  	 * @return matching results
50  	 */
51  	public static List<Match> search(BioPAXElement ele, Pattern pattern)
52  	{
53  		assert pattern.getStartingClass().isAssignableFrom(ele.getModelInterface());
54  
55  		Match m = new Match(pattern.size());
56  		m.set(ele, 0);
57  		return search(m, pattern);

60  	/**
61  	 * Continues searching with the mapped constraint at the given index.
62  	 * @param match match to start from
63  	 * @param mc mapped constraints of the pattern
64  	 * @param index index of the current mapped constraint
65  	 * @return matching results
66  	 */
67  	public static List<Match> searchRecursive(Match match, List<MappedConst> mc, int index) 
68  	{
69  		List<Match> result = new ArrayList<Match>();
70  
71  		Constraint con = mc.get(index).getConstr();
72  		int[] ind = mc.get(index).getInds();
73  		int lastInd = ind[ind.length-1];
74  
75  		if (con.canGenerate() && match.get(lastInd) == null)
76  		{
77  			Collection<BioPAXElement> elements = con.generate(match, ind);
78  
79  			for (BioPAXElement ele : elements)
80  			{
81  				match.set(ele, lastInd);
82  				
83  				if (mc.size() == index + 1)
84  				{
85  					result.add((Match) match.clone());
86  				}
87  				else
88  				{
89  					result.addAll(searchRecursive(match, mc, index + 1));
90  				}
91  				
92  				match.set(null, lastInd);
93  			}
94  		}
95  		else
96  		{
97  			if (con.satisfies(match, ind))
98  			{
99  				if (mc.size() == index + 1)
100 				{
101 					result.add((Match) match.clone());
102 				}
103 				else
104 				{
105 					result.addAll(searchRecursive(match, mc, index + 1));
106 				}
107 			}
108 		}
109 		return result;
110 	}

114 	 * list.

116 	 * @param pattern pattern to search for
117 	 * @return matching results

120 	{
121 		List<Match> list = new LinkedList<Match>();
122 
123 		Map<BioPAXElement, List<Match>> map = search(model, pattern);
124 		for (List<Match> matches : map.values())
125 		{

128 		return list;

161 	/**
162 	 * Searches the given pattern in the given model.
163 	 * @param model model to search in
164 	 * @param pattern pattern to search for
165 	 * @param prg progress watcher to keep track of the progress
166 	 * @return map from starting elements to the list matching results
167 	 */
168 	public static Map<BioPAXElement, List<Match>> search(final Model model, final Pattern pattern,
169 														 final ProgressWatcher prg)
170 	{
171 		final Map<BioPAXElement, List<Match>> map = new ConcurrentHashMap<BioPAXElement, List<Match>>();
172 		final ExecutorService exec = Executors.newFixedThreadPool(20);
173 
174 		Set<? extends BioPAXElement> eles = model.getObjects(pattern.getStartingClass());
175 		if (prg != null) prg.setTotalTicks(eles.size());
176 
177 		for (final BioPAXElement ele : eles)
178 		{
179 			exec.execute(new Runnable() {
180 				@Override
181 				public void run() {
182 					List<Match> matches = search(ele, pattern);
183 					if (!matches.isEmpty())
184 					{
185 						map.put(ele, matches);
186 					}
187 					if (prg != null) prg.tick(1);
188 				}
189 			});
190 		}
191 
192 		exec.shutdown();
193 		try {
194 			exec.awaitTermination(10, TimeUnit.MINUTES);
195 		} catch (InterruptedException e) {
196 			throw new RuntimeException("search, failed due to exec timed out.", e);
197 		}
198 
199 		return Collections.unmodifiableMap(map);
200 	}

202 	/**
203 	 * Searches the given pattern starting from the given elements.
204 	 * @param eles elements to start from
205 	 * @param pattern pattern to search for
206 	 * @return map from starting element to the matching results
207 	 */
208 	public static Map<BioPAXElement, List<Match>> search(final Collection<? extends BioPAXElement> eles,
209 														 final Pattern pattern)
210 	{
211 		final Map<BioPAXElement, List<Match>> map = new ConcurrentHashMap<BioPAXElement, List<Match>>();
212 		final ExecutorService exec = Executors.newFixedThreadPool(10);
213 
214 		for (final BioPAXElement ele : eles)
215 		{
216 			if (!pattern.getStartingClass().isAssignableFrom(ele.getModelInterface())) continue;
217 			
218 			exec.execute(new Runnable() {
219 				@Override
220 				public void run() {
221 					List<Match> matches = search(ele, pattern);
222 					if (!matches.isEmpty()) {
223 						map.put(ele, matches);
224 					}
225 				}
226 			});
227 		}
228 
229 		exec.shutdown();
230 		try {
231 			exec.awaitTermination(10, TimeUnit.MINUTES);
232 		} catch (InterruptedException e) {
233 			throw new RuntimeException("search, failed due to exec timed out.", e);
234 		}
235 
236 		return Collections.unmodifiableMap(new HashMap<BioPAXElement, List<Match>>(map));
237 	}

303 	 * Checks if there is any match for the given pattern if search starts from the given element.
304 	 * @param p pattern to search for
305 	 * @param ele element to start from
306 	 * @return true if there is a match
307 	 */
308 	public boolean hasSolution(Pattern p, BioPAXElement ... ele)
309 	{
310 		Match m = new Match(p.size());
311 		for (int i = 0; i < ele.length; i++)
312 		{
313 			m.set(ele[i], i);
314 		}
315 
316 		return !search(m, p).isEmpty();
317 	}

332 	/**
333 	 * Searches a pattern reading the model from the given file, and creates another model that is
334 	 * excised using the matching patterns. Users can limit the max number of starting element, and
335 	 * max number of matches for any starting element. These parameters is good for limiting the
336 	 * size of the result graph.
337 	 * @param p pattern to search for
338 	 * @param inFile filename for the model to search in
339 	 * @param outFile filename for the result model
340 	 * @param seedLimit max number of starting elements
341 	 * @param graphPerSeed max number of matches for a starting element
342 	 * @throws FileNotFoundException when no file exists
343 	 */
344 	public static void searchInFile(Pattern p, String inFile, String outFile, int seedLimit,
345 		int graphPerSeed) throws FileNotFoundException
346 	{
347 		SimpleIOHandler h = new SimpleIOHandler();
348 		Model model = h.convertFromOWL(new FileInputStream(inFile));
349 
350 		Map<BioPAXElement,List<Match>> matchMap = Searcher.search(model, p);
351 
352 		System.out.println("matching groups size = " + matchMap.size());
353 
354 		List<Set<Interaction>> inters = new LinkedList<Set<Interaction>>();
355 		Set<Integer> encountered = new HashSet<Integer>();
356 
357 		Set<BioPAXElement> toExise = new HashSet<BioPAXElement>();
358 
359 		int seedCounter = 0;
360 		for (BioPAXElement ele : matchMap.keySet())
361 		{
362 			if (seedCounter >= seedLimit) break;
363 
364 			int matchCounter = 0;
365 
366 			List<Match> matches = matchMap.get(ele);
367 
368 			if (!matches.isEmpty()) seedCounter++;
369 
370 			for (Match match : matches)
371 			{
372 				matchCounter++;
373 				
374 				if (matchCounter > graphPerSeed) break;
375 
376 				Set<Interaction> ints = getInter(match);
377 
378 				toExise.addAll(Arrays.asList(match.getVariables()));
379 				toExise.addAll(ints);
380 
381 				Integer hash = hashSum(ints);
382 				if (!encountered.contains(hash))
383 				{
384 					encountered.add(hash);
385 					inters.add(ints);
386 				}
387 			}
388 		}
389 
390 		System.out.println("created pathways = " + inters.size());
391 
392 		Model clonedModel = excise(toExise);
393 
394 		int i = 0;
395 		for (Set<Interaction> ints : inters)
396 		{
397 			Pathway pathway = clonedModel.addNew(Pathway.class,
398 				System.currentTimeMillis() + "PaxtoolsPatternGeneratedMatch" + (++i));
399 
400 			pathway.setDisplayName("Match " + getLeadingZeros(i, inters.size()) + i);
401 
402 			for (Interaction anInt : ints)
403 			{
404 				pathway.addPathwayComponent((Process) clonedModel.getByID(anInt.getUri()));
405 			}
406 		}
407 
408 		handler.convertToOWL(clonedModel, new FileOutputStream(outFile));
409 	}

412 	 * Prepares an int for printing with leading zeros for the given size.
413 	 * @param i the int to prepare
414 	 * @param size max value for i
415 	 * @return printable string for i with leading zeros
416 	 */
417 	private static String getLeadingZeros(int i, int size)
418 	{
419 		assert i <= size;
420 		int w1 = (int) Math.floor(Math.log10(size));
421 		int w2 = (int) Math.floor(Math.log10(i));
422 		
423 		String s = "";
424 
425 		for (int j = w2; j < w1; j++)
426 		{
427 			s += "0";
428 		}
429 		return s;
430 	}

455 	 * Excises a model to the given elements.
456 	 * @param result elements to excise to
457 	 * @return excised model
458 	 */
459 	private static Model excise(Set<BioPAXElement> result)
460 	{
461 		Completer c = new Completer(EM);
462 		result = c.complete(result);
463 		Cloner cln = new Cloner(EM, BioPAXLevel.L3.getDefaultFactory());
464 		return cln.clone(result);
465 	}

467 	/**
468 	 * Gets all interactions in a match.
469 	 * @param match match to search
470 	 * @return all interaction in the match
471 	 */
472 	private static Set<Interaction> getInter(Match match)
473 	{
474 		Set<Interaction> set = new HashSet<Interaction>();
475 		for (BioPAXElement ele : match.getVariables())
476 		{
477 			if (ele instanceof Interaction) 
478 			{
479 				set.add((Interaction) ele);
480 				addControlsRecursive((Interaction) ele, set);
481 			}
482 		}
483 		return set;
484 	}
485 
486 	/**
487 	 * Adds controls of the given interactions recursively to the given set.
488 	 * @param inter interaction to add its controls
489 	 * @param set set to add to
490 	 */
491 	private static void addControlsRecursive(Interaction inter, Set<Interaction> set)
492 	{
493 		for (Control ctrl : inter.getControlledOf())
494 		{
495 			set.add(ctrl);
496 			addControlsRecursive(ctrl, set);
497 		}
498 	}

500 	/**
501 	 * Creates a hash code for a set of interactions.
502 	 * @param set interactions
503 	 * @return sum of hashes
504 	 */
505 	private static Integer hashSum(Set<Interaction> set)
506 	{
507 		int x = 0;
508 		for (Interaction inter : set)
509 		{
510 			x += inter.hashCode();
511 		}
512 		return x;
513 	}
