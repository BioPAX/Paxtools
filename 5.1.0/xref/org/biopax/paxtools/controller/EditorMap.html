4   import org.biopax.paxtools.model.BioPAXLevel;

6   import java.util.Iterator;

9   /**
10   * This class contains methods that eases to use editors for
11   * specific or a set of property. Using the methods of this class
12   * editors of a class, editors of a property, and editors of a property
13   * with the given domain can be obtained; and these editors/this editor
14   * can be used to modify object's properties.
15   *
16   * The functionallity of this class plays key roles on several other classes'
17   * functionallity; e.g. {@link org.biopax.paxtools.io}, {@link org.biopax.paxtools.model},
18  
19   */
20  public interface EditorMap
21  {
22  
23  // -------------------------- OTHER METHODS --------------------------
24  
25      /**
26       * This method returns the <em>editor</em> intended to handle
27       * property named <em>property</em> of a class (<em>javaClass</em>).
28       * This editor can then be used to modify the property of
29       * an element of class <em>javaClass</em>.
30       *
31       * To put in other words, this methods returns the editor of which
32       * domain includes <em>javaClass</em>, and the editor that can handle
33       * the <em>property</em>.
34       *
35       * @param property name of the property for which editor will be called
36       * @param javaClass class of the element
37       * @param <D> domain
38       * @return null if there is no such editor
39       */
40  	<D extends BioPAXElement> PropertyEditor<? super D,?> getEditorForProperty(String property, Class<D> javaClass);
41  
42      /**
43       * This method returns the set of <em>editor</em>s intended to handle
44       * property named <em>property</em>. This editor can then be used
45       * to modify the property of an element which is in editor's domain list.
46       *
47       * In other words, this methods returns the set of the editors
48       * that can handle the <em>property</em>. Editors are not filtered for
49       * a specific domain class.
50       *
51       * @param property name of the property for which editor will be called
52       * @return empty set if there are no such editors
53       */
54      Set<PropertyEditor> getEditorsForProperty(String property);
55  
56  	/**
57  	  * This method returns the set of <em>editor</em>s intended to handle
58  	  * property named <em>property</em>. This editor can then be used
59  	  * to modify the property of an element which is in editor's domain list.
60  	  *
61  	  * In other words, this methods returns the set of the editors
62  	  * that can handle the <em>property</em>. Editors are not filtered for
63  	  * a specific domain class.
64  	  *
65  	  * @param property name of the property for which editor will be called
66  	  * @param domain biopax type/class the property belongs to
67  	  * @param <D> domain biopax type
68  	  * @return empty set if there are no such editors
69  	  */
70  
71  	public <D extends BioPAXElement> Set<PropertyEditor<? extends D, ?>> getSubclassEditorsForProperty(
72  			String property, Class<D> domain);
73  
74  	/**
75       * This method returns the set of <em>editor</em>s whose domain
76       * subsumes the class of given BioPAX element.
77       *
78  	 * @param bpe BioPAX element for which the available editors will be returned
79  	 * @return empty set if there are no such editors
80       */
81      Set<PropertyEditor> getEditorsOf(BioPAXElement bpe);
82  
83  	/**
84  	 * Properties in BioPAX specification is unidirectional. e.g. entityReference property that links a
85  	 * PhysicalEntity to EntityReference has no defined corresponding inverse property that links
86  	 * EntityReferences to their corresponding entities.
87  	 *
88  	 * Most OWL reasoners can query the inverse of a property at no additional cost, but for most OO implementations
89  	 * this would require an expensive O(n) lookup.
90  	 * An OO implementation requires keeping additional properties for efficiency purposes.
91  	 * Inverse editors are read-only editors that captures these "inverse" part of the bidirectional properties
92  	 * specifically implemented in Paxtools. They have the pattern <em>PropertyName</em>Of  e.g. entityReferenceOf.
93  	 * @param bpe  BioPAX element for which the available inverse editors will be returned.
94  	 * @return all inverse editors  for this entity's class type.
95  	 */
96  	Set<ObjectPropertyEditor> getInverseEditorsOf(BioPAXElement bpe);
97  
98  
99      /**
100      * Returns a set of sub classes of a given class. This method
101      * can be used for class filtering methods.
102      *
103      * @param javaClass the class whose subclasses will be returned
104      * @param <E> biopax type (biopax object model interface)
105      * @return an empty set if there are no such editors
106      */
107     <E extends BioPAXElement> Set<? extends Class<E>> getKnownSubClassesOf(Class<E> javaClass);
108 
109 
110     /**
111      * Returns the BioPAX level for which editor map is created. Different
112      * BioPAX levels have different editor maps.
113      *
114      * @return BioPAX Level
115      */
116     BioPAXLevel getLevel();
117 
118 	/**
119 	 * This method returns the set of <em>editor</em>s whose domain
120 	 * subsumes the given class
121 	 *
122 	 * @param domain BioPAX model interface for which the available editors will be returned
123 	 * @return empty set if there are no such editors
124 	 */
125 	Set<PropertyEditor> getEditorsOf(Class<? extends BioPAXElement> domain);
126 
127 	/**
128 	 * Properties in BioPAX specification is unidirectional. e.g. entityReference property that links a
129 	 * PhysicalEntity to EntityReference has no defined corresponding inverse property that links
130 	 * EntityReferences to their corresponding entities.
131 	 *
132 	 * Most OWL reasoners can query the inverse of a property at no additional cost, but for most OO implementations
133 	 * this would require an expensive O(n) lookup.
134 	 * An OO implementation requires keeping additional properties for efficiency purposes.
135 	 * Inverse editors are read-only editors that captures these "inverse" part of the bidirectional properties
136 	 * specifically implemented in Paxtools. They have the pattern <em>PropertyName</em>Of  e.g. entityReferenceOf.
137 	 * @param domain of the inverse property
138 	 * @return all inverse editors  for this class type.
139 	 */
140 	Set<ObjectPropertyEditor> getInverseEditorsOf(Class<? extends BioPAXElement> domain);
141 
142 	/**
143 	 * @return An iterator over all the properties in this EditorMap
144 	 */
145 	public Iterator<PropertyEditor> iterator();
146 
147 }
