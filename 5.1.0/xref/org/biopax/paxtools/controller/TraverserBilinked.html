3   import org.biopax.paxtools.model.BioPAXElement;
4   import org.biopax.paxtools.model.Model;
5   import org.biopax.paxtools.util.Filter;
6   import org.slf4j.Logger;
7   import org.slf4j.LoggerFactory;
8   
9   import java.util.HashSet;
10  import java.util.Set;
11  
12  /**
13   * A bi-directional BioPAX properties traverser.
14   * 
15   * To traverse both biopax element's standard properties, such as 'xref',  
16   * and (Paxtools') inverse properties, such as 'xrefOf', etc.
17   * 
18   * @author Ozgun Babur 
19   */
20  public class TraverserBilinked extends Traverser
21  {
22  	private boolean isInverseOnly = false;
23  	private final static Logger log = LoggerFactory.getLogger(TraverserBilinked.class);
24  	
25  	/**
26  	 * Constructor.
27  	 * 
28  	 * @param editorMap biopax property editors map
29  	 * @param visitor user's implementation; if it recursively 
30  	 * 		calls {@link #traverse(BioPAXElement, Model)} method, then care must be taken to prevent infinite loops.
31  	 * @param filters bidirectional property filters
32  	 */
33  	public TraverserBilinked(EditorMap editorMap, Visitor visitor, PropertyFilterBilinked... filters)
34  	{
35  		super(editorMap, visitor, filters);
36  	}
37  
38  	public boolean isInverseOnly() {
39  		return isInverseOnly;
40  	}
41  	public void setInverseOnly(boolean isInverseOnly) {
42  		this.isInverseOnly = isInverseOnly;
43  	}
44  	
45  	@Override
46  	public void traverse(BioPAXElement element, Model model)
47  	{
48  		if(!isInverseOnly)
49  			super.traverse(element, model);
50  
51  		// Now traverse the inverse links
52  
53  		Set<ObjectPropertyEditor> editors = editorMap.getInverseEditorsOf(element);
54  
55  		if(editors == null)
56  		{
57  			log.warn("No editors for : " + element.getModelInterface());
58  			return;
59  		}
60  		for (ObjectPropertyEditor editor : editors)
61  		{
62  			if (filterInverse(editor))
63  			{
64  					Set<BioPAXElement> valueSet = new HashSet(editor.getInverseAccessor().getValueFromBean(element));
65  					if (!valueSet.isEmpty()) for (BioPAXElement value : valueSet)
66  					{
67  						if(value != null) {
68  							//TODO how visitor knows whether it's called from inverse or normal property (e.g., to modify a value)?
69  							visitor.visit(element, value, model, editor); 
70  						}
71  					}
72  			}
73  		}
74  	}
75  
76  	protected boolean filterInverse(PropertyEditor editor)
77  	{
78  		for (Filter<PropertyEditor> filter : filters)
79  		{
80  			if (!((PropertyFilterBilinked) filter).filterInverse(editor))
81  			{
82  				return false;
83  			}
84  		}
85  		return true;
86  	}
87  
88  }
