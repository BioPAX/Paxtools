3   import org.biopax.paxtools.model.BioPAXElement;
4   import org.biopax.paxtools.model.level3.CovalentBindingFeature;
5   import org.biopax.paxtools.model.level3.SequenceModificationVocabulary;
6   
7   
8   public class CovalentBindingFeatureImpl extends BindingFeatureImpl implements CovalentBindingFeature
9   {
10  	private SequenceModificationVocabulary modificationType;
11  
12  	public CovalentBindingFeatureImpl() {
13  	}
14  	
15  	@Override
16  	public Class<? extends CovalentBindingFeature> getModelInterface()
17  	{
18  		return CovalentBindingFeature.class;
19  	}
20  
21  	public SequenceModificationVocabulary getModificationType()
22  	{
23  		return this.modificationType;
24  	}
25  
26  	public void setModificationType(SequenceModificationVocabulary modificationType)
27  	{
28  		this.modificationType = modificationType;
29  	}
30  
31  	@Override
32  	protected boolean semanticallyEquivalent(BioPAXElement element)
33  	{
34  		if(!(element instanceof CovalentBindingFeature))
35  			return false;
36  		
37  		CovalentBindingFeature that = (CovalentBindingFeature) element;
38  		SequenceModificationVocabulary type = that.getModificationType();
39  		boolean value = ((type == null) ?
40  		                this.getModificationType() == null :
41  		                this.getModificationType().equals(type));
42  		return value && super.semanticallyEquivalent(element);
43  	}
44  
45  	@Override
46  	public int equivalenceCode()
47  	{
48  		return super.equivalenceCode() +
49  		       29 * (modificationType == null ? 0 : modificationType.equivalenceCode());
50  	}
51  }
