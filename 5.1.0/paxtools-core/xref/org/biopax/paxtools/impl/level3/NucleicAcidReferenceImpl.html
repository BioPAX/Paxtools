7   import java.util.Set;
8   
9   public abstract class NucleicAcidReferenceImpl extends SequenceEntityReferenceImpl implements NucleicAcidReference
10  {
11  	private Set<NucleicAcidRegionReference> subRegion;
12  
13  	public NucleicAcidReferenceImpl()
14  	{
15  		this.subRegion = BPCollections.I.createSafeSet();
16  	}
17  
18  	public Set<NucleicAcidRegionReference> getSubRegion()
19  	{
20  		return subRegion;
21  	}
22  
23  	public void addSubRegion(NucleicAcidRegionReference regionReference)
24  	{
25  		if (regionReference != null)
26  		{
27  			subRegion.add(regionReference);
28  			regionReference.getSubRegionOf().add(this);
29  		}
30  	}
31  
32  	public void removeSubRegion(NucleicAcidRegionReference regionReference)
33  	{
34  		if (regionReference != null)
35  		{
36  			subRegion.remove(regionReference);
37  		}
38  	}
39  }
