21   * Upgrades BioPAX L1 and L2 to Level 3.
22   * 
23   * Notes:
24   * - it does not fix existing BioPAX errors
25   * - most but not all things are converted (e.g., complex.ORGANISM property cannot...)
26   * - phy. entities "clones" - because, during L1 or L2 data read, all re-used pEPs are duplicated... (TODO filter after the conversion)

100 	/**
101 	 * Converts a BioPAX Model, Level 1 or 2, to the Level 3.
102 	 *
103 	 * @param model BioPAX model to upgrade
104 	 * @return new Level3 model
105 	 */
106 	public Model filter(Model model) {
107 		if(model == null || model.getLevel() != BioPAXLevel.L2)
108 			return model; // nothing to do
109 
110 		preparePep2PEIDMap(model);
111 
112 		final Model newModel = factory.createModel();
113 		newModel.getNameSpacePrefixMap().putAll(model.getNameSpacePrefixMap());
114 		newModel.setXmlBase(model.getXmlBase());
115 		
116 		// facilitate the conversion
117 		normalize(model);
118 		
119 		// First, map classes (and only set ID) if possible (pre-processing)
120 		for(BioPAXElement bpe : model.getObjects()) {
121 			Level3Element l3element = mapClass(bpe);
122 			if(l3element != null) {
123 				newModel.add(l3element);
124 			} else {
125 				log.debug("Skipping " + bpe + " " + bpe.getModelInterface().getSimpleName());
126 			}
127 		}
128 		
129 		/* process each L2 element (mapping properties), 
130 		 * except for pEPs and oCVs that must be processed as values
131 		 * (anyway, we do not want dangling elements)
132 		 */
133 		for(BioPAXElement e : model.getObjects()) {
134 			if(e instanceof physicalEntityParticipant
135 				|| e instanceof openControlledVocabulary) {
136 				continue;
137 			}
138 			
139 			// map properties
140 			traverse((Level2Element) e, newModel);
141 		}
142 
143 		log.info("Done: new L3 model contains " + newModel.getObjects().size() + " BioPAX individuals.");
144 		
145 		// fix new model (e.g., add PathwayStep's processes to the pathway components ;))
146 		normalize(newModel);
147 		
148 		return newModel;
149 	}

152 	/*
153 	 * Fixes several problems in the BioPAX model
154 	 */
155 	private void normalize(Model model) {
156 		/* If a pE participates via pEP, no problem.
157 		 * Otherwise, - there is a special case. 
158 		 * 
159 		 * physicalEntity PARTICIPANT takes part in interactions
160 		 * either directly or via physicalEntityParticipant (pEP). 
161 		 * With pEP, the corresponding L3 PE and ER have been 
162 		 * already created (above). However, although (if no pEPs used) 
163 		 * we got Complex and basic PhysicalEntity objects created in L3,
164 		 * and others (protein, dna, etc.) became ER, we still have to 
165 		 * create another PEs for all of them and set their ER property (where exists).
166 		 */
167 		for (interaction itr : model.getObjects(interaction.class)) {
168 			if(itr instanceof conversion || itr instanceof control) 
169 				continue; // cannot have pE participants anyway (only pEPs)
170 			for (InteractionParticipant ip : itr.getPARTICIPANTS()) {
171 				if (ip instanceof physicalEntity) { 
172 					physicalEntity pe = (physicalEntity) ip;
173 					// create a new pEP
174 					String newId = itr.getUri() + "_"	+ getLocalId(pe);
175 					physicalEntityParticipant pep = model.addNew(physicalEntityParticipant.class, newId);
176 					pep.setPHYSICAL_ENTITY(pe);
177 					// no other properties though
178 					// reset participant
179 					itr.removePARTICIPANTS(pe);
180 					itr.addPARTICIPANTS(pep);
181 				}
182 			}
183 		}
184 		
185 		// ...adding step processes to the pathway components (L3) is NOT always required
186 		
187 		// TODO remove cloned UtilityClass elements
188 		
189 		// set 'name' for simple physical entities (because pEPs didn't have names)
190 		for(EntityReference er : model.getObjects(EntityReference.class)) {
191 			for(SimplePhysicalEntity spe : er.getEntityReferenceOf()) {
192 				// after the conversion, it's always empty.., but let's double-check
193 				if(spe.getName().isEmpty()) {
194 					spe.getName().addAll(er.getName());
195 				}
196 				
197 				if(spe.getDisplayName() == null || spe.getDisplayName().trim().length() == 0) {
198 					spe.setDisplayName(er.getDisplayName());
199 				}
200 			}
201 		}
202 		
203 	}

206 	// creates L3 classes and sets IDs
207 	private Level3Element mapClass(BioPAXElement bpe) {
208 		Level3Element newElement = null;
209 		
210 		if(bpe instanceof physicalEntityParticipant)
211 		{
212 			String id = pep2PE.get(bpe.getUri());
213 
214 			if(id == null) {
215 				log.warn("No mapping possible for " + bpe.getUri());
216 				return null;
217 			}
218 			else if (id.equals(bpe.getUri()))
219 			{
220 				// create a new simplePhysicalEntity
221 				//(excluding Complex and basic PhysicalEntity that map directly and have no ERs)
222 				newElement = createSimplePhysicalEntity((physicalEntityParticipant)bpe);
223 			}
224 		}
225 		else if(!(bpe instanceof openControlledVocabulary)) // skip oCVs
226 		{
227 			// using classesmap.properties to map other types
228 			String type = bpe.getModelInterface().getSimpleName();	
229 			String newType = classesmap.getProperty(type).trim();
230 			if (newType != null && factory.canInstantiate(factory.getLevel().getInterfaceForName(newType)))
231             {
232 				newElement = (Level3Element) factory.create(newType, bpe.getUri());
233 			} else {
234 				if(log.isDebugEnabled())
235 					log.debug("No mapping found for " + type);
236 				return null;
237 			}
238 		}
239 
240 		return newElement;
241 	}

246 	 *
247 	 * When pEP's PHYSICAL_ENTITY is either complex
248 	 * or basic physicalEntity, null will be the result.
249 	 */

251 		physicalEntity pe2 = pep.getPHYSICAL_ENTITY();
252 		return createSimplePhysicalEntity(pe2, pep.getUri());
253 	}

255 	private SimplePhysicalEntity createSimplePhysicalEntity(physicalEntity pe2, String id) {
256 		SimplePhysicalEntity e = null;
257 		if(pe2 instanceof protein) {
258 			e = factory.create(Protein.class, id);
259 		} else if(pe2 instanceof dna) {
260 			e = factory.create(DnaRegion.class, id);
261 		} else if (pe2 instanceof rna) {
262 			e = factory.create(RnaRegion.class, id);
263 		} else if (pe2 instanceof smallMolecule) {
264 			e = factory.create(SmallMolecule.class, id);
265 		}
266 		return e;
267 	}

274 	private ControlledVocabulary convertAndAddVocabulary(openControlledVocabulary value,
275 			Level2Element parent, Model newModel, PropertyEditor newEditor)
276 	{
277 		String id = ((BioPAXElement) value).getUri();
278 
279 		if (!newModel.containsID(id)) {
280 			if (newEditor != null) {
281 				newModel.addNew(newEditor.getRange(), id);
282 				// copy properties
283 				traverse(value, newModel);
284 			} else {
285 				log.warn("Cannot Convert CV: " + value + " (for prop.: " + newEditor + ")");
286 			}
287 		} 
288 		
289 		return (ControlledVocabulary) newModel.getByID(id);
290 	}

292 	// parent class's abstract method implementation
293 	protected void visit(Object value, BioPAXElement parent,
294 			Model newModel, PropertyEditor editor)
295 	{
296 			if(editor != null && editor.isUnknown(value)) {
297 				return;
298 			}
299 
300 			String parentType = parent.getModelInterface().getSimpleName();
301 			BioPAXElement newParent = null;
302 			Object newValue = value;
303 			String newProp = propsmap.getProperty(editor.getProperty());
304 
305 			// special case (PATHWAY-COMPONENTS maps to pathwayComponent or pathwayOrder)
306 			if(parent instanceof pathway && value instanceof pathwayStep
307 					&& editor.getProperty().equals("PATHWAY-COMPONENTS")) {
308 				newProp = "pathwayOrder";
309 			}
310 
311 			// for pEPs, getting the corresponding simple PE or Complex is different
312 			if(parent instanceof physicalEntityParticipant) {
313 				newParent = getMappedPE((physicalEntityParticipant) parent, newModel);
314 			} else {
315 				newParent = newModel.getByID(parent.getUri());
316 			}
317 
318 			// bug check!
319 			if(newParent == null) {
320 				throw new IllegalAccessError("Of " + value +
321 					", parent " + parentType + " : " + parent +
322 					" is not yet in the new model: ");
323 			}
324 
325 			PropertyEditor newEditor =
326 				SimpleEditorMap.L3.getEditorForProperty(newProp, newParent.getModelInterface());
327 
328 			if(value instanceof Level2Element)
329 			// not a String, Enum, or primitive type
330 			{
331 				// when pEP, create/add stoichiometry!
332 				if(value instanceof physicalEntityParticipant)
333 				{
334 					physicalEntityParticipant pep = (physicalEntityParticipant) value;
335 					newValue = getMappedPE(pep, newModel);
336 
337 					float coeff = (float) pep.getSTOICHIOMETRIC_COEFFICIENT();
338 					if (coeff > 1 ) { //!= BioPAXElement.UNKNOWN_DOUBLE) {
339 					  if(parent instanceof conversion || parent instanceof complex) {
340 						PhysicalEntity pe3 = (PhysicalEntity) newValue;
341 						Stoichiometry stoichiometry = factory
342 							.create(Stoichiometry.class,
343                                     pe3.getUri() + "-stoichiometry" + Math.random());
344 						stoichiometry.setStoichiometricCoefficient(coeff);
345 						stoichiometry.setPhysicalEntity(pe3);
346 						//System.out.println("parent=" + parent + "; phy.ent.=" + pep + "; coeff=" + coeff);
347 						if (parent instanceof conversion) {
348 							// (pep) value participates in the conversion interaction
349 							Conversion conv = (Conversion) newModel
350 								.getByID(parent.getUri());
351 							conv.addParticipantStoichiometry(stoichiometry);	
352 						} else {
353 							// this (pep) value is component of the complex
354 							Complex cplx = (Complex) newModel.getByID(parent.getUri());
355 							cplx.addComponentStoichiometry(stoichiometry);
356 						} 
357 						
358 						newModel.add(stoichiometry);
359 						
360 					  } else {
361 						if (log.isDebugEnabled())
362 							log.debug(pep + " STOICHIOMETRIC_COEFFICIENT is "
363 								+ coeff	+ ", but the pEP's parent is not a conversion or complex - " + parent);
364 					  }
365 					}
366 					
367 					traverse(pep, newModel);
368 				}
369 				else if(value instanceof openControlledVocabulary) 
370 				{
371 					// create the proper type ControlledVocabulary instance	
372 					newValue = convertAndAddVocabulary((openControlledVocabulary)value, 
373 						(Level2Element)parent, newModel, newEditor);
374 				}
375 				else
376 				{
377 					String id = ((Level2Element) value).getUri();
378 					newValue = newModel.getByID(id);
379 				}
380 			}
381 			else if (value.getClass().isEnum())
382 			{
383 				newValue = getMatchingEnum(value);
384 			}
385 			else
386 			{
387 				// relationshipXref.RELATIONSHIP-TYPE range changed (String -> RelationshipTypeVocabulaty)
388 				if(parent instanceof relationshipXref && editor.getProperty().equals("RELATIONSHIP-TYPE")) {
389 					String id = URLEncoder.encode(value.toString());
390 					if(!newModel.containsID(id)) {
391 						RelationshipTypeVocabulary cv = (RelationshipTypeVocabulary) 
392 							newModel.addNew(newEditor.getRange(), id);
393 						cv.addTerm(value.toString().toLowerCase());
394 						newValue = cv;
395 					} else {
396 						newValue = newModel.getByID(id);
397 					}
398 				}
399 			}
400 			
401 			if(newValue == null) {
402 				log.debug("Skipped for  " + parent + "." + editor.getProperty()
403 					+ "=" + value + " ==> " + newParent.getUri() + "." + newProp	+ " = NULL");
404 				return;
405 			}
406 			
407 			if (newProp != null) {
408 				if (newEditor != null){
409 					setNewProperty(newParent, newValue, newEditor);
410 				} else if(parent instanceof physicalEntity) {
411 					// Special mapping for 'AVAILABILITY' and 'DATA-SOURCE'
412 					// find parent pEP(s)
413 					Set<physicalEntityParticipant> ppeps = ((physicalEntity)parent).isPHYSICAL_ENTITYof();
414 					// if several pEPs use the same phy.entity, we get this property/value cloned...
415 					for(physicalEntityParticipant pep: ppeps) {
416 						//find proper L3 physical entity
417 						newParent = getMappedPE(pep, newModel);
418 						if(newParent != null) {
419 							newEditor = SimpleEditorMap.L3.getEditorForProperty(newProp, newParent.getModelInterface());
420 							setNewProperty(newParent, newValue, newEditor);
421 						} else { // bug!
422 							log.error("Cannot find converted PE to map the property " 
423 								+ editor.getProperty() + " of physicalEntity " + parent + " (" + parentType + ")");
424 						}
425 					}
426 				} else {
427 					log.debug("Skipped property " + editor.getProperty() + " in " + parentType + " to " +
428 						newParent.getModelInterface().getSimpleName() + " conversion (" + parent + ")");
429 				}
430 			} else {
431 				log.warn("No mapping defined for property: " + parentType + "." + editor.getProperty());
432 			} 
433 		}

445 	/*
446 	 * pEP->PE; pE->ER class mapping was done for "simple" entities;
447 	 * for complex and the "basic" pE, it is pE->PE mapping
448 	 * (although pEPs were skipped, their properties are to save anyway)
449 	 */
450 	private PhysicalEntity getMappedPE(physicalEntityParticipant pep, Model newModel)
451 	{
452 		String id = pep2PE.get(pep.getUri());
453 		physicalEntity pe2er = pep.getPHYSICAL_ENTITY();
454 		
455 		if(id == null || pe2er == null)
456 			throw new IllegalAccessError("Illegal pEP (cannot convert): " 
457 				+ pep.getUri());
458 		
459 		BioPAXElement pe = newModel.getByID(id);
460 		String inf = "pEP " + pep + " that contains " 
461 			+ pe2er.getModelInterface().getSimpleName();
462 		if(!isSimplePhysicalEntity(pe2er)) {
463 			if(pe == null) {
464 				if(log.isDebugEnabled())
465 					log.debug(inf + " gets new ID: " + pe2er.getUri());
466 				pe = newModel.getByID(pe2er.getUri());
467 			} else { // error: complex and basic p.entity's pEP 
468 				throw new IllegalAccessError("Illegal conversion of pEP: " + inf); 
469 			}
470 		} else if(pe == null){
471 			// for a pEP having a simple PE, a new PE should be created
472 			throw new IllegalAccessError("No PE for: " + inf
473 					+ " found in the new Model");
474 		}
475 		
476 		return (PhysicalEntity) pe;
477 	}

483 				|| pe2 instanceof smallMolecule);

487      * Gets the local part of the BioPAX element ID.
488      * 
489      * @param bpe BioPAX object
490      * @return id - local part of the URI
491      */
492    	static public String getLocalId(BioPAXElement bpe) {
493 		String id = bpe.getUri();
494 		return (id != null) ? id.replaceFirst("^.+[#/]", "") : null; //greedy pattern matches everything up to the last '/' or '#'

497 	protected Object getMatchingEnum(Object o)
498 	{
499 		assert o.getClass().isEnum();
500 
501 		if (enumMap.containsKey(o))
502 			return enumMap.get(o);
503 
504 		if (!propsmap.containsKey(o.toString()))
505 		{
506 			enumMap.put(o, null);
507 			return null;
508 		}
509 
510 		String l2Name = o.getClass().getName();
511 		l2Name = l2Name.substring(l2Name.lastIndexOf(".") + 1);
512 
513 		if (!classesmap.containsKey(l2Name))
514 		{
515 			log.error("There is no class mapping for enum " + o.getClass() + " in classesmap");
516 			return null;
517 		}
518 
519 		String l3Name = classesmap.getProperty(l2Name);
520 
521 		assert l3Name != null;
522 
523 		String l3value = propsmap.getProperty(o.toString());
524 
525 		assert l3value != null;
526 
527 		Class<?> cls = null;
528 		try
529 		{
530 			cls = Class.forName(l3PackageName + l3Name);
531 		}
532 		catch (ClassNotFoundException e)
533 		{
534 			log.error("Cannot find class " + l3PackageName + l3Name);
535 			//e.printStackTrace();
536 		}
537 
538 		assert cls != null;
539 
540 		Method meth = null;
541 		try
542 		{
543 			meth = cls.getMethod("valueOf", String.class);
544 		}
545 		catch (NoSuchMethodException e)
546 		{
547 			log.error("No valueOf method here. Is this possible ?!");
548 			e.printStackTrace();
549 		}
550 
551 		assert meth != null;
552 
553 		Object l3enum = null;
554 		try
555 		{
556 			l3enum = meth.invoke(null, l3value);
557 		}
558 		catch (IllegalAccessException e)
559 		{
560 			log.error("Cannot invoke method " 
561 				+ meth + " - " + e);
562 			//e.printStackTrace();
563 		}
564 		catch (InvocationTargetException e)
565 		{
566 			log.error("Cannot invoke method " + meth 
567 				+ " - " + e);
568 			//e.printStackTrace();
569 		}
570 
571 		enumMap.put(o, l3enum);
572 		return l3enum;
573 	}

575 	private List<Set<physicalEntityParticipant>> getPepsGrouped(physicalEntity pe)
576 	{
577 		List<Set<physicalEntityParticipant>> list = new ArrayList<Set<physicalEntityParticipant>>();
578 
579 		for (physicalEntityParticipant pep : pe.isPHYSICAL_ENTITYof())
580 		{
581 			boolean added = false;
582 
583 			for (Set<physicalEntityParticipant> group : list)
584 			{
585 				physicalEntityParticipant first = group.iterator().next();
586 
587 				if (first.isInEquivalentState(pep))
588 				{
589 					group.add(pep);
590 					added = true;
591 					break;
592 				}
593 			}
594 
595 			if (!added)
596 			{
597 				Set<physicalEntityParticipant> group = new HashSet<physicalEntityParticipant>();
598 				group.add(pep);
599 				list.add(group);
600 			}
601 		}
602 		return list;
603 	}

605 	private Map<String, String> getPep2StateIDMapping(physicalEntity pe)
606 	{
607 		List<Set<physicalEntityParticipant>> sets = getPepsGrouped(pe);
608 
609 		Map<String, String> map = new HashMap<String, String>();
610 
611 		for (Set<physicalEntityParticipant> set : sets)
612 		{
613 			physicalEntityParticipant first = set.iterator().next();
614 
615 			for (physicalEntityParticipant pep : set)
616 			{
617 				map.put(pep.getUri(), first.getUri());
618 			}
619 		}
620 		return map;
621 	}

623 	protected void preparePep2PEIDMap(Model model)
624 	{
625 		assert model.getLevel() == BioPAXLevel.L2;
626 
627 		pep2PE = new HashMap<String, String>();
628
