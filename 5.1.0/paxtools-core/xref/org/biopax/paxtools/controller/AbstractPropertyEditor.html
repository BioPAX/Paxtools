14  /**
15   * This is the base class for all property editors. Each property controller is responsible for
16   * manipulating a certain property for a given class of objects (domain).
17   */
18  public abstract class AbstractPropertyEditor<D extends BioPAXElement, R>
19  		extends SimplePropertyAccessor<D,R> implements PropertyEditor<D,R>
20  {

79  			boolean multipleCardinality)
80  	{
81  		super(domain, range, multipleCardinality, getMethod);
82  		this.property = property;

84  		try
85  		{
86  			detectMethods();

89  		{
90  			log.error("Failed at reflection, no method: " + e.getMessage());
91  		}
92  	}

97  	@Override
98  	public String toString()
99  	{
100 		String def = String.format("%s %s %s", domain.getSimpleName(), property, range.getSimpleName());
101 
102 		for (Class aClass : maxCardinalities.keySet())
103 		{
104 			Integer cardinality = maxCardinalities.get(aClass);
105 			def += " C:" + aClass.getSimpleName() + ":" + cardinality;
106 		}
107 
108 		return def;
109 	}

111 	/**
112 	 * This method creates a property reflecting on the domain and property. Proper subclass is chosen
113 	 * based on the range of the property.
114 	 * @param domain paxtools level2 interface that maps to the corresponding owl level2.
115 	 * @param property to be managed by the constructed controller.
116 	 * @param <D> domain
117 	 * @param <R> range
118 	 * @return a property controller to manipulate the beans for the given property.
119 	 */
120 	public static <D extends BioPAXElement, R> PropertyEditor<D, R> createPropertyEditor(Class<D> domain,
121 	                                                                                     String property)
122 	{
123 		PropertyEditor editor = null;
124 		try
125 		{
126 			Method getMethod = detectGetMethod(domain, property);
127 			boolean multipleCardinality = isMultipleCardinality(getMethod);
128 			Class<R> range = detectRange(getMethod);
129 
130 			if (range.isPrimitive() || range.equals(Boolean.class))
131 			{
132 				editor = new PrimitivePropertyEditor<D, R>(property, getMethod, domain, range, multipleCardinality);
133 			} else if (range.isEnum())
134 			{
135 				editor = new EnumeratedPropertyEditor(property, getMethod, domain, range, multipleCardinality);
136 			} else if (range.equals(String.class))
137 			{
138 				editor = new StringPropertyEditor(property, getMethod, domain, multipleCardinality);
139 			} else
140 			{
141 				editor = new ObjectPropertyEditor(property, getMethod, domain, range, multipleCardinality);
142 			}
143 		}
144 		catch (NoSuchMethodException e)
145 		{
146 			if (log.isWarnEnabled()) log.warn("Failed creating the controller for " + property + " on " + domain);
147 		}
148 		return editor;
149 	}

151 	private static Method detectGetMethod(Class beanClass, String property) throws NoSuchMethodException
152 	{
153 		String javaMethodName = getJavaName(property);
154 		//This is the name we are going to try, log it down
155 		if (log.isTraceEnabled())
156 		{
157 			log.trace("javaMethodName = get" + javaMethodName);
158 		}
159 
160 		//extract the get method
161 		return beanClass.getMethod("get" + javaMethodName);
162 	}

164 	/**
165 	 * Given the name of a property's name as indicated in the OWL file, this method converts the name
166 	 * to a Java compatible name.
167 	 * @param owlName the property name as a string
168 	 * @return the Java compatible name of the property
169 	 */
170 	private static String getJavaName(String owlName)
171 	{
172 		// Since java does not allow '-' replace them all with '_'
173 		String s = owlName.replaceAll("-", "_");
174 		s = s.substring(0, 1).toUpperCase() + s.substring(1);
175 		return s;
176 	}

178 	/**
179 	 * Given the multiple cardinality feature, the range of the get method is returned.
180 	 * @param getMethod default method
181 
182 	 * @return the range as a class
183 	 */
184 	protected static Class detectRange(Method getMethod)
185 	{
186 		Class range = getMethod.getReturnType();
187 		//if the return type is a collection then we have multiple cardinality
188 		if (Collection.class.isAssignableFrom(range))
189 		{
190 			//it is a collection, by default assume non parameterized.
191 			range = Object.class;
192 			//If the collection is parameterized, get it.
193 			Type genericReturnType = getMethod.getGenericReturnType();
194 			if (genericReturnType instanceof ParameterizedType)
195 			{
196 				try
197 				{
198 					range = (Class) ((ParameterizedType) genericReturnType).getActualTypeArguments()[0];
199 				}
200 				catch (Exception e)
201 				{
202 					e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
203 				}
204 				//Now this is required as autoboxing will not work with reflection
205 				if (range == Double.class)
206 				{
207 					range = double.class;
208 				}
209 				if (range == Float.class)
210 				{
211 					range = float.class;
212 				}
213 				if (range == Integer.class)
214 				{
215 					range = int.class;
216 				}
217 			}
218 			if (log.isTraceEnabled()) log.trace(range.getCanonicalName());
219 		}
220 
221 		return range;
222 	}

225 	/**
226 	 * Detects and sets the default methods for the property to which editor is associated. If property
227 	 * has multiple cardinality, {@link #setMethod}, {@link #addMethod}, and {@link #removeMethod} are
228 	 * set, otherwise only the {@link #setMethod}.
229 	 * @exception NoSuchMethodException if a method for the property does not exist
230 	 */
231 	private void detectMethods() throws NoSuchMethodException
232 	{
233 		String javaName = getJavaName(property);
234 		if (multipleCardinality)
235 		{
236 
237 			this.addMethod = domain.getMethod("add" + javaName, range);
238 			this.removeMethod = domain.getMethod("remove" + javaName, range);
239 		} else
240 		{
241 			this.setMethod = domain.getMethod("set" + javaName, range);
242 		}
243 	}

276 	@Override public void addMaxCardinalityRestriction(Class<? extends D> domain, int max)
277 	{
278 		if (multipleCardinality)
279 		{
280 			this.maxCardinalities.put(domain, max);
281 		} else
282 		{
283 			if (max == 1)
284 			{
285 				if (log.isInfoEnabled())
286 				{
287 					log.info("unnecessary use of cardinality restriction. " +
288 					         "Maybe you want to use functional instead?");
289 				}
290 			} else if (max == 0)
291 			{
292 				this.maxCardinalities.put(domain, max);
293 			} else
294 			{
295 				assert false;
296 			}
297 		}
298 	}

306 	 * Checks if <em>value</em> is an instance of one of the classes given in a set. This method
307 	 * becomes useful, when the restrictions have to be checked for a set of objects. e.g. check if the
308 	 * value is in the range of the editor.
309 	 *
310 	 * @param classes a set of classes to be checked
311 	 * @param value value whose class will be checked
312 	 * @return true if value belongs to one of the classes in the set
313 	 */
314 	protected boolean isInstanceOfAtLeastOne(Set<Class<? extends BioPAXElement>> classes, Object value)
315 	{
316 		boolean check = false;
317 		for (Class aClass : classes)
318 		{
319 			if (aClass.isInstance(value))
320 			{
321 				check = true;
322 				break;
323 			}
324 		}
325 		return check;
326 	}

336 	@Override public void removeValueFromBean(R value, D bean)
337 	{
338 		if(value == null)
339 			return;
340 
341 		try
342 		{
343 			if (removeMethod != null)
344 			{
345 				invokeMethod(removeMethod, bean, value);
346 			} else {
347 				assert !isMultipleCardinality() : "removeMethod is not defined " +
348 						"for the multiple cardinality property: " + property +
349 						". Here, this might add 'unknown' value while keeping exisiting one as well!";
350 				
351 				if(this.getValueFromBean(bean).contains(value))
352 				{
353 					this.setValueToBean(this.getUnknown(),bean);
354 				}
355 				else {
356 					log.error("Given value :" + value + 
357 						" is not equal to the existing value. " +
358 					         "remove value is ignored");
359 					assert getRange().isInstance(value) : "Range violation!";
360 					assert getDomain().isInstance(bean) : "Domain violation!";
361 				}
362 			}
363 		}
364 		catch (Exception e)
365 		{
366 			log.error("removeValueFromBean failed", e);
367 		}
368 	}

378 	/**
379 	 * Calls the <em>method</em> onto <em>bean</em> with the <em>value</em> as its parameter. In this
380 	 * context <em>method</em> can be one of these three: set, add, or remove.
381 	 * @param method method that is going to be called
382 	 * @param bean bean onto which the method is going to be applied
383 	 * @param value the value which is going to be used by method
384 	 */
385 	protected void invokeMethod(Method method, D bean, R value)
386 	{
387 		assert bean != null;
388 		try
389 		{
390 			method.invoke(domain.cast(bean), value);
391 		}
392 		catch (ClassCastException e)
393 		{
394 			String message = "Failed to set property: " + property;
395 			if (!domain.isAssignableFrom(bean.getClass()))
396 			{
397 				message += "  Invalid domain bean: " + domain.getSimpleName() + " is not assignable from " +
398 				           bean.getClass();
399 			}
400 			if (!range.isAssignableFrom(value.getClass()))
401 			{
402 				message += " Invalid range value: " + range + " is not assignable from " + value.getClass();
403 			}
404 			throw new IllegalBioPAXArgumentException(message, e);
405 		}
406 		catch (Exception e) //java.lang.reflect.InvocationTargetException
407 		{
408 			String valInfo = (value == null) ? null : value.getClass().getSimpleName() + ", " + value;
409 			String message = "Failed to set " + property + " with " + method.getName() + " on " 
410 				+ domain.getSimpleName() + " (" + bean.getClass().getSimpleName() + ", " + bean + ")" 
411 				+ " with range: " + range.getSimpleName() + " (" + valInfo + ")";
412 			throw new IllegalBioPAXArgumentException(message, e);
413 		}
414 	}

421 	@Override public void setValueToBean(R value, D bean)
422 	{
423 		if (this.getPrimarySetMethod() != null)
424 		{
425 			if (log.isTraceEnabled()) log.trace(
426 					this.getPrimarySetMethod().getName() + " bean:" + bean + " val:" + value);
427 		} else
428 		{
429 			log.error("setMethod is null; " + " bean:" + bean + " (" + bean.getUri() + ") val:" + value);
430 		}
431 
432 		// 'null' definitely means 'unknown'for single cardinality props
433 		if (value == null && !isMultipleCardinality())
434 			value = getUnknown(); // not null for primitive property editors
435 
436 		if (value instanceof String)
437 		{
438 			value = this.parseValueFromString(((String) value));
439 		}
440 		try
441 		{
442 			if (value != null && checkRestrictions.get()) checkRestrictions(value, bean);
443 			invokeMethod(this.getPrimarySetMethod(), bean, value);
444 		}
445 		catch (Exception e) {
446 			if(log.isDebugEnabled()) {
447 				log.debug("setValueToBean failed to set value: " + value
448 						+ ((value != null) ? " of type: " + value.getClass().getSimpleName() : "")
449 						+ " to " + bean.getClass().getSimpleName()
450 						+ " (" + bean.getUri() + ") using: " + getPrimarySetMethod().getName(), e);
451 			} else {
452 				log.error("setValueToBean, failed: " + e.getMessage());
453 			}
454 		}
455 	}

457 	@Override public void setValueToBean(Set<R> values, D bean)
458 	{
459 		if (values == null) setValueToBean(((R) null), bean);
460 
461 		else if (this.isMultipleCardinality() || values.size() < 2)
462 		{
463 			for (R r : values)
464 			{
465 				this.setValueToBean(r, bean);
466 			}
467 		} else throw new IllegalBioPAXArgumentException(
468 					this.getProperty() + " is single cardinality. Can not set" + "it with a set of size larger than" +
469 					" 1");
470 	}

473 	/**
474 	 * Checks if the <em>bean</em> and the <em>value</em> are consistent with the cardinality rules of
475 	 * the model. This method is important for validations.
476 	 * @param value Value that is related to the object
477 	 * @param bean Object that is related to the value
478 	 */
479 	protected void checkRestrictions(R value, D bean)
480 	{
481 		Integer max = this.maxCardinalities.get(value.getClass());
482 		if (max != null)
483 		{
484 			if (max == 0)
485 			{
486 				throw new IllegalBioPAXArgumentException("Cardinality 0 restriction violated");
487 			} else
488 			{
489 				assert multipleCardinality;
490 				Set values = this.getValueFromBean(bean);
491 				if (values.size() >= max)
492 				{
493 					throw new IllegalBioPAXArgumentException("Cardinality " + max + " restriction violated");
494 				}
495 			}
496 		}
497 	}
