5   
6   import java.lang.reflect.InvocationTargetException;
7   import java.lang.reflect.Method;
8   import java.util.Collections;
9   import java.util.Set;
10  
11  /**
12  
13   */
14  public class SimplePropertyAccessor<D extends BioPAXElement, R> extends PropertyAccessorAdapter<D, R>
15  {
16  	/**
17  	 * This variable stores the method to invoke for getting the value of the property on a given bean.
18  	 * In the case of multiple cardinality, this method is expected to return a {@link java.util.Set}
19  	 * otherwise an instance of {@link #range}
20  	 */
21  	protected final Method getMethod;
22  
23  	public SimplePropertyAccessor(Class<D> domain, Class<R> range, boolean multipleCardinality, Method getMethod)
24  	{
25  		super(domain, range, multipleCardinality);
26  		this.getMethod = getMethod;
27  	}
28  
29  	protected static boolean isMultipleCardinality(Method getMethod)
30  	{
31  		return Set.class.isAssignableFrom(getMethod.getReturnType());
32  	}
33  
34  	/**
35  	 * Returns the value of the <em>bean</em> using the default {@link #getMethod}.
36  	 * If the value is unknown returns null or an empty set depending on the cardinality.
37  	 * @param bean the object whose property is requested
38  	 * @return an object as the value
39  	 */
40  	@Override public Set<R> getValueFromBean(D bean) throws IllegalBioPAXArgumentException
41  	{
42  		Object value = null;
43  		try
44  		{
45  			if (this.getDomain().isInstance(bean)) value = this.getMethod.invoke(bean);
46  		}
47  		catch (IllegalAccessException e)
48  		{
49  			throw new IllegalBioPAXArgumentException(
50  					"Could not invoke get method " + getMethod.getName() + " for " + bean, e);
51  		}
52  		catch (IllegalArgumentException e)
53  		{
54  			throw new IllegalBioPAXArgumentException(
55  					"Could not invoke get method " + getMethod.getName() + " for " + bean, e);
56  		}
57  
58  		catch (InvocationTargetException e)
59  		{
60  			throw new IllegalBioPAXArgumentException(
61  					"Could not invoke get method " + getMethod.getName() + " for " + bean, e);
62  		}
63  
64  		if (value == null) return Collections.emptySet();
65  		else if (this.isMultipleCardinality())
66  		{
67  			return (Collections.unmodifiableSet(((Set<R>) value)));
68  		} else
69  		{
70  			return Collections.singleton(((R) value));
71  		}
72  	}
73  
74  	/**
75  	 * Checks if the <em>value</em> is unkown. In this context a <em>value</em> is regarded to be
76  	 * unknown if it is null (unset).
77  	 * @param value the value to be checked
78  	 * @return true if the value is unknown
79  	 */
80  	public boolean isUnknown(Object value)
81  	{
82  		return value == null || (value instanceof Set ? ((Set) value).isEmpty() : false);
83  	}
84  
85  	@Override
86  	public String toString() {
87  		return String.format("%s %s %s", domain.getSimpleName(), getMethod.getName(), range.getSimpleName());
88  	}
89  }
