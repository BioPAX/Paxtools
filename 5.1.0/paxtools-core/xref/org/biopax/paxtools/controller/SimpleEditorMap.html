15  /**
16   * Provides a simple editor map for a level with a given factory.
17   * 
18   * This class initializes 3 singletons( 1 for each level) from a tab delimited text resources that lists the
19   * properties and their domains. This is done to remove any dependencies to Jena.
20   * 
21   * The recommended usage is to use the {@link #get(org.biopax.paxtools.model.BioPAXLevel)} method.
22   * @author Emek Demir
23   */
24  public enum SimpleEditorMap implements EditorMap
25  {
26  
27  	L2(BioPAXLevel.L2),
28  	L3(BioPAXLevel.L3);
29  
30  	private static final Logger log = LoggerFactory.getLogger(EditorMapImpl.class);
31  
32  	private final SimpleEditorMapImpl impl;
33  
34  
35  	SimpleEditorMap(BioPAXLevel level)
36  	{
37  		this.impl = new SimpleEditorMapImpl(level);
38  	}
39  
40  	/**
41  	 * To obtain a copy of the editor map for the corresponding level, use the
42  	 * @param level the BioPAX level
43  	 * @return SimpleEditorMap implementation for the BioPAX L2 or L3
44  	 */
45  	public static SimpleEditorMap get(BioPAXLevel level)
46  	{
47  		for (SimpleEditorMap value : values())
48  		{
49  			if (value.getLevel().equals(level)) return value;
50  		}
51  		//should never reach here
52  		throw new IllegalBioPAXArgumentException("Unknown level:" + level);
53  	}

55  	static class SimpleEditorMapImpl extends EditorMapImpl implements EditorMap
56  	{
57  		SimpleEditorMapImpl(BioPAXLevel level)
58  		{
59  			super(level);
60  			InputStream stream = this.getClass().getResourceAsStream(level + "Editor.properties");
61  			readEditors(level, stream, this);
62  		}
63  	}

125 	static void readEditors(BioPAXLevel level, InputStream stream, EditorMapImpl map)
126 	{
127 		BufferedReader reader = new BufferedReader(new InputStreamReader(stream));
128 		try
129 		{
130 			String line = reader.readLine();
131 			StringTokenizer st = new StringTokenizer(line);
132 			while (st.hasMoreElements())
133 			{
134 				map.registerModelClass(st.nextToken());
135 			}
136 
137 			while ((line = reader.readLine()) != null)
138 			{
139 				if (!line.startsWith("#"))
140 				{
141 					st = new StringTokenizer(line);
142 					String domain = st.nextToken();
143 					Class<? extends BioPAXElement> domainInterface = map.getLevel().getInterfaceForName(domain);
144 
145 					String propertyName = st.nextToken();
146 					Map<Class<? extends BioPAXElement>, Set<Class<? extends BioPAXElement>>> rangeRestrictions =
147 							new HashMap<Class<? extends BioPAXElement>, Set<Class<? extends BioPAXElement>>>();
148 					while (st.hasMoreTokens())
149 					{
150 						String rToken = st.nextToken();
151 						if (rToken.startsWith("R:"))
152 						{
153 							StringTokenizer rt = new StringTokenizer(rToken.substring(2), "=");
154 							Class<? extends BioPAXElement> rDomain = level.getInterfaceForName(rt.nextToken());
155 							Set<Class<? extends BioPAXElement>> rRanges = new HashSet<Class<? extends
156 									BioPAXElement>>();
157 							for (StringTokenizer dt = new StringTokenizer(rt.nextToken(), ","); dt.hasMoreTokens(); )
158 							{
159 								rRanges.add(level.getInterfaceForName(dt.nextToken()));
160 							}
161 							rangeRestrictions.put(rDomain, rRanges);
162 						}
163 
164 					}
165 					map.createAndRegisterBeanEditor(propertyName, domainInterface, rangeRestrictions);
166 				}
167 			}
168 		}
169 		catch (IOException e)
170 		{
171 			log.error("Could not initialize " + "Editor Map", e);
172 		}
173 		finally
174 		{
175 			try
176 			{
177 				stream.close();
178 			}
179 			catch (IOException ignore)
180 			{
181 				log.error("Could not close stream! Exiting");
182 				System.exit(1);
183 			}
184 		}
185 
186 	}

188 	public static EditorMap buildCustomEditorMap(BioPAXLevel level, InputStream stream)
189 	{
190 		EditorMapImpl editorMap = new EditorMapImpl(level);
191 		readEditors(level,stream,editorMap);
192 		return editorMap;
193 
194 	}
