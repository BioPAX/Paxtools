4   /**
5    * <b>Definition</b>: An xref that defines a reference to an entity in an external resource that
6    * does not have the same biological identity as the referring entity.
7    *
8    * <b>Usage</b>: There is currently no controlled vocabulary of relationship types for BioPAX,
9    * although one will be created in the future if a need develops.
10   *
11   * <b>Examples</b>: A link between a gene G in a BioPAX data collection, and the protein product P
12   * of that gene in an external database. This is not a unification xref because G and P are
13   * different biological entities (one is a gene and one is a protein). Another example is a
14   * relationship xref for a protein that refers to the Gene Ontology biological process, e.g. 'immune
15   * response,' that the protein is involved in.
16   */
17  public interface RelationshipXref extends Xref
18  {
19  
20  	/**
21  	 * <b>Definition:</b>A controlled vocabulary term that defines the type of relationship that this xref defines.
22  	 *
23  	 * <b>Usage</b>: There is currently no controlled vocabulary of relationship types for BioPAX,
24  	 * although one will be created in the future as the usage of this property increases.
25  	 *
26  	 * @return The type of relationship
27  	 */
28  	RelationshipTypeVocabulary getRelationshipType();
29  
30  	/**
31  	 * <b>Definition:</b>A controlled vocabulary term that defines the type of relationship that this xref defines.
32  	 *
33  	 * <b>Usage</b>: There is currently no controlled vocabulary of relationship types for BioPAX,
34  	 * although one will be created in the future as the usage of this property increases.
35  	 *
36  	 * @param relationshipType The type of relationship that this xref defines
37  	 */
38  	void setRelationshipType(RelationshipTypeVocabulary relationshipType);
39  }
