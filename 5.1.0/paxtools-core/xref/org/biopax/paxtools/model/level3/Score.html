4   /**
5    * <b>Definition:</b> A score associated with a publication reference describing how the score was
6    * determined, the name of the method and a comment briefly describing the method.
7    *
8    * <b>Usage:</b>  The xref must contain at least one publication that describes the method used to
9    * determine the score value. There is currently no standard way of describing  values, so any
10   * string is valid.
11   *
12   * <b>Examples:</b> The statistical significance of a result, e.g. "p&lt;0.05".
13   */

15  {
16  
17  	/**
18  	 * This property defines the value of the score. This can be a numerical or categorical value.

20  	 * @return Numerical or categorical value of the score
21  	 */
22  	public String getValue();
23  
24  	/**
25  	 * This property defines the value of the score. This can be a numerical or categorical value.
26  	 *
27  	 * @param value Numerical or categorical value of the score
28  	 */
29  	public void setValue(String value);
30  
31  	/**
32  	 * This property defines the source of the scoring methodology.
33  	 *
34  	 * @return A publication or web site describing the scoring methodology and the range of values.

36  	public Provenance getScoreSource();

38  	/**
39  	 * This property defines the source of the scoring methodology.
40  	 *

44  	public void setScoreSource(Provenance scoreSource);
