3   /**
4    * Stoichiometric coefficient of a physical entity in the context of a conversion or complex. For
5    * each participating element there must be 0 or 1 stoichiometry element. A non-existing
6    * stoichiometric element is treated as unknown. This is an n-ary bridge for left, right and
7    * component properties.
8    */
9   public interface Stoichiometry extends UtilityClass
10  {
11  
12  
13  	/**
14  	 * @return the physical entity to be annotated with stoichiometry.
15  	 */
16  	PhysicalEntity getPhysicalEntity();
17  
18  	/**
19  	 * @param physicalEntity entity to be annotated with stoichiometry.
20  	 */
21  	void setPhysicalEntity(PhysicalEntity physicalEntity);
22  
23  
24  	/**
25  	 * Stoichiometric coefficient of the physicalEntity specified by {@link #getPhysicalEntity()} in
26  	 * the context of the owner interaction or complex. This value can be any rational number. Generic
27  	 * values such as "n" or "n+1" should not be used - polymers are currently not covered.
28  	 *
29  	 * @return Stoichiometric coefficient for one of the entities in an interaction or complex
30  	 */
31  	float getStoichiometricCoefficient();
32  
33  	/**
34  	 * Sets the stoichiometric coefficient of the physicalEntity specified by {@link
35  	 * #getPhysicalEntity()} in the context of the owner interaction or complex. { can be any
36  	 * rational number. Generic values such as "n" or "n+1" should not be used - polymers are currently
37  	 * not covered.
38  	 *
39  	 * @param newStoichiometricCoefficient of the physical entity
40  	 */
41  	void setStoichiometricCoefficient(float newStoichiometricCoefficient);
42  }
