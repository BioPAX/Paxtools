4   /**
5    * Definition: The biological source of an entity (e.g. protein, RNA or DNA). Some entities are
6    * considered source-neutral (e.g. small molecules), and the biological source of others can be
7    * deduced from their constituentss (e.g. complex, pathway). Examples: HeLa cells, human, and mouse
8    * liver tissue.
9    */
10  public interface BioSource extends UtilityClass, Named
11  {
12  
13  	/**
14  	 * A cell type, e.g. 'HeLa'. This should reference a term in a controlled vocabulary of cell types.
15  	 * Best practice is to refer to <a href="http://www.obofoundry.org/cgi-bin/detail.cgi?id=cell">OBOCell
16  	 * Ontology</a>.
17  	 * @return cell type if this biosource is a cell line or a specific tissue. Null otherwise.
18  	 */
19  	CellVocabulary getCellType();
20  
21  	/**
22  	 * A cell type, e.g. 'HeLa'. This should reference a term in a controlled vocabulary of cell types.
23  	 * Best practice is to refer to <a href="http://www.obofoundry.org/cgi-bin/detail.cgi?id=cell">OBOCell
24  	 * Ontology</a>.
25  	 * @param cellType if this biosource is a cell line or a specific tissue. Null for n/a.
26  	 */
27  	void setCellType(CellVocabulary cellType);
28  
29  
30  	/**
31  	 * An external controlled vocabulary of tissue types. A reference to the  
32  	 * <a href="http://www.brenda-enzymes.info/">BRENDA</a>
33  	 * @return An external controlled vocabulary of tissue types.
34  	 */
35  	TissueVocabulary getTissue();
36  
37  	/**
38  	 * An external controlled vocabulary of tissue types. A reference to the  
39  	 * <a href="http://www.brenda-enzymes.info/">BRENDA</a>
40  	 * @param tissue An external controlled vocabulary of tissue types.
41  	 */
42  	void setTissue(TissueVocabulary tissue);
43  }
