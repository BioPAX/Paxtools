136        {
137                StringBuilder sb = new StringBuilder();
138
139                int event = r.getEventType();
140                if (event == START_ELEMENT || event == END_ELEMENT || event == ENTITY_REFERENCE)
141                {
142                        sb.append(r.getLocalName());
143                }
144
145                if (r.getLocation() != null)
146                {
147                        sb.append(" line ");
148                        sb.append(r.getLocation().getLineNumber());
149                        sb.append(" column ");
150                        sb.append(r.getLocation().getColumnNumber());
151                }
152
153                return sb.toString();
154        }

193        @Override protected Map<String, String> readNameSpaces()
194        {
195                Map<String, String> ns = new HashMap<String, String>();
196                try
197                {
198                        if (r.getEventType() == START_DOCUMENT)
199                        {
200                                r.next();
201                        } else
202                        {
203                                throw new BioPaxIOException("Unexpected element at start");
204                        }
205
206                        // Skip any comment before we read the RDF headers
207                        while (r.getEventType() == COMMENT)
208                        {
209                                r.next();
210                        }
211
212                        if (r.getEventType() != START_ELEMENT)
213                        {
214                                throw new BioPaxIOException("Unexpected element at start: " + r.getEventType());
215                        }
216
217                        if (!r.getLocalName().equalsIgnoreCase("rdf"))
218                        {
219                                throw new BioPaxIOException("Unexpected element at start: " + r.getLocalName());
220                        }
221
222                        int count = r.getNamespaceCount();
223                        for (int i = 0; i < count; i++)
224                        {
225                                String pre = r.getNamespacePrefix(i);
226                                if (pre == null)
227                                {
228                                        pre = "";
229                                }
230                                String namespace = r.getNamespaceURI(pre);
231
232                                ns.put(pre, namespace);
233                        }
234
235                        base = null;
236                        for (int i = 0; i < r.getAttributeCount(); i++)
237                        {
238                                if ("base".equalsIgnoreCase(r.getAttributeLocalName(i)) &&
239                                    "xml".equalsIgnoreCase(r.getAttributePrefix(i)))
240                                {
241                                        base = r.getAttributeValue(i);
242                                }
243                        }
244                }
245                catch (XMLStreamException e)
246                {
247                        throw new BioPaxIOException(e.getClass().getSimpleName() + " " + e.getMessage() + "; " + e.getLocation());
248                }
249                return ns;
250
251        }

253        @Override protected void createAndBind(Model model)
254        {
255                try
256                {
257                        int type;
258                        while ((type = r.getEventType()) != END_DOCUMENT)
259                        {
260                                switch (type)
261                                {
262                                        case START_ELEMENT:
263                                                if (BioPAXLevel.isInBioPAXNameSpace(r.getName().getNamespaceURI()))
264                                                {
265                                                        String lname = r.getLocalName();
266                                                        BioPAXLevel level = getLevel();
267                                                        Class<? extends BioPAXElement> clazz;
268                                                        try
269                                                        {
270                                                                clazz = level.getInterfaceForName(lname);
271                                                        }
272                                                        catch (IllegalBioPAXArgumentException e)
273                                                        {
274                                                                log.error("Unknown BioPAX element location " + lname + " at " + r.getLocation());
275                                                                throw e; // for backward compatibility
276                                                        }
277                                                        if (this.getFactory().canInstantiate(clazz))
278                                                        {
279                                                                processIndividual(model);
280                                                        } else
281                                                        {
282                                                                if (log.isTraceEnabled())
283                                                                        log.trace("Ignoring element: " + lname);
284                                                                skip();
285                                                        }
286
287                                                }
288                                                break;
289                                        case CHARACTERS:
290                                                if (log.isTraceEnabled())
291                                                {
292                                                        StringBuilder sb = new StringBuilder("Ignoring text ");
293                                                        if (r.hasName()) sb.append(r.getLocalName());
294                                                        if (r.hasText()) sb.append(r.getText());
295                                                        if (log.isTraceEnabled())
296                                                                log.trace(sb.toString());
297                                                }
298                                                break;
299                                        case END_ELEMENT:
300                                                if (log.isTraceEnabled())
301                                                        log.trace(String.valueOf(r));
302                                                break;
303                                        default:
304                                                if (log.isTraceEnabled())
305                                                        log.trace("Test this!:" + type);
306                                }
307                                r.next();
308
309                        }
310                r.close();
311                }
312                catch (XMLStreamException e)
313                {
314                        throw new BioPaxIOException(e.getClass().getSimpleName() + " " + e.getMessage() + "; " + e.getLocation());
315                }
316
317                for (Triple triple : triples)
318                {
319                        try
320                        {
321                                bindValue(triple, model);
322                        }
323                        catch (IllegalBioPAXArgumentException e)
324                        {
325                                log.warn("Binding " + e);
326                        }
327                }
328
329        }

335         * @param triple A java object that represents an RDF Triple - domain-property-range.
336         * @param model that is being populated.

338        private void bindValue(Triple triple, Model model)
339        {
340                if (log.isDebugEnabled())
341                        log.debug(String.valueOf(triple));
342
343                BioPAXElement domain = model.getByID(triple.domain);

346
347                bindValue(triple.range, editor, domain, model);
348        }

351        private String processIndividual(Model model) throws XMLStreamException
352        {
353                String s = r.getLocalName();
354                String id;
355                try
356                {
357                        id = getId();
358                }
359                catch (NullPointerException e)
360                {
361                        throw new BioPaxIOException("Error processing individual " + s + ". rdf:ID or rdf:about not found!", e);
362                }
363
364                if (factory.canInstantiate((level.getInterfaceForName(s))))
365                {
366                        BioPAXElement bpe = model.getByID(id);
367                        if (!mergeDuplicates || bpe == null)
368                        {
369                                createBpe(s, id, model); //throws an exception when (mergeDuplicates==false && bpe!=null)
370                        } else if(!s.equals(bpe.getModelInterface().getSimpleName())) {
371                                //i.e., type mismatch,
372                                //whereas (mergeDuplicates==true && bpe != null) is true already -
373                                throw new BioPaxIOException(String.format("processIndividual: " +
374                                                "despite mergeDuplicates is True, failed/skipped creating an instance " +
375                                                "of %s, URI:%s, because previously added object (same URI) was of different type: %s",
376                                                s, id, bpe.getModelInterface().getSimpleName()));
377                        }
378                } else {
379                        if (r.hasText())
380                        {
381                                log.warn("Unknown class :" + r.getText());
382                        } else
383                        {
384                                log.warn("Unknown class :" + r);
385                        }
386                        skip();
387                }
388                propertyContext = true;
389                r.next();
390                while (r.getEventType() != END_ELEMENT)
391                {
392                        if (r.getEventType() == START_ELEMENT)
393                        {
394                                processProperty(model, id);
395                                propertyContext = true;
396
397                        }
398                        r.next();
399                }
400                return id;
401        }

403        private void createBpe(String s, String id, Model model)
404        {
405                BioPAXElement bpe = factory.create(s, id);
406                model.add(bpe);
407        }

409        private void skip() throws XMLStreamException
410        {
411                int depth = 1;
412                while (!(r.getEventType() == END_ELEMENT && depth == 0))
413                {
414                        r.next();
415                        switch (r.getEventType())
416                        {
417                                case START_ELEMENT:
418                                        depth++;
419                                        break;
420                                case END_ELEMENT:
421                                        depth--;
422                                        break;
423                        }
424                }
425        }

427        public String getId()
428        {
429                String id = r.getAttributeValue(rdf, "ID");
430                if (id == null)
431                {
432                        id = r.getAttributeValue(rdf, "about");
433                        if (id.startsWith("#"))
434                        {
435                                id = base + id.substring(1, id.length());
436
437                        }
438                } else if (base != null)
439                {
440                        id = base + id;
441                }
442
443                return id;
444        }

446        private void processProperty(Model model, String ownerID) throws XMLStreamException
447        {
448                if (rdfs.equals(r.getNamespaceURI()) && "comment".equals(r.getLocalName()))
449                {
450                        BioPAXElement paxElement = model.getByID(ownerID);
451                        AbstractPropertyEditor commentor = getRDFCommentEditor(paxElement);
452                        r.next();
453                        assert r.getEventType() == CHARACTERS;
454                        String text = r.getText();
455                        if(text!=null) text = text.trim();
456                        commentor.setValueToBean(text, paxElement);
457                        log.warn("rdfs:comment is converted into the bp:comment; " +
458                                 "however, this can be overridden " +
459                                 "if there exists another bp:comment (element: " +
460                                 paxElement.getUri() + " text: " + text + ")");
461                        gotoEndElement();
462                } else if (bp != null && bp.equals(r.getNamespaceURI()))
463                {
464                        String property = r.getLocalName();
465                        String resource = r.getAttributeValue(rdf, "resource");
466                        if (resource != null)
467                        {
468                                if (resource.startsWith("#"))
469                                {
470                                        resource = (base == null ? "" : base) + resource.substring(1, resource.length());
471                                }
472                                gotoEndElement();
473                        } else
474                        {
475                                r.next();
476                                boolean found = false;
477                                while (r.getEventType() != END_ELEMENT)
478                                {
479                                        if (!found && r.getEventType() == CHARACTERS)
480                                        {
481                                                StringBuilder buff = new StringBuilder(r.getText());
482                                                r.next();
483                                                while (r.getEventType() == CHARACTERS)
484                                                {
485                                                        buff.append(r.getText());
486                                                        r.next();
487                                                }
488                                                resource = buff.toString();
489
490                                        } else if (r.getEventType() == START_ELEMENT)
491                                        {
492                                                propertyContext = false;
493                                                resource = processIndividual(model);
494                                                found = true;
495                                                r.next();
496                                        } else r.next();
497                                }
498                                resource = (!found && resource != null) ? resource.replaceAll("[\n\r\t ]+", " ") : resource;
499                        }
500
501                        log.trace("setting = " + resource);
502                        triples.add(new Triple(ownerID, resource, property));
503                        propertyContext = false;
504                } else
505                {
506                        log.trace("ignoring unknown element " +
507                                  r.getNamespaceURI() + r.getLocalName());
508                        gotoEndElement();
509                }
510
511        }
512
513        private void gotoEndElement() throws XMLStreamException
514        {
515                while (r.getEventType() != END_ELEMENT)
516                {
517                        r.next();
518                }
519        }

543        /**
544         * Converts a model into BioPAX (OWL) format, and writes it into
545         * the outputStream. Saved data can be then read via {@link BioPAXIOHandler}
546         * interface (e.g., {@link SimpleIOHandler}).
547         *
548         * Note: When the model is incomplete (i.e., contains elements that refer externals,
549         * dangling BioPAX elements) and is exported by this method, it works; however one
550         * will find corresponding object properties set to NULL later,
551         * after converting such data back to Model.
552         * 
553         * Note: if the model is very very large, and the output stream is a byte array stream,
554         * then you can eventually get OutOfMemoryError "Requested array size exceeds VM limit"
555         * (max. array size is 2Gb)
556         * 
557         * @param model model to be converted into OWL format
558         * @param outputStream output stream into which the output will be written
559         * @throws BioPaxIOException in case of I/O problems
560         */
561        public void convertToOWL(Model model, OutputStream outputStream)
562        {
563                initializeExporter(model);
564
565                try
566                {
567                        Writer out = new BufferedWriter(new OutputStreamWriter(outputStream, "UTF-8"));
568                        writeObjects(out, model);
569                        out.close();
570                }
571                catch (IOException e)
572                {
573                        throw new BioPaxIOException("Cannot convert to OWL!", e);
574                }
575        }

578        /**
579         * Writes the XML representation of individual BioPAX element that
580         * is BioPAX-like but only for display or debug purpose (incomplete).
581         *
582         * Note: use {@link BioPAXIOHandler#convertToOWL(org.biopax.paxtools.model.Model, java.io.OutputStream)}
583         * convertToOWL(org.biopax.paxtools.model.Model, Object)} instead
584         * if you have a model and want to save and later restore it.
585         * @param out output
586         * @param bean BioPAX object
587         * @throws IOException when the output writer throws
588         */
589        public void writeObject(Writer out, BioPAXElement bean) throws IOException
590        {
591                String name = "bp:" + bean.getModelInterface().getSimpleName();
592                writeIDLine(out, bean, name);
593
594                Set<PropertyEditor> editors = editorMap.getEditorsOf(bean);
595                if (editors == null || editors.isEmpty())
596                {
597                        log.info("no editors for " + bean.getUri() + " | " + bean.getModelInterface().getSimpleName());
598                        out.write(newline + "</" + name + ">");
599                        return;
600                }
601
602                for (PropertyEditor editor : editors)
603                {
604                        Set value = editor.getValueFromBean(bean); //is never null
605                        for (Object valueElement : value)
606                        {
607                                if (!editor.isUnknown(valueElement)) writeStatementFor(bean, editor, valueElement, out);
608                        }
609                }
610
611                out.write(newline + "</" + name + ">");
612        }

615        private void writeObjects(Writer out, Model model) throws IOException
616        {
617                writeHeader(out);
618
619                Set<BioPAXElement> bioPAXElements = model.getObjects();
620                for (BioPAXElement bean : bioPAXElements)
621                {
622                        writeObject(out, bean);
623                }
624
625                out.write(newline + "</rdf:RDF>");
626        }

629        private void writeStatementFor(BioPAXElement bean, PropertyEditor editor, Object value, Writer out)
630                        throws IOException
631        {
632                assert (bean != null && editor != null && value != null);
633
634                //fix (for L3 only): skip 'name' if it's present in the displayName, etc..
635                if (editor.getProperty().equalsIgnoreCase("name") && bean instanceof Named)
636                { // the latter maybe not necessary...
637                        Named named = (Named) bean;
638                        if(value.equals(named.getDisplayName()) || value.equals(named.getStandardName()))
639                        {
640                                return;
641                        }
642                }
643                
644                if (editor.getProperty().equalsIgnoreCase("stepProcess") && bean instanceof BiochemicalPathwayStep)
645                {
646                        BiochemicalPathwayStep bps = (BiochemicalPathwayStep) bean;
647                        if(value.equals(bps.getStepConversion()))
648                        {
649                                return;
650                        }
651                }
652
653                String prop = "bp:" + editor.getProperty();
654                out.write(newline + " <" + prop);
655
656                if (value instanceof BioPAXElement)
657                {
658                        String id = ((BioPAXElement) value).getUri();
659                        assert id != null;
660                        if (!absoluteUris && base != null && id.startsWith(base))
661                        {
662                                id = '#' + id.substring(base.length());
663                        }
664                        out.write(" rdf:resource=\"" + id + "\" />");
665                } else
666                {
667                        String type = findLiteralType(editor);
668                        String valString = StringEscapeUtils.escapeXml(value.toString());
669                        out.write(" rdf:datatype = \"" + xsd + type + "\">" + valString +
670                                  "</" + prop + ">");
671                }
672        }

675        private String findLiteralType(PropertyEditor editor)
676        {
677                Class range = editor.getRange();
678                String type = null;
679                if (range.isEnum() || range.equals(String.class))
680                {
681                        type = "string";
682                } else if (range.equals(double.class) || range.equals(Double.class))
683                {
684                        type = "double";
685                } else if (range.equals(int.class) || range.equals(Integer.class))
686                {
687                        type = "int";
688                } else if (range.equals(float.class) || range.equals(Float.class))
689                {
690                        type = "float";
691                } else if (range.equals(boolean.class) || range.equals(Boolean.class))
692                {
693                        type = "boolean";
694                } else if (range.equals(long.class) || range.equals(Long.class))
695                {
696                        type = "long";
697                }
698                return type;
699        }

702        private void writeIDLine(Writer out, BioPAXElement bpe, String name) throws IOException
703        {
704
705                out.write(newline + newline + "<" + name + " ");
706                String s = bpe.getUri();
707                if (!absoluteUris &&  base != null && s.startsWith(base))
708                {
709                        String id = s.substring(base.length());
710                        out.write(RDF_ID + id + close);
711                } else
712                {
713                        out.write(RDF_about + s + close);
714                }

720        private void initializeExporter(Model model)
721        {
722                base = model.getXmlBase();
723                namespaces = new HashMap<String, String>(model.getNameSpacePrefixMap());
724
725                normalizeNameSpaces(); // - for this reader/exporter tool
726                // also save the changes to the model?
727                if (normalizeNameSpaces)
728                {
729                        model.getNameSpacePrefixMap().clear();
730                        model.getNameSpacePrefixMap().putAll(namespaces);
731                }
732
733                BioPAXLevel lev = model.getLevel();
734                if (lev != this.level) resetLevel(lev, lev.getDefaultFactory());
735        }

738        private void normalizeNameSpaces()
739        {
740                String owlPre = null;
741                String rdfPre = null;
742                String xsdPre = null;
743
744                for (String pre : namespaces.keySet())
745                {
746                        String ns = namespaces.get(pre);
747                        if (rdfNS.equalsIgnoreCase(ns))
748                        {
749                                rdfPre = pre;
750                        } else if (owlNS.equalsIgnoreCase(ns))
751                        {
752                                owlPre = pre;
753                        } else if (xsdNS.equalsIgnoreCase(ns))
754                        {
755                                xsdPre = pre;
756                        }
757                }
758
759                if (owlPre != null)
760                {
761                        namespaces.remove(owlPre);
762                }
763
764                if (rdfPre != null)
765                {
766                        namespaces.remove(rdfPre);
767                }
768
769                if (xsdPre != null)
770                {
771                        namespaces.remove(xsdPre);
772                }
773
774                namespaces.put("rdf", rdfNS);
775                namespaces.put("owl", owlNS);
776                namespaces.put("xsd", xsdNS);
777        }

780        private void writeHeader(Writer out) throws IOException
781        {
782                //Header
783                out.write("<?xml version=\"1.0\" encoding=\"UTF-8\"?>");
784                out.write(newline + "<rdf:RDF");
785                String bpns = this.editorMap.getLevel().getNameSpace();
786                for (String pre : namespaces.keySet())
787                {
788                        if (!pre.equals("bp"))
789                        {
790                                out.write(newline + " xmlns" +
791                                          (("".equals(pre)) ? "" : ":" + pre) + "=\"" + namespaces.get(pre) + "\"");
792                        }
793                }
794                // write 'xmlns:bp'
795                out.write(newline + " xmlns:bp" + "=\"" + bpns + "\"");
796
797                if (base != null)
798                {
799                        out.write(newline + " xml:base=\"" + base + "\"");
800                }
801
802                out.write(">");
803                out.write(newline + "<owl:Ontology rdf:about=\"\">");
804                out.write(newline + " <owl:imports rdf:resource=\"" + bpns + "\" />");
805                out.write(newline + "</owl:Ontology>");
806        }

864        /**
865         * Serializes a (not too large) BioPAX model to the RDF/XML (OWL) formatted string.
866         *
867         * @param model a BioPAX object model to convert to the RDF/XML format
868         * @return the BioPAX data in the RDF/XML format
869         * @throws IllegalArgumentException if model is null
870         * @throws OutOfMemoryError when it cannot be stored in a byte array (max 2Gb).
871         */
872        public static String convertToOwl(Model model)
873        {
874                if (model == null) throw new IllegalArgumentException();
875
876                ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
877
878                (new SimpleIOHandler(model.getLevel())).convertToOWL(model, outputStream);
879
880                try
881                {
882                        return outputStream.toString("UTF-8");
883                }
884                catch (UnsupportedEncodingException e)
885                {
886                        log.error("convertToOwl, outputStream.toString failed", e);
887                        return outputStream.toString();
888                }
889        }
