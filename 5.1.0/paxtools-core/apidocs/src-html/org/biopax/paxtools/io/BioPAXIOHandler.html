001package org.biopax.paxtools.io;
002
003import org.biopax.paxtools.controller.EditorMap;
004import org.biopax.paxtools.model.BioPAXFactory;
005import org.biopax.paxtools.model.BioPAXLevel;
006import org.biopax.paxtools.model.Model;
007
008import java.io.FileNotFoundException;
009import java.io.InputStream;
010import java.io.OutputStream;
011
012/**
013 * This interface defines IO related operations that can be performed on
014 * BioPAX models.
015 */
016public interface BioPAXIOHandler
017{
018
019 /**
020  * This option is only applicable two level 2 models.
021  * When enabled it will replicate illegally reused
022  * pysicalEntityParticipants in Level2 files.
023  * @param fixReusedPEPs true or false
024  */
025 void fixReusedPEPs(boolean fixReusedPEPs);
026
027 /**
028  * This method will read the OWL document given by the input stream
029  * and will convert it into an in memory BioPAX model.
030  * @param in a BioPAX data input stream (RDF/XML format)
031  * @return new BioPAX object Model
032  */
033 Model convertFromOWL(InputStream in);
034
035 /**
036  * This method will write the model to the output stream. Default encoding
037  * is RDF/XML.
038  * @param model a BioPAX model
039  * @param outputStream output stream
040  */
041 void convertToOWL(Model model, OutputStream outputStream);
042
043 /**
044  * This option is only applicable two level 2 models.
045  * When enabled it will replicate illegally reused
046  * pysicalEntityParticipants in Level2 files.
047  * @return true if this option is enabled.
048  */
049 boolean isFixReusedPEPs();
050
051 /**
052  * @return the factory that is used to create new BioPAX POJOs during a BioPAXIOHandler operation.
053  */
054 BioPAXFactory getFactory();
055
056 /**
057  * @param factory used for creating objects
058  */
059 void setFactory(BioPAXFactory factory);
060
061 /**
062  * @return EditorMap used for this handler.
063  */
064 EditorMap getEditorMap();
065
066 /**
067  * @param editorMap used for this handler.
068  */
069 void setEditorMap(EditorMap editorMap);
070
071 /**
072  * @return The level of the model that is being read.
073  */
074 BioPAXLevel getLevel();
075
076 /**
077  * This method will "excise" a new model from the given model that contains
078  * the objects with given ids and their dependents.
079  * @param model BioPAX object model to be exported to the output stream as RDF/XML
080  * @param outputStream the stream
081  * @param ids optional list of absolute URIs of BioPAX objects - roots/seeds for extracting a sub-model to be exported
082  */
083 void convertToOWL(Model model, OutputStream outputStream, String... ids);
084
085}
