060        /**
061         * Updates the level and factory for this I/O
062         * (final - because used in the constructor)
063         * @param level BioPAX Level
064         * @param factory concrete BioPAX factory impl.
065         */
066        protected final void resetLevel(BioPAXLevel level, BioPAXFactory factory)
067        {
068                this.level = (level != null) ? level : BioPAXLevel.L3;
069                this.factory = (factory != null) ? factory : this.level.getDefaultFactory();
070
071                // default flags
072                if (this.level == BioPAXLevel.L2)
073                {
074                        this.fixReusedPEPs = true;
075                }
076
077                bp = this.level.getNameSpace();
078
079                resetEditorMap(); //implemented by concrete subclasses
080        }

088        /**
089         * According the BioPAX documentation, it is illegal to reuse a Physical Entity Participant (PEP).

107        /**
108         * Workaround for a very common Level 2 issue. Most level2 exports in the past reused PhysicalEntityParticipants
109         * as if they represent states. This is problematic because PEPs also contain stoichiometry information which is
110         * specific to a reaction. BioPAX spec says that PEPs should not be reused across reactions.
111         *
112         * As Level2 exports getting obsolete this method is slated for deprecation.
113         * @return true if this Handler automatically splits reused PEPs to interaction specific PEPS.
114         */
115        public boolean isFixReusedPEPs()

157        /**
158         * Reads a BioPAX model from an OWL file input stream (<em>in</em>) and converts it to a model.
159         * @param in inputStream from which the model will be read
160         * @return an empty model in case of invalid input.
161         */
162        public Model convertFromOWL(InputStream in)
163        {
164                init(in);
165
166                //cache the namespaces.
167                namespaces = this.readNameSpaces();
168
169                autodetectBiopaxLevel(); // this may update level, editorMap and factory!
170
171                Model model = factory.createModel();
172
173                model.getNameSpacePrefixMap().putAll(namespaces);
174
175                model.setXmlBase(base);
176
177                boolean fixingPEPS = model.getLevel() == BioPAXLevel.L2 && this.isFixReusedPEPs();
178                if (fixingPEPS)
179                {
180                        reusedPEPHelper = new ReusedPEPHelper(model);
181                }
182
183                createAndBind(model);
184
185                if (fixingPEPS)
186                {
187                        this.getReusedPEPHelper().copyPEPFields();
188                }
189                
190                reset(in);
191
192                return model;
193        }
194
195        protected void reset(InputStream in)
196        {
197                try
198                {
199                        in.close();
200
201                }
202                catch (IOException e)
203                {
204                        throw new RuntimeException("Failed to close the file");
205                }
206
207        }
208
209        private void autodetectBiopaxLevel()
210        {
211                BioPAXLevel filelevel = null;
212                for (String namespaceValue : namespaces.values())
213                {
214                        filelevel = BioPAXLevel.getLevelFromNameSpace(namespaceValue);
215                        if (filelevel != null)
216                        {
217                                if (log.isDebugEnabled())
218                                        log.debug("Auto-detected biopax " + filelevel + " (current settings are for Level " + level + ")");
219                                break;
220                        }
221                }
222
223                if (filelevel == null)
224                {
225                        log.error("Cannot detect biopax level.");
226                        throw new BioPaxIOException("Cannot detect biopax level.");
227                } else if (level != filelevel)
228                {
229                        if (log.isDebugEnabled()) log.debug("Reset to the default factory for the detected BioPAX level.");
230                        resetLevel(filelevel, filelevel.getDefaultFactory());
231                }
232        }

234        /**
235         * This method is called by the reader for each OWL instance in the OWL model. It creates a POJO instance, with
236         * the given id and inserts it into the model. The inserted object is "clean" in the sense that its properties are
237         * not set yet.
238         *
239         * Implementers of this abstract class can override this method to inject code during object creation.
240         * @param model to be inserted
241         * @param id of the new object. The model should not contain another object with the same ID.
242         * @param localName of the class to be instantiated.
243         */
244        protected void createAndAdd(Model model, String id, String localName)
245        {
246                BioPAXElement bpe = this.getFactory().create(localName, id);
247
248                if (log.isTraceEnabled())
249                {
250                        log.trace("id:" + id + " " + localName + " : " + bpe);
251                }
252                /* null might occur here,
253                 * so the following is to prevent the NullPointerException
254                 * and to continue the model assembling.
255                 */
256                if (bpe != null)
257                {
258                        model.add(bpe);
259                } else
260                {
261                        log.warn("null object created during reading. It might not be an official BioPAX class.ID: " + id +
262                                 " Class " +
263                                 "name " + localName);
264                }
265        }

288        /**
289         * This method currently only fixes reusedPEPs if the option is set. As L2 is becoming obsolete this method will be
290         * slated for deprecation.
291         * @param bpe to be bound
292         * @param value to be assigned.
293         * @return a "fixed" value.
294         */
295        protected Object resourceFixes(BioPAXElement bpe, Object value)
296        {
297                if (this.isFixReusedPEPs() && value instanceof physicalEntityParticipant)
298                {
299                        value = this.getReusedPEPHelper().fixReusedPEP((physicalEntityParticipant) value, bpe);
300                }
301                return value;
302        }
303
304        /**
305         * This method binds the value to the bpe. Actual assignment is handled by the editor - but this method performs
306         * most of the workarounds and also error handling due to invalid parameters.
307         * @param valueString to be assigned
308         * @param editor that maps to the property
309         * @param bpe to be bound
310         * @param model to be populated.
311         */
312        protected void bindValue(String valueString, PropertyEditor editor, BioPAXElement bpe, Model model)
313        {
314                if (log.isDebugEnabled())
315                {
316                        log.debug("Binding: " + bpe + '(' + bpe.getModelInterface() + " has  " + editor + ' ' + valueString);
317                }
318
319                Object value = (valueString==null)?null:valueString.trim();
320
321                if (editor instanceof ObjectPropertyEditor)
322                {
323                        value = model.getByID(valueString);
324                        value = resourceFixes(bpe, value);
325                        if (value == null)
326                        {
327                                throw new IllegalBioPAXArgumentException(
328                                        "Illegal or Dangling Value/Reference: " + valueString + " (element: " + bpe.getUri() +
329                                        " property: " + editor.getProperty() + ")");
330                        }
331                }
332                if (editor == null)
333                {
334                        log.error("Editor is null. This probably means an invalid BioPAX property. Failed to set " + valueString);
335                } else //is either EnumeratedPropertyEditor or DataPropertyEditor
336                {
337                        editor.setValueToBean(value, bpe);
338                }
339        }

341        /**
342         * Paxtools maps BioPAX:comment (L3) and BioPAX:COMMENT (L2) to rdf:comment. This method handles that.
343         * @param bpe to be bound.
344         * @return a property editor responsible for editing comments.
345         */
346        protected StringPropertyEditor getRDFCommentEditor(BioPAXElement bpe)
347        {
348                StringPropertyEditor editor;
349                Class<? extends BioPAXElement> inter = bpe.getModelInterface();
350                if (this.getLevel().equals(BioPAXLevel.L3))
351                {
352                        editor = (StringPropertyEditor) this.getEditorMap().getEditorForProperty("comment", inter);
353                } else
354                {
355                        editor = (StringPropertyEditor) this.getEditorMap().getEditorForProperty("COMMENT", inter);
356                }
357                return editor;
358        }

360        /**
361         * Similar to {@link BioPAXIOHandler#convertToOWL(org.biopax.paxtools.model.Model,
362         * java.io.OutputStream)} (org.biopax.paxtools.model.Model, Object)}, but
363         * extracts a sub-model, converts it into BioPAX (OWL) format,
364         * and writes it into the outputStream.
365         * Saved data can be then read via {@link BioPAXIOHandler}
366         * interface (e.g., {@link SimpleIOHandler}).
367         * @param model model to be converted into OWL format
368         * @param outputStream output stream into which the output will be written
369         * @param ids optional list of "root" element absolute URIs;
370         *            direct/indirect child objects are auto-exported as well.
371         */
372        public void convertToOWL(Model model, OutputStream outputStream, String... ids)
373        {
374                if (ids.length == 0)
375                {
376                        convertToOWL(model, outputStream);
377                } else
378                {
379                        Model m = model.getLevel().getDefaultFactory().createModel();
380                        m.setXmlBase(model.getXmlBase());
381
382                        Fetcher fetcher = new Fetcher(SimpleEditorMap.get(model.getLevel())); //no Filters anymore
383
384                        for (String uri : ids)
385                        {
386                                BioPAXElement bpe = model.getByID(uri);
387                                if (bpe != null)
388                                {
389                                        fetcher.fetch(bpe, m);
390                                }
391                        }
392
393                        convertToOWL(m, outputStream);
394                }
395        }
