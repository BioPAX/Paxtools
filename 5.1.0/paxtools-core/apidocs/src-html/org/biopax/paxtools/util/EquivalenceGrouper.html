001package org.biopax.paxtools.util;
002
003import org.biopax.paxtools.model.BioPAXElement;
004import org.slf4j.Logger;
005import org.slf4j.LoggerFactory;
006
007import java.util.HashSet;
008import java.util.LinkedList;
009import java.util.List;
010import java.util.Set;
011
012/**
013 * Utility class for equivalence based comparison of a set of BioPAXElements.
014 *
015 * BioPAXElement by default uses equals and hash code methods based on the type and URI.
016 * On the other hand for many elements it is possible to determine semantic equivalence
017 * among elements. For example two entityFeatures with exactly the same type and location
018 * are equivalent. This logic is implemented in isEquivalent() and equivalenceCode()
019 * methods.
020 *
021 * For most Java collections that uses hashCode and equals there is no easy way to plug-in a
022 * comparator to switch to different comparison behavior. This is a simple Set implementation that uses
023 * equivalence codes when possible. It uses HashMap as the underlying implementation and will use a special bucket in
024 * the case of clashes.
025 *
026 */
027public class EquivalenceGrouper<T extends BioPAXElement>
028{
029        HashSet<EquivalanceBucket<T>> buckets;
030
031        private static final Logger log = LoggerFactory.getLogger(EquivalenceGrouper.class);
032
033        public EquivalenceGrouper(Set<? extends T> bpes)
034        {
035                this();
036                addAll(bpes);
037        }
038
039        public HashSet<? extends List<T>> getBuckets()
040        {
041                return buckets;
042        }
043
044        void addAll(Set<? extends T> bpes)
045        {
046                for (T bpe : bpes)
047                {
048                        add(bpe);
049                }
050        }
051
052        public EquivalanceBucket<T> access(final T element)
053        {
054                EquivalanceBucket<T> value =null;
055                if (element != null)
056                {
057                        value = access(element, EquivalanceBucket.EQUALITY);
058                        if (value==null) //now try with equivalence
059                        {
060                                value=access(element,EquivalanceBucket.EQUIVALENCE);
061                        }
062                }
063                return value;
064
065        }
066
067        private EquivalanceBucket<T> access(final T element, final boolean parity)
068        {
069                final Object[] trap = new Object[]{null};
070                if (this.buckets.contains(new Object()
071                {
072                        public int hashCode()
073                        {
074                                return parity?element.equivalenceCode():element.hashCode();
075                        }
076
077                        public boolean equals(Object other)
078                        {
079                                if(other != null && other.equals(element))
080                                {
081                                        trap[0] = other;
082                                        return true;
083                                }
084                                else return false;
085                        }
086                }))
087                {
088                        return (EquivalanceBucket<T>) trap[0];
089                }
090                return null;
091        }
092
093
094        public EquivalenceGrouper()
095        {
096                this.buckets = new HashSet<EquivalanceBucket<T>>();
097        }
098
099        public void add(T bpe)
100        {
101
102                // If this is the case then we will simply return false when
103                // we have something that matches the evcode
104                // AND if that something is a bucket contains something that is  equivalent to bpe
105                // AND if that something is not a bucket and it is equivalent to bpe
106                EquivalanceBucket<T> bucket = access(bpe);
107                if (bucket == null)
108                {
109                        bucket = new EquivalanceBucket(bpe);
110                        this.buckets.add(bucket);
111                } else
112                {
113                        for (T t : bucket)
114                        {
115                                if(t==bpe) return;
116                        }
117                        bucket.add(bpe);
118                }
119        }
120
121
122        private class EquivalanceBucket<T extends BioPAXElement> extends LinkedList<T>
123        {
124
125                static final boolean EQUALITY = false;
126
127                static final boolean EQUIVALENCE = true;
128
129                private final int code;
130
131                private boolean parity;
132
133                private EquivalanceBucket(T first)
134                {
135                        this.add(first);
136                        if (first.equivalenceCode() == 0 || first.equivalenceCode() == first.hashCode())
137                        {
138                                parity = EQUALITY;
139                                this.code = first.hashCode();
140                        } else
141                        {
142                                parity = EQUIVALENCE;
143                                this.code = first.equivalenceCode();
144                        }
145                }
146
147                @Override public int hashCode()
148                {
149                        return code;
150                }
151
152                @Override public boolean equals(Object o)
153                {
154                        if (o instanceof EquivalanceBucket)
155                        {
156                                return super.equals(o);
157                        } else
158                        {
159                                T t = this.get(0);
160                                if (parity) //EQUIVALENCE
161                                {
162                                        return t.isEquivalent((BioPAXElement) o);
163                                } else //EQUALITY
164                                {
165                                        return t.equals(o);
166                                }
167                        }
168                }
169
170        }
171}
