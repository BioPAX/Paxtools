001package org.biopax.paxtools.util;
002
003import org.biopax.paxtools.model.BioPAXElement;
004import org.slf4j.Logger;
005import org.slf4j.LoggerFactory;
006
007import java.util.HashSet;
008import java.util.List;
009import java.util.Set;
010
011/**
012 * This class performs set operations based on equivalence.
013 */
014public class SetEquivalenceChecker {
015
016        private static final Logger LOG = LoggerFactory.getLogger(SetEquivalenceChecker.class);
017
018        /**
019         * @param set1 First set to be checked.
020         * @param set2 Second set to be checked
021         * @param <T> Both sets should be of type that extends from T.
022         * @return true iff both sets are of equal size and all objects in set1 has an equivalent object in set2.
023         */
024        public static <T extends BioPAXElement> boolean isEquivalent(Set<? extends T> set1, Set<? extends T> set2)
025        {
026                if (set1 != null && !set1.isEmpty() && set2 != null && !set2.isEmpty())
027                {
028                        int size = set1.size();
029                        if (size == set2.size())
030                        {
031                                EquivalenceGrouper<T> grouper = new EquivalenceGrouper<T>();
032                                grouper.addAll(set1);
033                                if (grouper.getBuckets().size() == size)
034                                {
035                                        grouper.addAll(set2);
036                                        return (grouper.getBuckets().size() == size);
037                                }
038                        }
039                } 
040                
041                //now, only if both null or both empty - return true 
042                if( set1 == null && set2 == null //both null
043                        || // or both not null but empty
044                        set1 != null && set1.isEmpty() && set2 != null && set2.isEmpty()) 
045                { 
046                        return true;
047                }
048                
049                return false;   
050        }
051
052        /**
053         * @param set Set to test if it contains an element equivalent to query
054         * @param query BPE to look for equivalents in set
055         * @return true iff there is an element of set that is equivalent to query.
056         */
057        public static boolean containsEquivalent(Set<? extends BioPAXElement> set, BioPAXElement query)
058        {
059                if (set != null && query != null)
060                {
061                        for (BioPAXElement element2 : set)
062                        {
063                                if (query.isEquivalent(element2))
064                                {
065                                        return true;
066                                }
067                        }
068                }
069                return false;
070        }
071
072        /**
073         * @param set1 First set to be checked.
074         * @param set2 Second set to be checked
075         * @param <T> Both sets should be of type that extends from T.
076         * @return elements of set1 that has an equivalent element in set2
077         */
078        public static <T extends BioPAXElement> Set<T> findEquivalentIntersection(Set<? extends T> set1,
079                        Set<? extends T> set2)
080        {
081
082                Set<T> value = new HashSet<T>();
083
084                if (set1 != null && !set1.isEmpty() && set2 != null && !set2.isEmpty())
085                {
086                        EquivalenceGrouper<T> grouper = new EquivalenceGrouper<T>();
087                        grouper.addAll(set1);
088                        if (grouper.getBuckets().size() < set1.size())
089                        {
090                                throw new IllegalArgumentException("There should not be equivalent elements in a set");
091                        }
092                        grouper.addAll(set2);
093                        for (List<T> ts : grouper.getBuckets())
094                        {
095                                if (ts.size() > 1)
096                                {
097                                        for (T t : ts)
098                                        {
099                                                if (set1.contains(t))
100                                                {
101                                                        value.add(t);
102                                                }
103                                        }
104                                }
105                        }
106                }
107                return value;
108        }
109
110        /**
111         * @param set1 First set to be checked.
112         * @param set2 Second set to be checked
113         * @param <T> Both sets should be of type that extends from T.
114         * @return true iff there are at least one equivalent element between set1 and set2, or both sets are empty..
115         */
116        public static <T extends BioPAXElement> boolean hasEquivalentIntersection(Set<? extends T> set1,
117                        Set<? extends T> set2)
118        {
119                if (!set1.isEmpty() && !set2.isEmpty())
120                {
121                        EquivalenceGrouper<T> grouper1 = new EquivalenceGrouper<T>(set1);
122                        int size1 = grouper1.getBuckets().size();
123                        if(set1.size() > size1)
124                                LOG.warn("hasEquivalentIntersection: the first set already contains equivalent objects");
125                        EquivalenceGrouper<T> grouper2 = new EquivalenceGrouper<T>(set2);
126                        int size2 = grouper2.getBuckets().size();
127                        if(set2.size() > size2)
128                                LOG.warn("hasEquivalentIntersection: the second set already contains equivalent objects");
129                        //now add both sets into one grouper -
130                        EquivalenceGrouper<T> grouper = new EquivalenceGrouper<T>();
131                        grouper.addAll(set1);
132                        grouper.addAll(set2);
133                        return (grouper.getBuckets().size() < size1 + size2);
134                } else if(set1.isEmpty() ^ set2.isEmpty()) {
135                        //only one of sets is empty
136                        return false;
137                } else //both empty
138                        return true;
139        }
140
141}
