112        public synchronized void remove(BioPAXElement aBioPAXElement)
113        {               
114                if(this.contains(aBioPAXElement))
115                        this.idMap.remove(aBioPAXElement.getUri());
116        }

118        public synchronized <T extends BioPAXElement> T addNew(Class<T> c, String id)
119        {
120                T paxElement = factory.create(c, id);
121                this.add(paxElement);

125        /**
126         * This method returns true if given element 
127         * is the same object ("==") as the object stored in the model
128         * usually (for self-consistent models) but not necessarily under the element's ID.
129         * 
130         * @param aBioPAXElement BioPAX object (individual)
131         * @return true/false - whether this model contains the object or not
132         */
133        public synchronized boolean contains(BioPAXElement aBioPAXElement)
134        {
135                return this.idMap.get(aBioPAXElement.getUri()) == aBioPAXElement;
136        }

156        public synchronized void add(BioPAXElement aBioPAXElement)
157        {
158                String rdfId = aBioPAXElement.getUri();
159        if(!this.level.hasElement(aBioPAXElement))
160        {
161            throw new IllegalBioPAXArgumentException(
162                "Given object is of wrong level");
163        }
164        
165        if (rdfId == null)
166                {
167                        throw new IllegalBioPAXArgumentException(
168                                "null ID: every object must have an RDF ID");
169                }
170                else if (this.idMap.containsKey(rdfId))
171                {
172                        throw new IllegalBioPAXArgumentException(
173                                "I already have an object with the same ID: " + rdfId +
174                                        ". Try removing it first");
175                }
176                else if (this.contains(aBioPAXElement))
177                {
178                        throw new IllegalBioPAXArgumentException(
179                                "duplicate element:" + aBioPAXElement);
180                }
181                else
182                {
183                        this.idMap.put(rdfId, aBioPAXElement);
184                }
185        }

286    /**
287     * It does not automatically replace or clean up the old 
288     * element's object properties, therefore, some child 
289     * elements may become "dangling" if they were used by
290     * the replaced element only.
291     * 
292     * Can also clear object properties (- replace with null).
293     */
294        public synchronized void replace(final BioPAXElement existing, final BioPAXElement replacement) 
295        {
296                 ModelUtils.replace(this, Collections.singletonMap(existing, replacement));
297                 remove(existing);
298                 if(replacement != null)
299                         add(replacement);
300        }

303        /**
304         * This is default implementation that uses the 
305         * id-based merging ({@link SimpleMerger#merge(Model, Model...)})
306         * 
307         * NOTE: some applications, such as those dealing with persistence/transactions 
308         * or advanced BioPAX alignment/comparison algorithms (like the Patch), 
309         * may have to implement and use a more specific method instead.
310         * 
311         * @see SimpleMerger
312         * @see Model#merge(Model)
313         */
314        public synchronized void merge(Model source) {
315                SimpleMerger merger = new SimpleMerger(
316                        SimpleEditorMap.get(level));
317                if(source == null)
318                        merger.merge(this, this); //repairs itself
319                else
320                        merger.merge(this, source);
321        }
322
323        
324        /**
325         * 
326         * This implementation "repairs" the model 
327         * without unnecessarily copying objects:
328     * - recursively adds lost "children" (not null object property values
329     *   for which {@link Model#contains(BioPAXElement)} returns False)
330     * - updates object properties (should refer to model's elements)
331         * 
332         */
333        @Override
334        public synchronized void repair() {
335                // updates props and children
336                merge(null);
337        }
