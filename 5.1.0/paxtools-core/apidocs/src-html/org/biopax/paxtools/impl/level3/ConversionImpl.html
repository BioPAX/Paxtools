001package org.biopax.paxtools.impl.level3;
002
003import org.biopax.paxtools.model.BioPAXElement;
004import org.biopax.paxtools.model.level3.Conversion;
005import org.biopax.paxtools.model.level3.ConversionDirectionType;
006import org.biopax.paxtools.model.level3.PhysicalEntity;
007import org.biopax.paxtools.model.level3.Stoichiometry;
008import org.biopax.paxtools.util.BPCollections;
009import org.biopax.paxtools.util.SetEquivalenceChecker;
010
011import java.util.Set;
012
013
014public class ConversionImpl extends InteractionImpl
015                implements Conversion
016{
017// ------------------------------ FIELDS ------------------------------
018
019        private Set<PhysicalEntity> right;
020        private Set<PhysicalEntity> left;
021        private ConversionDirectionType conversionDirection;
022        private Set<Stoichiometry> participantStoichiometry;
023        private Boolean spontaneous;
024
025// --------------------------- CONSTRUCTORS ---------------------------
026
027        public ConversionImpl()
028        {
029                left = BPCollections.I.createSafeSet();
030                right = BPCollections.I.createSafeSet();
031                participantStoichiometry = BPCollections.I.createSafeSet();
032        }
033
034
035// --------------------- Interface BioPAXElement ---------------------
036
037        public Class<? extends Conversion> getModelInterface()
038        {
039                return Conversion.class;
040        }
041
042// --------------------- Interface Conversion ---------------------
043
044// --------------------- ACCESORS and MUTATORS---------------------
045
046        public Set<PhysicalEntity> getRight()
047        {
048                return right;
049        }
050
051        public void addRight(PhysicalEntity right)
052        {
053                if(right != null) {
054                        this.right.add(right);
055                        super.addParticipant(right);
056                }
057        }
058
059        public void removeRight(PhysicalEntity right)
060        {
061                if(right != null) {
062                        super.removeParticipant(right);
063                        this.right.remove(right);
064                }
065        }
066
067        public Set<PhysicalEntity> getLeft()
068        {
069                return left;
070        }
071
072        public void addLeft(PhysicalEntity left)
073        {
074                if(left != null) {
075                        this.left.add(left);
076                        super.addParticipant(left);
077                }
078        }
079
080        public void removeLeft(PhysicalEntity left)
081        {
082                if(left != null) {
083                        super.removeParticipant(left);
084                        this.left.remove(left);
085                }
086        }
087
088        
089        public Boolean getSpontaneous()
090        {
091                return spontaneous;
092        }
093
094        public void setSpontaneous(Boolean spontaneous)
095        {
096                this.spontaneous = spontaneous;
097        }
098
099        public Set<Stoichiometry> getParticipantStoichiometry()
100        {
101                return participantStoichiometry;
102        }
103
104        public void addParticipantStoichiometry(
105                        Stoichiometry participantStoichiometry)
106        {
107                this.participantStoichiometry.add(participantStoichiometry);
108        }
109
110        public void removeParticipantStoichiometry(
111                        Stoichiometry participantStoichiometry)
112        {
113                this.participantStoichiometry.remove(participantStoichiometry);
114        }
115
116        public ConversionDirectionType getConversionDirection()
117        {
118                return conversionDirection;
119        }
120
121        public void setConversionDirection(ConversionDirectionType spontanousType)
122        {
123                this.conversionDirection = spontanousType;
124        }
125
126
127        @Override
128        protected boolean semanticallyEquivalent(BioPAXElement element)
129        {
130                if(element.getModelInterface()== this.getModelInterface())
131                {
132                        Conversion that = (Conversion) element;
133                        if(that.getSpontaneous()==this.getSpontaneous() &&
134                       that.getConversionDirection() == this.getConversionDirection())
135                        {
136                                if(SetEquivalenceChecker.isEquivalent(this.getLeft(), that.getLeft()))
137                                {
138                                        return SetEquivalenceChecker.isEquivalent(this.getRight(), that.getRight());
139                                }
140                                else if(SetEquivalenceChecker.isEquivalent(this.getLeft(), that.getRight()))
141                                {
142                                        return(SetEquivalenceChecker.isEquivalent(this.getRight(), that.getLeft()));
143                                }
144                        }
145                }
146                return false;
147        }
148
149        @Override
150        public int equivalenceCode()
151        {
152                return getEqCodeForSet(this.getLeft())*getEqCodeForSet(this.getRight());
153
154        }
155
156        private int getEqCodeForSet(Set<PhysicalEntity> peSet)
157        {
158                int eqCode=0;
159                for (PhysicalEntity pe : peSet)
160                {
161                        eqCode+=pe.equivalenceCode();
162                }
163                return eqCode;
164        }
165
166}
