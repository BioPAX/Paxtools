001package org.biopax.paxtools.impl.level3;
002
003
004import org.biopax.paxtools.model.level3.*;
005import org.biopax.paxtools.model.level3.Process;
006import org.biopax.paxtools.util.BPCollections;
007import org.biopax.paxtools.util.IllegalBioPAXArgumentException;
008
009import java.util.Set;
010
011
012public class CatalysisImpl extends ControlImpl implements Catalysis
013{
014// ------------------------------ FIELDS ------------------------------
015
016        private CatalysisDirectionType catalysisDirection;
017
018        private Set<PhysicalEntity> cofactor;
019
020// --------------------------- CONSTRUCTORS ---------------------------
021
022        public CatalysisImpl()
023        {
024                this.cofactor = BPCollections.I.createSafeSet();
025        }
026
027// ------------------------ INTERFACE METHODS ------------------------
028
029
030// --------------------- Interface BioPAXElement ---------------------
031
032        public Class<? extends Catalysis> getModelInterface()
033        {
034                return Catalysis.class;
035        }
036
037// --------------------- Interface catalysis ---------------------
038
039// --------------------- ACCESORS and MUTATORS---------------------
040        public CatalysisDirectionType getCatalysisDirection()
041        {
042                return catalysisDirection;
043        }
044
045        public void setCatalysisDirection(CatalysisDirectionType catalysisDirection)
046        {
047                this.catalysisDirection = catalysisDirection;
048        }
049
050        public Set<PhysicalEntity> getCofactor()
051        {
052                return cofactor;
053        }
054
055        public void addCofactor(PhysicalEntity cofactor)
056        {
057                if (cofactor != null)
058                {
059                        this.cofactor.add(cofactor);
060                        super.addParticipant(cofactor);
061                }
062        }
063
064        public void removeCofactor(PhysicalEntity cofactor)
065        {
066                if (cofactor != null)
067                {
068                        super.removeParticipant(cofactor);
069                        this.cofactor.remove(cofactor);
070                }
071        }
072
073
074        @Override public void addController(Controller controller)
075        {
076                if (controller instanceof PhysicalEntity) super.addController(controller);
077                else throw new IllegalBioPAXArgumentException("Catalysis can only be controlled with a Physical Entity");
078
079        }
080
081        protected boolean checkControlled(Process controlled)
082        {
083                return controlled instanceof Conversion;
084        }
085}
