001package org.biopax.paxtools.impl.level3;
002
003import org.biopax.paxtools.model.level3.NucleicAcid;
004import org.biopax.paxtools.model.level3.PhysicalEntity;
005import org.biopax.paxtools.model.level3.TemplateDirectionType;
006import org.biopax.paxtools.model.level3.TemplateReaction;
007import org.biopax.paxtools.util.BPCollections;
008
009import java.util.Set;
010
011
012public class TemplateReactionImpl extends InteractionImpl implements TemplateReaction {
013    private Set<PhysicalEntity> product;
014    private NucleicAcid template;
015        private TemplateDirectionType templateDirection;
016
017        public TemplateReactionImpl()
018        {
019        this.product = BPCollections.I.createSafeSet();
020    }
021        
022    public Class<? extends TemplateReaction> getModelInterface()
023        {
024                return TemplateReaction.class;
025        }
026
027    public Set<PhysicalEntity> getProduct()
028    {
029        return product;
030    }
031
032    protected void setProduct(Set<PhysicalEntity> product)
033    {
034        this.product = product;
035    }
036
037    public void addProduct(PhysicalEntity product)
038    {
039        if(product != null) {
040                this.product.add(product);
041                super.addParticipant(product);
042        }
043    }
044
045    public void removeProduct(PhysicalEntity product)
046    {
047        if(product != null) {
048                super.removeParticipant(product);
049                this.product.remove(product);
050        }
051    }
052
053        public NucleicAcid getTemplate()
054     {
055         return this.template;
056     }
057
058    public void setTemplate(NucleicAcid template)
059    {
060         if(this.template!= null)
061         {
062            super.removeParticipant(this.template);
063         }
064         if(template != null) {
065                 this.template=template;
066                 super.addParticipant(template);
067         }
068    }
069
070        public TemplateDirectionType getTemplateDirection()
071        {
072                return templateDirection;
073        }
074
075        public void setTemplateDirection(TemplateDirectionType templateDirection)
076        {
077                this.templateDirection = templateDirection;
078        }
079
080}
