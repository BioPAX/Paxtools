001package org.biopax.paxtools.impl.level3;
002
003import org.biopax.paxtools.model.BioPAXElement;
004import org.biopax.paxtools.model.level3.*;
005import org.biopax.paxtools.util.BPCollections;
006import org.biopax.paxtools.util.SetEquivalenceChecker;
007
008import java.util.Set;
009
010
011public class ComplexImpl extends PhysicalEntityImpl implements Complex
012{
013// ------------------------------ FIELDS ------------------------------
014
015        private Set<PhysicalEntity> component;
016        private Set<Stoichiometry> componentStoichiometry;
017
018// --------------------------- CONSTRUCTORS ---------------------------
019
020        public ComplexImpl()
021        {
022                this.component = BPCollections.I.createSafeSet();
023                this.componentStoichiometry = BPCollections.I.createSafeSet();
024        }
025
026// ------------------------ INTERFACE METHODS ------------------------
027
028
029// --------------------- Interface BioPAXElement ---------------------
030
031        public Class<? extends Complex> getModelInterface()
032        {
033                return Complex.class;
034        }
035
036// --------------------- Interface Complex ---------------------
037
038// --------------------- ACCESORS and MUTATORS---------------------
039        public Set<PhysicalEntity> getComponent()
040        {
041                return component;
042        }
043
044        public void addComponent(PhysicalEntity component)
045        {
046                if (component != null) {
047                        this.component.add(component);
048                        component.getComponentOf().add(this);
049                }
050        }
051
052        public void removeComponent(PhysicalEntity component)
053        {
054                if (component != null) {
055                        this.component.remove(component);
056                        component.getComponentOf().remove(this);
057                }
058        }
059
060        protected void setComponent(Set<PhysicalEntity> component)
061        {
062                this.component = component;
063        }
064
065        public Set<Stoichiometry> getComponentStoichiometry()
066        {
067                return componentStoichiometry;
068        }
069
070        public void addComponentStoichiometry(
071                        Stoichiometry stoichiometry)
072        {
073                if(stoichiometry != null)
074                        this.componentStoichiometry.add(stoichiometry);
075        }
076
077        public void removeComponentStoichiometry(
078                        Stoichiometry stoichiometry)
079        {
080                if(stoichiometry != null)
081                        this.componentStoichiometry.remove(stoichiometry);
082        }
083
084        protected void setComponentStoichiometry(Set<Stoichiometry> stoichiometry)
085        {
086                this.componentStoichiometry = stoichiometry;
087        }
088
089        public Set<SimplePhysicalEntity> getSimpleMembers()
090        {
091                return getSimpleMembers(BPCollections.I.<SimplePhysicalEntity>createSet());
092        }
093
094        protected Set<SimplePhysicalEntity> getSimpleMembers(Set<SimplePhysicalEntity> set)
095        {
096                for (PhysicalEntity pe : this.getComponent())
097                {
098                        collectSimpleMembersRecursive(pe, set);
099                }
100                return set;
101        }
102
103        protected void collectSimpleMembersRecursive(PhysicalEntity pe, Set<SimplePhysicalEntity> set)
104        {
105                if (pe instanceof ComplexImpl && pe != this)
106                {
107                        ((ComplexImpl) pe).collectSimpleMembersRecursive(pe, set);
108                }
109                else if (pe instanceof SimplePhysicalEntity)
110                {
111                        set.add((SimplePhysicalEntity) pe);
112                }
113                for (PhysicalEntity mem : pe.getMemberPhysicalEntity())
114                {
115                        collectSimpleMembersRecursive(mem, set);
116                }
117        }
118
119        public Set<EntityReference> getMemberReferences()
120        {
121                Set<EntityReference> set = BPCollections.I.createSet();
122
123                for (SimplePhysicalEntity spe : getSimpleMembers())
124                {
125                        EntityReference er = spe.getEntityReference();
126                        if(er!=null) set.add(er);
127                }
128                return set;
129        }
130
131        @Override
132        protected boolean semanticallyEquivalent(BioPAXElement element)
133        {
134                return (element instanceof Complex) &&
135                                SetEquivalenceChecker.isEquivalent(this.getComponent(), ((Complex) element).getComponent())
136                                        && super.semanticallyEquivalent(element);
137        }
138}
