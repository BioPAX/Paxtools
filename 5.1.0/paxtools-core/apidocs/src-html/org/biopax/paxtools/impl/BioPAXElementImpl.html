001package org.biopax.paxtools.impl;
002
003import org.biopax.paxtools.model.BioPAXElement;
004import org.biopax.paxtools.util.BPCollections;
005
006import java.util.Map;
007
008public abstract class BioPAXElementImpl implements BioPAXElement
009{
010        private String uri;
011        
012        // anything extra can be stored in this map (not to persist in a DB usually)
013        private Map<String, Object> annotations;
014        
015        public BioPAXElementImpl() {
016                this.annotations = BPCollections.I.createMap();
017        }
018
019    public boolean isEquivalent(BioPAXElement element)
020    {
021        return this.equals(element) || this.semanticallyEquivalent(element);
022    }
023
024    protected boolean semanticallyEquivalent(BioPAXElement element)
025    {
026        return false;
027    }
028
029    public int equivalenceCode()
030    {
031        return uri.hashCode();
032    }
033
034    public String getUri()
035    {
036        return uri;
037    }
038
039    /**
040     * Private setter for the element's absolute URI.
041     * 
042     * Normally, URI should never be modified 
043     * after the object is created unless you know 
044     * what you're doing (and can use Java Reflection).
045     * 
046     * @param uri new absolute URI
047     */
048    @SuppressWarnings("unused")
049        private void setUri(String uri)
050    {
051        if(uri == null)
052                throw new IllegalArgumentException();
053        
054        this.uri = uri;
055    }
056
057    
058    public String toString()
059    {
060        return uri;
061    }
062
063
064    public Map<String, Object> getAnnotations() {
065                return annotations;
066        }
067    
068        
069        /**
070         * true if and only if the other obj has the same biopax type 
071         * (same {@link #getModelInterface()}, not a subclass) and 
072         * same URI. Other properties are not considered.
073         * 
074         */
075        @Override
076        public boolean equals(Object obj) {
077                return (obj instanceof BioPAXElement) 
078                        && this.getModelInterface() == ((BioPAXElement) obj).getModelInterface()
079                        && this.uri.equals(((BioPAXElement) obj).getUri());
080        }
081        
082        
083        /**
084         * This method is consistent with the 
085         * overridden {@link #equals(Object)} method
086         * (biopax type and URI are what matters) 
087         */
088        @Override
089        public int hashCode() {
090                return (getModelInterface() + uri).hashCode();
091        }
092
093}
