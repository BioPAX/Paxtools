006/**

009 * @see NucleicAcidRegionReference

011public interface NucleicAcidReference extends SequenceEntityReference
012{
013   Set<NucleicAcidRegionReference> getSubRegion();
014
015        void addSubRegion(NucleicAcidRegionReference regionReference);
016
017        void removeSubRegion(NucleicAcidRegionReference regionReference);
018}
