001package org.biopax.paxtools.model;
002
003
004import java.io.Serializable;
005import java.util.Map;
006import java.util.Set;
007
008/**
009 * A model acts as a container for BioPAX elements.
010 * Every object within a mode must have unique id.
011 * A single object can be contained in multiple models.
012 */
013public interface Model extends Serializable
014{
015    /**
016     * URI of this BioPAX Model.
017     *
018     * This can be used in many ways to refer to this model when:
019     *  - logging;
020     *  - merging several models;
021     *  - converting to or from another data format, such as GMT (GSEA);
022     *  - saving to or reading from RDF/XML file (e.g., save/read a special Provenance object)
023     *
024     * @return model's uri
025     */
026    String getUri();
027
028    void setUri(String modelUri);
029
030    /**
031     * URI of this BioPAX Model.
032     *
033     * This can be used in many ways; see {@link #getUri()}.
034     *
035     * @return model's name
036     */
037    String getName();
038
039    void setName(String modelName);
040
041    /**
042     * This method adds the given object to this model. If the object
043     * points to other objects that are not in the model, it is user's
044     * responsibility to add them into the model as well.
045     * If an object with the same id already exists, it will throw
046     * an {@link org.biopax.paxtools.util.IllegalBioPAXArgumentException}
047     *
048     * @param aBioPAXElement to be added
049     */
050
051    void add(BioPAXElement aBioPAXElement);
052
053
054    /**
055     * This method creates a new object using the model's factory, adds it
056     * to the model and returns it.
057     *
058     * @param <T> a BioPAX type
059     * @param aClass the BioPAX model interface class
060     * @param id     of the new object
061     * @return newly created object
062     */
063    <T extends BioPAXElement> T addNew(Class<T> aClass, String id);
064
065    /**
066     * This method returns true if the parameter is contained within
067     * this model. 
068     * 
069     * Note: the result can be 'false' when 
070     * {@link #containsID(String)} is 'true'
071     * (using the URI of the parameter) if, e.g., 
072     * model contains another object with the same URI.
073     * 
074     * @see #containsID(String)
075     *
076     * @param aBioPAXElement to be checked
077     * @return true if the parameter is in the object set
078     */
079    boolean contains(BioPAXElement aBioPAXElement);
080
081    /**
082     * This method returns the biopax element with the given id,
083     * returns null if the object with the given id does not exist
084     * in this model.
085     * @param id of the object to be retrieved.
086     * @return biopax element with the given id.
087     */
088    BioPAXElement getByID(String id);
089
090    /**
091     * This method checks for the biopax element with the given id,
092     * returns true if the object with the given id exists.
093     * in this model.
094     * @param id of the object to be retrieved.
095     * @return biopax element with the given id.
096     */
097    boolean containsID(String id);
098
099
100    /**
101     * This method returns a map of name space prefixes.
102     * This map can be modified.
103     * @return a map, mapping prefixes to full namespaces.
104     */
105    Map<String, String> getNameSpacePrefixMap();
106
107// --------------------- ACCESORS and MUTATORS---------------------
108
109    /**
110     * This method returns a set of objects in the model.
111     * Contents of this set can not be modified.
112     * @return an unmodifiable set of objects.
113     */
114    Set<BioPAXElement> getObjects();
115
116    /**
117     * This method returns a set of objects in the model of the given class.
118     * Contents of this set should not be modified.
119     *
120     * @param <T> a BioPAX type
121     * @param filterBy class to be used as a filter.
122     * @return an unmodifiable set of objects of the given class.
123     */
124    <T extends BioPAXElement> Set<T> getObjects(Class<T> filterBy);
125
126    /**
127     * This method removes the given BioPAX Element from the model.
128     * Other objects in the model can still point to this object.
129     * It is user's responsibility to properly excise these properties.
130     * @param aBioPAXElement to be removed.
131     */
132    void remove(BioPAXElement aBioPAXElement);
133
134    /**
135     * This method sets the factory this model will use for creating
136     * BioPAX objects. For example {@link #addNew(Class, String)} method
137     * uses this factory.
138     *
139     * @param factory this model will use for creating
140     * BioPAX objects.
141     */
142    void setFactory(BioPAXFactory factory);
143
144   /**
145     * This method returns the level of the objects that are
146     * contained within this model.
147     * @return level of the objects within this model.
148     */
149    BioPAXLevel getLevel();
150
151    /**
152     * When set to false, the model will not check for the dependent
153     * objects of a newly added object. When true it will traverse and
154     * add all dependent objects that are not already in the model.
155     * This feature is currently experimental.
156     *
157     * @param value defining the dependency adding behaviour
158     */
159    void setAddDependencies(boolean value);
160
161    /**
162     * When addDependencies is false, the model will not check for the dependent
163     * objects of a newly added object. When true it will traverse and
164     * add all dependent objects that are not already in the model.
165     * This feature is currently experimental.
166     *
167     * @return whether adding dependencies.
168     */
169    boolean isAddDependencies();
170
171    
172    /**
173     * Merges the source model into this one.
174     * 
175     * @param source a model to merge
176     */
177    void merge(Model source);
178    
179    
180    /**
181     * Replaces existing BioPAX element with another one,
182     * of the same or possibly equivalent type,
183     * and updates all the affected references to it (object properties).
184     * 
185     * @param existing object to be replaced
186     * @param replacement the replacement BioPAX object
187     */
188    void replace(BioPAXElement existing, BioPAXElement replacement);
189    
190    
191    /**
192     * Attempts to repair the model,
193     * i.e., make it self-consistent, integral.
194     */
195    void repair();
196
197    
198    /**
199     * Sets the xml:base to use when exporting a BioPAX model.
200     * Usually, is is a data-provider's URI prefix, e.g.,
201     * xml:base="http://www.pantherdb.org/pathways/biopax#" 
202     * Setting this to null makes the exporter print using absolute
203     * URIs (and rdf:about) instead of relative ones (and rdf:ID).
204     * 
205     * @param base a URI prefix or null.
206     */
207     void setXmlBase(String base);
208
209     
210     /**
211      * Gets the model's xml:base (URI prefix/namespace), which
212      * normally the majority of the BioPAX object's absolute URIs
213      * in the model begin with.
214      *
215      * Note: it's not required that all the BioPAX objects
216      * in the model have the same URI prefix/namespace;
217      * e.g., there are can be (and perfectly legal) objects
218      * that use other URI bases, such as http://identifiers.org/,
219      * http://purl.org/, etc. (- usually these are well-known
220      * standard xml bases, or these result from merging several BioPAX
221      * models of different data providers into one model.)
222      *
223      * @return xml:base value
224      */
225     String getXmlBase();
226}
