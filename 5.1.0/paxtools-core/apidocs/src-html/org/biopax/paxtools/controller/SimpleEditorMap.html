015/**
016 * Provides a simple editor map for a level with a given factory.
017 * 
018 * This class initializes 3 singletons( 1 for each level) from a tab delimited text resources that lists the
019 * properties and their domains. This is done to remove any dependencies to Jena.
020 * 
021 * The recommended usage is to use the {@link #get(org.biopax.paxtools.model.BioPAXLevel)} method.
022 * @author Emek Demir
023 */
024public enum SimpleEditorMap implements EditorMap
025{
026
027        L2(BioPAXLevel.L2),
028        L3(BioPAXLevel.L3);
029
030        private static final Logger log = LoggerFactory.getLogger(EditorMapImpl.class);
031
032        private final SimpleEditorMapImpl impl;
033
034
035        SimpleEditorMap(BioPAXLevel level)
036        {
037                this.impl = new SimpleEditorMapImpl(level);
038        }
039
040        /**
041         * To obtain a copy of the editor map for the corresponding level, use the
042         * @param level the BioPAX level
043         * @return SimpleEditorMap implementation for the BioPAX L2 or L3
044         */
045        public static SimpleEditorMap get(BioPAXLevel level)
046        {
047                for (SimpleEditorMap value : values())
048                {
049                        if (value.getLevel().equals(level)) return value;
050                }
051                //should never reach here
052                throw new IllegalBioPAXArgumentException("Unknown level:" + level);
053        }
054
055        static class SimpleEditorMapImpl extends EditorMapImpl implements EditorMap
056        {
057                SimpleEditorMapImpl(BioPAXLevel level)
058                {
059                        super(level);
060                        InputStream stream = this.getClass().getResourceAsStream(level + "Editor.properties");
061                        readEditors(level, stream, this);
062                }
063        }
064
065
066        public <D extends BioPAXElement> PropertyEditor<? super D, ?> getEditorForProperty(String property,
067                        Class<D> javaClass)
068
069        {
070                return impl.getEditorForProperty(property, javaClass);
071        }
072
073        @Override public Set<PropertyEditor> getEditorsForProperty(String property)
074        {
075                return impl.getEditorsForProperty(property);
076        }
077
078        @Override
079        public <D extends BioPAXElement> Set<PropertyEditor<? extends D, ?>> getSubclassEditorsForProperty(String property,
080                        Class<D> domain)
081        {
082                return impl.getSubclassEditorsForProperty(property, domain);
083        }
084
085        @Override public Set<PropertyEditor> getEditorsOf(BioPAXElement bpe)
086        {
087                return impl.getEditorsOf(bpe);
088
089        }
090
091        @Override public Set<ObjectPropertyEditor> getInverseEditorsOf(BioPAXElement bpe)
092        {
093                return impl.getInverseEditorsOf(bpe);
094        }
095
096        @Override public <E extends BioPAXElement> Set<? extends Class<E>> getKnownSubClassesOf(Class<E> javaClass)
097        {
098                return impl.getKnownSubClassesOf(javaClass);
099        }
100
101
102        public BioPAXLevel getLevel()
103        {
104                return impl.getLevel();
105
106        }
107
108        @Override public Set<PropertyEditor> getEditorsOf(Class<? extends BioPAXElement> domain)
109        {
110                return impl.getEditorsOf(domain);
111        }
112
113        @Override public Set<ObjectPropertyEditor> getInverseEditorsOf(Class<? extends BioPAXElement> domain)
114        {
115                return impl.getInverseEditorsOf(domain);
116        }
117
118        @Override
119        public Iterator<PropertyEditor> iterator()
120        {
121                return impl.iterator();
122        }
123
124
125        static void readEditors(BioPAXLevel level, InputStream stream, EditorMapImpl map)
126        {
127                BufferedReader reader = new BufferedReader(new InputStreamReader(stream));
128                try
129                {
130                        String line = reader.readLine();
131                        StringTokenizer st = new StringTokenizer(line);
132                        while (st.hasMoreElements())
133                        {
134                                map.registerModelClass(st.nextToken());
135                        }
136
137                        while ((line = reader.readLine()) != null)
138                        {
139                                if (!line.startsWith("#"))
140                                {
141                                        st = new StringTokenizer(line);
142                                        String domain = st.nextToken();
143                                        Class<? extends BioPAXElement> domainInterface = map.getLevel().getInterfaceForName(domain);
144
145                                        String propertyName = st.nextToken();
146                                        Map<Class<? extends BioPAXElement>, Set<Class<? extends BioPAXElement>>> rangeRestrictions =
147                                                        new HashMap<Class<? extends BioPAXElement>, Set<Class<? extends BioPAXElement>>>();
148                                        while (st.hasMoreTokens())
149                                        {
150                                                String rToken = st.nextToken();
151                                                if (rToken.startsWith("R:"))
152                                                {
153                                                        StringTokenizer rt = new StringTokenizer(rToken.substring(2), "=");
154                                                        Class<? extends BioPAXElement> rDomain = level.getInterfaceForName(rt.nextToken());
155                                                        Set<Class<? extends BioPAXElement>> rRanges = new HashSet<Class<? extends
156                                                                        BioPAXElement>>();
157                                                        for (StringTokenizer dt = new StringTokenizer(rt.nextToken(), ","); dt.hasMoreTokens(); )
158                                                        {
159                                                                rRanges.add(level.getInterfaceForName(dt.nextToken()));
160                                                        }
161                                                        rangeRestrictions.put(rDomain, rRanges);
162                                                }
163
164                                        }
165                                        map.createAndRegisterBeanEditor(propertyName, domainInterface, rangeRestrictions);
166                                }
167                        }
168                }
169                catch (IOException e)
170                {
171                        log.error("Could not initialize " + "Editor Map", e);
172                }
173                finally
174                {
175                        try
176                        {
177                                stream.close();
178                        }
179                        catch (IOException ignore)
180                        {
181                                log.error("Could not close stream! Exiting");
182                                System.exit(1);
183                        }
184                }
185
186        }
187
188        public static EditorMap buildCustomEditorMap(BioPAXLevel level, InputStream stream)
189        {
190                EditorMapImpl editorMap = new EditorMapImpl(level);
191                readEditors(level,stream,editorMap);
192                return editorMap;
193
194        }
