001package org.biopax.paxtools.controller;
002
003import org.biopax.paxtools.model.BioPAXElement;
004import org.biopax.paxtools.model.BioPAXFactory;
005import org.biopax.paxtools.model.Model;
006import org.slf4j.Logger;
007import org.slf4j.LoggerFactory;
008
009import java.util.HashSet;
010import java.util.Set;
011
012/**
013 * Specifically "Clones" the BioPAX elements set
014 * (traverses to obtain dependent elements), 
015 * puts them to the new model using the visitor and traverser framework;
016 * ignores elements that are not in the source list (compare to {@link Fetcher}).
017 *
018 * @see org.biopax.paxtools.controller.Visitor
019 * @see org.biopax.paxtools.controller.Traverser
020 */
021public class Cloner implements Visitor
022{
023        private static final Logger LOG = LoggerFactory.getLogger(Cloner.class);
024        
025        private final Traverser traverser;
026        private final BioPAXFactory factory;
027
028        public Cloner(EditorMap map, BioPAXFactory factory)
029        {
030                this.factory = factory;
031                this.traverser = new Traverser(map, this);
032        }
033
034        /**
035         * For each element from the 'toBeCloned' list,
036         * it creates a copy in the new model, setting all
037         * the data properties; however, object property values
038         * that refer to BioPAX elements not in 'toBeCloned' list
039         * are ignored.
040         *
041         * @deprecated use {@link #clone(Set)} instead ('source' model was not used in previous versions)
042         *
043         * @param source - actually, is never used... can be null all the same
044         * @param toBeCloned elements to clone
045         * @return a new model containing the cloned biopax objects
046         */
047        public Model clone(Model source, Set<BioPAXElement> toBeCloned) {
048                return clone(toBeCloned);
049        }
050
051        /**
052         * For each element from the 'toBeCloned' list,
053         * it creates a copy in the new model, setting all 
054         * the data properties; however, object property values
055         * that refer to BioPAX elements not in 'toBeCloned' list
056         * are ignored. As the result, e.g., a Pathway can be incomplete,
057         * missing some or all its sub-processes, xrefs, pathway steps.
058         * 
059         * @param toBeCloned elements to clone
060         * @return a new model containing the cloned biopax objects
061         */
062        public Model clone(Set<BioPAXElement> toBeCloned)
063        {
064                final Model targetModel = factory.createModel();
065
066                for (BioPAXElement bpe : new HashSet<BioPAXElement>(toBeCloned))
067                {
068                        // create a new empty object using the URI (all the biopax properties are empty;
069                        // addNew fails when there are several objects having the same URI)
070                        targetModel.addNew(bpe.getModelInterface(), bpe.getUri());
071                }
072
073                // a hack to avoid unnecessary checks for the valid sub-model being cloned, 
074                // and warnings when the Cloner copies BPS.stepProcess values, 
075                // and there is a Conversion among them (-always unless stepConversion is null).
076                AbstractPropertyEditor.checkRestrictions.set(false);
077                
078                for (BioPAXElement bpe : toBeCloned)
079                {
080                        traverser.traverse(bpe, targetModel);
081                }
082                
083                AbstractPropertyEditor.checkRestrictions.set(true); //back to the default mode
084
085                return targetModel;
086        }
087
088        //  Implement interface: Visitor
089        public void visit(BioPAXElement domain, Object range, Model targetModel, PropertyEditor editor)
090        {
091                BioPAXElement targetDomain = targetModel.getByID(domain.getUri());
092                
093                if (range instanceof BioPAXElement)
094                {
095                        BioPAXElement bpe = (BioPAXElement) range;
096                        BioPAXElement existing = targetModel.getByID(bpe.getUri());
097                        //set the property value if the value is already present in the target 
098                        if (existing != null) {
099                                editor.setValueToBean(existing, targetDomain);
100                        }
101                }
102                else
103                {
104                        editor.setValueToBean(range, targetDomain);
105                }
106        }
107}
