071        /**
072         * Replaces BioPAX elements in the model with ones from the map,
073         * updates corresponding BioPAX object references.
074         * 
075         * It does not neither remove the old nor add new elements in the model
076         * (if required, one can do this before/after this method, e.g., using
077         * the same 'subs' map)
078         * 
079         * This does visit all object properties of each "explicit" element
080         * in the model, but does not traverse deeper into one's sub-properties
081         * to replace something there as well (e.g., nested member entity references
082         * are not replaced unless parent entity reference present in the model)
083         * 
084         * This does not automatically move/migrate old (replaced) object's
085         * children to new objects (the replacement ones are supposed to have
086         * their own properties already set or to be set shortly; otherwise,
087         * consider using of something like {@link #fixDanglingInverseProperties(BioPAXElement, Model)} after.
088         * 
089         * @param model biopax model where the objects are to be replaced
090         * @param subs the replacements map (many-to-one, old-to-new)
091         * @exception IllegalBioPAXArgumentException if there is an incompatible type replacement object
092         */
093        public static void replace(Model model, final Map<? extends BioPAXElement, ? extends BioPAXElement> subs)
094        {
095                // update properties
096                Visitor visitor = new Visitor()
097                {
098                        @Override
099                        public void visit(BioPAXElement domain, Object range, Model model, PropertyEditor editor)
100                        {
101                                if (editor instanceof ObjectPropertyEditor && range != null && subs.containsKey(range))
102                                {
103                                        BioPAXElement value = (BioPAXElement) range;
104                                        // 'value' is to be replaced with the 'replacement'
105                                        BioPAXElement replacement = subs.get(range); //can get null (ok)
106                                        
107                                        // normal biopax property -
108                                        if (replacement != null && !editor.getRange().isInstance(replacement))
109                                        {
110                                                throw new IllegalBioPAXArgumentException(
111                                                        "Incompatible type! Attempted to replace " 
112                                                        + value.getUri() + " (" + value.getModelInterface().getSimpleName()
113                                                        + ") with " + replacement.getUri() + " ("
114                                                        + replacement.getModelInterface().getSimpleName() + "); "
115                                                        + "property: " + editor.getProperty() 
116                                                        + " of bean: " + domain.getUri() + " ("
117                                                        + domain.getModelInterface().getSimpleName() + ")");
118                                        }
119
120                                        if (replacement != value) 
121                                        {
122                                                editor.removeValueFromBean(value, domain);
123                                                editor.setValueToBean(replacement, domain);
124                                        } else {
125                                                LOG.debug("replace: skipped the identical: " + replacement.getUri());
126                                        }
127                                }
128                        }
129                };
130
131                Traverser traverser = new Traverser(em, visitor);
132                for (BioPAXElement bpe : new HashSet<BioPAXElement>(model.getObjects()))
133                {
134                        // update object properties and clear inverse properties using 'subs' map       
135                        traverser.traverse(bpe, null); //model is not needed
136                }
137        }

140        /**
141         * Finds "root" BioPAX objects that belong to a particular class (incl. sub-classes)
142         * in the model.
143         * 
144         * Note: however, such "root" elements may or may not be, a property of other
145         * elements, not included in the model.
146         * @param model biopax model to work with
147         * @param filterClass filter class (including subclasses)
148         * @param <T> biopax type
149         * @return set of the root biopax objects of given type
150         */
151        public static <T extends BioPAXElement> Set<T> getRootElements(final Model model, final Class<T> filterClass)
152        {
153                // copy all such elements (initially, we think all are roots...)
154                final Set<T> result = new HashSet<T>(model.getObjects(filterClass));
155
156                //"shallow" traverser (direct object properties only - Visitor.visit does not call traverse again) 
157                @SuppressWarnings("unchecked")
158                Traverser traverser = new Traverser(em, 
159                        new Visitor() {
160                                @Override
161                                public void visit(BioPAXElement parent, Object value, Model model,
162                                        PropertyEditor<?, ?> editor)
163                                {
164                                        if (filterClass.isInstance(value)) result.remove(value);
165                                }
166                        }, 
167                        new Filter<PropertyEditor>() {
168                                @Override
169                                public boolean filter(PropertyEditor pe) {
170                                        return (pe instanceof ObjectPropertyEditor);
171                                }
172                });
173                
174                // but we run from every element (all types)
175                for(BioPAXElement e : model.getObjects())
176                        traverser.traverse(e, null);
177
178                return result;
179        }

182        /**
183         * Iteratively removes "dangling" elements of given type and its sub-types,
184         * e.g. Xref.class objects, from the BioPAX model. 
185         * 
186         * If the "model" does not contain any root Entity class objects,
187         * and the second parameter is basic UtilityClass.class (i.e., not its sub-class), 
188         * then it simply logs a warning and quits shortly (otherwise, it would 
189         * remove everything from the model). Do not use basic Entity.class either
190         * (but a sub-class is OK) for the same reason (it would delete everything).
191         * 
192         * This, however, does not change relationships
193         * among objects, particularly, some inverse properties,
194         * such as entityReferenceOf or xrefOf, may still
195         * refer to a removed object.
196         * @param model to modify
197         * @param clazz filter-class (filter by this type and sub-classes)
198         * @param <T> biopax type
199         * @return removed objects
200         */
201        public static <T extends BioPAXElement> Set<BioPAXElement> removeObjectsIfDangling(Model model, Class<T> clazz)
202        {
203                final Set<BioPAXElement> removed = new HashSet<BioPAXElement>();
204                
205                // 'equals' below is used intentionally (isAssignableFrom() would be incorrect)
206                if(Entity.class.equals(clazz)) {
207                        LOG.warn("Ignored removeObjectsIfDangling call for: " +
208                                        "Entity.class (it would delete all)");
209                        return removed;
210                }
211                if(UtilityClass.class.equals(clazz) 
212                                && getRootElements(model, Entity.class).isEmpty()) 
213                {
214                        LOG.warn("Ignored removeObjectsIfDangling call: " +
215                                        "no root entities model; UtilityClass.class");
216                        return removed;
217                }
218                
219                Set<T> dangling = getRootElements(model, clazz);        
220                
221                // get rid of dangling objects
222                if (!dangling.isEmpty())
223                {
224                        LOG.info(dangling.size() + " " + clazz.getSimpleName() +
225                                " dangling objects will be deleted...");
226
227                        for (BioPAXElement thing : dangling)
228                        {
229                                model.remove(thing);
230                                removed.add(thing);
231                                LOG.debug("removed (dangling) " + thing.getUri() + " ("
232                                        + thing.getModelInterface().getSimpleName() + ") " + thing);
233                        }
234
235                        // some may have become dangling now, so check again...
236                        removed.addAll(removeObjectsIfDangling(model, clazz));
237                }
238                
239                return removed;
240        }

243        /**
244         * Cuts the BioPAX model off other models and BioPAX objects
245         * by essentially performing write/read to/from OWL.
246         * The resulting model contains new objects with same IDs
247         * and have object properties "fixed", i.e., dangling values
248         * become null/empty, and inverse properties (e.g. xrefOf)
249         * re-calculated. The original model is unchanged.
250         * 
251         * Note: this method will fail for very large models 
252         * (if resulting RDF/XML utf8 string is longer than approx. 1Gb)
253         * 
254         * @param model biopax model to process
255         * @return copy of the model
256         */
257        public static Model writeRead(Model model)
258        {
259                BioPAXIOHandler io = new SimpleIOHandler(model.getLevel());
260                ByteArrayOutputStream baos = new ByteArrayOutputStream();
261                io.convertToOWL(model, baos);
262                return io.convertFromOWL(new ByteArrayInputStream(baos.toByteArray()));
263        }

389        /**
390         * Calculates MD5 hash code (as 32-byte hex. string).
391         * 
392         * This method is not BioPAX specific. Can be
393         * used for many purposes, such as generating 
394         * new unique URIs, database primary keys, etc.
395         * 
396         * 
397         * @param id some identifier, e.g., URI
398         * @return the 32-byte digest string
399         */
400        public static String md5hex(String id)
401        {
402                byte[] digest = MD5_DIGEST.digest(id.getBytes());
403                StringBuffer sb = new StringBuffer();
404                for (byte b : digest)
405                {
406                        sb.append(Integer.toHexString((int) (b & 0xff) | 0x100).substring(1, 3));
407                }
408                String hex = sb.toString();
409                return hex;
410        }

416         * 
417         * @param bpe a biopax object
418         * @param model the model to look for objects in

420        public static void fixDanglingObjectProperties(BioPAXElement bpe, Model model)
421        {
422                final Visitor visitor = new Visitor()

425                        public void visit(BioPAXElement domain, Object range, Model model, PropertyEditor editor)

431                                                editor.removeValueFromBean(value, domain);
432                                }
433                        }
434                };
435
436                Traverser traverser = new Traverser(em, visitor);
437                traverser.traverse(bpe, model);
438        }

485        public static Set<EntityFeature> getFeatureSetByType(PhysicalEntity pe, FeatureType type)
486        {
487
488                Set<EntityFeature> modifiableSet = new HashSet<EntityFeature>();
489
490                switch (type)
491                {
492                        case FEATURE:
493                                modifiableSet.addAll(pe.getFeature());
494                                break;
495                        case NOT_FEATURE:
496                                modifiableSet.addAll(pe.getNotFeature());
497                                break;
498                        case UNKNOWN_FEATURE:
499                        {
500                                if (pe instanceof SimplePhysicalEntity)
501                                {
502                                        modifiableSet.addAll(((SimplePhysicalEntity) pe).getEntityReference().getEntityFeature());
503                                        modifiableSet.removeAll(pe.getFeature());
504                                        modifiableSet.removeAll(pe.getNotFeature());
505                                }
506                        }
507                }
508                return modifiableSet;
509        }

512        /**
513         * Finds and adds all (missing) entity features 
514         * to given entity reference from all its owner 
515         * simple physical entities ('feature' and 'notFeature' 
516         * properties).
517         * 
518         * Though, it neither checks for nor resolves any violations 
519         * of the 'entityFeature' property's inverse functional constraint
520         * (i.e., an EntityFeature instance can only belong to one and only one
521         * EntityReference object).  
522         * 
523         * @param er entity reference object
524         * @param fix flag
525         * @return true or false
526         */
527        public static boolean checkERFeatureSet(EntityReference er, boolean fix)
528        {
529                boolean check = true;
530                for (SimplePhysicalEntity spe : er.getEntityReferenceOf())
531                {
532                        for (EntityFeature ef : spe.getFeature())
533                        {
534                                check = scanAndAddToFeatureSet(er, fix, check, ef);
535                                // if not fixing return at first fail, otherwise go on;
536                                if (!fix && !check) return check;
537                        }
538                        for (EntityFeature ef : spe.getNotFeature())
539                        {
540                                check = scanAndAddToFeatureSet(er, fix, check, ef);
541                                // if not fixing return at first fail, otherwise go on;
542                                if (!fix && !check) return check;
543                        }
544                }
545                return check;
546        }

548        private static boolean scanAndAddToFeatureSet(EntityReference er, boolean fix, boolean check, EntityFeature ef)
549        {
550                if (!er.getEntityFeature().contains(ef))
551                {
552                        check = false;
553                        if (fix)
554                        {
555                                er.addEntityFeature(ef);
556                                //TODO resolve inverse functional prop. constraint violation (e.g., copy/replace the e.f. before adding if it has entityFeatureOf not null)?
557                        }
558                }
559                return check;
560        }

562        // TODO annotate
563        public static Set<EntityFeature> findFeaturesAddedToSecond(PhysicalEntity first, PhysicalEntity second,
564                        boolean fix)
565        {
566
567                if (checkCommonEntityReferenceForTwoPEs(first, second, fix)) return null;
568                Set<EntityFeature> explicit =
569                                getFeatureIntersection(first, FeatureType.NOT_FEATURE, second, FeatureType.FEATURE);
570                Set<EntityFeature> implicit =
571                                getFeatureIntersection(first, FeatureType.UNKNOWN_FEATURE, second, FeatureType.FEATURE);
572                Set<EntityFeature> negativeImplicit =
573                                getFeatureIntersection(first, FeatureType.NOT_FEATURE, second, FeatureType.UNKNOWN_FEATURE);
574
575                if (fix)
576                {
577                        for (EntityFeature implied : implicit)
578                        {
579                                LOG.info("The feature " + implied + "implied as a not-feature of " + first + ". " +
580                                         "Adding it to the not-feature list");
581                                first.addNotFeature(implied);
582                        }
583
584                        for (EntityFeature implied : negativeImplicit)
585                        {
586                                LOG.info("The feature " + implied + "implied as a feature of " + second + ". " +
587                                         "Adding it to the feature list");
588                                second.addFeature(implied);
589                        }
590
591                }
592                explicit.retainAll(implicit);
593                explicit.retainAll(negativeImplicit);
594                return explicit;
595        }

597        private static boolean checkCommonEntityReferenceForTwoPEs(PhysicalEntity first, PhysicalEntity second,
598                        boolean fix)
599        {
600                if (first instanceof SimplePhysicalEntity)
601                {
602                        EntityReference er = ((SimplePhysicalEntity) first).getEntityReference();
603                        if (!er.getEntityReferenceOf().contains(second))
604                        {
605                                LOG.warn("These two physicalEntities do not share an EntityReference. They can not be compared! " +
606                                         "Skipping");
607                                return false;
608                        } else if (!checkERFeatureSet(er, fix))
609                        {
610                                LOG.warn("ER feature set is incomplete!");
611                                if (!fix)
612                                {
613                                        LOG.warn("fixing...");
614                                } else
615                                {
616                                        LOG.warn("skipping");
617                                        return false;
618                                }
619                        }
620                        return true;
621                } else
622                {
623                        LOG.warn("These two physicalEntities do not share an EntityReference. They can not be compared! " +
624                                        "Skipping");
625                        return false;
626                }
627

672                final Map<Set<EntityReference>,EntityReference> memberMap = new HashMap<Set<EntityReference>,EntityReference>();
673                for (SimplePhysicalEntity pe : simplePEsToDo)
674                {
675                        try {
676                                createGenericEntityRef(model, pe, memberMap);
677                        }
678                        catch (Exception e) {
679                                LOG.error("createGenericEntityRef failed at " + pe.getUri(), e);
680                        }
681                }
682                //TODO: unlink and remove all pe.memberPhysicalEntity for each pe in simplePEsToDo?
683        }

685        private static void createGenericEntityRef(Model model, SimplePhysicalEntity spe,
686                        final Map<Set<EntityReference>, EntityReference> memberMap)
687        {
688                Set<EntityReference> members = spe.getGenericEntityReferences();
689                EntityReference er = memberMap.get(members);
690                if (er == null)
691                {
692                        //find an ER, if any...
693                        EntityReference firstEntityReference = spe.getEntityReference(); //null, but worth double-checking
694                        if(firstEntityReference == null) {
695                                for (PhysicalEntity pe : spe.getMemberPhysicalEntity()) {
696                                        //safe to cast to SimplePhysicalEntity, thanks to the memberPE property range restriction
697                                        firstEntityReference = ((SimplePhysicalEntity) pe).getEntityReference();
698                                        if (firstEntityReference != null)
699                                                break; //got one!
700                                }
701                        } else {
702                                return; //spe already has a ER
703                                //TODO: shall we add members (spe.getGenericEntityReferences()) to firstEntityReference?
704                        }
705
706                        if (firstEntityReference != null)
707                        {
708                                //generate a new URI in the same namespace (xml:base) and create and add a new EntityReference

710                                                firstEntityReference.getModelInterface().getSimpleName() + "_"+ md5hex(spe.getUri());

728                                memberMap.put(members, er);

789         * Copies names, xrefs, comments, evidence, data sources from source to target biopax object;
790         * it does not copy unification xrefs but instead adds relationship xrefs using the same
791         * db and id values as source's unification xrefs.
792         * 
793         * @param model the biopax model where the source and target objects belong
794         * @param source from
795         * @param target to
796         */
797        private static void copySimplePointers(Model model, Named source, Named target)
798        {
799                //copy names
800                if(source.getDisplayName()!=null)
801                        target.setDisplayName(source.getDisplayName());
802                if(source.getStandardName()!=null)
803                        target.setStandardName(source.getStandardName());
804                for (String name : source.getName()) {
805                        target.addName(name);
806                }
807
808                // copy xrefs, converting UXs to RXs on the go
809                for (Xref xref : source.getXref())
810                {
811                        if (xref instanceof UnificationXref)
812                        {
813                                // generate URI using model's xml:base and xref's properties
814                                String uri = model.getXmlBase() + "RelationshipXref_"+ md5hex(xref.getDb()+xref.getId());
815                                Xref byID = (Xref) model.getByID(uri);
816                                if (byID == null)
817                                {
818                                        RelationshipXref rref = model.addNew(RelationshipXref.class, uri);
819                                        rref.setDb(xref.getDb());
820                                        rref.setId(xref.getId());
821                                        rref.setDbVersion(xref.getDbVersion());
822                                        rref.setIdVersion(xref.getDbVersion());
823                                        xref = rref;
824                                } else
825                                {
826                                        xref = byID;
827                                }
828                        }
829                        target.addXref(xref);
830                }
831
832                //copy evidence and dataSource if (Named) source and target are same sub-type - either Entity or ER only:
833                if(source instanceof Entity && target instanceof Entity) {
834                        Entity src = (Entity) source;
835                        for (Evidence ev : src.getEvidence()) {
836                                ((Entity) target).addEvidence(ev);
837                        }
838                        for (Provenance prov : src.getDataSource()) {
839                                ((Entity) target).addDataSource(prov);
840                        }
841                        // copy comments
842                        for(String comm : source.getComment()) {
843                                target.addComment(comm);
844                        }
845                } else if(source instanceof EntityReference && target instanceof EntityReference) {
846                        EntityReference src = (EntityReference) source;
847                        for (Evidence ev : src.getEvidence()) {
848                                ((EntityReference) target).addEvidence(ev);
849                        }
850                        // copy comments
851                        for(String comm : source.getComment()) {
852                                target.addComment(comm);
853                        }
854                }

858        /**
859         * This method iterates over the features in a model and tries to find equivalent objects and merges them.
860         * @param model to be fixed
861         */
862        public static void replaceEquivalentFeatures(Model model)
863        {
864
865                EquivalenceGrouper<EntityFeature> equivalents = new EquivalenceGrouper<EntityFeature>();
866                HashMap<EntityFeature, EntityFeature> mapped = new HashMap<EntityFeature, EntityFeature>();
867                HashSet<EntityFeature> scheduled = new HashSet<EntityFeature>();
868
869                for (EntityFeature ef : model.getObjects(EntityFeature.class))
870                {
871                        if (ef.getEntityFeatureOf() == null)
872                                inferEntityReference(ef);
873
874                        equivalents.add(ef);
875                }
876                for (List<EntityFeature> bucket : equivalents.getBuckets())
877                {
878                        for (int i = 1; i < bucket.size(); i++)
879                        {
880                                EntityFeature ef = bucket.get(i);
881                                if (LOG.isWarnEnabled())
882                                {
883                                        LOG.warn("removing: "+ ef.getUri()+ " since it is equivalent to: "+ bucket.get(0));
884                                }
885                                scheduled.add(ef);
886                        }
887                }
888                for (EntityFeature entityFeature : scheduled)
889                {
890                        model.remove(entityFeature);
891                }
892                for (PhysicalEntity physicalEntity : model.getObjects(PhysicalEntity.class))
893                {
894                        Set<EntityFeature> features = new HashSet<EntityFeature>(physicalEntity.getFeature());
895                        for (EntityFeature feature : features)
896                        {
897                                EntityFeature that = mapped.get(feature);
898                                if (that != null && !that.equals(feature))
899                                {
900                                        LOG.debug(" replacing " + feature +
901                                                                          "{" + feature.getUri() + "} with " +
902                                                                          that + "{" + that.getUri() + "}");
903                                        physicalEntity.removeFeature(feature);
904                                        physicalEntity.addFeature(that);
905                                }
906                        }
907                }
908        }

932                                                er.addEntityFeature(ef);
933                                                break;
934                                        }
935                                }
936                        }
937                }

942         * Collects data type (not object) property
943         * values (can be then used for full-text indexing).
944         * 
945         * @param biopaxElement biopax object
946         * @param depth greater or equals 0: 0 means use this object's
947         *        data properties only; 1 - add child's data properties, etc.;
948         *        (the meaning is slightly different from that of Fetcher.fetch(..) method)
949         * @param dataPropertyFilters - biopax data property filters to optionally
950         *                        either skip e.g. properties 'sequence', 'temperature',
951         *                        or only accept 'term', 'comment', 'name', etc.
952         * @return set of keywords
953         */
954        public static Set<String> getKeywords(BioPAXElement biopaxElement, int depth,
955                                                                                  Filter<DataPropertyEditor>... dataPropertyFilters)
956        {
957                LOG.debug("getKeywords called: " + biopaxElement.getUri());
958                
959                EditorMap em = SimpleEditorMap.L3;
960                Set<String> ss = new HashSet<String>();
961
962                //if depth>0, fetch child biopax objects (ignoring PathwayStep.nextStep property)
963                Set<BioPAXElement> elms = (depth > 0)
964                        ? new Fetcher(em, Fetcher.nextStepFilter).fetch(biopaxElement, depth)
965                                : new HashSet<BioPAXElement>();
966
967                //add this one
968                elms.add(biopaxElement);
969                
970                for (BioPAXElement bpe : elms) {
971                        Set<PropertyEditor> props = em.getEditorsOf(bpe);
972                        for (PropertyEditor pe : props) {
973                                //skip for object prop. or one that fails to pass a filter
974                                if (pe instanceof ObjectPropertyEditor
975                                                || !filter((DataPropertyEditor)pe, dataPropertyFilters))
976                                        continue;
977
978                                Set values = pe.getValueFromBean(bpe);
979                                for (Object v : values) {
980                                        if (!pe.isUnknown(v)) {
981                                                ss.add(v.toString());
982                                        }
983                                }
984                        }
985                }
986                
987                return ss;
988        }

993                        return true;

996                        if (!pf.filter(pe)) {
997                                return false;
998                        }
999                }
1000
1001                return true;
1002        }

1123        // recursively finds all the parent pathways of the object, while escaping infinite loops
1124        private static Set<Pathway> getParentPathwaysRecursively(
1125                        final BioPAXElement biopaxElement, final Set<BioPAXElement> visited) {
1126
1127                final Set<Pathway> pathways = new HashSet<Pathway>();
1128                
1129                //shortcut, when bpe is null or already processed
1130                if(biopaxElement == null || !visited.add(biopaxElement)) {
1131                        LOG.debug("Ignored null or previously visited object:" + biopaxElement);
1132                        return pathways;
1133                }
1134                
1135                LOG.debug("getParentPathways called: " + biopaxElement.getUri());
1136
1137                if(biopaxElement instanceof Process) {
1138                        if(biopaxElement instanceof Pathway) // add itself
1139                                pathways.add((Pathway) biopaxElement);
1140                        // continue looking up to parent pathways (until all top ones reached)
1141                        //TODO: inf. loop happened here (StackOverFlow; e.g., using PC8 KEGG model)
1142                        for(Pathway pw : ((Process)biopaxElement).getPathwayComponentOf())
1143                                pathways.addAll(getParentPathwaysRecursively(pw, visited));
1144                        for(Interaction it : ((Process)biopaxElement).getParticipantOf())
1145                                pathways.addAll(getParentPathwaysRecursively(it, visited));
1146                        for(PathwayStep pt : ((Process)biopaxElement).getStepProcessOf())
1147                                pathways.addAll(getParentPathwaysRecursively(pt, visited));
1148                } else if(biopaxElement instanceof PathwayStep) {
1149                        pathways.addAll(getParentPathwaysRecursively(((PathwayStep) biopaxElement).getPathwayOrderOf(), visited));
1150                } else if(biopaxElement instanceof PhysicalEntity ) {
1151                        for(PhysicalEntity pe : ((PhysicalEntity)biopaxElement).getMemberPhysicalEntityOf())
1152                                pathways.addAll(getParentPathwaysRecursively(pe, visited));
1153                        for(Interaction it : ((Entity)biopaxElement).getParticipantOf())
1154                                pathways.addAll(getParentPathwaysRecursively(it, visited));
1155                        for(Complex c : ((PhysicalEntity)biopaxElement).getComponentOf())
1156                                pathways.addAll(getParentPathwaysRecursively(c, visited));
1157                } else if(biopaxElement instanceof EntityReference) {
1158                        for(EntityReference er : ((EntityReference) biopaxElement).getMemberEntityReferenceOf())
1159                                pathways.addAll(getParentPathwaysRecursively(er, visited));
1160                        for(SimplePhysicalEntity spe : ((EntityReference) biopaxElement).getEntityReferenceOf())
1161                                pathways.addAll(getParentPathwaysRecursively(spe, visited));
1162                } else if (biopaxElement instanceof Gene ) { 
1163                        for(Interaction it : ((Entity) biopaxElement).getParticipantOf())
1164                                pathways.addAll(getParentPathwaysRecursively(it, visited));
1165                } else {
1166                        // ignore
1167                }
1168                
1169                return pathways;
1170        }

1283        // base62 encoding/decoding for (not too long) URI shortening
1284        private static final String ALPHABET = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789";
1285        private static final int BASE = ALPHABET.length();

1287        public static String encodeBase62(String str) {
1288                // encode to base10 (long) first -
1289                char[] chars = new StringBuilder(str).reverse().toString().toCharArray();
1290                long n = 0;
1291                for (int i = chars.length - 1; i >= 0; i--) {
1292                        n += ALPHABET.indexOf(chars[i]) * Math.pow(BASE, i);
1293                }
1294
1295                // convert the long number to a base62 short string
1296                StringBuilder sb = new StringBuilder("");
1297                while (n > 0) {
1298                        int rem = (int)(n % BASE);
1299                        sb.append(ALPHABET.charAt(rem));
1300                        n = n / BASE;
1301                }
1302
1303                return sb.reverse().toString();
1304        }

1323         * Replaces the URI of a BioPAX object in the Model using java reflection.
1324         * If the element also belongs to other BioPAX models, those will become inconsistent
1325         * unless this method is called for each such model.
1326         *
1327         * Warnings:
1328         *  - one should not normally use this method at all;
1329         *  - but if you do, then don't use a URI of another object from the same model.
1330         *
1331         * @param model model (can be null; if the object in fact belongs to a model, the model will be inconsistent)
1332         * @param el biopax object
1333         * @param newUri URI - not null/empty URI
1334         */
1335        public static  void updateUri(Model model, BioPAXElement el, String newUri) {
1336                if(newUri == null || newUri.isEmpty())
1337                        throw new IllegalArgumentException("New URI is null or empty.");
1338
1339                if(model != null)
1340                        model.remove(el);
1341
1342                try {
1343                        Method m = BioPAXElementImpl.class.getDeclaredMethod("setUri", String.class);
1344                        m.setAccessible(true);
1345                        m.invoke(el, newUri);
1346                } catch (Exception e) {
1347                        e.printStackTrace();
1348                        throw new RuntimeException(e);
1349                }
1350
1351                if(model != null)
1352                        model.add(el);
1353        }

1356        /**
1357         * Removes cyclic pathway inclusions, non-trivial infinite loops, in 'pathwayComponent' biopax property.
1358         * Such loops usually do not make much sense and only can cause trouble in pathway data analysis.
1359         * This tool recursively removes parent pathways from sub pathways' pathwayComponent set.
1360         *
1361         * @param model a model that contains Pathways; will be modified as the result
1362         */
1363        public static void breakPathwayComponentCycle(final Model model) {
1364                for(Pathway pathway : model.getObjects(Pathway.class))
1365                        breakPathwayComponentCycle(new HashSet<Pathway>(), pathway, pathway);
1366        }

1368        //Recursively, though, avoiding infinite loops (KEGG pathways can cause it),
1369        // removes cyclic pathway inclusions like pathwayA/pathwayA or pathwayA/pathwayB/pathwayA, etc.
1370        private static void breakPathwayComponentCycle(final Set<Pathway> visited, final Pathway rootPathway,
1371                                                                                        final Pathway currentPathway)
1372        {
1373                if(!visited.add(currentPathway))
1374                        return; // already processed
1375
1376                if(currentPathway.getPathwayComponent().contains(rootPathway)) {
1377                        currentPathway.removePathwayComponent(rootPathway);
1378                }
1379
1380                for(Process proc : currentPathway.getPathwayComponent())
1381                        if(proc instanceof Pathway)
1382                                breakPathwayComponentCycle(visited, rootPathway, (Pathway) proc);
1383        }
