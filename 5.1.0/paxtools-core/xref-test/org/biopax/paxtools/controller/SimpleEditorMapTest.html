37  	/**
38  	 * Currently (before 10-Apr-2011), it seems impossible
39  	 * to clear a single-cardinality property by
40  	 * using the corresponding property editor
41  	 * TODO Is this what we really want? (i.e., cannot set to 'unknown'/null)
42  	 */
43      @Test
44      public void testClearSingularProperty() {    	
45      	BioPAXFactory fac = BioPAXLevel.L3.getDefaultFactory();
46      	EditorMap em = SimpleEditorMap.L3;
47      	
48      	// test - for a singular object property
49      	ProteinReference pr = fac.create(ProteinReference.class, "PR");
50      	BioSource bs = fac.create(BioSource.class, "BS");
51      	PropertyEditor editor = em.getEditorForProperty("organism", ProteinReference.class);
52      	pr.setOrganism(bs);
53      	
54      	pr.setOrganism(null);
55      	assertNull(pr.getOrganism());
56      	
57      	pr.setOrganism(bs);
58      	
59      	editor.removeValueFromBean(bs, pr);
60      	assertNull(pr.getOrganism());
61      	
62      	editor.setValueToBean(null, pr);
63      	/*
64      	assertNotNull(pr.getOrganism()); // no effect! Is this the behavior we want?
65      	*/
66      	// after re-factoring (AbstractPropertyEditor)
67      	assertNull(pr.getOrganism());
68      	assertTrue(editor.isUnknown(pr.getOrganism()));
69      	
70      	// test for a singular primitive property
71      	editor = em.getEditorForProperty("ph", DeltaG.class);
72      	
73      	DeltaG dg = fac.create(DeltaG.class, "DG");
74      	assertNotNull(dg.getPh());
75      	assertTrue(editor.isUnknown(dg.getPh()));
76      	
77      	dg.setPh(6.5f);
78      	assertFalse(editor.isUnknown(dg.getPh()));
79      	
80      	editor.setValueToBean(null, dg);
81      	//assertFalse(editor.isUnknown(dg.getPh())); // no effect: cannot go back to 'unknown' by using null!
82      	assertTrue(editor.isUnknown(dg.getPh())); // after re-factoring
83      	
84      	editor.setValueToBean(BioPAXElement.UNKNOWN_FLOAT, dg);
85      	assertTrue(editor.isUnknown(dg.getPh()));
86      }

88  	@Test
89  	public final void testComments() {
90      	BioPAXFactory fac = BioPAXLevel.L3.getDefaultFactory();
91      	EditorMap em = SimpleEditorMap.L3;
92  		ProteinReference pr1 = fac.create(ProteinReference.class, "pr1"); 
93  		pr1.addComment("one");
94  		pr1.addComment("two");
95  		assertEquals(2, pr1.getComment().size());
96  		
97  		pr1.getComment().clear();
98  		assertEquals(0, pr1.getComment().size());
99  		
100 		PropertyEditor commEditor = em.getEditorForProperty("comment", pr1.getModelInterface());
101 		assertNotNull(commEditor);
102 		assertTrue(commEditor.isMultipleCardinality());
103 		commEditor.setValueToBean("one", pr1);
104 		commEditor.setValueToBean("two", pr1);
105 		assertEquals(2, pr1.getComment().size());
106 	}

109 	 * This test helps understand paxtools simple IO
110 	 * BioPAX property editors better and use them 
111 	 * with confidence.

113 	@Test
114 	public void testClearInverseProperty() {    	
115 		BioPAXFactory fac = BioPAXLevel.L3.getDefaultFactory();
116 	    EditorMap em = SimpleEditorMap.L3;
117 	    
118 	    ProteinReference pr = fac.create(ProteinReference.class, "PR");
119 	    Protein p = fac.create(Protein.class, "P");
120 	    p.setEntityReference(pr);
121 	    assertNotNull(p.getEntityReference());
122 	    assertFalse(pr.getEntityReferenceOf().isEmpty());
123 	    
124 	    Set<ObjectPropertyEditor> prInvPropertyEditors = em.getInverseEditorsOf(pr);
125 //	    System.out.println("PR inv. props are:" + prInvPropertyEditors); 
126 	    ObjectPropertyEditor editor = null;
127 	    PropertyEditor foo = null;
128 	    for(PropertyEditor pe : prInvPropertyEditors) {
129 	    	if(pe.getProperty().equals("entityReference")) {
130 	    		// an inverse editor is actually normal editor, 
131 	    		// for which it's possible to get the inverse prop. accessor
132 	    		editor = (ObjectPropertyEditor) pe;
133 	    	} else if (pe.getProperty().equals("name")) {
134 	    		foo = pe;
135 	    	}
136 	    }
137 	    assertNotNull(editor);
138 	    assertNull(foo);
139 	    
140 	    assertTrue(editor.isInverseMultipleCardinality());
141 	    assertFalse(editor.isMultipleCardinality());
142 	    
143 	    ObjectPropertyEditor bar = null;
144 	    for(PropertyEditor pe : em.getEditorsOf(p)) {
145 	    	if(pe.getProperty().equals("entityReference")) {
146 	    		bar = (ObjectPropertyEditor) pe;
147 	    	} 
148 	    }
149 	    assertNotNull(bar);
150 	    // an inverse editor is actually normal editor
151 	    // (it matters how to use it though)
152 	    assertEquals(editor, bar);	    
153 	    
154 	    Set<ProteinReference> prs = editor.getValueFromBean(p);
155 	    assertEquals(1, prs.size());
156 	    assertEquals(pr, prs.iterator().next());
157 	    
158 	    Set<Protein> ps = editor.getInverseAccessor().getValueFromBean(pr);
159 	    assertEquals(1, ps.size());
160 	    assertEquals(p, ps.iterator().next());
161 	    
162 	    
163 	    // what if one wants to clear pr.entityReferenceOf() this way 
164 	    // (not having a reference to parent 'p' yet) 
165 	    for(Protein val : ps) {
166 	    	assertEquals(p, val); // loop body is called once anyway
167 	    	
168 	    	//a) should not change anything or fails
169 	    	
170 	    	if(editor.isInverseMultipleCardinality()) {//- wrong/useless use; it may lead to surprising results and (shelved for later...)
171 	    		try {
172 	    			editor.removeValueFromBean(val, pr); //wrong order! (it might not throw any exception...)
173 	    		} catch (Throwable e) {
174 	    			//assertions enabled or method changed (throws anexception now)
175 				}
176 	    		assertNotNull(p.getEntityReference()); //unchanged
177 	    		assertFalse(pr.getEntityReferenceOf().isEmpty());  //unchanged
178 	    	}
179 	    	
180 	    	if(editor.isMultipleCardinality()) {
181 	    		editor.removeValueFromBean(pr, val); 
182 	    		fail("correct use but never happens ;-)");
183 	    	}
184 	    	
185 	    	// let's clear it "by mistake";
186 	    	// it actually calls setValueToBean internally
187 	    	// so can be used always!
188 	    	editor.removeValueFromBean(pr, val); 
189 	    	
190     		assertNull(p.getEntityReference());
191     		assertTrue(pr.getEntityReferenceOf().isEmpty());
192 	    	
193     		// well, let's set and clear it normal way, at last -
194     		p.setEntityReference(pr);
195     		if(!editor.isMultipleCardinality()) {
196     			editor.setValueToBean(null, val); //good!
197 	    	} else 
198 	    		fail("impossible!");
199     		
200 			assertNull(p.getEntityReference());
201 			assertTrue(pr.getEntityReferenceOf().isEmpty());
202     		
203 	    	break;
204 	    }
