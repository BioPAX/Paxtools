50  	@Test
51  	public final void testForEndlessLoopOrOutOfMemory() {
52  
53  		Pathway pw1 = (Pathway) model.getByID("pathway1");
54  		Fetcher fetcher = new Fetcher(SimpleEditorMap.L3);
55  		Set<Protein> proteins = fetcher.fetch(pw1, Protein.class);
56  		assertEquals(2, proteins.size());
57  
58  		Set<BioPAXElement> bpes = new Fetcher(SimpleEditorMap.L3).fetch(pw1);
59  		assertEquals(8, bpes.size()); //because pw1 is child of itself via pw3 sub-pathway...
60  
61  
62  		//when skip sub-pathways is enabled -
63  		fetcher.setSkipSubPathways(true);
64  		proteins = fetcher.fetch(pw1, Protein.class);
65  		assertEquals(1, proteins.size());
66  		bpes = fetcher.fetch(pw1);
67  		assertEquals(5, bpes.size());
68  		//pw1 is not included; other pw are there without their children, such as con1 is not there
69  	}

72  	public final void testFetchDepth() {
73  
74  		Pathway pw1 = (Pathway) model.getByID("pathway1");
75  		Set<BioPAXElement> elements;
76  
77  		final Fetcher fetcher = new Fetcher(SimpleEditorMap.L3);
78  
79  		//exception
80  		try {
81  			fetcher.fetch(pw1, 0);
82  			fail();
83  		} catch (IllegalArgumentException e) {
84  		}
85  		try {
86  			fetcher.fetch(pw1, -1);
87  			fail();
88  		} catch (IllegalArgumentException e) {
89  		}
90  
91  		//depth=1
92  		elements = fetcher.fetch(pw1, 1);
93  		assertEquals(4, elements.size());
94  
95  		//depth=2
96  		elements = fetcher.fetch(pw1, 2);
97  		assertEquals(6, elements.size()); //that's it
98  
99  		//depth=3 - all objects except the pw1 itself (despite it belongs to its sub-pathway - loop)
100 		elements = fetcher.fetch(pw1, 3);
101 		assertEquals(7, elements.size());
102 
103 		//depth > 3 all the same...
104 		elements = fetcher.fetch(pw1, 4);
105 		assertEquals(7, elements.size());
106 
107 		//depth=2, skipping sub-pathways
108 		fetcher.setSkipSubPathways(true);
109 		elements = fetcher.fetch(pw1, 2);
110 		assertEquals(5, elements.size()); // 'con1' (component of pw4) is not there
111 		elements = fetcher.fetch(pw1, 4);
112 		assertEquals(5, elements.size());// 'con1', 'p2' are not there
113 	}
