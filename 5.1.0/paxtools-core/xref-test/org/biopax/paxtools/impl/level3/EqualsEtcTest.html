21  	public final void testEquals() {
22  		BioPAXFactory factory = BioPAXLevel.L3.getDefaultFactory();
23  
24  		// a BioPAX object (e.g., xref)
25  		UnificationXref x1 = factory.create(UnificationXref.class, "x1");
26  		x1.setDb("MI");
27  		x1.setId("MI:0492");
28  		
29  		// another xref - different ID, different properties
30      	UnificationXref x2 = factory.create(UnificationXref.class, "x2");
31  		x2.setDb("MI");
32  		x2.setId("MI:0493");
33  		
34  		// - same ID (x1), different property value
35      	UnificationXref x3 = factory.create(UnificationXref.class, "x1");
36  		x3.setDb("MI");
37  		x3.setId("MI:0493");
38  		
39  		// - a copy of x1 (a different java object)
40      	UnificationXref x4 = factory.create(UnificationXref.class, "x1");
41  		x4.setDb("MI");
42  		x4.setId("MI:0492");
43  		
44  		// - different ID, same properties (as of x1)
45      	UnificationXref x5 = factory.create(UnificationXref.class, "x5");
46  		x5.setDb("MI");
47  		x5.setId("MI:0492");
48  		
49  		
50      	UnificationXref x6 = null;
51      	UnificationXref x7 = null;
52      	assertTrue(x6 == x7);
53      	
54      	x6 = x1;
55      	assertFalse(x6 == x7); 
56      	assertTrue(x1 == x6); // same reference (address)
57      	assertTrue(x1.equals(x6)); // - for sure
58      	
59      	x7 = x1;
60      	assertTrue(x6 == x7);
61      	assertTrue(x6.equals(x7));
62      	assertTrue(x1.equals(x7));
63      	assertTrue(x1.equals(x6));
64      	
65      	assertFalse(x1 == x3);   	
66      	assertTrue(x1.equals(x3)); //same type and URI (prop. are ignored)
67      	assertTrue(x1.isEquivalent(x3)); // not the same db/id, same URI - equivalent because equals (above)
68      	
69      	assertFalse(x1 == x4);
70      	assertTrue(x1.equals(x4)); //same type and URI
71      	assertTrue(x1.isEquivalent(x4)); // x4 is a copy of x1
72      	
73      	assertTrue(x1.isEquivalent(x5)); // x5 has different rdfId but same db/id
74  	}

97

114 	@Test
115 	public final void testBiopaxSafeSet() {
116 		BioPAXFactory factory = BioPAXLevel.L3.getDefaultFactory();
117 
118 		UnificationXref x1 = factory.create(UnificationXref.class, "x1");
119 		x1.setDb("foo");
120 		x1.setId("foo");
121 		//another xref, same URI
122     	UnificationXref x2 = factory.create(UnificationXref.class, "x1");
123 		x2.setDb("bar");
124 		x2.setId("bar");
125 		
126 		Set<Xref> col = new BiopaxSafeSet<Xref>();
127 		col.add(x1);
128 		assertTrue(col.contains(x1));
129 		assertEquals(x1, ((BiopaxSafeSet<Xref>)col).get("x1"));		
130 		
131 		// BiopaxSafeSet overrides 'contains'
132 		assertFalse(col.contains(x2));
133 		col.add(x2); //duplicate URI is ignored (warning is logged)
134 		assertTrue(col.size() == 1);
135 		
136 		x2 = factory.create(UnificationXref.class, "x2");
137 		x2.setDb("bar");
138 		x2.setId("bar");
139 		col.add(x2);
140 		assertTrue(col.size() == 2);
141 	
142 		Set<Xref> col1 = new BiopaxSafeSet<Xref>();
143 		col1.addAll(col);
144 		assertTrue(col1.size() == 2);
145 		assertTrue(col1.contains(x1));
146 		assertTrue(col1.contains(x2));
147 		
148 		col1.remove(x2);
149 		assertTrue(col1.size() == 1);
150 		assertTrue(col1.contains(x1));
151 		assertFalse(col1.contains(x2));
152 		assertEquals("foo", ((BiopaxSafeSet<Xref>)col1).get("x1").getDb());
153 			
154 	}
