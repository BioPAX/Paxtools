015 * This class generates a blacklist for the given model. It is important that the given model is the

043        public BlacklistGenerator2()
044        {
045                this(new Decider()
046                {
047                        @Override
048                        public boolean isUbique(int neighborSize, int upstrOnly, int dwstrOnly)
049                        {
050                                return neighborSize >= 50;
051                        }
052
053                        @Override
054                        public int getScore(int neighborSize, int upstrOnly, int dwstrOnly)
055                        {
056                                return neighborSize;
057                        }
058
059                        @Override
060                        public RelType getContext(int neighborSize, int upstrOnly, int dwstrOnly)
061                        {
062                                if (upstrOnly > 10 * dwstrOnly) return RelType.OUTPUT;
063                                else if (dwstrOnly > 10 * upstrOnly) return RelType.INPUT;
064                                else return null;
065                        }
066                });
067        }

069        /**
070         * Generates the blacklist.
071         * @param model model to use
072         * @return the blacklist
073         * @throws IOException when there is an I/O problem
074         */
075        public Blacklist generateBlacklist(Model model) throws IOException
076        {
077                Map<String, String> nameMapping = readNameMapping();
078                if (nameMapping == null)
079                {
080                        generateNameMappingFileToCurate(model);
081                        throw new RuntimeException("Small molecule name mapping file not found. Generated a " +
082                                "mapping file, but it needs manual curation before use.\nPlease go over some top " +
083                                "portion of this file and delete invalid lines and any uncurated bottom part.\n" +
084                                "After that, you can rerun this method.");
085                }
086
087                SIFSearcher searcher = new SIFSearcher(new Fetcher(nameMapping), SIFEnum.USED_TO_PRODUCE);
088
089                Set<SIFInteraction> sifs = searcher.searchSIF(model);
090
091                // read interactions into maps
092
093                Map<String, Set<String>> upstrMap = new HashMap<String, Set<String>>();
094                Map<String, Set<String>> dwstrMap = new HashMap<String, Set<String>>();
095                final Map<String, Set<String>> neighMap = new HashMap<String, Set<String>>();
096
097                for (SIFInteraction sif : sifs)
098                {
099                        String source = sif.sourceID;
100                        String target = sif.targetID;
101
102                        if (!neighMap.containsKey(source)) neighMap.put(source, new HashSet<String>());
103                        if (!neighMap.containsKey(target)) neighMap.put(target, new HashSet<String>());
104                        if (!dwstrMap.containsKey(source)) dwstrMap.put(source, new HashSet<String>());
105                        if (!dwstrMap.containsKey(target)) dwstrMap.put(target, new HashSet<String>());
106                        if (!upstrMap.containsKey(source)) upstrMap.put(source, new HashSet<String>());
107                        if (!upstrMap.containsKey(target)) upstrMap.put(target, new HashSet<String>());
108
109                        neighMap.get(source).add(target);
110                        neighMap.get(target).add(source);
111                        dwstrMap.get(source).add(target);
112                        upstrMap.get(target).add(source);
113                }
114
115                // remove intersection of upstream and downstream
116
117                for (String name : neighMap.keySet())
118                {
119                        if (!upstrMap.containsKey(name) || !dwstrMap.containsKey(name)) continue;
120
121                        Set<String> upstr = upstrMap.get(name);
122                        Set<String> dwstr = dwstrMap.get(name);
123
124                        Set<String> temp = new HashSet<String>(upstr);
125                        upstr.removeAll(dwstr);
126                        dwstr.removeAll(temp);
127                }
128
129//              writeTheGuideRankingToTuneTheDecider(model, nameMapping, upstrMap, dwstrMap, neighMap);
130//              if (true) return null;
131
132                Set<String> white = readWhitelist();
133
134                Blacklist blacklist = new Blacklist();
135
136                // populate the blacklist
137
138                Fetcher nameFetcher = new Fetcher(nameMapping);
139                for (SmallMoleculeReference smr : model.getObjects(SmallMoleculeReference.class))
140                {
141                        Set<String> names = nameFetcher.fetchID(smr);
142                        if (names.isEmpty()) continue;
143                        String name = names.iterator().next();
144
145                        if (white != null && white.contains(name)) continue;
146
147                        int neighSize = neighMap.containsKey(name) ? neighMap.get(name).size() : 0;
148                        int upstrOnly = upstrMap.containsKey(name) ? upstrMap.get(name).size() : 0;
149                        int dwstrOnly = dwstrMap.containsKey(name) ? dwstrMap.get(name).size() : 0;
150
151                        if (decider.isUbique(neighSize, upstrOnly, dwstrOnly))
152                        {
153                                blacklist.addEntry(smr.getUri(),
154                                        decider.getScore(neighSize, upstrOnly, dwstrOnly),
155                                        decider.getContext(neighSize, upstrOnly, dwstrOnly));
156                        }
157                }
158
159                return blacklist;
160        }

162        private void writeTheGuideRankingToTuneTheDecider(Model model, Map<String, String> nameMapping, Map<String, Set<String>> upstrMap, Map<String, Set<String>> dwstrMap, final Map<String, Set<String>> neighMap) throws IOException
163        {
164                // Sort to degree
165
166                List<String> names = new ArrayList<String>(neighMap.keySet());
167                Collections.sort(names, new Comparator<String>()
168                {
169                        @Override
170                        public int compare(String o1, String o2)
171                        {
172                                return new Integer(neighMap.get(o2).size()).compareTo(neighMap.get(o1).size());
173                        }
174                });
175
176                SIFSearcher searcher2 = new SIFSearcher(new Fetcher(nameMapping), new ChemicalAffectsThroughControlMiner());
177
178                Set<SIFInteraction> sifs2 = searcher2.searchSIF(model);
179
180                // read interactions into maps
181
182                final Map<String, Set<String>> affectMap = new HashMap<String, Set<String>>();
183
184                for (SIFInteraction sif : sifs2)
185                {
186                        String source = sif.sourceID;
187                        String target = sif.targetID;
188
189                        if (!affectMap.containsKey(source)) affectMap.put(source, new HashSet<String>());
190                        affectMap.get(source).add(target);
191                }
192
193                BufferedWriter writer = new BufferedWriter(new FileWriter("ubique-stats.txt"));
194
195                for (String name : names)
196                {
197                        writer.write(name + "\t" + neighMap.get(name).size() + "\t" +
198                                upstrMap.get(name).size() + "\t" + dwstrMap.get(name).size() + "\t|\t" +
199                                (affectMap.containsKey(name) ? affectMap.get(name).size() : 0) +  "\n");
200                }
201
202                writer.close();
203        }

205        private void generateNameMappingFileToCurate(Model model) throws IOException
206        {
207                SIFSearcher searcher = new SIFSearcher(new Fetcher(null), SIFEnum.USED_TO_PRODUCE);
208
209                Set<SIFInteraction> sifs = searcher.searchSIF(model);
210
211                // read interactions into maps
212
213                final Map<String, Set<String>> neighMap = new HashMap<String, Set<String>>();
214
215                for (SIFInteraction sif : sifs)
216                {
217                        String source = sif.sourceID;
218                        String target = sif.targetID;
219
220                        if (!neighMap.containsKey(source)) neighMap.put(source, new HashSet<String>());
221                        if (!neighMap.containsKey(target)) neighMap.put(target, new HashSet<String>());
222
223                        neighMap.get(source).add(target);
224                        neighMap.get(target).add(source);
225                }
226
227                // Sort to degree
228
229                List<String> names = new ArrayList<String>(neighMap.keySet());
230                Collections.sort(names, new Comparator<String>()
231                {
232                        @Override
233                        public int compare(String o1, String o2)
234                        {
235                                return new Integer(neighMap.get(o2).size()).compareTo(neighMap.get(o1).size());
236                        }
237                });
238
239                Set<Set<String>> bag = collectNameSets(model);
240                Map<String, Set<Set<String>>> nameToSets = getNameToSetsMap(bag);
241                Map<Set<String>, Set<Set<String>>> intersectionMap = getIntersectionMap(bag);
242
243                // Find and record name mapping
244
245                String dir = System.getProperty("java.io.tmpdir") + File.separator;
246
247                BufferedWriter writer = new BufferedWriter(new FileWriter(dir + MAPPING_FILE));
248
249                List<String> remaining = new ArrayList<String>();
250                for (String name : names)
251                {
252                        remaining.add(name.toLowerCase());
253                }
254
255                for (String name : names)
256                {
257                        name = name.toLowerCase();
258                        remaining.remove(name);
259
260                        Map<String, Set<String>[]> mappings = getMappingsAndBasis(
261                                name, remaining, nameToSets, intersectionMap);
262
263                        for (String mapped : mappings.keySet())
264                        {
265                                Set<String>[] s = mappings.get(mapped);
266                                writer.write(name + "\t" + mapped + "\t" + s[0] + "\t" + s[1] + "\t" + s[2] + "\n");
267                        }
268                }
269
270                writer.close();
271                System.out.println("A small molecule name mapping file is generated (" + dir +
272                        MAPPING_FILE + "). Please curate the top portion of this file, delete invalid lines, " +
273                        "copy the curated file into the working directory, and re-run this code.");
274        }

276        private Set<Set<String>> collectNameSets(Model model)
277        {
278                Set<Set<String>> bag = new HashSet<Set<String>>();
279                for (SmallMoleculeReference smr : model.getObjects(SmallMoleculeReference.class))
280                {
281                        Set<String> set = new HashSet<String>();
282
283                        for (String name : smr.getName())
284                        {
285                                set.add(name.toLowerCase());
286                        }
287
288                        for (SimplePhysicalEntity sm : smr.getEntityReferenceOf())
289                        {
290                                for (String name : sm.getName())
291                                {
292                                        set.add(name.toLowerCase());
293                                }
294                        }
295
296                        doSomeCleaning(set);
297                        enrichWithModifications(set);
298                        bag.add(set);
299                }
300                return bag;
301        }
302
303        private void doSomeCleaning(Set<String> set)
304        {
305                Set<String> rem = new HashSet<String>();
306                for (String s : set)
307                {
308                        if (s.contains("pathway") || s.contains("participant")) rem.add(s);
309                }
310                set.removeAll(rem);
311        }
312
313        private void enrichWithModifications(Set<String> set)
314        {
315                for (String s : new HashSet<String>(set))
316                {
317                        if (s.endsWith("-)") || s.endsWith("+)"))
318                        {
319                                set.add(s.substring(0, s.lastIndexOf("(")).trim());
320                        }
321                        else if (s.endsWith(" zwitterion"))
322                        {
323                                set.add(s.substring(0, s.lastIndexOf(" ")).trim());
324                        }
325                }
326        }

328        private Map<String, Set<Set<String>>> getNameToSetsMap(Set<Set<String>> bag)
329        {
330                Map<String, Set<Set<String>>> map = new HashMap<String, Set<Set<String>>>();
331
332                for (Set<String> set : bag)
333                {
334                        for (String name : set)
335                        {
336                                if (!map.containsKey(name)) map.put(name, new HashSet<Set<String>>());
337                                map.get(name).add(set);
338                        }
339                }
340                return map;
341        }

343        private Set<String> getCommon(Set<String> set1, Set<String> set2)
344        {
345                Set<String> comm = new HashSet<String>(set1);
346                comm.retainAll(set2);
347                return comm;
348        }
349
350        private List<String> getCommon(List<String> list1, Set<String> set2)
351        {
352                List<String> comm = new ArrayList<String>();
353                for (String s : list1)
354                {
355                        if (set2.contains(s)) comm.add(s);
356                }
357                return comm;
358        }

360        private Map<Set<String>, Set<Set<String>>> getIntersectionMap(Set<Set<String>> bag)
361        {
362                Map<Set<String>, Set<Set<String>>> map = new HashMap<Set<String>, Set<Set<String>>>();
363
364                for (Set<String> set : bag)
365                {
366                        for (Set<String> oth : bag)
367                        {
368                                if (set == oth) continue;
369
370                                if (!getCommon(set, oth).isEmpty())
371                                {
372                                        if (!map.containsKey(set)) map.put(set, new HashSet<Set<String>>());
373                                        map.get(set).add(oth);
374                                }
375                        }
376                }
377                return map;
378        }

380        private Map<String, Set<String>[]> getMappingsAndBasis(String name, List<String> consider,
381                Map<String, Set<Set<String>>> nameToSets, Map<Set<String>, Set<Set<String>>> intersectionMap)
382        {
383                Map<String, Set<String>[]> map = new HashMap<String, Set<String>[]>();
384
385                for (Set<String> set : nameToSets.get(name))
386                {
387                        if (intersectionMap.containsKey(set))
388                        {
389                                for (Set<String> other : intersectionMap.get(set))
390                                {
391                                        List<String> common = getCommon(consider, other);
392
393                                        for (String target : common)
394                                        {
395                                                map.put(target, new Set[]{getCommon(set, other), set, other});
396                                        }
397                                }
398                        }
399                }
400                return map;
401        }

403        private Map<String, String> readNameMapping() throws FileNotFoundException
404        {
405                if (!new File(MAPPING_FILE).exists()) return null;
406
407                Map<String, String> map = new HashMap<String, String>();
408
409                Scanner sc = new Scanner(new File(MAPPING_FILE));
410                while (sc.hasNextLine())
411                {
412                        String[] token = sc.nextLine().split("\t");
413
414                        map.put(token[1], token[0]);
415                }
416                return map;
417        }

419        private Set<String> readWhitelist() throws FileNotFoundException
420        {
421                if (!new File(WHITELIST_FILE).exists())
422                {
423                        System.out.println("No whitelist file found (" + WHITELIST_FILE + "). " +
424                                "Not whitelisting anything.");
425                        return null;
426                }
427                Set<String> names = new HashSet<String>();
428                Scanner sc = new Scanner(new File(WHITELIST_FILE));
429                while (sc.hasNextLine())
430                {
431                        String name = sc.nextLine().split("\t")[0];
432                        names.add(name);
433                }
434                return names;
435        }

437        /**
438         * Class to fetch the ID of the small molecule.
439         */
440        class Fetcher extends CommonIDFetcher
441        {
442                Map<String, String> nameMap;
443
444                public Fetcher(Map<String, String> nameMap)
445                {
446                        this.nameMap = nameMap;
447                }
448
449                @Override
450                public Set<String> fetchID(BioPAXElement ele)
451                {
452                        if (ele instanceof SmallMoleculeReference)
453                        {
454                                SmallMoleculeReference smr = (SmallMoleculeReference) ele;
455                                String name = null;
456                                if (smr.getDisplayName() != null) name = smr.getDisplayName();
457                                else if (smr.getStandardName() != null) name = smr.getStandardName();
458                                else if (!smr.getName().isEmpty()) name = smr.getName().iterator().next();
459
460                                if (name != null)
461                                {
462                                        name = name.toLowerCase();
463                                        if (nameMap != null && nameMap.containsKey(name))
464                                                name = nameMap.get(name);
465                                        return Collections.singleton(name);
466                                }
467                        }
468
469                        return super.fetchID(ele);
470                }
471        }

473        /**
474         * The class to decide if a molecule is ubique, its score and its context of ubiquity.
475         */
476        static interface Decider
477        {
478                /**
479                 * Tells if the molecule is ubique in at least one context.
480                 * @param neighborSize number of neighbors in the used-to-produce network
481                 * @param upstrOnly number of upstream neighbors in the used-to-produce network, that are not also at downstream
482                 * @param dwstrOnly number of downstream neighbors in the used-to-produce network, that are not also at upstream
483                 */
484                public boolean isUbique(int neighborSize, int upstrOnly, int dwstrOnly);
485
486                /**
487                 * Gets the ubiquity score of the ubique molecule. This score is used for comparing ubiques
488                 * and deciding the most essential reactants of a reaction if all reactants are ubique.
489                 * @param neighborSize number of neighbors in the used-to-produce network
490                 * @param upstrOnly number of upstream neighbors in the used-to-produce network, that are not also at downstream
491                 * @param dwstrOnly number of downstream neighbors in the used-to-produce network, that are not also at upstream
492                 */
493                public int getScore(int neighborSize, int upstrOnly, int dwstrOnly);
494
495                /**
496                 * Gets the context of ubiquity. A molecule can be ubiquitously consumed, or can be
497                 * ubiquitously produced, or both. When it is both, this method has to return null.
498                 * @param neighborSize number of neighbors in the used-to-produce network
499                 * @param upstrOnly number of upstream neighbors in the used-to-produce network, that are not also at downstream
500                 * @param dwstrOnly number of downstream neighbors in the used-to-produce network, that are not also at upstream
501                 */
502                public RelType getContext(int neighborSize, int upstrOnly, int dwstrOnly);
503        }
