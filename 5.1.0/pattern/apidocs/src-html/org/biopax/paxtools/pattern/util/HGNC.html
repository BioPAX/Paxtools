001package org.biopax.paxtools.pattern.util;
002
003import java.io.BufferedReader;
004import java.io.FileNotFoundException;
005import java.io.IOException;
006import java.io.InputStreamReader;
007import java.util.HashMap;
008import java.util.Map;
009import java.util.Set;
010
011/**
012 * This class provides a mapping between HGNC IDs and Symbols.
013 * We periodically, manually download
014 * <a href="http://www.genenames.org/cgi-bin/download?col=gd_hgnc_id&col=gd_app_sym&col=gd_prev_sym&status=Approved&status_opt=2&where=&order_by=gd_app_sym_sort&format=text&limit=&hgnc_dbtag=on&submit=submit">
015 * the ID mapping file from genenames.org</a>
016 *
017 * @author Ozgun Babur
018 */
019public class HGNC
020{
021        private static Map<String, String> sym2id;
022        private static Map<String, String> id2sym;
023        private static Map<String, String> old2new;
024
025        public static String getSymbol(String idOrSymbol)
026        {
027                if (id2sym.containsKey(idOrSymbol)) return id2sym.get(idOrSymbol);
028                else if (sym2id.containsKey(idOrSymbol)) return idOrSymbol;
029                else if (old2new.containsKey(idOrSymbol)) return old2new.get(idOrSymbol);
030                else if (!idOrSymbol.toUpperCase().equals(idOrSymbol))
031                        return getSymbol(idOrSymbol.toUpperCase());
032                else return null;
033        }
034
035        static
036        {
037                try
038                {
039                        sym2id = new HashMap<String, String>();
040                        id2sym = new HashMap<String, String>();
041                        old2new = new HashMap<String, String>();
042                        BufferedReader reader = new BufferedReader(new InputStreamReader(
043                                HGNC.class.getResourceAsStream("hgnc.txt")));
044
045                        reader.readLine(); //skip header
046                        for (String line = reader.readLine(); line != null; line = reader.readLine())
047                        {
048                                String[] token = line.split("\t");
049                                String sym = token[1].trim();
050                                String id = token[0].trim();
051                                sym2id.put(sym, id);
052                                id2sym.put(id, sym);
053
054                                if (token.length > 2)
055                                {
056                                        String olds = token[2];
057                                        for (String old : olds.split(","))
058                                        {
059                                                old = old.trim();
060                                                old2new.put(old, sym);
061                                        }
062                                }
063                        }
064                        reader.close();
065                }
066                catch (FileNotFoundException e)
067                {
068                        e.printStackTrace();
069                }
070                catch (IOException e)
071                {
072                        e.printStackTrace();
073                }
074        }
075}
