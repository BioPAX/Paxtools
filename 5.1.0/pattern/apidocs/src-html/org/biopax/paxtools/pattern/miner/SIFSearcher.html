010import java.io.IOException;
011import java.io.OutputStream;
012import java.io.OutputStreamWriter;
013import java.util.*;

019/**
020 * Searches a model and generates SIF network using the pattern matches.
021 *
022 * @author Ozgun Babur
023 */
024public class SIFSearcher
025{
026        /**
027         * SIF miners to use.
028         */
029        private List<SIFMiner> miners;
030
031        /**
032         * SIF types to mine.
033         */
034        private Set<SIFType> types;
035
036        /**
037         * Used for getting an ID out of BioPAX elements.
038         */
039        private IDFetcher idFetcher;
040
041        /**
042         * Blacklist used for detecting ubiquitous molecules.
043         */
044        private Blacklist blacklist;
045
046        /**
047         * Constructor with binary interaction types.
048         * @param types sif types
049         */
050        public SIFSearcher(SIFType... types)
051        {
052                this(null, types);
053        }
054
055        /**
056         * Constructor with miners.
057         * @param miners sif miners
058         */
059        public SIFSearcher(SIFMiner... miners)
060        {
061                this(null, miners);
062        }
063
064        /**
065         * Constructor with ID fetcher and binary interaction types.
066         *
067         * @param idFetcher ID fetcher
068         * @param types sif types
069         */
070        public SIFSearcher(IDFetcher idFetcher, SIFType... types)
071        {
072                this.idFetcher = idFetcher;
073                this.types = new HashSet<SIFType>(Arrays.asList(types));
074
075                if (idFetcher == null) this.idFetcher = new SimpleIDFetcher();//gets URIs
076        }
077
078        /**
079         * Constructor with ID fetcher and miners.
080         *
081         * @param idFetcher ID fetcher
082         * @param miners sif miners
083         */
084        public SIFSearcher(IDFetcher idFetcher, SIFMiner... miners)
085        {
086                this.idFetcher = idFetcher;
087                this.miners = Arrays.asList(miners);
088
089                if (idFetcher == null) this.idFetcher = new SimpleIDFetcher();
090        }
091
092        private void initMiners()
093        {
094                try
095                {
096                        this.miners = new ArrayList<SIFMiner>();
097
098                        for (SIFType type : types)
099                        {
100                                for (Class<? extends SIFMiner> clazz : type.getMiners())
101                                {
102                                        SIFMiner miner = clazz.newInstance();
103                                        miner.setBlacklist(blacklist);
104                                        miner.setIDFetcher(idFetcher);
105                                        miners.add(miner);
106                                }
107                        }
108                }
109                catch (InstantiationException e)
110                {
111                        e.printStackTrace();
112                }
113                catch (IllegalAccessException e)
114                {
115                        e.printStackTrace();
116                }
117        }
118
119        /**
120         * Sets the blacklist that manages IDs of ubique molecules. This is not mandatory but need if
121         * ubiquitous small molecules are needed to be handled.
122         * @param blacklist for identifying ubiquitous small molecules
123         */
124        public void setBlacklist(Blacklist blacklist)
125        {
126                this.blacklist = blacklist;
127        }
128
129        /**
130         * Searches the given model with the contained miners.
131         * @param model model to search
132         * @return sif interactions
133         */
134        public AdjacencyMatrix searchSIFGetMatrix(Model model)
135        {
136                Set<SIFInteraction> sifInts = searchSIF(model);
137                return new AdjacencyMatrix(
138                        SIFInteraction.convertToAdjacencyMatrix(sifInts),
139                        SIFInteraction.getSortedGeneNames(sifInts));
140        }
141
142        /**
143         * Searches the given model with the contained miners.
144         * @param model model to search
145         * @return sif interactions
146         */
147        public Set<SIFInteraction> searchSIF(final Model model)
148        {
149                if (miners == null) initMiners();
150
151                final Map<SIFInteraction, SIFInteraction> map = new ConcurrentHashMap<SIFInteraction, SIFInteraction>();
152
153                for (final SIFMiner miner : miners)
154                {
155                        if (miner instanceof MinerAdapter)
156                                ((MinerAdapter) miner).setIdMap(new HashMap<BioPAXElement, Set<String>>());
157
158                        Map<BioPAXElement,List<Match>> matches = Searcher.search(model, miner.getPattern());
159                        for (final List<Match> matchList : matches.values())
160                        {
161                                for (Match m : matchList)
162                                {
163                                        Set<SIFInteraction> sifs = miner.createSIFInteraction(m, idFetcher);
164                                        for (SIFInteraction sif : sifs)
165                                        {
166                                                if ( sif != null && sif.hasIDs() && !sif.sourceID.equals(sif.targetID)
167                                                                && (types == null || types.contains(sif.type)) )
168                                                {
169                                                        SIFInteraction existing = map.get(sif);
170                                                        if(existing != null)
171                                                                existing.mergeWith(sif);
172                                                        else
173                                                                map.put(sif, sif);
174                                                }
175                                        }
176                                }
177                        }
178                }
179
180                return new HashSet<SIFInteraction>(map.values());
181        }
182
183        /**
184         * Searches the given model with the contained miners.
185         * Writes the textual result to the given output stream.
186         * Closes the stream at the end.
187         *
188         * Produces the simplest version of SIF format.
189         *
190         * @param model model to search
191         * @param out stream to write
192         * @return true if any output produced successfully
193         */
194        public boolean searchSIF(Model model, OutputStream out)
195        {
196                return searchSIF(model, out, new SIFToText() {
197                        @Override
198                        public String convert(SIFInteraction inter)
199                        {
200                                return inter.toString();
201                        }
202                });
203        }
204
205        /**
206         * Searches the given model with the contained miners. Writes the textual result to the given
207         * output stream. Closes the stream at the end.
208         * @param model model to search
209         * @param out stream to write
210         * @param stt sif to text converter
211         * @return true if any output produced successfully
212         */
213        public boolean searchSIF(Model model, OutputStream out, SIFToText stt)
214        {
215                Set<SIFInteraction> inters = searchSIF(model);
216
217                if (!inters.isEmpty())
218                {
219                        List<SIFInteraction> interList = new ArrayList<SIFInteraction>(inters);
220                        Collections.sort(interList);
221                        try
222                        {
223                                boolean first = true;
224                                OutputStreamWriter writer = new OutputStreamWriter(out);
225                                for (SIFInteraction inter : interList)
226                                {
227                                        if (first) first = false;
228                                        else writer.write("\n");
229
230                                        writer.write(stt.convert(inter));
231                                }
232                                writer.close();
233                                return true;
234                        }
235                        catch (IOException e)
236                        {
237                                e.printStackTrace();
238                        }
239                }
240                return false;
241        }
242}
