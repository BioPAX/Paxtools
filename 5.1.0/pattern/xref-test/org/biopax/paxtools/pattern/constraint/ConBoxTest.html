199 	public void testNotUbique() throws Exception
200 	{
201 		Pattern p = new Pattern(Complex.class, "Complex");
202 		p.add(ConBox.complexMembers(), "Complex", "member");
203 		Blacklist blacklist = new Blacklist();
204 		blacklist.addEntry("http://pid.nci.nih.gov/biopaxpid_679", 0, null); // GDP ER ID
205 		p.add(new NonUbique(blacklist), "member"); // not GDP
206 
207 		List<Match> list = Searcher.search(model.getByID("http://pid.nci.nih.gov/biopaxpid_35409"), // Gi family/GNB1/GNG2/GDP
208 			p);
209 
210 		Assert.assertTrue(list.size() == 3);
211 		Assert.assertFalse(collect(list, 1).contains(model.getByID("http://pid.nci.nih.gov/biopaxpid_678"))); //GDP PE ID
