19  /**
20   * Miner for the directed relations between proteins.
21   * @author Ozgun Babur
22   */
23  public class DirectedRelationMiner extends MinerAdapter
24  {
25  	/**
26  	 * Constructor that sets name and description.
27  	 */
28  	public DirectedRelationMiner()
29  	{
30  		super("directed-relations", "Finds relations between proteins where " +
31  			"the first one is controlling an interaction where the second protein is participant.");
32  	}
33  
34  	/**
35  	 * Constructs the pattern.
36  	 * @return pattern
37  	 */
38  	@Override
39  	public Pattern constructPattern()
40  	{
41  		Pattern p = new Pattern(SequenceEntityReference.class, "controller ER");
42  		p.add(isHuman(), "controller ER");
43  		p.add(linkedER(true), "controller ER", "controller generic ER");
44  		p.add(erToPE(), "controller generic ER", "controller simple PE");
45  		p.add(linkToComplex(), "controller simple PE", "controller PE");
46  		p.add(peToControl(), "controller PE", "Control");
47  		p.add(controlToInter(), "Control", "Interaction");
48  		p.add(new NOT(participantER()), "Interaction", "controller ER");
49  		p.add(participant(), "Interaction", "affected PE");
50  		p.add(linkToSpecific(), "affected PE", "affected simple PE");
51  		p.add(new Type(SequenceEntity.class), "affected simple PE");
52  		p.add(peToER(), "affected simple PE", "affected generic ER");
53  		p.add(peToER(), "affected generic ER", "affected ER");
54  
55  		return p;
56  	}
57  
58  	/**
59  	 * Writes the result as "A B", where A and B are gene symbols, and whitespace is tab.
60  	 * @param matches pattern search result
61  	 * @param out output stream
62  	 */
63  	@Override
64  	public void writeResult(Map<BioPAXElement, List<Match>> matches, OutputStream out)
65  		throws IOException
66  	{
67  		writeResultAsSIF(matches, out, true, getSourceLabel(), getTargetLabel());
68  	}
69  
70  	/**
71  	 * Sets header of the output.
72  	 * @return header
73  	 */
74  	@Override
75  	public String getHeader()
76  	{
77  		return "Upstream\tDownstream";
78  	}
79  
80  	public String getSourceLabel()
81  	{
82  		return "controller ER";
83  	}
84  
85  	public String getTargetLabel()
86  	{
87  		return "affected ER";
88  	}
89  
90  	@Override
91  	public String[] getMediatorLabels()
92  	{
93  		return new String[]{"Control", "Interaction"};
94  	}
95  }
