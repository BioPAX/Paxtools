10  import java.util.Collection;
11  import java.util.HashSet;
12  import java.util.Set;

14  /**
15   * After traversing a PhysicalEntity and the Conversion it participates, this constraint takes us to
16   * the Control. The given relation between PhysicalEntity and Conversion is used to filter out
17   * unrelated controls.
18   *
19   * Var0 - PhysicalEntity (participant)
20   * Var1 - Conversion
21   * Var2 - related Control
22   *
23   * @author Ozgun Babur
24   */
25  public class RelatedControl extends ConstraintAdapter
26  {
27  	/**
28  	 * Relation type between PhysicalEntity and Conversion.
29  	 */
30  	RelType peType;
31  
32  	/**
33  	 * Accessor to controller of Control recursively.
34  	 */
35  	PathAccessor controlledOf = new PathAccessor("Conversion/controlledOf*:Control");
36  
37  	/**
38  	 * Constructor with the relation type between PhysicalEntity and Conversion.
39  	 * @param peType relation type between PhysicalEntity and Conversion
40  	 */
41  	public RelatedControl(RelType peType)
42  	{
43  		this(peType, null);
44  	}
45  
46  	/**
47  	 * Constructor with the relation type between PhysicalEntity and Conversion, and the blacklist.
48  	 * @param peType relation type between PhysicalEntity and Conversion
49  	 * @param blacklist to detect ubiquitous small molecules
50  	 */
51  	public RelatedControl(RelType peType, Blacklist blacklist)
52  	{
53  		super(3, blacklist);
54  		this.peType = peType;
55  	}
56  
57  	/**
58  	 * This is a generative constraint.
59  	 * @return true
60  	 */
61  	@Override
62  	public boolean canGenerate()
63  	{
64  		return true;
65  	}
66  
67  	/**
68  	 * According to the relation between PhysicalEntity and the Conversion, checks of the Control is
69  	 * relevant. If relevant it is retrieved.
70  	 * @param match current pattern match
71  	 * @param ind mapped indices
72  	 * @return related controls
73  	 */
74  	@Override
75  	public Collection<BioPAXElement> generate(Match match, int... ind)
76  	{
77  		PhysicalEntity pe = (PhysicalEntity) match.get(ind[0]);
78  		Conversion conv = (Conversion) match.get(ind[1]);
79  
80  		if (!((peType == RelType.INPUT && getConvParticipants(conv, RelType.INPUT).contains(pe)) ||
81  			(peType == RelType.OUTPUT && getConvParticipants(conv, RelType.OUTPUT).contains(pe))))
82  			throw new IllegalArgumentException("peType = " + peType +
83  				", and related participant set does not contain this PE. Conv dir = " +
84  				getDirection(conv) + " conv.id=" + conv.getUri() + " pe.id=" +pe.getUri());
85  
86  		boolean rightContains = conv.getRight().contains(pe);
87  		boolean leftContains = conv.getLeft().contains(pe);
88  		
89  		assert rightContains || leftContains : "PE is not a participant.";
90  
91  		Set<BioPAXElement> result = new HashSet<BioPAXElement>();
92  
93  		ConversionDirectionType avoidDir = (leftContains && rightContains) ? null : peType == RelType.OUTPUT ?
94  			(leftContains ? ConversionDirectionType.LEFT_TO_RIGHT : ConversionDirectionType.RIGHT_TO_LEFT) :
95  			(rightContains ? ConversionDirectionType.LEFT_TO_RIGHT : ConversionDirectionType.RIGHT_TO_LEFT);
96  
97  		for (Object o : controlledOf.getValueFromBean(conv))
98  		{
99  			Control ctrl = (Control) o;
100 			ConversionDirectionType dir = getDirection(conv, ctrl);
101 
102 			if (avoidDir != null && dir == avoidDir) continue;
103 
104 			// don't collect this if the pe is ubique in the context of this control
105 			if (blacklist != null && blacklist.isUbique(pe, conv, dir, peType)) continue;
106 
107 			result.add(ctrl);
108 		}
109 		return result;
110 	}
111 
112 }
