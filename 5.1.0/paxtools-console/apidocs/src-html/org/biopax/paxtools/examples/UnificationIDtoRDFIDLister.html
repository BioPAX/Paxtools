001package org.biopax.paxtools.examples;
002
003import org.biopax.paxtools.io.BioPAXIOHandler;
004import org.biopax.paxtools.io.SimpleIOHandler;
005import org.biopax.paxtools.model.Model;
006import org.biopax.paxtools.model.level2.physicalEntity;
007import org.biopax.paxtools.model.level2.unificationXref;
008import org.biopax.paxtools.util.ClassFilterSet;
009import org.slf4j.Logger;
010import org.slf4j.LoggerFactory;
011
012import java.io.File;
013import java.io.FileInputStream;
014import java.io.FileNotFoundException;
015import java.lang.reflect.InvocationTargetException;
016import java.util.Set;
017
018/**
019 * A basic example that shows how to list all unification xrefs.
020 *
021 * NOTE: This method is now outdated as it is easier to do this now with the new PathAccessors. I did not remove this
022 * example, however, as it demonstrates many "low-level" operations of Paxtools.
023
024 */
025public class UnificationIDtoRDFIDLister
026{
027        private static Logger log = LoggerFactory.getLogger(UnificationIDtoRDFIDLister.class);
028
029        static BioPAXIOHandler handler = new SimpleIOHandler();
030
031        public static void main(String[] args)
032                throws IllegalAccessException, InvocationTargetException
033        {
034
035        for (String arg : args)
036        {
037            log.info(arg);
038            if (arg.toLowerCase().endsWith("owl"))
039            {
040                try
041                {
042                    processXrefs(arg);
043                }
044                catch (FileNotFoundException e)
045                {
046                    e.printStackTrace();
047                }
048            }
049
050        }
051        }
052
053        private static void processXrefs(String arg) throws
054                FileNotFoundException,
055                IllegalAccessException,
056                InvocationTargetException
057        {
058                FileInputStream in =
059                        new FileInputStream(new File(arg));
060                Model level2 = handler.convertFromOWL(in);
061
062                Set<unificationXref> unis =
063                        level2.getObjects(unificationXref.class);
064                for (unificationXref uni : unis)
065                {
066                        ClassFilterSet referrables = new ClassFilterSet(uni.isXREFof(),
067                                physicalEntity.class);
068                        for (Object referrable : referrables)
069                        {
070                                System.out
071                                        .print(uni.getDB() + " : " + uni.getID() + " refers to " +
072                                                ((physicalEntity) referrable).getUri());
073                        }
074                }
075        }
076}
