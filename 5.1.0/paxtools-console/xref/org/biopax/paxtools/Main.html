10   * PaxtoolsMain console application
11   * (very useful BioPAX utilities).
12   */
13  public final class Main {

16      public static void main(String[] argv) throws Exception
17  	{
18          if (argv.length == 0) {
19              help();
20          } else {
21  	        String command = argv[0];
22  	        Command.valueOf(command).run(argv);
23          }
24      }

26      enum Command {
27          merge("<file1> <file2> <output>\n" +
28  					"\t- merges file2 into file1 and writes it into output")
29  		        {public void run(String[] argv) throws IOException{merge(argv);} },
30          toSIF("<input> <output> [-extended] [-andSif] [\"include=SIFType,..\"] [\"exclude=SIFType,..\"]" +
31  					" [\"seqDb=db,..\"] [\"chemDb=db,..\"] [-dontMergeInteractions] [-useNameIfNoId] [<property_accessor> ...]\n" +
32  					"\t- converts a BioPAX model to SIF (default) or custom SIF-like text format;\n" +
33  					"\t  will use blacklist.txt (recommended) file in the current directory, if present.\n" +
34  					"\t- Include or exclude to/from the analysis one or more relationship types by \n" +
35  					"\t  using 'include=' and/or 'exclude=', respectively, e.g., exclude=NEIGHBOR_OF,INTERACTS_WITH\n" +
36  					"\t  (mind using underscore instead of minus sign in the SIF type names; the default is to use all types).\n" +
37  					"\t- With 'seqDb=' and 'chemDb=', you can specify standard sequence/gene/chemical ID type(s)\n" +
38  					"\t  (can be just a unique prefix) to match actual xref.db values in the BioPAX model,\n" +
39  					"\t  e.g., \"seqDb=uniprot,hgnc,refseq\", and in that order, means: if a UniProt entity ID is found,\n" +
40  					"\t  other ID types ain't used; otherwise, if an 'hgnc' ID/Symbol is found... and so on;\n" +
41  					"\t  when not specified, then 'hgnc' (in fact, 'HGNC Symbol') for bio-polymers - \n" +
42  					"\t  and ChEBI IDs or name (if '-useNameIfNoId' is set) for chemicals - are selected.\n" +
43  					"\t- With '-extended' flag, the output will be the Pathway Commons TXT (Extended SIF) format:\n" +
44  					"\t  two sections separated with one blank line - first come inferred SIF interactions -\n" +
45  					"\t  'A\trelationship-type\tB' plus RESOURCE, PUBMED, PATHWAY, MEDIATOR extra columns, \n" +
46  					"\t  followed by interaction participants description section).\n" +
47  					"\t- If '-andSif' flag is present (only makes sense together with '-extended'), then the \n" +
48  					"\t  classic SIF output file is also created (will have '.sif' extension).\n" +
49  					"\t- Finally, <property_accessor>... list is to specify 4th, 5th etc. custom output columns;\n" +
50  					"\t  use pre-defined column names (accessors): \n" +
51  						"\t\tMEDIATOR,\n" +
52  						"\t\tPUBMED,\n" +
53  						"\t\tPMC,\n" +
54  						"\t\tCOMMENTS,\n" +
55  						"\t\tPATHWAY,\n" +
56  						"\t\tPATHWAY_URI,\n" +
57  						"\t\tRESOURCE,\n" +
58  						"\t\tSOURCE_LOC,\n" +
59  						"\t\tTARGET_LOC\n" +
60  					"\t  or custom biopax property path accessors (XPath-like expressions to apply to each mediator entity; \n" +
61  					"\t  see https://github.com/BioPAX/Paxtools/wiki/PatternBinaryInteractionFramework)")
62  		        {public void run(String[] argv) throws IOException{toSifnx(argv);} },
63          toSBGN("<biopax.owl> <output.sbgn> [-nolayout]\n" +
64          		"\t- converts model to the SBGN format and applies COSE layout unless optional -nolayout flag is set.")
65                  {public void run(String[] argv) throws IOException { toSbgn(argv); } },
66          validate("<path> <out> [xml|html|biopax] [auto-fix] [only-errors] [maxerrors=n] [notstrict]\n" +
67          		"\t- validate BioPAX file/directory (up to ~25MB in total size, -\n" +
68          		"\totherwise download and run the stand-alone validator)\n" +
69          		"\tin the directory using the online validator service\n" +
70          		"\t(generates html or xml report, or gets the processed biopax\n" +
71          		"\t(cannot be perfect though) see http://www.biopax.org/validator)")
72  		        {public void run(String[] argv) throws IOException{validate(argv);} },
73          integrate("<file1> <file2> <output>\n" +
74          		"\t- integrates file2 into file1 and writes it into output (experimental)")
75  		        {public void run(String[] argv) throws IOException{integrate(argv);} },
76          toLevel3("<input> <output> [-psimiToComplexes]\n" +
77          		"\t- converts BioPAX level 1 or 2, PSI-MI 2.5 and PSI-MITAB to the level 3 file;\n" +
78          		"\t-psimiToComplexes forces PSI-MI Interactions become BioPAX Complexes instead MolecularInteractions.")
79  		        {public void run(String[] argv) throws IOException{toLevel3(argv);} },
80          toGSEA("<input> <output> <db> [-crossSpecies] [-subPathways] [-notPathway] [organisms=9606,human,rat,..]\n" +
81          		"\t- converts BioPAX data to the GSEA software format (GMT); options/flags:\n"
82                  + "\t<db> - gene/protein ID type; values: uniprot, hgnc, refseq, etc. (a name or prefix to match\n"
83  				+ "\t  ProteinReference/xref/db property values in the input BioPAX model).\n"
84                  + "\t-crossSpecies - allows printing on the same line gene/protein IDs from different species;\n"
85  				+ "\t-subPathways - traverse into sub-pathways to collect all protein IDs for a pathway.\n"
86  				+ "\t-notPathway - also list those protein/gene IDs that cannot be reached from pathways.\n"
87  				+ "\torganisms - optional filter; a comma-separated list of taxonomy IDs and/or names\n")
88  		        {public void run(String[] argv) throws IOException{toGSEA(argv);} },
89          fetch("<input> <output> [uris=URI1,..] [-absolute] \n" +
90          		"\t- extracts a self-integral BioPAX sub-model from file1 and writes to the output; options:\n" +
91  				"\turi=... - an optional list of existing in the model BioPAX elements' full URIs;\n" +
92  				"\t-absolute - set this flag to write full/absolute URIs to the output (i.e., 'rdf:about' instead 'rdf:ID').")
93  		        {public void run(String[] argv) throws IOException{fetch(argv);} },
94          getNeighbors("<input> <id1,id2,..> <output>\n" +
95          		"\t- nearest neighborhood graph query (id1,id2 - of Entity sub-class only)")
96  		        {public void run(String[] argv) throws IOException{getNeighbors(argv);} },
97          summarize("<input> <output> [--model] [--pathways] [--hgnc-ids] [--uniprot-ids] [--chebi-ids]\n" +
98          		"\t- (experimental) summary of the input BioPAX model;\n " +
99  				"\truns one or several analyses and writes to the output file;\n " +
100 				"\t'--model' - (default) BioPAX classes, properties and values summary;\n " +
101 				"\t'--pathways' - pathways and sub-pathways hierarchy;\n " +
102 				"\t'--hgnc-ids' - HGNC IDs/Symbols that occur in sequence entity references;\n " +
103 				"\t'--uniprot-ids' - UniProt IDs in protein references;\n " +
104 				"\t'--chebi-ids' - ChEBI IDs in small molecule references;\n " +
105 				"\t'--uri-ids' - URI,type,name(s) and standard identifiers (in JSON format) for each physical entity;\n " +
106 				"\tthe options' order defines the results output order.")
107 		        {public void run(String[] argv) throws IOException{summarize(argv);} },
108 		blacklist("<input> <output>\n" +
109 		        "\t- creates a blacklist of ubiquitous small molecules, like ATP, \n"
110 		        + "\tfrom the BioPAX model and writes it to the output file. The blacklist can be used with\n "
111 		        + "\tpaxtools graph queries or when converting from the SAME BioPAX data to the SIF formats.")
112 				{public void run(String[] argv) throws IOException{blacklist(argv);} },
113 		pattern("\n\t- BioPAX pattern search tool (opens a new dialog window)")
114 				{public void run(String[] argv){pattern(argv);} },
115         help("\n\t- prints this screen and exits\n")
116 				{public void run(String[] argv){ help();} };
117 
118         String description;
119 
120         Command(String description) {
121             this.description = description;
122         }
123 
124         public abstract void run(String[] argv) throws IOException;
125     }

127 	private static void help() {
128 		System.out.println("(PaxtoolsMain Console) Available Operations:\n");
129 		for (Command cmd : Command.values()) {
130 			System.out.println(cmd.name() + " " + cmd.description);
131 		}
132 		System.out.println("Commands can also use compressed input files (only '.gz').\n");
133 	}
