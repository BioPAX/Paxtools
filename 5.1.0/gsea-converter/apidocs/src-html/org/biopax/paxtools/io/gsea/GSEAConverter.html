019/**
020 * Converts a BioPAX model to the GMT format (used by GSEA software).
021 * 
022 * It creates GSEA entries from sequence entity reference xrefs
023 * in the BioPAX model as follows:
024 *
025 *   Each entry (row) consists of three columns (tab separated):
026 * name (we use pathway URI),
027 * description (e.g. "name: Apoptosis; datasource: reactome; organism: 9606 idtype: uniprot"),
028 * and the list of identifiers (of the same type). For participants not associated with any pathway,
029 * "other" is used for the pathway name and uri.
030 *
031 *   The list may have one or more IDs of the same type per entity reference,
032 * e.g., UniProt IDs or HGNC Symbols; entity references that do not have any xref of
033 * given db/id type are ignored. Optionally, if there are less than three protein
034 * references per entry, it will not be printed.
035 * 
036 * Note, to effectively enforce cross-species violation, 
037 * 'organism' property and pathways must be set
038 * to a BioSource object that has a valid unification xref: 
039 * db="Taxonomy" and id= some valid taxonomy id.
040 *
041 * Note, this code assumes that the model has successfully been validated
042 * and perhaps normalized (using the BioPAX Validator, Paxtools Normalizer).
043 * A BioPAX L1 or L2 model is first converted to the L3.
044 */
045public class GSEAConverter
046{

163        /**
164         * Converts model to GSEA (GMT) and writes to out.
165         * See class declaration for more information.
166         *
167         * @param model Model
168         * @param out output stream to write the result to
169         * @throws IOException when there's an output stream error
170         */
171        public void writeToGSEA(final Model model, OutputStream out) throws IOException
172        {
173                Collection<GMTEntry> entries = convert(model);
174                if (entries.size() > 0)
175                {
176                        Writer writer = new OutputStreamWriter(out);
177                        for (GMTEntry entry : entries) {
178                                if ((minNumOfGenesPerEntry <= 1 && !entry.identifiers().isEmpty())
179                                                || entry.identifiers().size() >= minNumOfGenesPerEntry)
180                                {
181                                        writer.write(entry.toString() + "\n");
182                                }
183                        }
184                        writer.flush();
185                }
186        }

188        /**
189         * Creates GSEA entries from the pathways contained in the model.
190         * @param model Model
191         * @return a set of GSEA entries
192         */
193        public Collection<GMTEntry> convert(final Model model)
194        {
195                final Collection<GMTEntry> toReturn = new TreeSet<GMTEntry>(new Comparator<GMTEntry>() {
196                        @Override
197                        public int compare(GMTEntry o1, GMTEntry o2) {
198                                return o1.toString().compareTo(o2.toString());
199                        }
200                });
201
202                Model l3Model;
203                // convert to level 3 in necessary
204                if (model.getLevel() == BioPAXLevel.L2)
205                        l3Model = (new LevelUpgrader()).filter(model);
206                else
207                        l3Model = model;
208                
209                //a modifiable copy of the set of all PRs in the model - 
210                //after all, it has all the PRs that do not belong to any pathway
211                final Set<SequenceEntityReference> sequenceEntityReferences =
212                                new HashSet<SequenceEntityReference>(l3Model.getObjects(SequenceEntityReference.class));
213
214                final Set<Pathway> pathways = l3Model.getObjects(Pathway.class);
215                for (Pathway pathway : pathways) 
216                {
217                        String name = (pathway.getDisplayName() == null) ? pathway.getStandardName() : pathway.getDisplayName();
218                        if(name == null || name.isEmpty()) 
219                                name = pathway.getUri();
220
221                        final Pathway currentPathway = pathway;
222                        final String currentPathwayName = name;
223
224                        final boolean ignoreSubPathways =
225                                (!skipSubPathwaysOf.isEmpty() && shareSomeObjects(currentPathway.getDataSource(), skipSubPathwaysOf))
226                                        || skipSubPathways;
227
228                        LOG.debug("Begin converting " + currentPathwayName + " pathway, uri=" + currentPathway.getUri());
229
230                        // collect sequence entity references from current pathway
231                        //TODO: add Fetcher.evidenceFilter?
232                        Fetcher fetcher = new Fetcher(SimpleEditorMap.L3, Fetcher.nextStepFilter);
233                        fetcher.setSkipSubPathways(ignoreSubPathways);
234                        Set<SequenceEntityReference> pathwaySers = fetcher.fetch(currentPathway, SequenceEntityReference.class);
235
236                        if(!pathwaySers.isEmpty()) {
237                                LOG.debug("For pathway: " + currentPathwayName + " (" + currentPathway.getUri()
238                                                + "), got " + pathwaySers.size() + " sERs; now - grouping by organism...");
239                                Map<String,Set<SequenceEntityReference>> orgToPrsMap = organismToProteinRefsMap(pathwaySers);
240                                // create GSEA/GMT entries - one entry per organism (null organism also makes one)
241                                String dataSource = getDataSource(currentPathway.getDataSource());
242                                Collection<GMTEntry> entries = createGseaEntries(currentPathway.getUri(),
243                                                currentPathwayName, dataSource, orgToPrsMap);
244                                if(!entries.isEmpty())
245                                        toReturn.addAll(entries);
246                                sequenceEntityReferences.removeAll(pathwaySers);//keep not processed PRs (a PR can be processed multiple times)
247                                LOG.debug("- collected " + entries.size() + "entries.");
248                        }
249                }
250                
251                //when there're no pathways, only empty pathays, pathways w/o PRs, then use all/rest of PRs -
252                //organize PRs by species (GSEA s/w can handle only same species identifiers in a data row)
253                if(!sequenceEntityReferences.isEmpty() && !skipOutsidePathways) {
254                        LOG.info("Creating entries for the rest of PRs (outside any pathway)...");
255                        Map<String,Set<SequenceEntityReference>> orgToPrsMap = organismToProteinRefsMap(sequenceEntityReferences);
256                        if(!orgToPrsMap.isEmpty()) {
257                                // create GSEA/GMT entries - one entry per organism (null organism also makes one)
258                                if(model.getUri()==null) {
259                                        toReturn.addAll(createGseaEntries(
260                                                "other", "other", getDataSource(l3Model.getObjects(Provenance.class)), orgToPrsMap));
261                                } else {
262                                        toReturn.addAll(createGseaEntries(model.getUri()
263                                                , model.getName(), getDataSource(l3Model.getObjects(Provenance.class)), orgToPrsMap));
264                                }
265                        }
266                }
267                                        
268                return toReturn;
269        }
270
271        private Collection<GMTEntry> createGseaEntries(String uri, final String name, final String dataSource,
272                                                                                                   final Map<String, Set<SequenceEntityReference>> orgToPrsMap)
273        {
274                // generate GSEA entries for each species
275                final Collection<GMTEntry> toReturn = new ArrayList<GMTEntry>();
276                for (final String org : orgToPrsMap.keySet()) {
277                        if(orgToPrsMap.get(org).size() > 0) {
278                                GMTEntry GMTEntry = new GMTEntry(uri, org, idType,
279                                                String.format("name: %s; datasource: %s", name, dataSource));
280                                processEntityReferences(orgToPrsMap.get(org), GMTEntry);
281                                toReturn.add(GMTEntry);
282                        }
283                }
284                return toReturn;
285        }

288        //warn: there can be many equivalent BioSource objects (same taxonomy id, different URIs)
289        private Map<String, Set<SequenceEntityReference>> organismToProteinRefsMap(Set<SequenceEntityReference> seqErs)
290        {
291                Map<String,Set<SequenceEntityReference>> map = new HashMap<String, Set<SequenceEntityReference>>();
292
293                if(seqErs.isEmpty())
294                        throw new IllegalArgumentException("Empty set");
295                
296                if (crossSpeciesCheckEnabled) {
297                        for (SequenceEntityReference r : seqErs) {
298                                String key = getOrganismKey(r.getOrganism()); // null org. is ok (key == "")
299                                //collect PRs only from allowed organisms
300                                if(allowedOrganisms==null || allowedOrganisms.isEmpty() || allowedOrganisms.contains(key)) {
301                                        Set<SequenceEntityReference> sers = map.get(key);
302                                        if (sers == null) {
303                                                sers = new HashSet<SequenceEntityReference>();
304                                                map.put(key, sers);
305                                        }
306                                        sers.add(r);
307                                }
308                        }
309                } else {
310                        final Set<SequenceEntityReference> sers = new HashSet<SequenceEntityReference>();
311                        for (SequenceEntityReference r : seqErs) {
312                                String key = getOrganismKey(r.getOrganism());
313                                //collect PRs only from allowed organisms
314                                if(allowedOrganisms==null || allowedOrganisms.isEmpty() || allowedOrganisms.contains(key)) {
315                                        sers.add(r);
316                                }
317                        }
318                        map.put("", sers);
319                }
320                                
321                return map;
322        }

325        void processEntityReferences(Set<SequenceEntityReference> sers, GMTEntry targetEntry)
326        {
327                prs_loop: for (SequenceEntityReference ser : sers)
328                {
329                        // process PRs that belong to the same species (as targetEntry's) if crossSpeciesCheckEnabled==true
330                        if (crossSpeciesCheckEnabled && !targetEntry.taxID().equals(getOrganismKey(ser.getOrganism())))
331                                continue;
332                                
333                        if (idType != null && !idType.isEmpty())
334                        {
335                                final String db = idType.toLowerCase();
336                                // a shortcut if we are converting validated normalized BioPAX model:
337                                // get the primary ID from the URI of the ProteinReference
338                                final String uri = ser.getUri();
339                                if (uri.startsWith("http://identifiers.org/") && uri.contains(db))
340                                {
341                                        String accession = ser.getUri();
342                                        accession = accession.substring(accession.lastIndexOf("/") + 1);
343                                        targetEntry.identifiers().add(accession);
344                                }
345                                else {
346                                        int added = 0;
347                                        for (Xref aXref : ser.getXref())
348                                        {
349                                                if(aXref instanceof UnificationXref
350                                                                && aXref.getId() != null && aXref.getDb() != null
351                                                                && aXref.getDb().toLowerCase().startsWith(db))
352                                                {
353                                                        targetEntry.identifiers().add(aXref.getId());
354                                                        ++added;
355                                                }
356                                        }
357                                        if(added == 0) for (Xref aXref : ser.getXref())
358                                        {
359                                                if(aXref instanceof RelationshipXref
360                                                                && aXref.getId() != null && aXref.getDb() != null
361                                                                && aXref.getDb().toLowerCase().startsWith(db))
362                                                {
363                                                        targetEntry.identifiers().add(aXref.getId());
364                                                        ++added;
365                                                }
366                                        }
367
368                                        if(added > 12)
369                                                LOG.info("In GSEA entry: " + targetEntry.taxID() + " " + targetEntry.name() +
370                                                                ", sER " + ser.getUri() + " got " + added + " '" + db + "' identifiers...");
371                                }
372                        } else {
373                                // fallback - use URI (not really useful for the GSEA software)
374                                targetEntry.identifiers().add(ser.getUri());
375                        }
376                }
377        }

379        /*
380         * Gets datasource names, if any, in a consistent way/order, excl. duplicates
381         */
382        private String getDataSource(Set<Provenance> provenances)
383        {
384                if(provenances.isEmpty()) return "N/A";
385                
386                Set<String> dsNames = new TreeSet<String>();
387                for (Provenance provenance : provenances)
388                {
389                        String name = provenance.getDisplayName();
390                        if(name == null) 
391                                name = provenance.getStandardName();
392                        if(name == null && !provenance.getName().isEmpty()) 
393                                name = provenance.getName().iterator().next();
394                        if (name != null && name.length() > 0)
395                                dsNames.add(name.toLowerCase());
396                }
397                
398                return StringUtils.join(dsNames, ";");
399        }

402        private String getOrganismKey(BioSource org) {
403                String key = ""; //default value: unspecified/all species
404
405                if (org != null) {
406                        Set<Xref> xrefs = org.getXref();
407
408                        if(!xrefs.isEmpty()) {
409                                for (Xref xref : xrefs) {
410                                        if (xref instanceof UnificationXref
411                                                && xref.getDb().equalsIgnoreCase("taxonomy")) {
412                                                        if(key.isEmpty())
413                                                                key = xref.getId();
414                                                        else
415                                                                LOG.warn("BioSource " + org + " has multiple taxonomy unification xrefs; " +
416                                                                                "I will use " + key);
417                                        }
418                                }
419                        }
420
421                        //when there're no Taxonomy xrefs - use a name
422                        if(key.isEmpty()) {
423                                if (org.getStandardName()!=null) {
424                                        key = org.getStandardName().toLowerCase();
425                                } else if(org.getDisplayName()!=null) {
426                                        key = org.getDisplayName().toLowerCase();
427                                } else if(!org.getName().isEmpty()) {
428                                        key = org.getName().iterator().next().toLowerCase();
429                                }
430                        }
431                }
432
433                return key;
434        }

436        private boolean shareSomeObjects(Set<?> setA, Set<?> setB) {
437                return (!setA.isEmpty() && !setB.isEmpty())     ? !CollectionUtils.intersection(setA, setB).isEmpty() : false;
438        }
