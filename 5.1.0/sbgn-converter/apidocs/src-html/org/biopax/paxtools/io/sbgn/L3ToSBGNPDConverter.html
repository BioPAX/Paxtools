034/**
035 * This class converts BioPAX L3 model into SBGN PD (SBGN-ML XML).
036 * It optionally applies a special COSE layout.
037 *
038 * Currently, this converter ignores several BioPAX types/properties:
039 * <ul>
040 * <li>Parent-child relationship between entity references</li>
041 * <li>Binding features and covalent binding features of physical entities</li>
042 * <li>Pathway, PathwayStep, Evidence, etc.</li>
043 * </ul>
044 *
045 * Also note that:
046 * <ul>
047 * <li>Compartment is just a controlled vocabulary in BioPAX, so nesting and neighborhood relations
048 * between compartments are not handled here.</li>
049 * <li>Control structures in BioPAX and in SBGN PD are a little different. We use AND and NOT
050 * glyphs to approximate controls in BioPAX. However, AND-ing everything is not really proper,
051 * because BioPAX does not imply a logical operator between controllers.</li>
052 * </ul>
053 *
054 * @author Ozgun Babur
055 */
056public class L3ToSBGNPDConverter
057{

105        /**
106         * SBGN process glyph can be used to show reversible reactions. In that case two ports of the
107         * process will only have product glyphs. However, this creates an incompatibility with BioPAX:
108         * reversible biochemical reactions can have catalysis with a direction. But if we use a single
109         * glyph for the process and direct that catalysis to it, then the direction of the catalysis
110         * will be lost. If we use two process nodes for the reversible reaction (one for left-to-right
111         * and another for right-to-left), then we can direct the directed catalysis to only the
112         * relevant process glyph.
113         *
114         * Also note that the layout do not support ports. If layout is run, the ports are removed. In
115         * that case it will be impossible to distinguish left and right components of the reversible
116         * process glyph because they will all be product edges.
117         */
118        protected boolean useTwoGlyphsForReversibleConversion;

261        public void writeSBGN(Model model, OutputStream stream)
262        {
263                Sbgn sbgn = createSBGN(model);
264
265                try {
266                        JAXBContext context = JAXBContext.newInstance("org.sbgn.bindings");
267                        Marshaller marshaller = context.createMarshaller();

269                        marshaller.marshal(sbgn, stream);

271                catch (JAXBException e) {
272                        throw new RuntimeException("writeSBGN: JAXB marshalling failed", e);
273                }
274        }

276        /**
277         * Creates an SBGN object from the given BioPAX L3 model.
278         *
279         * Currently, it converts physical entities (interaction participants)
280         * and Conversion, Control, including their sub-classes, TemplateReaction,
281         * MolecularInteraction, GeneticInteraction, and base Interaction.class types.
282         * I does not convert other BioPAX entities and utility classes, such as:
283         * Pathway, Evidence, PathwayStep.
284         *
285         * @param model model to convert to SBGN
286         * @return SBGN representation of the BioPAX model
287         */
288        public Sbgn createSBGN(Model model)
289        {
290                assert model.getLevel().equals(BioPAXLevel.L3) : "This method only supports L3 graphs";
291
292                glyphMap = new HashMap<String, Glyph>();
293                compartmentMap = new HashMap<String, Glyph>();
294                arcMap = new HashMap<String, Arc>();
295                ubiqueSet = new HashSet<Glyph>();
296
297                int n = 0; //approximate number of SBGN nodes
298
299                // Create glyphs for Physical Entities
300                for (Entity entity : model.getObjects(Entity.class))
301                {
302                        if (needsToBeCreatedInitially(entity))
303                        {
304                                createGlyph(entity);
305                                ++n;
306                        }
307                }
308
309                // Create glyph for conversions and link with arcs
310                for (Interaction interaction : model.getObjects(Interaction.class))
311                {
312                        if(interaction.getParticipant().isEmpty())
313                                continue;
314
315                        // For each conversion we check if we need to create a left-to-right and/or right-to-left process.
316                        if(interaction instanceof Conversion) {
317                                Conversion conv = (Conversion) interaction;
318                                if (conv.getConversionDirection() == null ||
319                                                conv.getConversionDirection().equals(ConversionDirectionType.LEFT_TO_RIGHT) ||
320                                                (conv.getConversionDirection().equals(ConversionDirectionType.REVERSIBLE) &&
321                                                                useTwoGlyphsForReversibleConversion)) {
322                                        createProcessAndConnections(conv, ConversionDirectionType.LEFT_TO_RIGHT);
323                                } else if (conv.getConversionDirection() != null &&
324                                                (conv.getConversionDirection().equals(ConversionDirectionType.RIGHT_TO_LEFT) ||
325                                                                (conv.getConversionDirection().equals(ConversionDirectionType.REVERSIBLE)) &&
326                                                                                useTwoGlyphsForReversibleConversion)) {
327                                        createProcessAndConnections(conv, ConversionDirectionType.RIGHT_TO_LEFT);
328                                } else if (conv.getConversionDirection() != null &&
329                                                conv.getConversionDirection().equals(ConversionDirectionType.REVERSIBLE) &&
330                                                !useTwoGlyphsForReversibleConversion) {
331                                        createProcessAndConnections(conv, ConversionDirectionType.REVERSIBLE);
332                                }
333                        } else if(interaction instanceof TemplateReaction) {
334                                // Create glyph for template reactions and link with arcs
335                                createProcessAndConnections((TemplateReaction) interaction);
336                        } else if(interaction instanceof MolecularInteraction) {
337                                createMiProcess((MolecularInteraction) interaction);
338                        } else if(interaction instanceof GeneticInteraction) {
339                                createGiProcess((GeneticInteraction) interaction);
340                        } else if(!(interaction instanceof Control)) {
341                                createBasicProcess(interaction);
342                        }
343
344                        ++n;
345                }
346
347                // Register created objects into sbgn construct
348                final Sbgn sbgn = factory.createSbgn();
349                org.sbgn.bindings.Map map = new org.sbgn.bindings.Map();
350                sbgn.setMap(map);
351                map.setLanguage(Language.PD.toString());
352                map.getGlyph().addAll(getRootGlyphs(glyphMap.values()));
353                map.getGlyph().addAll(getRootGlyphs(ubiqueSet));
354                map.getGlyph().addAll(compartmentMap.values());
355                map.getArc().addAll(arcMap.values());
356
357                //TODO: store the BioPAX model's metadata (e.g., uri, name, provenance) in the SBGN Notes in JSON format
358//experimental - storing metadata as custom XML within Notes element
359//              biopaxMetaDoc.setDocumentURI(model.getUri()); //can be null
360//              Element elt = biopaxMetaDoc.createElementNS(BIOPAX_NS, "Model");
361//              elt.setPrefix("bp");
362//              elt.setAttribute("name", model.getName()); //can be null/empty too
363//              SBGNBase.Notes modelNotes = new SBGNBase.Notes();
364//              sbgn.setNotes(modelNotes);
365//              modelNotes.getAny().add(elt);
366
367                final boolean layout = doLayout && n < this.maxNodes && !arcMap.isEmpty();
368                try {
369                        //Must call this, although actual layout might never run;
370                        //in some real data tests, skipping createLayout method
371                        //led to malformed SBGN model, unfortunately...
372                        (new SBGNLayoutManager()).createLayout(sbgn, layout);
373                } catch (Exception e) {
374                        throw new RuntimeException("SBGN Layout of " + model.getXmlBase()
375                                        + ((model.getName()==null) ? "" : model.getName()) + " failed.", e);
376                }
377                if(!layout) log.warn(String.format("No layout, for either " +
378                                "it's disabled: %s, or ~ no. nodes > %s: %s, or - no edges: %s",
379                                !doLayout, maxNodes, n>maxNodes, arcMap.isEmpty()));
380
381                return sbgn; //modified sbgn (even when no layout is run)
382        }

410        /**
411         * Initially, we don't want to represent every PhysicalEntity or Gene node.
412         * For example, if a Complex is nested under another Complex,
413         * and if it is not a participant of any interaction,
414         * then we don't want to draw it separately; also skip for dangling entities.
415         *
416         * @param ent physical entity or gene (it returns false for other entity types) to test
417         * @return true if we want to draw this entity in SBGN; false - otherwise, or - to be auto-created later
418         */
419        private boolean needsToBeCreatedInitially(Entity ent)
420        {
421                boolean create = false;
422
423                if(ent instanceof PhysicalEntity || ent instanceof Gene) {
424                        if(ubiqueDet != null && ubiqueDet.isUbique(ent))
425                                create = false; // ubiques will be created where they are actually used.
426                        else if (!ent.getParticipantOf().isEmpty())
427                                create = true;
428                        else if(ent instanceof Complex && ((Complex) ent).getComponentOf().isEmpty()
429                                        && ((Complex) ent).getMemberPhysicalEntityOf().isEmpty())
430                                create = true; //do make a root/top complex despite it's dangling
431                }
432
433                return create;
434        }

436        /**
437         * Creates a glyph representing the given PhysicalEntity.
438         *
439         * @param e PhysicalEntity or Gene to represent
440         * @return the created glyph
441         */
442        private Glyph createGlyph(Entity e)
443        {
444                String id = convertID(e.getUri());
445                if (glyphMap.containsKey(id))
446                        return glyphMap.get(id);
447
448                // Create its glyph and register
449                Glyph g = createGlyphBasics(e, true);
450                glyphMap.put(g.getId(), g);
451
452                //TODO: export metadata (e.g., from bpe.getAnnotations() map) using the SBGN Extension feature
453//              SBGNBase.Extension ext = new SBGNBase.Extension();
454//              g.setExtension(ext);
455//              Element el = biopaxMetaDoc.createElementNS(BIOPAX_NS, e.getModelInterface().getSimpleName());
456//              el.setPrefix("bp");
457//              el.setAttribute("uri", e.getUri());
458//              ext.getAny().add(el);
459
460                if (g.getClone() != null)
461                        ubiqueSet.add(g);
462
463                if(e instanceof PhysicalEntity) {
464                        PhysicalEntity pe = (PhysicalEntity) e;
465                        assignLocation(pe, g);
466                        if("or".equalsIgnoreCase(g.getClazz())) {
467                                buildGeneric(pe, g, null);
468                        } else if (pe instanceof Complex) {
469                                createComplexContent((Complex) pe, g);
470                        }
471                }
472
473                return g;
474        }

491        /**
492         * This method creates a glyph for the given PhysicalEntity, sets its title and state variables
493         * if applicable.
494         *
495         * @param e PhysicalEntity or Gene to represent
496         * @param idIsFinal if ID is final, then it is recorded for future reference
497         * @return the glyph
498         */
499        private Glyph createGlyphBasics(Entity e, boolean idIsFinal)
500        {
501                Glyph g = factory.createGlyph();
502                g.setId(convertID(e.getUri()));
503
504                String s = typeMatchMap.get(e.getModelInterface());
505                if(( //use 'or' sbgn class for special generic physical entities
506                        e instanceof Complex && !((Complex)e).getMemberPhysicalEntity().isEmpty()
507                                && ((Complex) e).getComponent().isEmpty())
508                        ||
509                        (e instanceof SimplePhysicalEntity && ((SimplePhysicalEntity) e).getEntityReference()==null
510                                && !((SimplePhysicalEntity) e).getMemberPhysicalEntity().isEmpty()))
511                {
512                        s = OR.getClazz();
513                }
514                g.setClazz(s);
515
516                // Set the label
517                Label label = factory.createLabel();
518                label.setText(findLabelFor(e));
519                g.setLabel(label);
520
521                // Detect if ubique
522                if (ubiqueDet != null && ubiqueDet.isUbique(e))
523                {
524                        g.setClone(factory.createGlyphClone());
525                }
526
527                // Put on state variables
528                if (!g.getClazz().equals(OR.getClazz()))
529                {
530                        g.getGlyph().addAll(getInformation(e));
531                }
532
533                // Record the mapping
534                if (idIsFinal)
535                {
536                        Set<String> uris = new HashSet<String>();
537                        uris.add(e.getUri());
538                        sbgn2BPMap.put(g.getId(), uris);
539                }
540                return g;
541        }

543        /**
544         * Gets the representing glyph of the PhysicalEntity or Gene.
545         * @param e PhysicalEntity or Gene to get its glyph
546         * @param linkID Edge id, used if the Entity is ubique
547         * @return Representing glyph
548         */
549        private Glyph getGlyphToLink(Entity e, String linkID)
550        {
551                if (ubiqueDet == null || !ubiqueDet.isUbique(e))
552                {
553                        return glyphMap.get(convertID(e.getUri()));
554                }
555                else {
556                        // Create a new glyph for each use of ubique
557                        Glyph g = createGlyphBasics(e, false);
558                        g.setId(convertID(e.getUri()) + "_" + ModelUtils.md5hex(linkID));
559                        Set<String> uris = new HashSet<String>();
560                        uris.add(e.getUri());
561                        sbgn2BPMap.put(g.getId(), uris);
562                        if(e instanceof PhysicalEntity && ((PhysicalEntity)e).getCellularLocation() != null) {
563                                assignLocation((PhysicalEntity) e, g);
564                        }
565                        ubiqueSet.add(g);
566                        return g;
567                }
568        }

570        /**
571         * Fills in the content of a complex.
572         *
573         * @param cx Complex to be filled
574         * @param cg its glyph
575         */
576        private void createComplexContent(Complex cx, Glyph cg)
577        {
578                if (flattenComplexContent)
579                {
580                        for (PhysicalEntity mem : getFlattenedMembers(cx))
581                        {
582                                createComplexMember(mem, cg);
583                        }
584                }
585                else {
586                        for (PhysicalEntity mem : cx.getComponent())
587                        {
588                                if (mem instanceof Complex)
589                                {
590                                        addComplexAsMember((Complex) mem, cg);
591                                }
592                                else
593                                {
594                                        createComplexMember(mem, cg);
595                                }
596                        }
597                }
598        }

600        private void buildGeneric(PhysicalEntity generic, Glyph or, Glyph container)
601        {
602                assert "or".equalsIgnoreCase(or.getClazz()) : "must be 'or' glyph class";
603
604                for (PhysicalEntity m : generic.getMemberPhysicalEntity())
605                {
606                        Glyph g = createGlyphBasics(m,false);
607                        if(container!=null)
608                                container.getGlyph().add(g);
609                        String gid = g.getId() + "_" + ModelUtils.md5hex("memberof_" + or.getId());
610                        g.setId(gid);
611                        glyphMap.put(gid, g);
612                        Set<String> uris =  new HashSet<String>();
613                        uris.add(m.getUri());
614                        sbgn2BPMap.put(gid, uris);
615                        assignLocation(m, g);
616                        createArc(g, or, LOGIC_ARC.getClazz(), null);
617                        if(m instanceof Complex)
618                                createComplexContent((Complex) m, g);
619                }
620        }

623        /**
624         * Recursive method for creating the content of a complex. A complex may contain other complexes
625         * (bad practice), but SBGN needs them flattened. If an inner complex is empty, then we
626         * represent it using a glyph. Otherwise we represent only the members of this inner complex,
627         * merging them with the most outer complex.
628         *
629         * @param cx inner complex to add as member
630         * @param container glyph for most outer complex
631         */
632        private void addComplexAsMember(Complex cx, Glyph container)
633        {
634                // Create a glyph for the inner complex
635                Glyph inner = createComplexMember(cx, container);
636
637                for (PhysicalEntity mem : cx.getComponent())
638                {
639                        if (mem instanceof Complex)
640                        {
641                                // Recursive call for inner complexes
642                                addComplexAsMember((Complex) mem, inner);
643                        }
644                        else
645                        {
646                                createComplexMember(mem, inner);
647                        }
648                }
649        }

651        /**
652         * Gets the members of the Complex that needs to be displayed in a flattened view.
653         * @param cx to get members
654         * @return members to display
655         */
656        private Set<PhysicalEntity> getFlattenedMembers(Complex cx)
657        {
658                Set<PhysicalEntity> set = new HashSet<PhysicalEntity>();
659
660                for (PhysicalEntity mem : cx.getComponent())
661                {
662                        if (mem instanceof Complex)
663                        {
664                                if (!hasNonComplexMember((Complex) mem))
665                                {
666                                        set.add(mem);
667                                }

724        /**
725         * Looks for the display name of this PhysicalEntity. If there is none, then it looks for the
726         * display name of its EntityReference. If still no name at hand, it tries the standard
727         * name, and then first element in name lists.
728         *
729         * A good BioPAX file will use a short and specific name (like HGNC symbols) as displayName.
730         *
731         * @param pe PhysicalEntity or Gene to find a name
732         * @return a name for labeling
733         */
734        private String findLabelFor(Entity pe)
735        {
736                // Use gene symbol of PE
737                for (Xref xref : pe.getXref())
738                {
739                        String sym = extractGeneSymbol(xref);
740                        if (sym != null) return sym;
741                }
742
743                // Use gene symbol of ER
744                EntityReference er = null;
745
746                if (pe instanceof SimplePhysicalEntity)
747                {
748                        er = ((SimplePhysicalEntity) pe).getEntityReference();
749                }
750
751                if (er != null)
752                {
753                        for (Xref xref : er.getXref())
754                        {
755                                String sym = extractGeneSymbol(xref);
756                                if (sym != null) return sym;
757                        }
758                }
759                
760                // Use display name of entity
761                String name = pe.getDisplayName();
762
763                if (name == null || name.trim().isEmpty())
764                {
765                        if (er != null)
766                        {
767                                // Use display name of reference
768                                name = er.getDisplayName();
769                        }
770
771                        if (name == null || name.trim().isEmpty())
772                        {
773                                // Use standard name of entity
774                                name = pe.getStandardName();
775
776                                if (name == null || name.trim().isEmpty())
777                                {
778                                        if (er != null)
779                                        {
780                                                // Use standard name of reference
781                                                name = er.getStandardName();
782                                        }
783
784                                        if (name == null || name.trim().isEmpty())
785                                        {
786                                                if (!pe.getName().isEmpty())
787                                                {
788                                                        // Use first name of entity
789                                                        name = pe.getName().iterator().next();
790                                                }
791                                                else if (er != null && !er.getName().isEmpty())
792                                                {
793                                                        // Use first name of reference
794                                                        name = er.getName().iterator().next();
795                                                }
796                                        }
797                                }
798                        }
799                }
800
801                // Search for the shortest name of chemicals
802                if (pe instanceof SmallMolecule)
803                {
804                        String shortName = getShortestName((SmallMolecule) pe);
805
806                        if (shortName != null)
807                        {
808                                if (name == null || (shortName.length() < name.length() &&
809                                        !shortName.isEmpty()))
810                                {
811                                        name = shortName;
812                                }
813                        }
814                }
815                
816                if (name == null || name.trim().isEmpty())
817                {
818                        // Don't leave it without a name
819                        name = "noname";
820                }
821
822                return name;
823        }

851        /**
852         * Searches for gene symbol in Xref.
853         * @param xref Xref to search
854         * @return gene symbol
855         */
856        private String extractGeneSymbol(Xref xref)
857        {
858                if (xref.getDb() != null && (
859                        xref.getDb().equalsIgnoreCase("HGNC Symbol") ||
860                        xref.getDb().equalsIgnoreCase("Gene Symbol") ||
861                        xref.getDb().equalsIgnoreCase("HGNC")))
862                {
863                        String ref = xref.getId();
864
865                        if (ref != null)
866                        {
867                                ref = ref.trim();
868                                if (ref.contains(":")) ref = ref.substring(ref.indexOf(":") + 1);
869                                if (ref.contains("_")) ref = ref.substring(ref.indexOf("_") + 1);
870
871                                // if the reference is an HGNC ID, then convert it to a symbol
872                                if (!HGNC.containsSymbol(ref) && Character.isDigit(ref.charAt(0)))
873                                {
874                                        ref = HGNC.getSymbol(ref);
875                                }
876                        }
877                        return ref;
878                }
879                return null;
880        }

1239        /**
1240         * Creates or gets the glyph to connect to the control arc.
1241         *
1242         * @param ctrl Control to represent
1243         * @return glyph representing the controller tree
1244         */
1245        private Glyph createControlStructure(Control ctrl)
1246        {
1247                Glyph cg;
1248
1249                Set<PhysicalEntity> controllers = getControllers(ctrl);
1250
1251                // If no representable controller found, skip this control
1252                if (controllers.isEmpty())
1253                {
1254                        cg = null;
1255                }
1256                else if (controllers.size() == 1 && getControllerSize(ctrl.getControlledOf()) == 0)
1257                { // If there is only one controller with no modulator, put an arc for controller
1258                        cg = getGlyphToLink(controllers.iterator().next(), convertID(ctrl.getUri()));
1259                }
1260                else
1261                {
1262                        // This list will contain handles for each participant of the AND structure
1263                        List<Glyph> toConnect = new ArrayList<Glyph>();
1264
1265                        // Bundle controllers if necessary
1266                        Glyph gg = handlePEGroup(controllers, convertID(ctrl.getUri()));
1267                        if(gg != null)
1268                                toConnect.add(gg);
1269
1270                        // Create handles for each controller
1271                        for (Control ctrl2 : ctrl.getControlledOf())
1272                        {
1273                                Glyph g = createControlStructure(ctrl2);
1274                                if (g != null)
1275                                {
1276                                        // If the control is negative, add a NOT in front of it
1277                                        if (getControlType(ctrl2).equals(INHIBITION.getClazz()))
1278                                        {
1279                                                g = addNOT(g);
1280                                        }
1281
1282                                        toConnect.add(g);
1283                                }
1284                        }
1285
1286                        // Handle co-factors of catalysis
1287                        if (ctrl instanceof Catalysis)
1288                        {
1289                                Set<PhysicalEntity> cofs = ((Catalysis) ctrl).getCofactor();
1290                                Glyph g = handlePEGroup(cofs, convertID(ctrl.getUri()));
1291                                if (g != null) 
1292                                        toConnect.add(g);
1293                        }
1294
1295                        if (toConnect.isEmpty()) {
1296                                return null;
1297                        }
1298                        else if (toConnect.size() == 1)
1299                        {
1300                                cg = toConnect.iterator().next();
1301                        }
1302                        else
1303                        {
1304                                cg = connectWithAND(toConnect);
1305                        }
1306                }
1307
1308                return cg;
1309        }

1355        /**
1356         * Creates an AND glyph downstream of the given glyphs.
1357         *
1358         * @param gs upstream glyph list
1359         * @return AND glyph
1360         */
1361        private Glyph connectWithAND(List<Glyph> gs)
1362        {
1363                // Compose an ID for the AND glyph
1364                StringBuilder sb = new StringBuilder();
1365                Iterator<Glyph> iterator = gs.iterator();
1366                if(iterator.hasNext())
1367                    sb.append(iterator.next());
1368                while (iterator.hasNext()) {
1369                        sb.append("-AND-").append(iterator.next().getId());
1370                }
1371                String id = ModelUtils.md5hex(sb.toString()); //shorten a very long id
1372
1373                // Create the AND glyph if not exists
1374                Glyph and = glyphMap.get(id);
1375                if (and == null) {
1376                        and = factory.createGlyph();
1377                        and.setClazz(AND.getClazz());
1378                        and.setId(id);
1379                        glyphMap.put(and.getId(), and);
1380                }
1381
1382                // Connect upstream to the AND glyph
1383                for (Glyph g : gs) {
1384                        createArc(g, and, LOGIC_ARC.getClazz(), null);
1385                }
1386
1387                return and;
1388        }

1417         * Converts the control type of the Control to the SBGN classes.
1418         *
1419         * @param ctrl Control to get its type
1420         * @return SBGN type of the Control
1421         */
1422        private String getControlType(Control ctrl)
1423        {
1424                if (ctrl instanceof Catalysis)
1425                {
1426                        // Catalysis has its own class
1427                        return CATALYSIS.getClazz();
1428                }
1429
1430                ControlType type = ctrl.getControlType();
1431                if (type == null)
1432                {
1433                        // Use stimulation as the default control type
1434                        return STIMULATION.getClazz();
1435                }
1436
1437                // Map control type to stimulation or inhibition
1438
1439                switch (type)
1440                {
1441                        case ACTIVATION:
1442                        case ACTIVATION_ALLOSTERIC:
1443                        case ACTIVATION_NONALLOSTERIC:
1444                        case ACTIVATION_UNKMECH:

1446                        case INHIBITION:

1452                        case INHIBITION_UNCOMPETITIVE:
1453                        case INHIBITION_NONCOMPETITIVE:
1454                                return INHIBITION.getClazz();
1455                }
1456
1457                throw new RuntimeException("Invalid control type: " + type);

1513        /*
1514         * Creates an arc from the source to the target, and sets its class to the specified clazz.
1515         * Puts the new arc in the sullied arcMap.
1516         *
1517         * @param source source of the arc -- either Glyph or Port
1518         * @param target target of the arc -- either Glyph or Port
1519         * @param clazz class of the arc
1520         */
1521        private void createArc(Object source, Object target, String clazz, Stoichiometry stoic)
1522        {
1523                assert source instanceof Glyph || source instanceof Port : "source = " + source;
1524                assert target instanceof Glyph || target instanceof Port : "target = " + target;
1525
1526                Arc arc = factory.createArc();
1527                arc.setSource(source);
1528                arc.setTarget(target);
1529                arc.setClazz(clazz);
1530                
1531                String sourceID = source instanceof Glyph ?
1532                        ((Glyph) source).getId() : ((Port) source).getId();
1533                String targetID = target instanceof Glyph ?
1534                        ((Glyph) target).getId() : ((Port) target).getId();
1535
1536                arc.setId(sourceID + "--TO--" + targetID);
1537
1538                if (stoic != null && stoic.getStoichiometricCoefficient() > 1)
1539                {
1540                        Glyph card = factory.createGlyph();
1541                        card.setClazz(CARDINALITY.getClazz());
1542                        Label label = factory.createLabel();
1543                        label.setText(new DecimalFormat("0.##").format(stoic.getStoichiometricCoefficient()));
1544                        card.setLabel(label);
1545                        arc.getGlyph().add(card);
1546//                      card.setId(convertID(stoic.getUri()+"_"+arc.getId()));
1547                }
1548
1549                Arc.Start start = new Arc.Start();
1550                start.setX(0);
1551                start.setY(0);
1552                arc.setStart(start);
1553
1554                Arc.End end = new Arc.End();
1555                end.setX(0);
1556                end.setY(0);
1557                arc.setEnd(end);
1558
1559                arcMap.put(arc.getId(), arc);
1560        }
