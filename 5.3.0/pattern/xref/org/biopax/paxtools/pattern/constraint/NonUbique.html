1   package org.biopax.paxtools.pattern.constraint;

3   import org.biopax.paxtools.model.level3.PhysicalEntity;
4   import org.biopax.paxtools.pattern.Match;
5   import org.biopax.paxtools.pattern.util.Blacklist;
6   
7   /**
8    * This is a non-generative constraint that checks if the small molecule is ubiquitous in any
9    * context.
10   *
11   * Var0 - PhysicalEntity
12   *
13   * @author Ozgun Babur
14   */
15  public class NonUbique extends ConstraintAdapter
16  {
17  	public NonUbique(Blacklist blacklist)
18  	{
19  		super(1, blacklist);
20  	}
21  
22  	@Override
23  	public boolean satisfies(Match match, int... ind)
24  	{
25  		PhysicalEntity pe = (PhysicalEntity) match.get(ind[0]);
26  		return !blacklist.isUbique(pe);
27  	}
28  }
