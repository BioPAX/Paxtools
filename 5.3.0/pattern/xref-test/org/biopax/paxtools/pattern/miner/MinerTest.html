14  /**
15   * This class demonstrates a typical use of this framework. A user is defining a new pattern and
16   * how to extract the result in a text file, and launches the search dialog.
17   *
18   * The pattern captures two proteins appear to be associated with the same molecular complex.
19   *
20   * @author Ozgun Babur
21   */
22  @Ignore
23  public class MinerTest
24  {
25  	public static void main(String[] args)
26  	{
27  		// Define and initialize the miner.
28  
29  		Miner miner = new MinerAdapter("Appear-in-same-complex", "The pattern captures two " +
30  			"proteins appear to be members of the same complex. There may be a nesting hierarchy " +
31  			"in the complex, and the proteins can be represented with generic entities, again " +
32  			"through multiple generic-member relations.")
33  		{
34  			/**
35  			 * The pattern is composed of two proteins associated to a complex as members. The
36  			 * relation can be through nested memberships or through generic relations.
37  			 */
38  			public Pattern constructPattern()
39  			{
40  				Pattern p = new Pattern(ProteinReference.class, "Protein 1");
41  				p.add(ConBox.erToPE(), "Protein 1", "SPE1");
42  				p.add(ConBox.linkToComplex(), "SPE1", "Complex");
43  				p.add(new Type(Complex.class), "Complex");
44  				p.add(ConBox.linkToSpecific(), "Complex", "SPE2");
45  				p.add(ConBox.peToER(), "SPE2", "Protein 2");
46  				p.add(ConBox.equal(false), "Protein 1", "Protein 2");
47  				p.add(new Type(ProteinReference.class), "Protein 2");
48  
49  				return p;
50  			}
51  
52  			/**
53  			 * Writes the result as "P1 P2", where P1 and P2 are gene symbols of proteins and the
54  			 * whitespace is tab. The relation is undirected, so "P2 P1" is treated as the same
55  			 * relation.
56  			 */
57  			public void writeResult(Map<BioPAXElement, List<Match>> matches, OutputStream out)
58  				throws IOException
59  			{
60  				writeResultAsSIF(matches, out, false, "Protein 1", "Protein 2");
61  			}
62  		};
63  
64  		// Launch the GUI that will assist choosing a source model, and output file name
65  		Dialog d = new Dialog(miner);
66  		d.setVisible(true);
67  	}
68  }
