113 	private void printOrdered(final Map<String, Integer> map, int min)
114 	{
115 		List<String> list = new ArrayList<String>(map.keySet());
116 		Collections.sort(list, new Comparator<String>()
117 		{
118 			@Override
119 			public int compare(String o1, String o2)
120 			{
121 				return new Integer(map.get(o2)).compareTo(map.get(o1));
122 			}
123 		});
124 
125 		for (String s : list)
126 		{
127 			Integer cnt = map.get(s);
128 			if (cnt < min) break;
129 			System.out.println(s + "\t" + cnt);
130 		}
131 	}

227 	{
228 		Set<String> set = new HashSet<String>();
229 		for (String s : normal)
230 		{
231 			String[] tok = s.split("\t");
232 			set.add(tok[1] + "\t" + tok[0]);
233 		}

247 	private int conflict(Set<String> set1, Set<String> set2)
248 	{
249 		int cnt  = 0;
250 		for (String s : set1)
251 		{
252 			String r = s.substring(s.indexOf("\t") + 1) + "\t" + s.substring(0, s.indexOf("\t"));
253 			if (!set1.contains(r) && set2.contains(r) && !set2.contains(s)) cnt ++;
254 		}
255 		return cnt;
256 	}

259 	{
260 		int cnt = 0;
261 		for (String s : set)
262 		{
263 			String[] tok = s.split("\t");
264 			String a = tok[1] + "\t" + tok[0];
265 			if (set.contains(a)) cnt ++;
266 		}
267 		return cnt / 2;
268 	}

350 	private Set<String>[] parseSpikeControl(String id)
351 	{
352 		int ind = id.indexOf("PROMOTES");
353 		if (ind < 0) ind = id.indexOf("INHIBITS");
354 		if (ind < 0) ind = id.indexOf("UNKNOWN");
355 
356 		if (ind < 0)
357 		{
358 			System.err.println("no kywd = " + id);
359 			return null;
360 		}
361 
362 		String up = id.substring(0, ind);
363 		String down = id.substring(id.indexOf(".", ind));
364 
365 		return new Set[]{collectSymbols(up), collectSymbols(down)};
366 	}

368 	private Set<String> collectSymbols(String s)
369 	{
370 		Set<String> set = new HashSet<String>();
371 
372 		s = s.replaceAll("\\.", " ");
373 		s = s.replaceAll(",", " ");
374 
375 		for (String x : s.split(" "))
376 		{
377 			x = HGNC.getSymbol(x);
378 			if (x != null) set.add(x);
379 		}
380 		return set;
381 	}
