3   import org.biopax.paxtools.model.level3.Entity;

5   import java.util.Set;
6   
7   /**
8    * Detects ubiquitous molecules using a given ID set.
9    *
10   * @author Ozgun Babur
11   */
12  public class ListUbiqueDetector implements UbiqueDetector
13  {
14  	Set<String> uris;
15  
16  	/**
17  	 * Contructor.
18  	 *
19  	 * @param uris URIs of ubiques
20  	 */
21  	public ListUbiqueDetector(Set<String> uris)
22  	{
23  		this.uris = uris;
24  	}
25  
26  	/**
27  	 * Checks if the ID of the PhysicalEntity is in the set.
28  	 * @param e PhysicalEntity or Gene to check
29  	 * @return true if ubique
30  	 */
31  	public boolean isUbique(Entity e)
32  	{
33  		return uris.contains(e.getUri());
34  	}
35  }
