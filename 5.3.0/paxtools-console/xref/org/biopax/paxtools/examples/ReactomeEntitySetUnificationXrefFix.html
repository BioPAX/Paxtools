17  /**
18   * This is a class I wrote for fixing the unification xref problem in Reactome entity sets. It
19   * reads in the model, goes over physical entities and looks for the converted from
20   * reactome entity set comment. For each such generic entities it converts all of its unification
21   * xrefs to relationship xrefs.
22   *
23   * NOTE: This method is now outdated as it is easier to do this now with the new PathAccessors. I did not remove this
24   * example, however, as it demonstrates many "low-level" operations of Paxtools.
25   */
26  public class ReactomeEntitySetUnificationXrefFix
27  {
28  	public static void main(String[] args) throws IOException
29  	{
30  		if (args.length == 2)
31  		{
32  			fixReactome(new FileInputStream(new File(args[0])),
33  					new FileOutputStream(new File(args[1])));
34  		}
35  		else
36  		fixReactome(System.in, System.out);
37  	}
38  
39  	public static void fixReactome(InputStream in, OutputStream out) throws IOException
40  	{
41  		BioPAXIOHandler io = new SimpleIOHandler();
42  		Model level2 = io.convertFromOWL(in);
43  		Set<physicalEntity> physicalEntitySet = new HashSet<physicalEntity>();
44  		physicalEntitySet.addAll(level2.getObjects(physicalEntity.class));
45  		for (physicalEntity pe : physicalEntitySet)
46  		{
47  			boolean entitySet = false;
48  			for (String comment : pe.getCOMMENT())
49  			{
50  				if (comment.contains("Converted from EntitySet in Reactome"))
51  				{
52  					entitySet = true;
53  					break;
54  
55  				}
56  
57  			}
58  			if (entitySet)
59  			{
60  				Set<unificationXref> unis = new HashSet<unificationXref>();
61  				unis.addAll(new ClassFilterSet<xref,unificationXref>(pe.getXREF(), unificationXref.class));
62  				
63  				for (unificationXref uni : unis)
64  				{
65  					relationshipXref rel;
66  					String rid = uni.getUri() + "-r";
67  					BioPAXElement exists = level2.getByID(rid);
68  					if (exists != null)
69  					{
70  						rel = (relationshipXref) exists;
71  					}
72  					else
73  					{
74  						rel = level2.addNew(relationshipXref.class, rid);
75  						rel.setDB(uni.getDB());
76  						rel.setID(uni.getID());
77  					}
78  					pe.removeXREF(uni);
79  					if (uni.isXREFof().isEmpty())
80  					{
81  						level2.remove(uni);
82  					}
83  					pe.addXREF(rel);
84  
85  				}
86  			}
87  
88  		}
89  
90  		io.convertToOWL(level2, out);
91  		out.close();
92  	}
93  }
