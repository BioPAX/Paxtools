187 	@Override protected Map<String, String> readNameSpaces()
188 	{
189 		Map<String, String> ns = new HashMap<String, String>();
190 		try
191 		{
192 			if (r.getEventType() == START_DOCUMENT)
193 			{
194 				r.next();
195 			} else
196 			{
197 				throw new BioPaxIOException("Unexpected element at start");
198 			}
199 
200 			// Skip any comment before we read the RDF headers
201 			while (r.getEventType() == COMMENT)
202 			{
203 				r.next();
204 			}
205 
206 			if (r.getEventType() != START_ELEMENT)
207 			{
208 				throw new BioPaxIOException("Unexpected element at start: " + r.getEventType());
209 			}
210 
211 			if (!r.getLocalName().equalsIgnoreCase("rdf"))
212 			{
213 				throw new BioPaxIOException("Unexpected element at start: " + r.getLocalName());
214 			}
215 
216 			int count = r.getNamespaceCount();
217 			for (int i = 0; i < count; i++)
218 			{
219 				String pre = r.getNamespacePrefix(i);
220 				if (pre == null)
221 				{
222 					pre = "";
223 				}
224 				String namespace = r.getNamespaceURI(pre);
225 
226 				ns.put(pre, namespace);
227 			}
228 
229 			base = null;
230 			for (int i = 0; i < r.getAttributeCount(); i++)
231 			{
232 				if ("base".equalsIgnoreCase(r.getAttributeLocalName(i)) &&
233 				    "xml".equalsIgnoreCase(r.getAttributePrefix(i)))
234 				{
235 					base = r.getAttributeValue(i);
236 				}
237 			}
238 		}
239 		catch (XMLStreamException e)
240 		{
241 			throw new BioPaxIOException(e.getClass().getSimpleName() + " " + e.getMessage() + "; " + e.getLocation());
242 		}
243 		return ns;
244 
245 	}

247 	@Override protected void createAndBind(Model model)
248 	{
249 		try
250 		{
251 			int type;
252 			while ((type = r.getEventType()) != END_DOCUMENT)
253 			{
254 				switch (type)
255 				{
256 					case START_ELEMENT:
257 						if (BioPAXLevel.isInBioPAXNameSpace(r.getName().getNamespaceURI()))
258 						{
259 							String lname = r.getLocalName();
260 							BioPAXLevel level = getLevel();
261 							Class<? extends BioPAXElement> clazz;
262 							try
263 							{
264 								clazz = level.getInterfaceForName(lname);
265 							}
266 							catch (IllegalBioPAXArgumentException e)
267 							{
268 								log.error("Unknown BioPAX element location " + lname + " at " + r.getLocation());
269 								throw e; // for backward compatibility
270 							}
271 							if (this.getFactory().canInstantiate(clazz))
272 							{
273 								processIndividual(model);
274 							} else
275 							{
276 								if (log.isTraceEnabled())
277 									log.trace("Ignoring element: " + lname);
278 								skip();
279 							}
280 
281 						}
282 						break;
283 					case CHARACTERS:
284 						if (log.isTraceEnabled())
285 						{
286 							StringBuilder sb = new StringBuilder("Ignoring text ");
287 							if (r.hasName()) sb.append(r.getLocalName());
288 							if (r.hasText()) sb.append(r.getText());
289 							if (log.isTraceEnabled())
290 								log.trace(sb.toString());
291 						}
292 						break;
293 					case END_ELEMENT:
294 						if (log.isTraceEnabled())
295 							log.trace(String.valueOf(r));
296 						break;
297 					default:
298 						if (log.isTraceEnabled())
299 							log.trace("Test this!:" + type);
300 				}
301 				r.next();
302 
303 			}
304 		r.close();
305 		}
306 		catch (XMLStreamException e)
307 		{
308 			throw new BioPaxIOException(e.getClass().getSimpleName() + " " + e.getMessage() + "; " + e.getLocation());
309 		}
310 
311 		for (Triple triple : triples)
312 		{
313 			try
314 			{
315 				bindValue(triple, model);
316 			}
317 			catch (IllegalBioPAXArgumentException e)
318 			{
319 				log.warn("Binding " + e);
320 			}
321 		}
322 
323 	}

345 	private String processIndividual(Model model) throws XMLStreamException
346 	{
347 		String s = r.getLocalName();
348 		String id;
349 		try
350 		{
351 			id = getId();
352 		}
353 		catch (NullPointerException e)
354 		{
355 			throw new BioPaxIOException("Error processing individual " + s + ". rdf:ID or rdf:about not found!", e);
356 		}
357 
358 		if (factory.canInstantiate((level.getInterfaceForName(s))))
359 		{
360 			BioPAXElement bpe = model.getByID(id);
361 			if (!mergeDuplicates || bpe == null)
362 			{
363 				createBpe(s, id, model); //throws an exception when (mergeDuplicates==false && bpe!=null)
364 			} else if(!s.equals(bpe.getModelInterface().getSimpleName())) {
365 				//i.e., type mismatch,
366 				//whereas (mergeDuplicates==true && bpe != null) is true already -
367 				throw new BioPaxIOException(String.format("processIndividual: " +
368 						"despite mergeDuplicates is True, failed/skipped creating an instance " +
369 						"of %s, URI:%s, because previously added object (same URI) was of different type: %s",
370 						s, id, bpe.getModelInterface().getSimpleName()));
371 			}
372 		} else {
373 			if (r.hasText())
374 			{
375 				log.warn("Unknown class :" + r.getText());
376 			} else
377 			{
378 				log.warn("Unknown class :" + r);
379 			}
380 			skip();
381 		}
382 		propertyContext = true;
383 		r.next();
384 		while (r.getEventType() != END_ELEMENT)
385 		{
386 			if (r.getEventType() == START_ELEMENT)
387 			{
388 				processProperty(model, id);
389 				propertyContext = true;
390 
391 			}
392 			r.next();
393 		}
394 		return id;
395 	}

403 	private void skip() throws XMLStreamException
404 	{
405 		int depth = 1;
406 		while (!(r.getEventType() == END_ELEMENT && depth == 0))
407 		{
408 			r.next();
409 			switch (r.getEventType())
410 			{
411 				case START_ELEMENT:
412 					depth++;
413 					break;
414 				case END_ELEMENT:
415 					depth--;
416 					break;
417 			}
418 		}
419 	}

421 	public String getId()
422 	{
423 		String id = r.getAttributeValue(rdf, "ID");
424 		if (id == null)
425 		{
426 			id = r.getAttributeValue(rdf, "about");
427 			if (id.startsWith("#"))
428 			{
429 				id = base + id.substring(1, id.length());
430 
431 			}
432 		} else if (base != null)
433 		{
434 			id = base + id;
435 		}
436 
437 		return id;
438 	}

440 	private void processProperty(Model model, String ownerID) throws XMLStreamException
441 	{
442 		if (rdfs.equals(r.getNamespaceURI()) && "comment".equals(r.getLocalName()))
443 		{
444 			BioPAXElement paxElement = model.getByID(ownerID);
445 			AbstractPropertyEditor commentor = getRDFCommentEditor(paxElement);
446 			r.next();
447 			assert r.getEventType() == CHARACTERS;
448 			String text = r.getText();
449 			if(text!=null) text = text.trim();
450 			commentor.setValueToBean(text, paxElement);
451 			log.warn("rdfs:comment is converted into the bp:comment; " +
452 			         "however, this can be overridden " +
453 			         "if there exists another bp:comment (element: " +
454 			         paxElement.getUri() + " text: " + text + ")");
455 			gotoEndElement();
456 		} else if (bp != null && bp.equals(r.getNamespaceURI()))
457 		{
458 			String property = r.getLocalName();
459 			String resource = r.getAttributeValue(rdf, "resource");
460 			if (resource != null)
461 			{
462 				if (resource.startsWith("#"))
463 				{
464 					resource = (base == null ? "" : base) + resource.substring(1, resource.length());
465 				}
466 				gotoEndElement();
467 			} else
468 			{
469 				r.next();
470 				boolean found = false;
471 				while (r.getEventType() != END_ELEMENT)
472 				{
473 					if (!found && r.getEventType() == CHARACTERS)
474 					{
475 						StringBuilder buff = new StringBuilder(r.getText());
476 						r.next();
477 						while (r.getEventType() == CHARACTERS)
478 						{
479 							buff.append(r.getText());
480 							r.next();
481 						}
482 						resource = buff.toString();
483 
484 					} else if (r.getEventType() == START_ELEMENT)
485 					{
486 						propertyContext = false;
487 						resource = processIndividual(model);
488 						found = true;
489 						r.next();
490 					} else r.next();
491 				}
492 				resource = (!found && resource != null) ? resource.replaceAll("[\n\r\t ]+", " ") : resource;
493 			}
494 
495 			log.trace("setting = " + resource);
496 			triples.add(new Triple(ownerID, resource, property));
497 			propertyContext = false;
498 		} else
499 		{
500 			log.trace("ignoring unknown element " +
501 			          r.getNamespaceURI() + r.getLocalName());
502 			gotoEndElement();
503 		}
504 
505 	}
506 
507 	private void gotoEndElement() throws XMLStreamException
508 	{
509 		while (r.getEventType() != END_ELEMENT)
510 		{
511 			r.next();
512 		}
513 	}

537 	/**
538 	 * Converts a model into BioPAX (OWL) format, and writes it into
539 	 * the outputStream. Saved data can be then read via {@link BioPAXIOHandler}
540 	 * interface (e.g., {@link SimpleIOHandler}).
541 	 *
542 	 * Note: When the model is incomplete (i.e., contains elements that refer externals,
543 	 * dangling BioPAX elements) and is exported by this method, it works; however one
544 	 * will find corresponding object properties set to NULL later,
545 	 * after converting such data back to Model.
546 	 * 
547 	 * Note: if the model is too large, and the output stream is a byte array stream,
548 	 * then you can eventually get OutOfMemoryError "Requested array size exceeds VM limit"
549 	 * (max. array size is 2Gb)
550 	 * 
551 	 * @param model model to be converted into OWL format
552 	 * @param outputStream output stream into which the output will be written
553 	 * @throws BioPaxIOException in case of I/O problems
554 	 */
555 	public void convertToOWL(Model model, OutputStream outputStream)
556 	{
557 		initializeExporter(model);
558 
559 		try
560 		{
561 			Writer out = new BufferedWriter(new OutputStreamWriter(outputStream, "UTF-8"));
562 			writeObjects(out, model);
563 			out.close();
564 		}
565 		catch (IOException e)
566 		{
567 			throw new BioPaxIOException("Cannot convert to OWL!", e);
568 		}
569 	}

572 	/**
573 	 * Writes the XML representation of individual BioPAX element that
574 	 * is BioPAX-like but only for display or debug purpose (incomplete).
575 	 *
576 	 * Note: use {@link BioPAXIOHandler#convertToOWL(org.biopax.paxtools.model.Model, java.io.OutputStream)}
577 	 * convertToOWL(org.biopax.paxtools.model.Model, Object)} instead
578 	 * if you have a model and want to save and later restore it.
579 	 * @param out output
580 	 * @param bean BioPAX object
581 	 * @throws IOException when the output writer throws
582 	 */
583 	public void writeObject(Writer out, BioPAXElement bean) throws IOException
584 	{
585 		String name = "bp:" + bean.getModelInterface().getSimpleName();
586 		writeIDLine(out, bean, name);
587 
588 		Set<PropertyEditor> editors = editorMap.getEditorsOf(bean);
589 		if (editors == null || editors.isEmpty())
590 		{
591 			log.info("no editors for " + bean.getUri() + " | " + bean.getModelInterface().getSimpleName());
592 			out.write(newline + "</" + name + ">");
593 			return;
594 		}
595 
596 		for (PropertyEditor editor : editors)
597 		{
598 			Set value = editor.getValueFromBean(bean); //is never null
599 			for (Object valueElement : value)
600 			{
601 				if (!editor.isUnknown(valueElement)) writeStatementFor(bean, editor, valueElement, out);
602 			}
603 		}
604 
605 		out.write(newline + "</" + name + ">");
606 	}

609 	private void writeObjects(Writer out, Model model) throws IOException
610 	{
611 		writeHeader(out);
612 
613 		Set<BioPAXElement> bioPAXElements = model.getObjects();
614 		for (BioPAXElement bean : bioPAXElements)
615 		{
616 			writeObject(out, bean);
617 		}
618 
619 		out.write(newline + "</rdf:RDF>");
620 	}

623 	private void writeStatementFor(BioPAXElement bean, PropertyEditor editor, Object value, Writer out)
624 			throws IOException
625 	{
626 		assert (bean != null && editor != null && value != null);
627 
628 		//fix (for L3 only): skip 'name' if it's present in the displayName, etc..
629 		if (editor.getProperty().equalsIgnoreCase("name") && bean instanceof Named)
630 		{ // the latter maybe not necessary...
631 			Named named = (Named) bean;
632 			if(value.equals(named.getDisplayName()) || value.equals(named.getStandardName()))
633 			{
634 				return;
635 			}
636 		}
637 		
638 		if (editor.getProperty().equalsIgnoreCase("stepProcess") && bean instanceof BiochemicalPathwayStep)
639 		{
640 			BiochemicalPathwayStep bps = (BiochemicalPathwayStep) bean;
641 			if(value.equals(bps.getStepConversion()))
642 			{
643 				return;
644 			}
645 		}
646 
647 		String prop = "bp:" + editor.getProperty();
648 		out.write(newline + " <" + prop);
649 
650 		if (value instanceof BioPAXElement)
651 		{
652 			String id = ((BioPAXElement) value).getUri();
653 			assert id != null;
654 			if (!absoluteUris && base != null && id.startsWith(base))
655 			{
656 				id = '#' + id.substring(base.length());
657 			}
658 			out.write(" rdf:resource=\"" + id + "\" />");
659 		} else
660 		{
661 			String type = findLiteralType(editor);
662 			String valString = StringEscapeUtils.escapeXml(value.toString());
663 			out.write(" rdf:datatype = \"xsd:"  + type + "\">" + valString +
664 			          "</" + prop + ">");
665 		}
666 	}

669 	private String findLiteralType(PropertyEditor editor)
670 	{
671 		Class range = editor.getRange();
672 		String type = null;
673 		if (range.isEnum() || range.equals(String.class))
674 		{
675 			type = "string";
676 		} else if (range.equals(double.class) || range.equals(Double.class))
677 		{
678 			type = "double";
679 		} else if (range.equals(int.class) || range.equals(Integer.class))
680 		{
681 			type = "int";
682 		} else if (range.equals(float.class) || range.equals(Float.class))
683 		{
684 			type = "float";
685 		} else if (range.equals(boolean.class) || range.equals(Boolean.class))
686 		{
687 			type = "boolean";
688 		} else if (range.equals(long.class) || range.equals(Long.class))
689 		{
690 			type = "long";
691 		}
692 		return type;
693 	}

696 	private void writeIDLine(Writer out, BioPAXElement bpe, String name) throws IOException
697 	{
698 
699 		out.write(newline + newline + "<" + name + " ");
700 		String s = bpe.getUri();
701 		if (!absoluteUris &&  base != null && s.startsWith(base))
702 		{
703 			String id = s.substring(base.length());
704 			out.write(RDF_ID + id + close);
705 		} else
706 		{
707 			out.write(RDF_about + s + close);
708 		}
709 
710 
711 	}

714 	private void initializeExporter(Model model)
715 	{
716 		base = model.getXmlBase();
717 		namespaces = new HashMap<String, String>(model.getNameSpacePrefixMap());
718 
719 		normalizeNameSpaces(); // - for this reader/exporter tool
720 		// also save the changes to the model?
721 		if (normalizeNameSpaces)
722 		{
723 			model.getNameSpacePrefixMap().clear();
724 			model.getNameSpacePrefixMap().putAll(namespaces);
725 		}
726 
727 		BioPAXLevel lev = model.getLevel();
728 		if (lev != this.level) resetLevel(lev, lev.getDefaultFactory());
729 	}

732 	/*
733 	 Detect and normalize some of the xml namespaces.
734 	 Required due to input BioPAX OWL (RDF+XML) data files may declare e.g.
735 	 xmlns:xs instead xmlns:xsd for the schema; xmlns:biopax instead of usual xmlns:bp for biopax, etc.
736 	 */
737 	private void normalizeNameSpaces()
738 	{
739 		String owlPre = null;
740 		String rdfPre = null;
741 		String xsdPre = null;
742 
743 		for (String pre : namespaces.keySet())
744 		{
745 			String ns = namespaces.get(pre);
746 			if (rdf.equalsIgnoreCase(ns))
747 			{
748 				rdfPre = pre;
749 			} else if (owl.equalsIgnoreCase(ns))
750 			{
751 				owlPre = pre;
752 			} else if (xsd.equalsIgnoreCase(ns))
753 			{
754 				xsdPre = pre;
755 			}
756 		}
757 
758 		if (owlPre != null)
759 		{
760 			namespaces.remove(owlPre);
761 		}
762 
763 		if (rdfPre != null)
764 		{
765 			namespaces.remove(rdfPre);
766 		}
767 
768 		if (xsdPre != null)
769 		{
770 			namespaces.remove(xsdPre);
771 		}
772 
773 		namespaces.put("rdf", rdf);
774 		namespaces.put("owl", owl);
775 		namespaces.put("xsd", xsd);
776 	}

779 	private void writeHeader(Writer out) throws IOException
780 	{
781 		//Header
782 		out.write("<?xml version=\"1.0\" encoding=\"UTF-8\"?>");
783 		out.write(newline + "<rdf:RDF");
784 		String bpns = this.editorMap.getLevel().getNameSpace();
785 		for (String pre : namespaces.keySet())
786 		{
787 			if (!pre.equals("bp"))
788 			{
789 				out.write(newline + " xmlns" +
790 				          (("".equals(pre)) ? "" : ":" + pre) + "=\"" + namespaces.get(pre) + "\"");
791 			}
792 		}
793 		// write 'xmlns:bp'
794 		out.write(newline + " xmlns:bp" + "=\"" + bpns + "\"");
795 
796 		if (base != null)
797 		{
798 			out.write(newline + " xml:base=\"" + base + "\"");
799 		}
800 
801 		out.write(">");
802 		out.write(newline + "<owl:Ontology rdf:about=\"\">");
803 		out.write(newline + " <owl:imports rdf:resource=\"" + bpns + "\" />");
804 		out.write(newline + "</owl:Ontology>");
805 	}

863 	/**
864 	 * Serializes a (not too large) BioPAX model to the RDF/XML (OWL) formatted string.
865 	 *
866 	 * @param model a BioPAX object model to convert to the RDF/XML format
867 	 * @return the BioPAX data in the RDF/XML format
868 	 * @throws IllegalArgumentException if model is null
869 	 * @throws OutOfMemoryError when it cannot be stored in a byte array (max 2Gb).
870 	 */
871 	public static String convertToOwl(Model model)
872 	{
873 		if (model == null) throw new IllegalArgumentException();
874 
875 		ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
876 
877 		(new SimpleIOHandler(model.getLevel())).convertToOWL(model, outputStream);
878 
879 		try
880 		{
881 			return outputStream.toString("UTF-8");
882 		}
883 		catch (UnsupportedEncodingException e)
884 		{
885 			log.error("convertToOwl, outputStream.toString failed", e);
886 			return outputStream.toString();
887 		}
888 	}
