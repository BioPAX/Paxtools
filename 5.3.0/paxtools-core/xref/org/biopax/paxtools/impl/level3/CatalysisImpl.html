38

45  	public void setCatalysisDirection(CatalysisDirectionType catalysisDirection)
46  	{

48  	}
49  
50  	public Set<PhysicalEntity> getCofactor()
51  	{
52  		return cofactor;
53  	}
54  
55  	public void addCofactor(PhysicalEntity cofactor)
56  	{
57  		if (cofactor != null)
58  		{
59  			this.cofactor.add(cofactor);
60  			super.addParticipant(cofactor);
61  		}
62  	}
63  
64  	public void removeCofactor(PhysicalEntity cofactor)
65  	{
66  		if (cofactor != null)
67  		{
68  			super.removeParticipant(cofactor);
69  			this.cofactor.remove(cofactor);
70  		}
71  	}
72  
73  
74  	@Override public void addController(Controller controller)
75  	{
76  		if (controller instanceof PhysicalEntity) super.addController(controller);
77  		else throw new IllegalBioPAXArgumentException("Catalysis can only be controlled with a Physical Entity");
78  
79  	}
