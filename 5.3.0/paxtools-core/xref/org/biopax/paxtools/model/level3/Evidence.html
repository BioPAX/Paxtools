6   /**
7    * Definition: The support for a particular assertion, such as the existence of an interaction or pathway.
8    * 
9    * Usage: At least one of confidence, evidenceCode, or experimentalForm must be instantiated when creating an
10   * evidence instance. XREF may reference a publication describing the experimental evidence using a publicationXref
11   * or may store a description of the experiment in an experimental description database using a unificationXref (if
12   * the referenced experiment is the same) or relationshipXref (if it is not identical,
13   * but similar in some way e.g. similar in protocol). Evidence is meant to provide more information than just an xref
14   * to the source paper.
15   * 
16   * Examples: A description of a molecular binding assay that was used to detect a protein-protein interaction.
17   */
18  public interface Evidence extends UtilityClass, XReferrable
19  {
20  
21  
22  	/**
23  	 * Confidence in the containing instance.  Usually a statistical measure.
24  	 * @return a set of scores representing confidence
25  	 */
26  	Set<Score> getConfidence();
27  
28  	/**
29  	 * Confidence in the containing instance.  Usually a statistical measure.
30  	 * @param confidence a new confidence measure to add
31  	 */
32  	void addConfidence(Score confidence);
33  
34  	/**
35  	 * Confidence in the containing instance.  Usually a statistical measure.
36  	 * @param confidence a confidence measure to be removed.
37  	 */
38  	void removeConfidence(Score confidence);
39  
40  
41  	/**
42  	 * A pointer to a term in an external controlled vocabulary, such as the GO, PSI-MI or BioCyc
43  	 * evidence codes, that describes the nature of the support, such as 'traceable author statement'
44  	 * or 'yeast two-hybrid'.
45  	 * Contents of this set should not be modified. Use add/remove methods instead.
46  	 * @return a set of evidence codes  for this evidence type.
47  	 */
48  	Set<EvidenceCodeVocabulary> getEvidenceCode();
49  
50  	/**
51  	 * A pointer to a term in an external controlled vocabulary, such as the GO, PSI-MI or BioCyc
52  	 * evidence codes, that describes the nature of the support, such as 'traceable author statement'
53  	 * or 'yeast two-hybrid'.
54  	 * @param evidenceCode a new evidence code  for this evidence.
55  	 */
56  	void addEvidenceCode(EvidenceCodeVocabulary evidenceCode);
57  
58  	/**
59  	 * A pointer to a term in an external controlled vocabulary, such as the GO, PSI-MI or BioCyc
60  	 * evidence codes, that describes the nature of the support, such as 'traceable author statement'
61  	 * or 'yeast two-hybrid'.
62  	 * @param evidenceCode to be removed
63  	 */
64  	void removeEvidenceCode(EvidenceCodeVocabulary evidenceCode);
65  
66  
67  	/**
68  	 * Contents of this set should not be modified. Use add/remove methods instead.
69  	 * @return The experimental forms associated with an evidence instance.
70  	 */
71  	Set<ExperimentalForm> getExperimentalForm();
72  
73  	/**
74  	 * Adds an experimental form.
75  	 * @param experimentalForm associated with an evidence instance.
76  	 */
77  	void addExperimentalForm(ExperimentalForm experimentalForm);
78  
79  	/**
80  	 * Removes an experimental form.
81  	 * @param experimentalForm associated with an evidence instance.
82  	 */
83  	void removeExperimentalForm(ExperimentalForm experimentalForm);
84  
85  }
