7   /**
8    * Enumeration type for BioPAX levels.

10  public enum BioPAXLevel
11  {
12  	// define enum constants using default BioPAX factories
13  	// (L1 is not supported anymore)
14  	L2("biopax-level2.owl", new Level2FactoryImpl(), "org.biopax.paxtools.model.level2"),
15  	L3("biopax-level3.owl", new Level3FactoryImpl(), "org.biopax.paxtools.model.level3");

23  	private final String packageName;
24  	
25  	// default L2 factory implementation
26  	private static class Level2FactoryImpl extends BioPAXFactory {
27  	    @Override
28  	    public BioPAXLevel getLevel() {
29  	    	return BioPAXLevel.L2;
30  	    }
31  	    
32  	    public String mapClassName(Class<? extends BioPAXElement> aClass) 
33  	    {
34  	        String name = "org.biopax.paxtools.impl.level2."
35  	                + aClass.getSimpleName()
36  	                + "Impl";
37  	        return name;
38  	    }  
39  	}
40  	
41  	// default L3 factory implementation
42  	private static class Level3FactoryImpl extends BioPAXFactory {
43  	    @Override
44  	    public BioPAXLevel getLevel() {
45  	    	return BioPAXLevel.L3;
46  	    }
47  	    
48  	    public String mapClassName(Class<? extends BioPAXElement> aClass) 
49  	    {
50  	        String name = "org.biopax.paxtools.impl.level3."
51  	                + aClass.getSimpleName()
52  	                + "Impl";
53  	        return name;
54  	    }  
55  	}

112 	/**
113 	 * This method returns the namespace defined for this level.
114 	 * @return namespace defined for this level.
115 	 */
116 	public String getNameSpace()
117 	{
118 		return BP_PREFIX + filename + "#";
119 	}
120 
121 	/**
122 	 * This method returns true if the given string starts with the
123 	 * BP_PREFIX
124 	 * @param nameSpace to be checked
125 	 * @return rue if the given string starts with the BP_PREFIX
126 	 */
127 	public static boolean isInBioPAXNameSpace(String nameSpace)
128 	{
129 		return nameSpace != null && nameSpace.startsWith(BP_PREFIX);
130 	}
131 
132 	public static BioPAXLevel getLevelFromNameSpace(String namespace)
133 	{
134 		if (isInBioPAXNameSpace(namespace))
135 		{
136 			{
137 				for (BioPAXLevel level : BioPAXLevel.values())
138 				{
139 					if (namespace.equalsIgnoreCase(level.getNameSpace()))
140 					{
141 						return level;
142 					}
143 				}
144 			}
145 		}
146 		return null;
147 	}
148 
149 	public String getPackageName()
150 	{
151 		return packageName;
152 	}
153 
154 	public boolean hasElement(BioPAXElement element)
155 	{
156 		return element.getModelInterface().getPackage().getName().equals(this.packageName);
157 	}
158 
159 
160 	/**
161 	 * Gets the BioPAX type (java interface) by name.
162 	 * @param localName a BioPAX type name
163 	 * @return the BioPAX interface class
164 	 * @throws IllegalBioPAXArgumentException when there is no such type / class not found.
165 	 */
166 	public Class<? extends BioPAXElement> getInterfaceForName(String localName)
167 	{
168         try
169         {
170             Class modelInterface = Class.forName(this.packageName + "." + localName);
171 
172             if (BioPAXElement.class.isAssignableFrom(modelInterface))
173             {
174                 return modelInterface;
175             } else
176             {
177                 throw new IllegalBioPAXArgumentException(
178                         "BioPAXElement is not assignable from class:" + modelInterface.getSimpleName());
179             }
180         }
181         catch (ClassNotFoundException e)
182         {
183             throw new IllegalBioPAXArgumentException("Could not locate interface for:" + localName);
184         }
185     }
