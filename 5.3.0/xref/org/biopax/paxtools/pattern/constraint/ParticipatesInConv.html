4   import org.biopax.paxtools.model.BioPAXElement;
5   import org.biopax.paxtools.model.level3.Conversion;
6   import org.biopax.paxtools.model.level3.ConversionDirectionType;
7   import org.biopax.paxtools.model.level3.Interaction;
8   import org.biopax.paxtools.model.level3.PhysicalEntity;
9   import org.biopax.paxtools.pattern.util.Blacklist;
10  import org.biopax.paxtools.pattern.util.RelType;
11  
12  import java.util.Collection;
13  import java.util.HashSet;
14  
15  /**
16   * Gets the related Conversion where the PhysicalEntity is input or output, whichever is desired.
17   *
18   * var0 is a PE
19   * var1 is a Conversion
20   *
21   * @author Ozgun Babur
22   */
23  public class ParticipatesInConv extends ConstraintAdapter
24  {
25  	/**
26  	 * Input or output.
27  	 */
28  	private RelType type;
29  
30  	/**
31  	 * Constructor with parameters.
32  	 * @param type input or output conversion
33  	 * @param blacklist for detecting blacklisted small molecules
34  	 */
35  	public ParticipatesInConv(RelType type, Blacklist blacklist)
36  	{
37  		super(2, blacklist);
38  		this.type = type;
39  	}
40  
41  	/**
42  	 * Constructor with parameters.
43  	 * @param type input or output
44  	 * conversion
45  	 */
46  	public ParticipatesInConv(RelType type)
47  	{
48  		this(type, null);
49  	}
50  
51  	/**
52  	 * This is a generative constraint.
53  	 * @return true
54  	 */
55  	@Override
56  	public boolean canGenerate()
57  	{
58  		return true;
59  	}
60  
61  	/**
62  	 * Identifies the conversion direction and gets the related participants.
63  	 * @param match current pattern match
64  	 * @param ind mapped indices
65  	 * @return input or output participants
66  	 */
67  	@Override
68  	public Collection<BioPAXElement> generate(Match match, int... ind)
69  	{
70  		Collection<BioPAXElement> result = new HashSet<BioPAXElement>();
71  
72  		PhysicalEntity pe = (PhysicalEntity) match.get(ind[0]);
73  
74  		for (Interaction inter : pe.getParticipantOf())
75  		{
76  			if (inter instanceof Conversion)
77  			{
78  				Conversion cnv = (Conversion) inter;
79  				ConversionDirectionType dir = getDirection(cnv);
80  
81  				// do not get blacklisted small molecules
82  				if (blacklist != null && blacklist.isUbique(pe, cnv, dir, type)) continue;
83  
84  				if (dir == ConversionDirectionType.REVERSIBLE)
85  				{
86  					result.add(cnv);
87  				}
88  				else if (dir == ConversionDirectionType.RIGHT_TO_LEFT &&
89  					(type == RelType.INPUT ? cnv.getRight().contains(pe) : cnv.getLeft().contains(pe)))
90  				{
91  					result.add(cnv);
92  				}
93  				// Note that null direction is treated as if LEFT_TO_RIGHT. This is not a best
94  				// practice, but it is a good approximation.
95  				else if ((dir == ConversionDirectionType.LEFT_TO_RIGHT || dir == null) &&
96  					(type == RelType.INPUT ? cnv.getLeft().contains(pe) : cnv.getRight().contains(pe)))
97  				{
98  					result.add(cnv);
99  				}
100 			}
101 		}
102 
103 		return result;
104 	}
105 }
