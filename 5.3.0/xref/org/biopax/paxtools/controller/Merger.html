53  	 * value of the equivalent element for <em>bpe</em> by using the specific <em>editor</em>.
54  	 *
55  	 * @param domain owner 
56  	 * @param range property value
57  	 * @param model  model containing the equivalent element's equivalent
58  	 * @param editor biopax property editor specific for the value type to be updated
59  	 */
60  	public void visit(BioPAXElement domain, Object range, Model model, PropertyEditor editor)
61  	{
62  		if (range instanceof BioPAXElement)
63  		{
64  			BioPAXElement bpe = (BioPAXElement) range;
65  
66  			// do nothing if you already inserted this
67  			if (!model.contains(bpe))
68  			{
69  				//if there is an identical
70  				if (model.getByID(bpe.getUri()) != null)
71  				{
72  					if (editor.isMultipleCardinality())
73  					{
74  						editor.removeValueFromBean(bpe, domain);
75  					}
76  					editor.setValueToBean(getIdentical(bpe), domain);
77  				}
78  			}
79  		}
80  	}

108 	/**
109 	 * Merges the <em>source</em> models into <em>target</em> model.
110 	 *
111 	 * @param target  model into which merging process will be done
112 	 * @param sources model(s) that are going to be merged with <em>target</em>
113 	 */
114 	public void merge
115 			(Model target, Model... sources)
116 	{
117 		// Empty merged and added elements sets
118 		mergedElements.clear();
119 		addedElements.clear();
120 
121 		// Fill equivalence map with objects from  target model
122 		Set<BioPAXElement> targetElements = target.getObjects();
123 		for (BioPAXElement t_bpe : targetElements)
124 		{
125 			this.addIntoEquivalanceMap(t_bpe);
126 		}
127 
128 		// Try to insert every biopax element in every source one by one
129 		for (Model source : sources)
130 		{
131 			Set<BioPAXElement> sourceElements = source.getObjects();
132 			for (BioPAXElement bpe : sourceElements)
133 			{
134 				insert(target, bpe);
135 			}
136 		}
137 	}

139 	/**
140 	 * Inserts a BioPAX element into the <em>target</em> model if it does not contain an equivalent;
141 	 * but if does, than it updates the equivalent using this element's values.
142 	 *
143 	 * @param target model into which bpe will be inserted
144 	 * @param bpe    BioPAX element to be inserted into target
145 	 */
146 	private void insert(Model target, BioPAXElement bpe)
147 	{
148 		// do nothing if you already inserted this
149 		if (!target.contains(bpe))
150 		{
151 			//if there is an identical (in fact, "equal") object
152 			BioPAXElement ibpe = target.getByID(bpe.getUri());
153 			if (ibpe != null && ibpe.equals(bpe))
154 			/* - ibpe.equals(bpe) can be 'false' here, because, 
155 			 * even though the above !target.contains(bpe) is true,
156 			 * it probably compared objects using '==' operator
157 			 * (see the ModelImpl for details)
158 			 */
159 			{
160 				updateObjectFields(bpe, ibpe, target);
161 				// We have a merged element, add it into the tracker
162 				mergedElements.add(ibpe);
163 			}
164 			else
165 			{
166 				target.add(bpe);
167 				this.addIntoEquivalanceMap(bpe);
168 				traverser.traverse(bpe, target);
169 				// We have a new element, add it into the tracker
170 				addedElements.add(bpe);
171 			}
172 		}
173 	}

184 		int key = bpe.hashCode();
185 		List<BioPAXElement> list = equivalenceMap.get(key);
186 		if (list != null)

226 	/**
227 	 * Updates the value of <em>existing</em> element, using the value of <em>update</em>.
228 	 * Editor is used for the modification.
229 	 *
230 	 * @param editor   editor for the specific value to be updated
231 	 * @param update   BioPAX element of which value is used for the update
232 	 * @param existing BioPAX element to be updated
233 	 * @param target
234 	 */
235 	private void updateObjectFieldsForEditor(PropertyEditor editor,
236 	                                         BioPAXElement update,
237 	                                         BioPAXElement existing,
238 	                                         Model target)
239 	{
240 		if (editor.isMultipleCardinality())
241 		{
242 			for (Object updateValue : editor.getValueFromBean(update))
243 			{
244 				updateField(editor, updateValue, existing, target);
245 			}
246 		}
247 		else
248 		{
249 			Set existingValue = editor.getValueFromBean(existing);
250 			Set updateValue = editor.getValueFromBean(update);
251 			if (editor.isUnknown(existingValue))
252 			{
253 				if (!editor.isUnknown(updateValue))
254 				{
255 					updateField(editor, updateValue.iterator().next(), existing, target);
256 				}
257 			}
258 			else {
259 				if (!existingValue.equals(updateValue))
260 					log.info("Keep existing single cardinality property value: " +
261 						existingValue + " (ignore: " + updateValue + ")" );
262 			}
263 		}
264 	}

267 	 * Updates <em>existing</em>, using the <em>updateValue</em> by the editor.
268 	 *
269 	 * @param editor      editor for the specific value to be updated
270 	 * @param updateValue the value for the update
271 	 * @param existing    BioPAX element to be updated
272 	 * @param target
273 	 */
274 	private void updateField(PropertyEditor editor, Object updateValue,
275 	                         BioPAXElement existing, Model target)
276 	{
277 		if (updateValue instanceof BioPAXElement)
278 		{
279 			BioPAXElement bpe = (BioPAXElement) updateValue;
280 			//Now there are two possibilities
281 
282 
283 			BioPAXElement ibpe = target.getByID(bpe.getUri());
284 			//1. has an identical in the target
285 			if (ibpe != null)
286 			{
287 				updateValue = ibpe;
288 			}
289 			//2. it has no identical in the target
290 			//I do not have to do anything as it will eventually
291 			//be moved.
292 
293 		}
294 		editor.setValueToBean(updateValue, existing);
295 	}

297 	/**
298 	 * Adds a BioPAX element into the equivalence map.
299 	 *
300 	 * @param bpe BioPAX element to be added into the map.
301 	 */
302 	private void addIntoEquivalanceMap(BioPAXElement bpe)
303 	{
304 		int key = bpe.hashCode();
305 		List<BioPAXElement> list = equivalenceMap.get(key);
306 		if (list == null)
307 		{
308 			list = new ArrayList<BioPAXElement>();
309 			equivalenceMap.put(key, list);
310 		}
311 		list.add(bpe);
312 	}
