3   import java.io.BufferedReader;
4   import java.io.FileNotFoundException;
5   import java.io.IOException;
6   import java.io.InputStreamReader;
7   import java.util.HashMap;
8   import java.util.Map;
9   
10  /**
11   * This class provides a mapping Entrez Gene IDs and gene symbols.
12   *
13   * @author Ozgun Babur
14   */
15  public class EntrezGene
16  {
17  	private static Map<String, String> sym2id;
18  	private static Map<String, String> id2sym;
19  
20  	public static void main(String[] args)
21  	{
22  		System.out.println("getSymbol(\"367\") = " + getSymbol("367"));
23  		System.out.println("getID(\"AR\") = " + getID("AR"));
24  	}
25  
26  	/**
27  	 * Provides Entrez Gene ID of the given gene symbol.
28  	 * @param symbol gene symbol
29  	 * @return EG ID
30  	 */
31  	public static String getID(String symbol)
32  	{
33  		return sym2id.get(symbol);
34  	}
35  
36  	public static String getSymbol(String id)
37  	{
38  		return id2sym.get(id);
39  	}
40  
41  	public static boolean containsID(String id)
42  	{
43  		return id2sym.containsKey(id);
44  	}
45  
46  	public static boolean containsSymbol(String symbol)
47  	{
48  		return sym2id.containsKey(symbol);
49  	}
50  
51  	static
52  	{
53  		try
54  		{
55  			sym2id = new HashMap<String, String>();
56  			BufferedReader reader = new BufferedReader(new InputStreamReader(
57  				HGNC.class.getResourceAsStream("EntrezGene.txt")));
58  			for (String line = reader.readLine(); line != null; line = reader.readLine())
59  			{
60  				String[] token = line.split("\t");
61  
62  				if (token.length < 2) continue;
63  
64  				String sym = token[0];
65  				String id = token[1];
66  				if (sym.length() > 0 && id.length() > 0) sym2id.put(sym, id);
67  			}
68  			reader.close();
69  
70  			id2sym = new HashMap<String, String>();
71  			for (String key : sym2id.keySet())
72  			{
73  				id2sym.put(sym2id.get(key), key);
74  			}
75  
76  		}
77  		catch (FileNotFoundException e)
78  		{
79  			e.printStackTrace();
80  		}
81  		catch (IOException e)
82  		{
83  			e.printStackTrace();
84  		}
85  	}
86  }
