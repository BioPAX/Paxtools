28  	@Test
29  	public void testWriteL2GSEA() throws Exception {
30  		// write the output
31  		out.println("testWriteL2GSEA:");
32  		InputStream in = getClass().getResourceAsStream("/L2/biopax_id_557861_mTor_signaling.owl");
33  		Model level2 = handler.convertFromOWL(in);
34  		GSEAConverter gseaConverter = new GSEAConverter("GENE_SYMBOL", true);
35  		Collection<GMTEntry> entries = gseaConverter.convert(level2);
36  		// assert some things
37  		//for(GMTEntry gseaEntry : entries) System.out.println("gsea: " + gseaEntry.toString());
38  		assertEquals(1, entries.size());
39  		GMTEntry entry = entries.iterator().next();
40  		//CPATH-557861
41  		assertEquals("http://cbio.mskcc.org/cpathCPATH-557861", entry.name());
42  		assertTrue(entry.description().contains("mTOR signaling pathway"));
43  		assertTrue(entry.description().contains("nci-nature"));
44  		assertEquals("9606", entry.taxID());
45  		assertEquals("GENE_SYMBOL", entry.idType());
46  		assertEquals(27, entry.identifiers().size());
47  		// dump the output
48  		(new GSEAConverter("GENE_SYMBOL", true)).writeToGSEA(level2, out);
49  
50  		// NO more hacks that enabled using xref.id prefixes, like 'NP', 'GO', instead of true 'DB' names...
51  		gseaConverter = new GSEAConverter("NP", true);
52  		entries.clear();
53  		entries = gseaConverter.convert(level2);
54  		assertEquals(1, entries.size());
55  		entry = entries.iterator().next();
56  		assertEquals("http://cbio.mskcc.org/cpathCPATH-557861", entry.name());
57  		assertTrue(entry.identifiers().isEmpty());
58  
59  		gseaConverter = new GSEAConverter("ref_seq", true); //test data has this non-standard name (instead RefSeq)
60  		entries.clear();
61  		entries = gseaConverter.convert(level2);
62  		// assert some things
63  		assertEquals(1, entries.size());
64  		entry = entries.iterator().next();
65  		assertEquals("http://cbio.mskcc.org/cpathCPATH-557861", entry.name());
66  		assertTrue(entry.description().contains("nci-nature"));
67  		assertEquals("9606", entry.taxID());
68  		assertEquals("ref_seq", entry.idType());
69  		assertEquals(33, entry.identifiers().size());
70  		(new GSEAConverter("NP", true)).writeToGSEA(level2, out);
71  	}
