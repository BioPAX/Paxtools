014/**
015 * This is the base class for all property editors. Each property controller is responsible for
016 * manipulating a certain property for a given class of objects (domain).
017 */
018public abstract class AbstractPropertyEditor<D extends BioPAXElement, R>
019                extends SimplePropertyAccessor<D,R> implements PropertyEditor<D,R>
020{
021// ------------------------------ FIELDS ------------------------------
022
023        protected static final Logger log = LoggerFactory.getLogger(AbstractPropertyEditor.class);

025        /**
026         * This variable stores the method to invoke for setting a property to the to the given value. If
027         * {@link #multipleCardinality multiple cardinality}, the returned method is expected to have a
028         * {@link java.util.Set} as its only parameter.
029         */
030        protected Method setMethod;
031
032        /**
033         * This variable stores the method to invoke for adding the given value to the property managed by
034         * this commander. In the case of multiple cardinality, the method is expected to have a {@link
035         * #range} as its only parameter, otherwise expected to be null.
036         */
037        protected Method addMethod;
038
039        /**
040         * This variable stores the method for removing the value of the property on a given bean. In the
041         * case of multiple cardinality, this method is expected to have a {@link #range} as its only
042         * parameter, otherwise expected to be null
043         */
044        protected Method removeMethod;
045
046        /**
047         * Local OWL name of the property
048         */
049        protected final String property;

070         * Constructor.
071         * 
072         * @param property biopax property name
073         * @param getMethod getter
074         * @param domain class the property belongs to
075         * @param range property values type/class
076         * @param multipleCardinality whether more than one value is allowed

078    public AbstractPropertyEditor(String property, Method getMethod, Class<D> domain, Class<R> range,
079                        boolean multipleCardinality)
080        {
081                super(domain, range, multipleCardinality, getMethod);
082                this.property = property;
083
084                try
085                {
086                        detectMethods();
087                }
088                catch (NoSuchMethodException e)
089                {
090                        log.error("Failed at reflection, no method: " + e.getMessage());
091                }
092        }

097        @Override
098        public String toString()
099        {
100                String def = String.format("%s %s %s", domain.getSimpleName(), property, range.getSimpleName());
101
102                //TODO cardinalities are not being read!
103                for (Class aClass : maxCardinalities.keySet())
104                {
105                        Integer cardinality = maxCardinalities.get(aClass);
106                        def += " C:" + aClass.getSimpleName() + ":" + cardinality;
107                }
108
109                return def;
110        }

112        /**
113         * This method creates a property reflecting on the domain and property. Proper subclass is chosen
114         * based on the range of the property.
115         * @param domain paxtools level2 interface that maps to the corresponding owl level2.
116         * @param property to be managed by the constructed controller.
117         * @param <D> domain
118         * @param <R> range
119         * @return a property controller to manipulate the beans for the given property.
120         */
121        public static <D extends BioPAXElement, R> PropertyEditor<D, R> createPropertyEditor(Class<D> domain,
122                                                                                             String property)
123        {
124                PropertyEditor editor = null;
125                try
126                {
127                        Method getMethod = detectGetMethod(domain, property);
128                        boolean multipleCardinality = isMultipleCardinality(getMethod);
129                        Class<R> range = detectRange(getMethod);
130
131                        if (range.isPrimitive() || range.equals(Boolean.class))
132                        {
133                                editor = new PrimitivePropertyEditor<D, R>(property, getMethod, domain, range, multipleCardinality);
134                        } else if (range.isEnum())
135                        {
136                                editor = new EnumeratedPropertyEditor(property, getMethod, domain, range, multipleCardinality);
137                        } else if (range.equals(String.class))
138                        {
139                                editor = new StringPropertyEditor(property, getMethod, domain, multipleCardinality);
140                        } else
141                        {
142                                editor = new ObjectPropertyEditor(property, getMethod, domain, range, multipleCardinality);
143                        }
144                }
145                catch (NoSuchMethodException e)
146                {
147                        if (log.isWarnEnabled()) log.warn("Failed creating the controller for " + property + " on " + domain);
148                }
149                return editor;
150        }
151
152        private static Method detectGetMethod(Class beanClass, String property) throws NoSuchMethodException
153        {
154                String javaMethodName = getJavaName(property);
155                //This is the name we are going to try, log it down
156                if (log.isTraceEnabled())
157                {
158                        log.trace("javaMethodName = get" + javaMethodName);
159                }
160
161                //extract the get method
162                return beanClass.getMethod("get" + javaMethodName);
163        }
164
165        /**
166         * Given the name of a property's name as indicated in the OWL file, this method converts the name
167         * to a Java compatible name.
168         * @param owlName the property name as a string
169         * @return the Java compatible name of the property
170         */
171        private static String getJavaName(String owlName)
172        {
173                // Since java does not allow '-' replace them all with '_'
174                String s = owlName.replaceAll("-", "_");
175                s = s.substring(0, 1).toUpperCase() + s.substring(1);
176                return s;
177        }
178
179        /**
180         * Given the multiple cardinality feature, the range of the get method is returned.
181         * @param getMethod default method
182
183         * @return the range as a class
184         */
185        protected static Class detectRange(Method getMethod)
186        {
187                Class range = getMethod.getReturnType();
188                //if the return type is a collection then we have multiple cardinality
189                if (Collection.class.isAssignableFrom(range))
190                {
191                        //it is a collection, by default assume non parameterized.
192                        range = Object.class;
193                        //If the collection is parameterized, get it.
194                        Type genericReturnType = getMethod.getGenericReturnType();
195                        if (genericReturnType instanceof ParameterizedType)
196                        {
197                                try
198                                {
199                                        range = (Class) ((ParameterizedType) genericReturnType).getActualTypeArguments()[0];
200                                }
201                                catch (Exception e)
202                                {
203                                        e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
204                                }
205                                //Now this is required as autoboxing will not work with reflection
206                                if (range == Double.class)
207                                {
208                                        range = double.class;
209                                }
210                                if (range == Float.class)
211                                {
212                                        range = float.class;
213                                }
214                                if (range == Integer.class)
215                                {
216                                        range = int.class;
217                                }
218                        }
219                        if (log.isTraceEnabled()) log.trace(range.getCanonicalName());
220                }
221
222                return range;
223        }

226        /**
227         * Detects and sets the default methods for the property to which editor is associated. If property
228         * has multiple cardinality, {@link #setMethod}, {@link #addMethod}, and {@link #removeMethod} are
229         * set, otherwise only the {@link #setMethod}.
230         * @exception NoSuchMethodException if a method for the property does not exist
231         */
232        private void detectMethods() throws NoSuchMethodException
233        {
234                String javaName = getJavaName(property);
235                if (multipleCardinality)
236                {
237
238                        this.addMethod = domain.getMethod("add" + javaName, range);
239                        this.removeMethod = domain.getMethod("remove" + javaName, range);
240                } else
241                {
242                        this.setMethod = domain.getMethod("set" + javaName, range);
243                }
244        }

277        @Override public void addMaxCardinalityRestriction(Class<? extends D> domain, int max)
278        {
279                if (multipleCardinality)
280                {
281                        this.maxCardinalities.put(domain, max);
282                } else
283                {
284                        if (max == 1)
285                        {
286                                if (log.isInfoEnabled())
287                                {
288                                        log.info("unnecessary use of cardinality restriction. " +
289                                                 "Maybe you want to use functional instead?");
290                                }
291                        } else if (max == 0)
292                        {
293                                this.maxCardinalities.put(domain, max);
294                        } else
295                        {
296                                assert false;
297                        }
298                }
299        }

301        @Override public Integer getMaxCardinality(Class<? extends D> restrictedDomain)
302        {
303                return this.maxCardinalities.get(restrictedDomain);
304        }

306        /**
307         * Checks if <em>value</em> is an instance of one of the classes given in a set. This method
308         * becomes useful, when the restrictions have to be checked for a set of objects. e.g. check if the
309         * value is in the range of the editor.
310         *
311         * @param classes a set of classes to be checked
312         * @param value value whose class will be checked
313         * @return true if value belongs to one of the classes in the set
314         */
315        protected boolean isInstanceOfAtLeastOne(Set<Class<? extends BioPAXElement>> classes, Object value)
316        {
317                boolean check = false;
318                for (Class aClass : classes)
319                {
320                        if (aClass.isInstance(value))
321                        {
322                                check = true;
323                                break;
324                        }
325                }
326                return check;
327        }

337        @Override public void removeValueFromBean(R value, D bean)
338        {
339                if(value == null)
340                        return;
341
342                try
343                {
344                        if (removeMethod != null)
345                        {
346                                invokeMethod(removeMethod, bean, value);
347                        } else {
348                                assert !isMultipleCardinality() : "removeMethod is not defined " +
349                                                "for the multiple cardinality property: " + property +
350                                                ". Here, this might add 'unknown' value while keeping exisiting one as well!";
351                                
352                                if(this.getValueFromBean(bean).contains(value))
353                                {
354                                        this.setValueToBean(this.getUnknown(),bean);
355                                }
356                                else { 
357                                        //TODO throw an exception if range is violated rather than always log the following message
358                                        log.error("Given value :" + value + 
359                                                " is not equal to the existing value. " +
360                                                 "remove value is ignored");
361                                        assert getRange().isInstance(value) : "Range violation!";
362                                        assert getDomain().isInstance(bean) : "Domain violation!";
363                                }
364                        }
365                }
366                catch (Exception e)
367                {
368                        log.error("removeValueFromBean failed", e);
369                }
370        }
371
372
373        @Override public void removeValueFromBean(Set<R> values, D bean) {
374                for(R r : values) {
375                        removeValueFromBean(r, bean);
376                }
377        }

380        /**
381         * Calls the <em>method</em> onto <em>bean</em> with the <em>value</em> as its parameter. In this
382         * context <em>method</em> can be one of these three: set, add, or remove.
383         * @param method method that is going to be called
384         * @param bean bean onto which the method is going to be applied
385         * @param value the value which is going to be used by method
386         */
387        protected void invokeMethod(Method method, D bean, R value)
388        {
389                assert bean != null;
390                try
391                {
392                        method.invoke(domain.cast(bean), value);
393                }
394                catch (ClassCastException e)
395                {
396                        String message = "Failed to set property: " + property;
397                        if (!domain.isAssignableFrom(bean.getClass()))
398                        {
399                                message += "  Invalid domain bean: " + domain.getSimpleName() + " is not assignable from " +
400                                           bean.getClass();
401                        }
402                        if (!range.isAssignableFrom(value.getClass()))
403                        {
404                                message += " Invalid range value: " + range + " is not assignable from " + value.getClass();
405                        }
406                        throw new IllegalBioPAXArgumentException(message, e);
407                }
408                catch (Exception e) //java.lang.reflect.InvocationTargetException
409                {
410                        String valInfo = (value == null) ? null : value.getClass().getSimpleName() + ", " + value;
411                        String message = "Failed to set " + property + " with " + method.getName() + " on " 
412                                + domain.getSimpleName() + " (" + bean.getClass().getSimpleName() + ", " + bean + ")" 
413                                + " with range: " + range.getSimpleName() + " (" + valInfo + ")";
414                        throw new IllegalBioPAXArgumentException(message, e);
415                        //TODO actual exceptions thrown by the biopax setter/method are lost, unfortunately...
416                }
417        }

424        @Override public void setValueToBean(R value, D bean)
425        {
426                if (this.getPrimarySetMethod() != null)
427                {
428                        if (log.isTraceEnabled()) log.trace(
429                                        this.getPrimarySetMethod().getName() + " bean:" + bean + " val:" + value);
430                } else
431                {
432                        log.error("setMethod is null; " + " bean:" + bean + " (" + bean.getUri() + ") val:" + value);
433                }
434
435                // 'null' definitely means 'unknown'for single cardinality props
436                if (value == null && !isMultipleCardinality())
437                        value = getUnknown(); // not null for primitive property editors
438
439                if (value instanceof String)
440                {
441                        value = this.parseValueFromString(((String) value));
442                }
443                try
444                {
445                        if (value != null && checkRestrictions.get()) checkRestrictions(value, bean);
446                        invokeMethod(this.getPrimarySetMethod(), bean, value);
447                }
448                catch (Exception e) {
449                        if(log.isDebugEnabled()) {
450                                log.debug("setValueToBean failed to set value: " + value
451                                                + ((value != null) ? " of type: " + value.getClass().getSimpleName() : "")
452                                                + " to " + bean.getClass().getSimpleName()
453                                                + " (" + bean.getUri() + ") using: " + getPrimarySetMethod().getName(), e);
454                        } else {
455                                log.error("setValueToBean, failed: " + e.getMessage());
456                        }
457                }
458        }
459
460        @Override public void setValueToBean(Set<R> values, D bean)
461        {
462                if (values == null) setValueToBean(((R) null), bean);
463
464                else if (this.isMultipleCardinality() || values.size() < 2)
465                {
466                        for (R r : values)
467                        {
468                                this.setValueToBean(r, bean);
469                        }
470                } else throw new IllegalBioPAXArgumentException(
471                                        this.getProperty() + " is single cardinality. Can not set" + "it with a set of size larger than" +
472                                        " 1");
473        }

476        /**
477         * Checks if the <em>bean</em> and the <em>value</em> are consistent with the cardinality rules of
478         * the model. This method is important for validations.
479         * @param value Value that is related to the object
480         * @param bean Object that is related to the value
481         */
482        protected void checkRestrictions(R value, D bean)
483        {
484                Integer max = this.maxCardinalities.get(value.getClass());
485                if (max != null)
486                {
487                        if (max == 0)
488                        {
489                                throw new IllegalBioPAXArgumentException("Cardinality 0 restriction violated");
490                        } else
491                        {
492                                assert multipleCardinality;
493                                Set values = this.getValueFromBean(bean);
494                                if (values.size() >= max)
495                                {
496                                        throw new IllegalBioPAXArgumentException("Cardinality " + max + " restriction violated");
497                                }
498                        }
499                }
500        }
501
502        @Override public Method getPrimarySetMethod()
503        {
504                return multipleCardinality ? addMethod : setMethod;
505        }
