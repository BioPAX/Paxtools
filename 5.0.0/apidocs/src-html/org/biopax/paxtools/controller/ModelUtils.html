071        /**
072         * Replaces BioPAX elements in the model with ones from the map,
073         * updates corresponding BioPAX object references.
074         * 
075         * It does not neither remove the old nor add new elements in the model
076         * (if required, one can do this before/after this method, e.g., using
077         * the same 'subs' map)
078         * 
079         * This does visit all object properties of each "explicit" element
080         * in the model, but does not traverse deeper into one's sub-properties
081         * to replace something there as well (e.g., nested member entity references
082         * are not replaced unless parent entity reference present in the model)
083         * 
084         * This does not automatically move/migrate old (replaced) object's
085         * children to new objects (the replacement ones are supposed to have
086         * their own properties already set or to be set shortly; otherwise,
087         * consider using of something like {@link #fixDanglingInverseProperties(BioPAXElement, Model)} after.
088         * 
089         * @param model biopax model where the objects are to be replaced
090         * @param subs the replacements map (many-to-one, old-to-new)
091         * @exception IllegalBioPAXArgumentException if there is an incompatible type replacement object
092         */
093        public static void replace(Model model, final Map<? extends BioPAXElement, ? extends BioPAXElement> subs)
094        {
095                // update properties
096                Visitor visitor = new Visitor()
097                {
098                        @Override
099                        public void visit(BioPAXElement domain, Object range, Model model, PropertyEditor editor)
100                        {
101                                if (editor instanceof ObjectPropertyEditor && range != null && subs.containsKey(range))
102                                {
103                                        BioPAXElement value = (BioPAXElement) range;
104                                        // 'value' is to be replaced with the 'replacement'
105                                        BioPAXElement replacement = subs.get(range); //can get null (ok)
106                                        
107                                        // normal biopax property -
108                                        if (replacement != null && !editor.getRange().isInstance(replacement))
109                                        {
110                                                throw new IllegalBioPAXArgumentException(
111                                                        "Incompatible type! Attempted to replace " 
112                                                        + value.getUri() + " (" + value.getModelInterface().getSimpleName()
113                                                        + ") with " + replacement.getUri() + " ("
114                                                        + replacement.getModelInterface().getSimpleName() + "); "
115                                                        + "property: " + editor.getProperty() 
116                                                        + " of bean: " + domain.getUri() + " ("
117                                                        + domain.getModelInterface().getSimpleName() + ")");
118                                        }
119
120                                        if (replacement != value) 
121                                        {
122                                                editor.removeValueFromBean(value, domain);
123                                                editor.setValueToBean(replacement, domain);
124                                        } else {
125                                                LOG.debug("replace: skipped the identical: " + replacement.getUri());
126                                        }
127                                }
128                        }
129                };
130
131                Traverser traverser = new Traverser(em, visitor);
132                for (BioPAXElement bpe : new HashSet<BioPAXElement>(model.getObjects()))
133                {
134                        // update object properties and clear inverse properties using 'subs' map       
135                        traverser.traverse(bpe, null); //model is not needed
136                }
137        }

140        /**
141         * Finds "root" BioPAX objects that belong to a particular class (incl. sub-classes)
142         * in the model.
143         * 
144         * Note: however, such "root" elements may or may not be, a property of other
145         * elements, not included in the model.
146         * @param model biopax model to work with
147         * @param filterClass filter class (including subclasses)
148         * @param <T> biopax type
149         * @return set of the root biopax objects of given type
150         */
151        public static <T extends BioPAXElement> Set<T> getRootElements(final Model model, final Class<T> filterClass)
152        {
153                // copy all such elements (initially, we think all are roots...)
154                final Set<T> result = new HashSet<T>(model.getObjects(filterClass));
155
156                //"shallow" traverser (direct object properties only - Visitor.visit does not call traverse again) 
157                @SuppressWarnings("unchecked")
158                Traverser traverser = new Traverser(em, 
159                        new Visitor() {
160                                @Override
161                                public void visit(BioPAXElement parent, Object value, Model model,
162                                        PropertyEditor<?, ?> editor)
163                                {
164                                        if (filterClass.isInstance(value)) result.remove(value);
165                                }
166                        }, 
167                        new Filter<PropertyEditor>() {
168                                @Override
169                                public boolean filter(PropertyEditor pe) {
170                                        return (pe instanceof ObjectPropertyEditor);
171                                }
172                });
173                
174                // but we run from every element (all types)
175                for(BioPAXElement e : model.getObjects())
176                        traverser.traverse(e, null);
177
178                return result;
179        }

182        /**
183         * Iteratively removes "dangling" elements of given type and its sub-types,
184         * e.g. Xref.class objects, from the BioPAX model. 
185         * 
186         * If the "model" does not contain any root Entity class objects,
187         * and the second parameter is basic UtilityClass.class (i.e., not its sub-class), 
188         * then it simply logs a warning and quits shortly (otherwise, it would 
189         * remove everything from the model). Do not use basic Entity.class either
190         * (but a sub-class is OK) for the same reason (it would delete everything).
191         * 
192         * This, however, does not change relationships
193         * among objects, particularly, some inverse properties,
194         * such as entityReferenceOf or xrefOf, may still
195         * refer to a removed object.
196         * @param model to modify
197         * @param clazz filter-class (filter by this type and sub-classes)
198         * @param <T> biopax type
199         * @return removed objects
200         */
201        public static <T extends BioPAXElement> Set<BioPAXElement> removeObjectsIfDangling(Model model, Class<T> clazz)
202        {
203                final Set<BioPAXElement> removed = new HashSet<BioPAXElement>();
204                
205                // 'equals' below is used intentionally (isAssignableFrom() would be incorrect)
206                if(Entity.class.equals(clazz)) {
207                        LOG.warn("Ignored removeObjectsIfDangling call for: " +
208                                        "Entity.class (it would delete all)");
209                        return removed;
210                }
211                if(UtilityClass.class.equals(clazz) 
212                                && getRootElements(model, Entity.class).isEmpty()) 
213                {
214                        LOG.warn("Ignored removeObjectsIfDangling call: " +
215                                        "no root entities model; UtilityClass.class");
216                        return removed;
217                }
218                
219                Set<T> dangling = getRootElements(model, clazz);        
220                
221                // get rid of dangling objects
222                if (!dangling.isEmpty())
223                {
224                        LOG.info(dangling.size() + " " + clazz.getSimpleName() +
225                                " dangling objects will be deleted...");
226
227                        for (BioPAXElement thing : dangling)
228                        {
229                                model.remove(thing);
230                                removed.add(thing);
231                                LOG.debug("removed (dangling) " + thing.getUri() + " ("
232                                        + thing.getModelInterface().getSimpleName() + ") " + thing);
233                        }
234
235                        // some may have become dangling now, so check again...
236                        removed.addAll(removeObjectsIfDangling(model, clazz));
237                }
238                
239                return removed;
240        }

243        /**
244         * Cuts the BioPAX model off other models and BioPAX objects
245         * by essentially performing write/read to/from OWL.
246         * The resulting model contains new objects with same IDs
247         * and have object properties "fixed", i.e., dangling values
248         * become null/empty, and inverse properties (e.g. xrefOf)
249         * re-calculated. The original model is unchanged.
250         * 
251         * Note: this method will fail for very large models 
252         * (if resulting RDF/XML utf8 string is longer than approx. 1Gb)
253         * 
254         * @param model biopax model to process
255         * @return copy of the model
256         */
257        public static Model writeRead(Model model)
258        {
259                BioPAXIOHandler io = new SimpleIOHandler(model.getLevel());
260                ByteArrayOutputStream baos = new ByteArrayOutputStream();
261                io.convertToOWL(model, baos);
262                return io.convertFromOWL(new ByteArrayInputStream(baos.toByteArray()));
263        }

293         * (using the "tuned" {@link Fetcher}) and adds them to a
294         * new model.
295         * @param bpe biopax object
296         * @param filters property filters (e.g., for Fetcher to skip some properties). Default is to skip 'nextStep'.

298         */
299        public static Model getAllChildren(BioPAXElement bpe, 
300                @SuppressWarnings("rawtypes") Filter<PropertyEditor>... filters)
301        {
302                Model m = factory.createModel();
303                if (filters.length == 0)
304                {
305                        new Fetcher(em, Fetcher.nextStepFilter).fetch(bpe, m);
306                } else
307                {
308                        new Fetcher(em, filters).fetch(bpe, m);
309                }
310                m.remove(bpe); // remove the parent
311
312                return m;

350                {

388        /**
389         * Calculates MD5 hash code (as 32-byte hex. string).
390         * 
391         * This method is not BioPAX specific. Can be
392         * used for many purposes, such as generating 
393         * new unique URIs, database primary keys, etc.
394         * 
395         * 
396         * @param id some identifier, e.g., URI
397         * @return the 32-byte digest string
398         */
399        public static String md5hex(String id)
400        {
401                byte[] digest = MD5_DIGEST.digest(id.getBytes());
402                StringBuffer sb = new StringBuffer();
403                for (byte b : digest)
404                {
405                        sb.append(Integer.toHexString((int) (b & 0xff) | 0x100).substring(1, 3));
406                }
407                String hex = sb.toString();
408                return hex;
409        }

413         * Unlinks <em>object properties</em> of the BioPAX object
414         * from values the model does not have.
415         * 
416         * @param bpe a biopax object
417         * @param model the model to look for objects in
418         */
419        public static void fixDanglingObjectProperties(BioPAXElement bpe, Model model)
420        {
421                final Visitor visitor = new Visitor()
422                {
423                        @Override
424                        public void visit(BioPAXElement domain, Object range, Model model, PropertyEditor editor)
425                        {
426                                if (editor instanceof ObjectPropertyEditor)
427                                {
428                                        BioPAXElement value = (BioPAXElement) range;
429                                        if (value != null && !model.containsID(value.getUri()))
430                                                editor.removeValueFromBean(value, domain);
431                                }
432                        }
433                };
434
435                Traverser traverser = new Traverser(em, visitor);
436                traverser.traverse(bpe, model);
437        }

444         * @param model where to look for other objects

483        // TODO annotate
484        public static Set<EntityFeature> getFeatureSetByType(PhysicalEntity pe, FeatureType type)
485        {
486
487                Set<EntityFeature> modifiableSet = new HashSet<EntityFeature>();
488
489                switch (type)
490                {
491                        case FEATURE:
492                                modifiableSet.addAll(pe.getFeature());
493                                break;
494                        case NOT_FEATURE:
495                                modifiableSet.addAll(pe.getNotFeature());
496                                break;
497                        case UNKNOWN_FEATURE:
498                        {
499                                if (pe instanceof SimplePhysicalEntity)
500                                {
501                                        modifiableSet.addAll(((SimplePhysicalEntity) pe).getEntityReference().getEntityFeature());
502                                        modifiableSet.removeAll(pe.getFeature());
503                                        modifiableSet.removeAll(pe.getNotFeature());
504                                }
505                        }
506                }
507                return modifiableSet;
508        }

511        /**
512         * Finds and adds all (missing) entity features 
513         * to given entity reference from all its owner 
514         * simple physical entities ('feature' and 'notFeature' 
515         * properties).
516         * 
517         * Though, it neither checks for nor resolves any violations 
518         * of the 'entityFeature' property's inverse functional constraint
519         * (i.e., an EntityFeature instance can only belong to one and only one
520         * EntityReference object).  
521         * 
522         * @param er entity reference object
523         * @param fix flag
524         * @return true or false
525         */
526        public static boolean checkERFeatureSet(EntityReference er, boolean fix)
527        {
528                boolean check = true;
529                for (SimplePhysicalEntity spe : er.getEntityReferenceOf())
530                {
531                        for (EntityFeature ef : spe.getFeature())
532                        {
533                                check = scanAndAddToFeatureSet(er, fix, check, ef);
534                                // if not fixing return at first fail, otherwise go on;
535                                if (!fix && !check) return check;
536                        }
537                        for (EntityFeature ef : spe.getNotFeature())
538                        {
539                                check = scanAndAddToFeatureSet(er, fix, check, ef);
540                                // if not fixing return at first fail, otherwise go on;
541                                if (!fix && !check) return check;
542                        }
543                }
544                return check;
545        }
546
547        private static boolean scanAndAddToFeatureSet(EntityReference er, boolean fix, boolean check, EntityFeature ef)
548        {
549                if (!er.getEntityFeature().contains(ef))
550                {
551                        check = false;
552                        if (fix)
553                        {
554                                er.addEntityFeature(ef);
555                                //TODO resolve inverse functional prop. constraint violation (e.g., copy/replace the e.f. before adding if it has entityFeatureOf not null)?
556                        }
557                }
558                return check;
559        }

561        // TODO annotate
562        public static Set<EntityFeature> findFeaturesAddedToSecond(PhysicalEntity first, PhysicalEntity second,
563                        boolean fix)
564        {
565
566                if (checkCommonEntityReferenceForTwoPEs(first, second, fix)) return null;
567                Set<EntityFeature> explicit =
568                                getFeatureIntersection(first, FeatureType.NOT_FEATURE, second, FeatureType.FEATURE);
569                Set<EntityFeature> implicit =
570                                getFeatureIntersection(first, FeatureType.UNKNOWN_FEATURE, second, FeatureType.FEATURE);
571                Set<EntityFeature> negativeImplicit =
572                                getFeatureIntersection(first, FeatureType.NOT_FEATURE, second, FeatureType.UNKNOWN_FEATURE);
573
574                if (fix)
575                {
576                        for (EntityFeature implied : implicit)
577                        {
578                                LOG.info("The feature " + implied + "implied as a not-feature of " + first + ". " +
579                                         "Adding it to the not-feature list");
580                                first.addNotFeature(implied);
581                        }
582
583                        for (EntityFeature implied : negativeImplicit)
584                        {
585                                LOG.info("The feature " + implied + "implied as a feature of " + second + ". " +
586                                         "Adding it to the feature list");
587                                second.addFeature(implied);
588                        }
589
590                }
591                explicit.retainAll(implicit);
592                explicit.retainAll(negativeImplicit);
593                return explicit;
594        }

596        private static boolean checkCommonEntityReferenceForTwoPEs(PhysicalEntity first, PhysicalEntity second,
597                        boolean fix)
598        {
599                if (first instanceof SimplePhysicalEntity)
600                {
601                        EntityReference er = ((SimplePhysicalEntity) first).getEntityReference();
602                        if (!er.getEntityReferenceOf().contains(second))
603                        {
604                                LOG.warn("These two physicalEntities do not share an EntityReference. They can not be compared! " +
605                                         "Skipping");
606                                return false;
607                        } else if (!checkERFeatureSet(er, fix))
608                        {
609                                LOG.warn("ER feature set is incomplete!");
610                                if (!fix)
611                                {
612                                        LOG.warn("fixing...");
613                                } else
614                                {
615                                        LOG.warn("skipping");
616                                        return false;
617                                }
618                        }
619                        return true;
620                } else
621                {
622                        LOG.warn("These two physicalEntities do not share an EntityReference. They can not be compared! " +
623                                        "Skipping");
624                        return false;
625                }
626
627        }

630        /**
631         * Converts generic simple physical entities, 
632         * i.e., physical entities except Complexes 
633         * that have not empty memberPhysicalEntity property,
634         * into equivalent physical entities
635         * with generic entity references (which have members);
636         * this is a better and less error prone way to model
637         * generic molecules in BioPAX L3. 
638         * 
639         * Notes:
640         * Generic Complexes could be normalized in a similar way,
641         * but they do not have entityReference property and might
642         * contain generic (incl. not yet normalized) components, which
643         * makes it complicated.
644         * 
645         * So, please avoid using 'memberPhysicalEntity' in your BioPAX L3 models
646         * unless absolutely required, for there is a better alternative way
647         * (using PhysicalEntity/entityReference/memberEntityReference).
648         * 
649         * @param model biopax model to fix
650         */
651        public static void normalizeGenerics(Model model)
652        {
653                final Set<SimplePhysicalEntity> simplePEsToDo = new HashSet<SimplePhysicalEntity>();
654
655                for (SimplePhysicalEntity spe : model.getObjects(SimplePhysicalEntity.class))
656                        if (spe.getEntityReference() == null && !spe.getMemberPhysicalEntity().isEmpty())
657                                simplePEsToDo.add(spe);
658
659                final Map<Set<EntityReference>,EntityReference> memberMap = new HashMap<Set<EntityReference>,EntityReference>();
660                for (SimplePhysicalEntity pe : simplePEsToDo)
661                {
662                        try {
663                                createGenericEntityRef(model, pe, memberMap);
664                        }
665                        catch (Exception e) {
666                                LOG.error("createGenericEntityRef failed at " + pe.getUri(), e);
667                        }
668                }
669        }
670        
671        private static void createGenericEntityRef(Model model, SimplePhysicalEntity spe,
672                        final Map<Set<EntityReference>, EntityReference> memberMap)
673        {
674                Set<EntityReference> members = spe.getGenericEntityReferences();
675                EntityReference er = memberMap.get(members);
676                if (er == null)
677                {
678                        //find an ER, if any...
679                        EntityReference firstEntityReference = spe.getEntityReference(); //null, but worth double-checking
680                        if(firstEntityReference == null) {
681                                for (PhysicalEntity pe : spe.getMemberPhysicalEntity()) {
682                                        //safe to cast to SimplePhysicalEntity, thanks to the memberPE property range restriction
683                                        firstEntityReference = ((SimplePhysicalEntity) pe).getEntityReference();
684                                        if (firstEntityReference != null)
685                                                break; //got one!
686                                }
687                        } else {
688                                return;
689                        }
690
691                        if (firstEntityReference != null)
692                        {
693                                //generate a new URI in the same namespace (xml:base) and create and add a new EntityReference
694                                String syntheticId = model.getXmlBase() +

770        /**
771         * Copies names, xrefs, comments, evidence, data sources from source to target biopax object;
772         * it does not copy unification xrefs but instead adds relationship xrefs using the same
773         * db and id values as source's unification xrefs.
774         * 
775         * @param model the biopax model where the source and target objects belong
776         * @param source from
777         * @param target to
778         */
779        private static void copySimplePointers(Model model, Named source, Named target)
780        {
781                //copy names
782                if(source.getDisplayName()!=null)
783                        target.setDisplayName(source.getDisplayName());
784                if(source.getStandardName()!=null)
785                        target.setStandardName(source.getStandardName());
786                for (String name : source.getName()) {
787                        target.addName(name);
788                }
789
790                // copy xrefs, converting UXs to RXs on the go
791                for (Xref xref : source.getXref())
792                {
793                        if (xref instanceof UnificationXref)
794                        {
795                                // generate URI using model's xml:base and xref's properties
796                                String uri = model.getXmlBase() + "RelationshipXref_"+ md5hex(xref.getDb()+xref.getId());
797                                Xref byID = (Xref) model.getByID(uri);
798                                if (byID == null)
799                                {
800                                        RelationshipXref rref = model.addNew(RelationshipXref.class, uri);
801                                        rref.setDb(xref.getDb());
802                                        rref.setId(xref.getId());
803                                        rref.setDbVersion(xref.getDbVersion());
804                                        rref.setIdVersion(xref.getDbVersion());
805                                        xref = rref;
806                                } else
807                                {
808                                        xref = byID;
809                                }
810                        }
811                        target.addXref(xref);
812                }
813
814                //copy evidence and dataSource if (Named) source and target are same sub-type - either Entity or ER only:
815                if(source instanceof Entity && target instanceof Entity) {
816                        Entity src = (Entity) source;
817                        for (Evidence ev : src.getEvidence()) {
818                                ((Entity) target).addEvidence(ev);
819                        }
820                        for (Provenance prov : src.getDataSource()) {
821                                ((Entity) target).addDataSource(prov);
822                        }
823                        // copy comments
824                        for(String comm : source.getComment()) {
825                                target.addComment(comm);
826                        }
827                } else if(source instanceof EntityReference && target instanceof EntityReference) {
828                        EntityReference src = (EntityReference) source;
829                        for (Evidence ev : src.getEvidence()) {
830                                ((EntityReference) target).addEvidence(ev);
831                        }
832                        // copy comments
833                        for(String comm : source.getComment()) {
834                                target.addComment(comm);
835                        }
836                }
837        }

840        /**
841         * This method iterates over the features in a model and tries to find equivalent objects and merges them.
842         * @param model to be fixed
843         */
844        public static void replaceEquivalentFeatures(Model model)
845        {
846
847                EquivalenceGrouper<EntityFeature> equivalents = new EquivalenceGrouper<EntityFeature>();
848                HashMap<EntityFeature, EntityFeature> mapped = new HashMap<EntityFeature, EntityFeature>();
849                HashSet<EntityFeature> scheduled = new HashSet<EntityFeature>();
850
851                for (EntityFeature ef : model.getObjects(EntityFeature.class))
852                {
853                        if (ef.getEntityFeatureOf() == null)
854                                inferEntityReference(ef);
855
856                        equivalents.add(ef);
857                }
858                for (List<EntityFeature> bucket : equivalents.getBuckets())
859                {
860                        for (int i = 1; i < bucket.size(); i++)
861                        {
862                                EntityFeature ef = bucket.get(i);
863                                if (LOG.isWarnEnabled())
864                                {
865                                        LOG.warn("removing: "+ ef.getUri()+ " since it is equivalent to: "+ bucket.get(0));
866                                }
867                                scheduled.add(ef);
868                        }
869                }
870                for (EntityFeature entityFeature : scheduled)
871                {
872                        model.remove(entityFeature);
873                }
874                for (PhysicalEntity physicalEntity : model.getObjects(PhysicalEntity.class))
875                {
876                        Set<EntityFeature> features = new HashSet<EntityFeature>(physicalEntity.getFeature());
877                        for (EntityFeature feature : features)
878                        {
879                                EntityFeature that = mapped.get(feature);
880                                if (that != null && !that.equals(feature))
881                                {
882                                        LOG.debug(" replacing " + feature +
883                                                                          "{" + feature.getUri() + "} with " +
884                                                                          that + "{" + that.getUri() + "}");
885                                        physicalEntity.removeFeature(feature);
886                                        physicalEntity.addFeature(that);
887                                }
888                        }
889                }
890        }

913                                        if (er != null) {
914                                                er.addEntityFeature(ef);
915                                                break;
916                                        }
917                                }
918                        }
919                }

924         * Collects data type (not object) property
925         * values (can be then used for full-text indexing).
926         * 
927         * @param biopaxElement biopax object
928         * @param depth greater or equals 0: 0 means use this object's
929         *        data properties only; 1 - add child's data properties, etc.;
930         *        (the meaning is slightly different from that of Fetcher.fetch(..) method)
931         * @param dataPropertyFilters - biopax data property filters to optionally
932         *                        either skip e.g. properties 'sequence', 'temperature',
933         *                        or only accept 'term', 'comment', 'name', etc.
934         * @return set of keywords
935         */
936        public static Set<String> getKeywords(BioPAXElement biopaxElement, int depth,
937                                                                                  Filter<DataPropertyEditor>... dataPropertyFilters)
938        {
939                LOG.debug("getKeywords called: " + biopaxElement.getUri());
940                
941                EditorMap em = SimpleEditorMap.L3;
942                Set<String> ss = new HashSet<String>();
943
944                //if depth>0, fetch child biopax objects (ignoring PathwayStep.nextStep property)
945                Set<BioPAXElement> elms = (depth > 0)
946                        ? new Fetcher(em, Fetcher.nextStepFilter).fetch(biopaxElement, depth)
947                                : new HashSet<BioPAXElement>();
948
949                //add this one
950                elms.add(biopaxElement);
951                
952                for (BioPAXElement bpe : elms) {
953                        Set<PropertyEditor> props = em.getEditorsOf(bpe);
954                        for (PropertyEditor pe : props) {
955                                //skip for object prop. or one that fails to pass a filter
956                                if (pe instanceof ObjectPropertyEditor
957                                                || !filter((DataPropertyEditor)pe, dataPropertyFilters))
958                                        continue;
959
960                                Set values = pe.getValueFromBean(bpe);
961                                for (Object v : values) {
962                                        if (!pe.isUnknown(v)) {
963                                                ss.add(v.toString());
964                                        }
965                                }
966                        }
967                }
968                
969                return ss;
970        }

1105        // recursively finds all the parent pathways of the object, while escaping infinite loops
1106        private static Set<Pathway> getParentPathwaysRecursively(
1107                        final BioPAXElement biopaxElement, final Set<BioPAXElement> visited) {
1108
1109                final Set<Pathway> pathways = new HashSet<Pathway>();
1110                
1111                //shortcut, when bpe is null or already processed
1112                if(biopaxElement == null || !visited.add(biopaxElement)) {
1113                        LOG.debug("Ignored null or previously visited object:" + biopaxElement);
1114                        return pathways;
1115                }
1116                
1117                LOG.debug("getParentPathways called: " + biopaxElement.getUri());
1118
1119                if(biopaxElement instanceof Process) {
1120                        if(biopaxElement instanceof Pathway) // add itself
1121                                pathways.add((Pathway) biopaxElement);
1122                        // continue looking up to parent pathways (until all top ones reached)
1123                        //TODO: inf. loop happened here (StackOverFlow; e.g., using PC8 KEGG model)
1124                        for(Pathway pw : ((Process)biopaxElement).getPathwayComponentOf())
1125                                pathways.addAll(getParentPathwaysRecursively(pw, visited));
1126                        for(Interaction it : ((Process)biopaxElement).getParticipantOf())
1127                                pathways.addAll(getParentPathwaysRecursively(it, visited));
1128                        for(PathwayStep pt : ((Process)biopaxElement).getStepProcessOf())
1129                                pathways.addAll(getParentPathwaysRecursively(pt, visited));
1130                } else if(biopaxElement instanceof PathwayStep) {
1131                        pathways.addAll(getParentPathwaysRecursively(((PathwayStep) biopaxElement).getPathwayOrderOf(), visited));
1132                } else if(biopaxElement instanceof PhysicalEntity ) {
1133                        for(PhysicalEntity pe : ((PhysicalEntity)biopaxElement).getMemberPhysicalEntityOf())
1134                                pathways.addAll(getParentPathwaysRecursively(pe, visited));
1135                        for(Interaction it : ((Entity)biopaxElement).getParticipantOf())
1136                                pathways.addAll(getParentPathwaysRecursively(it, visited));
1137                        for(Complex c : ((PhysicalEntity)biopaxElement).getComponentOf())
1138                                pathways.addAll(getParentPathwaysRecursively(c, visited));
1139                } else if(biopaxElement instanceof EntityReference) {
1140                        for(EntityReference er : ((EntityReference) biopaxElement).getMemberEntityReferenceOf())
1141                                pathways.addAll(getParentPathwaysRecursively(er, visited));
1142                        for(SimplePhysicalEntity spe : ((EntityReference) biopaxElement).getEntityReferenceOf())
1143                                pathways.addAll(getParentPathwaysRecursively(spe, visited));
1144                } else if (biopaxElement instanceof Gene ) { 
1145                        for(Interaction it : ((Entity) biopaxElement).getParticipantOf())
1146                                pathways.addAll(getParentPathwaysRecursively(it, visited));
1147                } else {
1148                        // ignore
1149                }
1150                
1151                return pathways;
1152        }

1265        // base62 encoding/decoding for (not too long) URI shortening
1266        private static final String ALPHABET = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789";
1267        private static final int BASE = ALPHABET.length();
1268
1269        public static String encodeBase62(String str) {
1270                // encode to base10 (long) first -
1271                char[] chars = new StringBuilder(str).reverse().toString().toCharArray();
1272                long n = 0;
1273                for (int i = chars.length - 1; i >= 0; i--) {
1274                        n += ALPHABET.indexOf(chars[i]) * Math.pow(BASE, i);
1275                }
1276
1277                // convert the long number to a base62 short string
1278                StringBuilder sb = new StringBuilder("");
1279                while (n > 0) {
1280                        int rem = (int)(n % BASE);
1281                        sb.append(ALPHABET.charAt(rem));
1282                        n = n / BASE;
1283                }
1284
1285                return sb.reverse().toString();
1286        }

1304        /**
1305         * Replaces the URI of a BioPAX object in the Model using java reflection.
1306         * If the element also belongs to other BioPAX models, those will become inconsistent
1307         * unless this method is called for each such model.
1308         *
1309         * Warnings:
1310         *  - one should not normally use this method at all;
1311         *  - but if you do, then don't use a URI of another object from the same model.
1312         *
1313         * @param model model (can be null; if the object in fact belongs to a model, the model will be inconsistent)
1314         * @param el biopax object
1315         * @param newUri URI - not null/empty URI
1316         */
1317        public static  void updateUri(Model model, BioPAXElement el, String newUri) {
1318                if(newUri == null || newUri.isEmpty())
1319                        throw new IllegalArgumentException("New URI is null or empty.");
1320
1321                if(model != null)
1322                        model.remove(el);
1323
1324                try {
1325                        Method m = BioPAXElementImpl.class.getDeclaredMethod("setUri", String.class);
1326                        m.setAccessible(true);
1327                        m.invoke(el, newUri);
1328                } catch (Exception e) {
1329                        e.printStackTrace();
1330                        throw new RuntimeException(e);
1331                }
1332
1333                if(model != null)
1334                        model.add(el);
1335        }

1338        /**
1339         * Removes cyclic pathway inclusions, non-trivial infinite loops, in 'pathwayComponent' biopax property.
1340         * Such loops usually do not make much sense and only can cause trouble in pathway data analysis.
1341         * This tool recursively removes parent pathways from sub pathways' pathwayComponent set.
1342         *
1343         * @param model a model that contains Pathways; will be modified as the result
1344         */
1345        public static void breakPathwayComponentCycle(final Model model) {
1346                for(Pathway pathway : model.getObjects(Pathway.class))
1347                        breakPathwayComponentCycle(new HashSet<Pathway>(), pathway, pathway);
1348        }
1349
1350        //Recursively, though, avoiding infinite loops (KEGG pathways can cause it),
1351        // removes cyclic pathway inclusions like pathwayA/pathwayA or pathwayA/pathwayB/pathwayA, etc.
1352        private static void breakPathwayComponentCycle(final Set<Pathway> visited, final Pathway rootPathway,
1353                                                                                        final Pathway currentPathway)
1354        {
1355                if(!visited.add(currentPathway))
1356                        return; // already processed
1357
1358                if(currentPathway.getPathwayComponent().contains(rootPathway)) {
1359                        currentPathway.removePathwayComponent(rootPathway);
1360                }
1361
1362                for(Process proc : currentPathway.getPathwayComponent())
1363                        if(proc instanceof Pathway)
1364                                breakPathwayComponentCycle(visited, rootPathway, (Pathway) proc);
1365        }
