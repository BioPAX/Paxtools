12  /**
13   * Specifically "Clones" the BioPAX elements set
14   * (traverses to obtain dependent elements), 
15   * puts them to the new model using the visitor and traverser framework;
16   * ignores elements that are not in the source list (compare to {@link Fetcher}).
17   *
18   * @see org.biopax.paxtools.controller.Visitor
19   * @see org.biopax.paxtools.controller.Traverser
20   */
21  public class Cloner implements Visitor
22  {
23  	private static final Logger LOG = LoggerFactory.getLogger(Cloner.class);
24  	
25  	private final Traverser traverser;
26  	private final BioPAXFactory factory;
27  
28  	public Cloner(EditorMap map, BioPAXFactory factory)
29  	{
30  		this.factory = factory;
31  		this.traverser = new Traverser(map, this);
32  	}
33  
34  
35  	/**
36  	 * For each element from the 'toBeCloned' list,
37  	 * it creates a copy in the new model, setting all
38  	 * the data properties; however, object property values
39  	 * that refer to BioPAX elements not in 'toBeCloned' list
40  	 * are ignored.
41  	 *
42  	 *	@deprecated use {@link #clone(Set)} instead ('source' model was not used in previous versions)
43  	 *
44  	 * @param source - actually, is never used... can be null all the same
45  	 * @param toBeCloned elements to clone
46  	 * @return a new model containing the cloned biopax objects
47  	 */
48  	public Model clone(Model source, Set<BioPAXElement> toBeCloned) {
49  		return clone(toBeCloned);
50  	}
51  
52  	/**
53  	 * For each element from the 'toBeCloned' list,
54  	 * it creates a copy in the new model, setting all 
55  	 * the data properties; however, object property values
56  	 * that refer to BioPAX elements not in 'toBeCloned' list
57  	 * are ignored.
58  	 * 
59  	 * @param toBeCloned elements to clone
60  	 * @return a new model containing the cloned biopax objects
61  	 */
62  	public Model clone(Set<BioPAXElement> toBeCloned)
63  	{
64  		final Model targetModel = factory.createModel();
65  
66  		for (BioPAXElement bpe : new HashSet<BioPAXElement>(toBeCloned))
67  		{
68  			// make a copy (all properties are empty except for ID)
69  			if(targetModel.containsID(bpe.getUri())) {
70  				throw new RuntimeException("There're different objects having the same URI"
71  						+ " in the target/cloned model and input set:" + bpe.getUri());
72  			} else {
73  				targetModel.addNew(bpe.getModelInterface(), bpe.getUri());
74  			}
75  		}
76  
77  		// a hack to avoid unnecessary checks for the valid sub-model being cloned, 
78  		// and warnings when the Cloner copies BPS.stepProcess values, 
79  		// and there is a Conversion among them (-always unless stepConversion is null).
80  		AbstractPropertyEditor.checkRestrictions.set(false);
81  		
82  		for (BioPAXElement bpe : toBeCloned)
83  		{
84  			traverser.traverse(bpe, targetModel);
85  		}
86  		
87  		AbstractPropertyEditor.checkRestrictions.set(true); //back to the default mode
88  
89  		return targetModel;
90  	}
91  
92  	//  Implement interface: Visitor
93  	public void visit(BioPAXElement domain, Object range, Model targetModel, PropertyEditor editor)
94  	{
95  		BioPAXElement targetDomain = targetModel.getByID(domain.getUri());
96  		
97  		if (range instanceof BioPAXElement)
98  		{
99  			BioPAXElement bpe = (BioPAXElement) range;
100 			BioPAXElement existing = targetModel.getByID(bpe.getUri());
101 			//set the property value if the value is already present in the target 
102 			if (existing != null) {
103 				editor.setValueToBean(existing, targetDomain);
104 			}
105 		}
106 		else
107 		{
108 			editor.setValueToBean(range, targetDomain);
109 		}
110 	}
111 }
