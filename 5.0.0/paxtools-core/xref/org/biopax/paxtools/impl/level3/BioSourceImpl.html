16  public class BioSourceImpl extends NamedImpl implements BioSource
17  {
18  	private final static Logger LOG = LoggerFactory.getLogger(BioSourceImpl.class);
19  	
20  	private CellVocabulary celltype;
21  	private TissueVocabulary tissue;
22  
23  
24  	public BioSourceImpl(){
25  	}
26  
27  	//
28  	// BioPAXElement interface implementation
29  	//
30  	////////////////////////////////////////////////////////////////////////////
31  	public Class<? extends BioSource> getModelInterface()
32  	{
33  		return BioSource.class;
34  	}
35  
36  	protected boolean semanticallyEquivalent(BioPAXElement element)
37  	{
38  		if(!(element instanceof BioSource))
39  			return false;
40  		
41  		final BioSource bioSource = (BioSource) element;
42  
43  		return
44  			(celltype != null ?
45  				celltype.isEquivalent(bioSource.getCellType()) :
46  				bioSource.getCellType() == null)
47  			&&
48  				(tissue != null ?
49  					tissue.isEquivalent(bioSource.getTissue()) :
50  					bioSource.getTissue() == null)
51  			&& hasEquivalentIntersection(
52  				new ClassFilterSet<Xref, UnificationXref>(getXref(), UnificationXref.class),
53  				new ClassFilterSet<Xref, UnificationXref>(bioSource.getXref(), UnificationXref.class));
54      }
55  
56  	public int equivalenceCode()
57  	{
58  		int result = 29 *super.equivalenceCode() + (celltype != null ? celltype.hashCode() : 0);
59  		result = 29 * result + (tissue != null ? tissue.hashCode() : 0);
60  		return result;
61  	}
62  
63  	//
64  	// BioSource interface implementation
65  	//
66  	////////////////////////////////////////////////////////////////////////////
67  	public CellVocabulary getCellType()
68  	{
69  		return celltype;
70  	}
71  
72  	public void setCellType(CellVocabulary celltype)
73  	{
74  		this.celltype = celltype;
75  	}
76  
77  	public TissueVocabulary getTissue()
78  	{
79  		return tissue;
80  	}
81  
82  	public void setTissue(TissueVocabulary tissue)
83  	{
84  		this.tissue = tissue;
85  	}

88  	@Override
89  	public String toString() {
90  		try {
91  			StringBuilder sb = new StringBuilder();
92  			sb.append(getUri()).append(" ");
93  			sb.append(getName().toString());
94  			if (tissue != null)
95  				sb.append(" tissue: ").append(tissue.getTerm().toString());
96  			if (celltype != null)
97  				sb.append(" celltype: ").append(celltype.getTerm().toString());
98  			sb.append(" xrefs: ").append(getXref().toString());
99  			return sb.toString();
100 		} catch (Exception e) {
101 			// possible issues - when in a persistent context (e.g., lazy collections init...)
102 			LOG.warn("Error in toString(): ", e);
103 			return getUri();
104 		}
105 	}
106 }
