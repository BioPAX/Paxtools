5   /**
6    * Definition: An interaction in which molecules of one or more {@link PhysicalEntity} pools are physically
7    * transformed and become a member of one or more other PhysicalEntity pools.
8    * 
9    * Rationale: Conversion and pools of entities are two central abstractions of chemistry. They can be quantized with
10   * rate and concentration respectively. A conversion in BioPAX, similar to chemistry,
11   * is stoichiometric and closed world, i.e. it is assumed that all of the participants are listed. These properties
12   * are due to the law of mass conservation. This differs from Control for example where multiple entities might be
13   * controlling the controlled and everything that is not listed is assumed to be unknown. Conversions in BioPAX can
14   * be reversible -- the property names Left and Right were preferred specifically because they are direction neutral
15   * as opposed to substrate and product or input and output.
16   * 
17   * Usage: Subclasses of conversion represent different types of transformation reflected by the properties of
18   * different physicalEntity. {@link BiochemicalReaction}s will change the {@link ModificationFeature}s on a
19   * {@link PhysicalEntity}, {@link Transport} will change the Cellular Location and {@link ComplexAssembly} will
20   * change {@link BindingFeature}s. Generic Conversion class should only be used when the modification does not fit
21   * into a any of these classes.
22   * 
23   * Example: Opening of a voltage gated channel.
24   */
25  public interface Conversion extends Interaction
26  {
27  
28  
29  	/**
30  	 * The participants on the left side of the conversion interaction. Since conversion interactions may proceed
31  	 * in either the left-to-right or right-to-left direction, occupants of the left property may be either
32  	 * reactants or products. left is a sub-property of participants.
33  	 * @return The participants on the left side of the conversion interaction.
34  	 */
35  	public Set<PhysicalEntity> getLeft();
36  
37  	/**
38  	 * Adds a participant to the left side of the conversion interaction. Since conversion interactions may proceed
39  	 * in either the left-to-right or right-to-left direction, occupants of the left property may be either
40  	 * reactants or products. left is a sub-property of participants.
41  	 * @param left participants to be added to the left side of the conversion interaction.
42  	 */
43  	void addLeft(PhysicalEntity left);
44  
45  	/**
46  	 * Removes a participant from the left side of the conversion interaction. Since conversion interactions may
47  	 * proceed
48  	 * in either the left-to-right or right-to-left direction, occupants of the left property may be either
49  	 * reactants or products. left is a sub-property of participants.
50  	 * @param left participants to be removed from the left side of the conversion interaction.
51  	 */
52  	void removeLeft(PhysicalEntity left);
53  
54  
55  	/**
56  	 * Stoichiometry of the left ({@link #getLeft()}) and right({@link #getRight()}) participants.
57  	 * Note: This is a "bridge" workaround for the n-ary relationship problem. There is no default stoichiometry (
58  	 * e.g. 1). Leaving this out will create reactions with unknown stoichiometry.
59  	 * It is a best practice to define the stoichiometry of each participant.
60  	 * It is invalid to define more than one stoichiometry per participant.
61  	 * It is invalid to reuse stoichiometry instances across conversions.
62  	 * @return Stoichiometry of the left ({@link #getLeft()}) and right({@link #getRight()}) participants.
63  	 */
64  	Set<Stoichiometry> getParticipantStoichiometry();
65  
66  	/**
67  	 * This method adds a stoichiometry for one of the participants of this conversion.
68  	 * @param stoichiometry to be added
69  	 */
70  	void addParticipantStoichiometry(Stoichiometry stoichiometry);
71  
72  	/**
73  	 * This method removes a stoichiometry for one of the participants of this conversion.
74  	 * @param stoichiometry to be removed
75  	 */
76  	void removeParticipantStoichiometry(Stoichiometry stoichiometry);
77  
78

82  	 * The participants on the right side of the conversion interaction. Since conversion interactions may proceed
83  	 * in either the right-to-right or right-to-right direction, occupants of the right property may be either
84  	 * reactants or products. right is a sub-property of participants.
85  	 * @return The participants on the right side of the conversion interaction.
86  	 */
87  	public Set<PhysicalEntity> getRight();
88  
89  	/**
90  	 * Adds a participant to the right side of the conversion interaction. Since conversion interactions may proceed
91  	 * in either the right-to-right or right-to-right direction, occupants of the right property may be either
92  	 * reactants or products. right is a sub-property of participants.
93  	 * @param right participants to be added to the right side of the conversion interaction.
94  	 */
95  	void addRight(PhysicalEntity right);
96  
97  	/**
98  	 * Removes a participant from the right side of the conversion interaction. Since conversion interactions may
99  	 * proceed
100 	 * in either the right-to-right or right-to-right direction, occupants of the right property may be either
101 	 * reactants or products. right is a sub-property of participants.
102 	 * @param right participants to be removed from the right side of the conversion interaction.
103 	 */
104 	void removeRight(PhysicalEntity right);

108 	 * Specifies whether a conversion occurs spontaneously or not.
109 	 * 
110 	 * If the spontaneity is not known,this property should not be specified.
111 	 * @return whether if this conversion is spontaneous.
112 	 */
113 	public Boolean getSpontaneous();
114 
115 	/**
116 	 * Specifies whether a conversion occurs spontaneously or not.
117 	 * 
118 	 * If the spontaneity is not known,this property should not be specified.
119 	 * @param spontaneous Whether if this conversion is spontaneous.
120 	 */
121 	public void setSpontaneous(Boolean spontaneous);
122 
123 
124 	/**
125 	 * This property represents the direction of the reaction. If a reaction will run in a single direction under
126 	 * all biological contexts then it is considered irreversible and has a direction. Otherwise it is reversible.
127 	 * @return One of REVERSIBLE, LEFT-TO-RIGHT or RIGHT-TO-LEFT
128 	 */
129 	public ConversionDirectionType getConversionDirection();
130 
131 	/**
132 	 * This property represents the direction of the reaction. If a reaction will run in a single direction under
133 	 * all biological contexts then it is considered irreversible and has a direction. Otherwise it is reversible.
134 	 * @param conversionDirection One of REVERSIBLE, LEFT-TO-RIGHT or RIGHT-TO-LEFT --  null for unknown.
135 	 */
136 	public void setConversionDirection(ConversionDirectionType conversionDirection);
