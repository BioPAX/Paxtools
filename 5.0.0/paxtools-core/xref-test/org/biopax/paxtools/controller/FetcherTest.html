50  	@Test
51  	public final void testForEndlessLoopOrOutOfMemory() {
52  
53  		Pathway pw1 = (Pathway) model.getByID("pathway1");
54  		Set<Protein> proteins = new Fetcher(SimpleEditorMap.L3).fetch(pw1, Protein.class);
55  		assertEquals(2, proteins.size());
56  
57  		Set<BioPAXElement> bpes = new Fetcher(SimpleEditorMap.L3).fetch(pw1);
58  		assertEquals(8, bpes.size());
59  	}

61  	@Test
62  	public final void testFetchDepth() {
63  
64  		Pathway pw1 = (Pathway) model.getByID("pathway1");
65  		Set<BioPAXElement> elements;
66  
67  		//exception
68  		try {
69  			new Fetcher(SimpleEditorMap.L3).fetch(pw1, 0);
70  			fail();
71  		} catch (IllegalArgumentException e) {
72  		}
73  		try {
74  			new Fetcher(SimpleEditorMap.L3).fetch(pw1, -1);
75  			fail();
76  		} catch (IllegalArgumentException e) {
77  		}
78  
79  		//depth=1
80  		elements = new Fetcher(SimpleEditorMap.L3).fetch(pw1, 1);
81  		assertEquals(4, elements.size());
82  
83  		//depth=2
84  		elements = new Fetcher(SimpleEditorMap.L3).fetch(pw1, 2);
85  		assertEquals(6, elements.size()); //that's it
86  
87  		//depth=3 - all objects except the pw1 itself (despite it belongs to its sub-pathway - loop)
88  		elements = new Fetcher(SimpleEditorMap.L3).fetch(pw1, 3);
89  		assertEquals(7, elements.size());
90  
91  		//depth > 3 all the same...
92  		elements = new Fetcher(SimpleEditorMap.L3).fetch(pw1, 4);
93  		assertEquals(7, elements.size());
94  	}
