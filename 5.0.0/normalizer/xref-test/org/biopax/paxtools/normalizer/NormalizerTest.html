28  	@Test
29  	public final void testUri() {
30  		 // using null or non-standard db
31  		 assertEquals(Normalizer.uri("test/", "foo", "bar", UnificationXref.class), Normalizer.uri("test/", "FOo", "bar", UnificationXref.class));
32  		 //'pubchem' is a ambigous synonym (correct ones are: pubchem-substance, pubchem-compound, etc.)
33  		 assertEquals(Normalizer.uri("", "pubchem", "bar", UnificationXref.class), Normalizer.uri("", "PubChem", "bar", UnificationXref.class));
34  		 assertEquals(Normalizer.uri("", null, "bar", UnificationXref.class), Normalizer.uri(null, null, "bar", UnificationXref.class));
35  		 assertFalse(Normalizer.uri(null, "foo", "bar", UnificationXref.class).equals(Normalizer.uri(null, "foo", "BAR", UnificationXref.class)));
36  		 assertFalse(Normalizer.uri(null, "foo", "bar", UnificationXref.class).equals(Normalizer.uri(null, "foo", "bar", PublicationXref.class)));
37  		 
38  		 // using standard db names (Miriam is used to normalize name and/or get identifiers.org URI) -
39  		 assertEquals(Normalizer.uri("test/", "pubmed", "12345", PublicationXref.class), Normalizer.uri("test/", "PubMED", "12345", PublicationXref.class));
40  		 assertEquals("http://identifiers.org/pubmed/12345", Normalizer.uri("test/", "PubMED", "12345", PublicationXref.class));
41  		 assertFalse("http://identifiers.org/pubmed/12345".equals(Normalizer.uri(null, "PubMED", "12345", RelationshipXref.class))); //- not PublicationXref
42  		 
43  		 assertEquals("http://identifiers.org/chebi/CHEBI:12345",Normalizer.uri("", "chebi", "CHEBI:12345", SmallMoleculeReference.class));
44  		 assertEquals("http://identifiers.org/pubchem.substance/12345",Normalizer.uri("", "pubchem-substance", "12345", SmallMoleculeReference.class));
45  		 
46  		 System.setProperty("biopax.normalizer.uri.strategy", Normalizer.VALUE_NORMALIZER_URI_STRATEGY_SIMPLE);
47  		 assertEquals("SequenceModificationVocabulary_protein_modification_ontology_MOD_12345",Normalizer.uri("", "PSI-mod", "MOD:12345", SequenceModificationVocabulary.class));
48  		 assertEquals("ControlledVocabulary_protein_modification_ontology_MOD_12345",Normalizer.uri("", "MOD", "MOD:12345", ControlledVocabulary.class));
49  		 System.setProperty("biopax.normalizer.uri.strategy", Normalizer.VALUE_NORMALIZER_URI_STRATEGY_MD5);
50  		 //wrong id (case-sens.)
51  		 assertFalse("http://identifiers.org/chebi/CHEBI:12345".equals(Normalizer.uri("", "chebi", "chebi:12345", SmallMoleculeReference.class)));
52  		 //no 'pubchem' namespace there
53  		 assertFalse("http://identifiers.org/pubchem/12345".equals(Normalizer.uri("", "pubchem-substance", "12345", UnificationXref.class)));
54  		 
55  		 //when there're special symbols, spaces in the 'id' part
56  		 assertEquals("UnificationXref_foo_bar", Normalizer.uri(null, null, "foo bar", UnificationXref.class));
57  
58  		assertEquals("http://identifiers.org/taxonomy/9606", Normalizer.uri(null, "taxonomy", "9606", BioSource.class));
59  		assertEquals("http://identifiers.org/taxonomy/9606", Normalizer.uri(null, "NCBI Taxonomy", "9606", BioSource.class));
60  		assertEquals("http://identifiers.org/taxonomy/9606", Normalizer.uri(null, "NEWT", "9606", BioSource.class));
61  		//when organism's id is not taxID (e.g., if the BioSource has tissue, cellType CVs...)
62  		assertNotSame("http://identifiers.org/taxonomy/9606",Normalizer.uri(null, "taxonomy", "9606_blah_blah", BioSource.class));
63  
64  		String uri = Normalizer.uri("", "UniProt", "W0C7J9", UnificationXref.class);
65  		assertEquals("UnificationXref_uniprot_knowledgebase_W0C7J9", uri);
66  	}

344 	@Test
345 	public final void testNormalizeInoh() {
346 		//there're two Protein objects that have entityReference 
347 		//rdf:ID="IMR_0100366_G_alpha_s_Canonical" (in fact a generic ProteinReference inproperly defined);
348 		//and that PR has two UniProt UnificationXref: P63092 (human), P63095 (rat).
349 		//Identifiers.org URI for the PR should NOT be generated.
350 		//
351 		//Also, there was a bug when replacing the PR's URI (warning: IllegalBioPAXArgumentException: Incompatible type!..)
352 		//The cause is that there was a (weird, invalid) PublicationXref having db:id as UniProt:P63092 that gets the same URI!
353 		//To avoid such issues altogether (despite illegal pub. xrefs), the Normalizer won't use Identifiers.org for PublicationXrefs anymore.
354 		
355 		Model model = simpleIO.convertFromOWL(getClass().getResourceAsStream("/test-inoh.owl"));
356 		Normalizer normalizer = new Normalizer();
357 		normalizer.setXmlBase("");
358 		normalizer.normalize(model);
359 
360 		//A weird PublicationXref that uses a UniProt ID won't be normalized:
361 		assertFalse(model.containsID("http://identifiers.org/uniprot/P63092"));
362 
363 		//A PR with two UniProt IDs/unif.xrefs - human, rat - won't be normalized!
364 		assertTrue(model.containsID(model.getXmlBase() + "IMR_0100366_G_alpha_s_Canonical"));
365 		assertEquals("ProteinReference", model.getByID(model.getXmlBase() + "IMR_0100366_G_alpha_s_Canonical")
366 				.getModelInterface().getSimpleName());
367 	}

380 				+ ", id=" + x.getId() + ", idVer=" + x.getIdVersion());
381 			for(XReferrable rx : x.getXrefOf()) {
382 				System.out.println("model=" + m.contains(rx) + ":\t" 
383 					+ "    xrefOf: " + rx);
384 			}
385 		}
