1   package org.biopax.paxtools.pattern.constraint;
2   
3   import org.biopax.paxtools.model.BioPAXElement;
4   import org.biopax.paxtools.pattern.MappedConst;
5   import org.biopax.paxtools.pattern.Match;
6   
7   import java.util.Collection;
8   import java.util.HashSet;
9   import java.util.Set;
10  
11  /**
12   * Checks if generated elements has or has not a specific value for a field, or the field value of
13   * another set of generated elements.
14   *
15   * @author Ozgun Babur
16   */
17  public class FieldOfMultiple extends Field
18  {
19  	/**
20  	 * Generative constraint for first group of generated elements, to use on first mapped element.
21  	 */
22  	protected MappedConst con1;
23  
24  	/**
25  	 * Generative constraint for second group of generated elements, to use on second mapped
26  	 * element.
27  	 */
28  	protected MappedConst con2;
29  
30  	/**
31  	 * Constructor with accessor string for the field value of the element and the desired value. If
32  	 * the desired value is EMPTY, then emptiness is checked. If it is USE_SECOND_ARG, then the
33  	 * second mapped element is used as the desired value. If a filed of the second element is
34  	 * desired then the other constructor should be used.
35  	 *
36  	 * @param con the generative constraint of size 2 whose generated values will be checked
37  	 * @param accessorString accessor string for the element
38  	 * @param oper operation
39  	 * @param value desired value
40  	 */
41  	public FieldOfMultiple(MappedConst con, String accessorString, Operation oper, Object value)
42  	{
43  		super(accessorString, oper, value);
44  		con1 = con;
45  	}
46  
47  	/**
48  	 * Constructor with accessor strings for the field value of the element and the desired value
49  	 * that will be reached from the second element.
50  	 *
51  	 * @param con1 the generative constraint of size 2 for the first group of elements
52  	 * @param accessorString1 accessor string for the first element
53  	 * @param con2 the generative constraint of size 2 for the second group of elements
54  	 * @param accessorString2 accessor string for the second element
55  	 * @param oper operation
56  	 */
57  	public FieldOfMultiple(MappedConst con1, String accessorString1, MappedConst con2,
58  		String accessorString2, Operation oper)
59  	{
60  		super(accessorString1, accessorString2, oper);
61  		this.con1 = con1;
62  		this.con2 = con2;
63  	}
64  
65  	/**
66  	 * Size of this constraint is one less than con1 if con2 is null, otherwise it is two less than
67  	 * the total of size of con1 and con2.
68  	 * @return the size based on con1 and con2
69  	 */
70  	@Override
71  	public int getVariableSize()
72  	{
73  		return con1.getVariableSize() + (con2 != null ? con2.getVariableSize() - 2 : -1);
74  	}
75  
76  	/**
77  	 * Checks if the generated elements from the first mapped element has either the desired value,
78  	 * or has some value in common with the elements generated from second mapped element.
79  	 * @param match current pattern match
80  	 * @param ind mapped indices
81  	 * @return true if a value match is found
82  	 */
83  	@Override
84  	public boolean satisfies(Match match, int... ind)
85  	{
86  		assertIndLength(ind);
87  
88  		// Collect values of the element group
89  		Set values = new HashSet();
90  		for (BioPAXElement gen : con1.generate(match, ind))
91  		{
92  			values.addAll(pa1.getValueFromBean(gen));
93  		}
94  
95  		// If emptiness is desired, check that
96  		if (value == EMPTY) return values.isEmpty();
97  
98  		// If cannot be empty, check it
99  		if (oper == Operation.NOT_EMPTY_AND_NOT_INTERSECT && values.isEmpty()) return false;
100 
101 		// If the second element is desired value, check that
102 		else if (value == USE_SECOND_ARG)
103 		{
104 			BioPAXElement q = match.get(ind[1]);
105 			return oper == Operation.INTERSECT ? values.contains(q) : !values.contains(q);
106 		}
107 
108 		// If element group is compared to preset value, but the value is actually a collection,
109 		// then iterate the collection, see if any of them matches
110 		else if (value instanceof Collection)
111 		{
112 			Collection query = (Collection) value;
113 			values.retainAll(query);
114 
115 			if (oper == Operation.INTERSECT) return !values.isEmpty();
116 			else return values.isEmpty();
117 		}
118 
119 		// If two set of elements should share a field value, check that
120 		else if (pa2 != null)
121 		{
122 			// Collect values of the second group
123 			Set others = new HashSet();
124 			for (BioPAXElement gen : con2.generate(match, ind))
125 			{
126 				others.addAll(pa2.getValueFromBean(gen));
127 			}
128 
129 			switch (oper)
130 			{
131 				case INTERSECT:
132 					others.retainAll(values);
133 					return !others.isEmpty();
134 				case NOT_INTERSECT:
135 					others.retainAll(values);
136 					return others.isEmpty();
137 				case NOT_EMPTY_AND_NOT_INTERSECT:
138 					if (others.isEmpty()) return false;
139 					others.retainAll(values);
140 					return others.isEmpty();
141 				default: throw new RuntimeException("Unhandled operation: " + oper);
142 			}
143 		}
144 
145 		// Check if the element field values contain the parameter value
146 		else if (oper == Operation.INTERSECT) return values.contains(value);
147 		else return !values.contains(value);
148 	}
149 }
