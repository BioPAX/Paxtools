3   import org.biopax.paxtools.pattern.MappedConst;
4   import org.biopax.paxtools.pattern.Match;
5   import org.biopax.paxtools.model.BioPAXElement;
6   
7   import java.util.Collection;
8   import java.util.HashSet;
9   
10  /**
11   * Used for getting logical AND of a set of constraints.
12   *
13   * @author Ozgun Babur
14   */
15  public class AND extends OR
16  {
17  	/**
18  	 * Constructor with the mapped constraints.
19  	 * @param con mapped constraints
20  	 */
21  	public AND(MappedConst... con)
22  	{
23  		super(con);
24  	}
25  
26  	/**
27  	 * Checks if all the constraints satisfy.
28  	 * @param match match to validate
29  	 * @param ind mapped indices
30  	 * @return true if all satisfy
31  	 */
32  	@Override
33  	public boolean satisfies(Match match, int... ind)
34  	{
35  		for (MappedConst mc : con)
36  		{
37  			if (!mc.satisfies(match, ind)) return false;
38  		}
39  		return true;
40  	}
41  
42  	/**
43  	 * Gets intersection of the generated elements by the member constraints.
44  	 * @param match match to process
45  	 * @param ind mapped indices
46  	 * @return satisfying elements
47  	 */
48  	@Override
49  	public Collection<BioPAXElement> generate(Match match, int... ind)
50  	{
51  		Collection<BioPAXElement> gen = new HashSet<BioPAXElement> (
52  			con[0].generate(match, ind));
53  
54  		for (int i = 1; i < con.length; i++)
55  		{
56  			if (gen.isEmpty()) break;
57  
58  			gen.retainAll(con[i].generate(match, ind));
59  		}
60  		return gen;
61  	}
62  }
