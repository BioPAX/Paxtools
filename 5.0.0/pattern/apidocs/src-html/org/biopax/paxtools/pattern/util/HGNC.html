001package org.biopax.paxtools.pattern.util;
002
003import java.io.BufferedReader;
004import java.io.FileNotFoundException;
005import java.io.IOException;
006import java.io.InputStreamReader;
007import java.util.HashMap;
008import java.util.Map;
009import java.util.Set;
010
011/**
012 * This class provides a mapping between HGNC IDs and Symbols.
013 *
014 * @author Ozgun Babur
015 */
016public class HGNC
017{
018        private static Map<String, String> sym2id;
019        private static Map<String, String> id2sym;
020        private static Map<String, String> old2new;
021
022        public static String getSymbol(String idOrSymbol)
023        {
024                if (id2sym.containsKey(idOrSymbol)) return id2sym.get(idOrSymbol);
025                else if (sym2id.containsKey(idOrSymbol)) return idOrSymbol;
026                else if (old2new.containsKey(idOrSymbol)) return old2new.get(idOrSymbol);
027                else if (!idOrSymbol.toUpperCase().equals(idOrSymbol))
028                        return getSymbol(idOrSymbol.toUpperCase());
029                else return null;
030        }
031
032        static
033        {
034                try
035                {
036                        sym2id = new HashMap<String, String>();
037                        id2sym = new HashMap<String, String>();
038                        old2new = new HashMap<String, String>();
039                        BufferedReader reader = new BufferedReader(new InputStreamReader(
040                                HGNC.class.getResourceAsStream("hgnc.txt")));
041
042                        reader.readLine(); //skip header
043                        for (String line = reader.readLine(); line != null; line = reader.readLine())
044                        {
045                                String[] token = line.split("\t");
046                                String sym = token[1];
047                                String id = token[0];
048                                sym2id.put(sym, id);
049                                id2sym.put(id, sym);
050
051                                if (token.length > 2)
052                                {
053                                        String olds = token[2];
054                                        for (String old : olds.split(","))
055                                        {
056                                                old = old.trim();
057                                                old2new.put(old, sym);
058                                        }
059                                }
060                        }
061                        reader.close();
062                }
063                catch (FileNotFoundException e)
064                {
065                        e.printStackTrace();
066                }
067                catch (IOException e)
068                {
069                        e.printStackTrace();
070                }
071        }
072}
