16      @Override
17      public String toString()
18      {
19      	StringBuilder sb = new StringBuilder();
20      	 	
21          if(modificationType!=null)
22          	sb.append("ModificationFeature: ").append(modificationType.getTerm());
23          
24          if(getFeatureLocation() != null) {
25          	if(sb.length() == 0)
26          		sb.append("ModificationFeature: ");
27          	sb.append("@"+getFeatureLocation());
28          }
29          
30          if(sb.length() == 0)
31          	sb.append(super.toString());
32          
33          return sb.toString();
34      }

51  	@Override
52  	protected boolean semanticallyEquivalent(BioPAXElement element)
53  	{
54  		if(!(element instanceof ModificationFeature))
55  			return false;
56  		
57  		ModificationFeature that = (ModificationFeature) element;
58  		boolean value = super.atEquivalentLocation(that);
59  		if (value)
60  		{
61  			SequenceModificationVocabulary yourType = that.getModificationType();
62  			SequenceModificationVocabulary myType = getModificationType();
63  			value = (yourType == null) ?
64  			        myType == null :
65  			        yourType.isEquivalent(myType);
66  		}
67  
68  		return value;
69  	}
70  
71  	@Override
72  	public int equivalenceCode()
73  	{
74  		SequenceModificationVocabulary myType = this.getModificationType();
75  		int code = myType == null ?0:myType.hashCode();
76  		return code+13*super.locationCode();
77  	}
