14  /**
15   * Implements breadth-first search. Takes a set of source nodes, distance limit and labels nodes
16   * towards one direction, with their breadth distances.
17   *
18   * @author Ozgun Babur
19   * @author Merve Cakir
20   */
21  public class BFS
22  {
23  	private static Logger LOG = LoggerFactory.getLogger(BFS.class);
24  	
25  	/**
26  	 * Distance labels. Missing label interpreted as infinitive.
27  	 */
28  	protected Map<GraphObject, Integer> dist;
29  
30  	/**
31  	 * Color labels. Missing color interpreted as white.
32  	 */
33  	protected Map<GraphObject, Integer> colors;
34  
35  	/**
36  	 * BFS starts from source nodes. They get the label 0.
37  	 */
38  	protected Set<Node> sourceSet;
39  
40  	/**
41  	 * BFS will not further traverse neighbors of any node in the stopSet.
42  	 */
43  	protected Set<Node> stopSet;
44  
45  	/**
46  	 * Whether the direction is FORWARD, it is REVERSE otherwise.
47  	 */
48  	protected Direction direction;
49  
50  	/**
51  	 * Stop distance.
52  	 */
53  	protected int limit;
54  
55  	/**
56  	 * BFS queue.
57  	 */
58  	protected LinkedList<Node> queue;
59  
60  	/**
61  	 * Constructor with all parameters.
62  	 * @param sourceSet Seed of BFS
63  	 * @param stopSet Nodes that won't be traversed
64  	 * @param direction Direction of the traversal
65  	 * @param limit Distance limit
66  	 */
67  	public BFS(Set<Node> sourceSet, Set<Node> stopSet, Direction direction, int limit)
68  	{
69  		if (direction != Direction.UPSTREAM && direction != Direction.DOWNSTREAM)
70  			throw new IllegalArgumentException("Direction has to be either upstream or downstream");
71  
72  		this.sourceSet = sourceSet;
73  		this.stopSet = stopSet;
74  		this.direction = direction;
75  		this.limit = limit;
76  	}
77  
78  	/**
79  	 * Empty constructor for other possible uses.
80  	 */
81  	public BFS()
82  	{
83  	}
84  
85  	/**
86  	 * Executes the algorithm.
87  	 * @return BFS tree
88  	 */
89  	public Map<GraphObject, Integer> run()
90  	{
91  		initMaps();
92  
93  		// Add all source nodes to the queue if traversal is needed
94  
95  		if (limit > 0)
96  		{
97  			queue.addAll(sourceSet);
98  		}
99  
100 		// Initialize dist and color of source set
101 
102 		for (Node source : sourceSet)
103 		{
104 			setLabel(source, 0);
105 			setColor(source, GRAY);
106 
107 			labelEquivRecursive(source, UPWARD, 0, true, false);
108 			labelEquivRecursive(source, DOWNWARD, 0, true, false);
109 		}
110 
111 		// Process the queue
112 
113 		while (!queue.isEmpty())
114 		{
115 			Node current = queue.remove(0);
116 
117 			processNode(current);
118 
119 			// Current node is processed
120 			setColor(current, BLACK);
121 		}
122 
123 		return dist;
124 	}
125 
126 	/**
127 	 * Initializes maps used during query.
128 	 */
129 	protected void initMaps()
130 	{
131 		// Initialize label, maps and queue
132 
133 		dist = new HashMap<GraphObject, Integer>();
134 		colors = new HashMap<GraphObject, Integer>();
135 		queue = new LinkedList<Node>();
136 	}
137 
138 	/**
139 	 * Processes a node.
140 	 * @param current The current node
141 	 */
142 	protected void processNode(Node current)
143 	{
144 		// Do not process the node if it is ubique
145 
146 		if (current.isUbique())
147 		{
148 			setColor(current, BLACK);
149 			return;
150 		}
151 
152 //		System.out.println("processing = " + current);
153 
154 		// Process edges towards the direction
155 
156 		for (Edge edge : direction == Direction.DOWNSTREAM ?
157 			current.getDownstream() : current.getUpstream())
158 		{
159 			assert edge != null;
160 
161 			// Label the edge considering direction of traversal and type of current node
162 
163 			if (direction == Direction.DOWNSTREAM || !current.isBreadthNode())
164 			{
165 				setLabel(edge, getLabel(current));
166 			}
167 			else
168 			{
169 				setLabel(edge, getLabel(current) + 1);
170 			}
171 
172 			// Get the other end of the edge
173 			Node neigh = direction == Direction.DOWNSTREAM ?
174 				edge.getTargetNode() : edge.getSourceNode();
175 
176 			assert neigh != null;
177 
178 			// Decide neighbor label according to the search direction and node type
179 			int dist = getLabel(edge);
180 			if (neigh.isBreadthNode() && direction == Direction.DOWNSTREAM) dist++;
181 
182 			// Check if we need to stop traversing the neighbor, enqueue otherwise
183 			boolean further = (stopSet == null || !isEquivalentInTheSet(neigh, stopSet)) &&
184 				(!neigh.isBreadthNode() || dist < limit) && !neigh.isUbique();
185 
186 			// Process the neighbor if not processed or not in queue
187 
188 			if (getColor(neigh) == WHITE)
189 			{
190 				// Label the neighbor
191 				setLabel(neigh, dist);
192 
193 				if (further)
194 				{
195 					setColor(neigh, GRAY);
196 
197 					// Enqueue the node according to its type
198 
199 					if (neigh.isBreadthNode())
200 					{
201 						queue.addLast(neigh);
202 					}
203 					else
204 					{
205 						// Non-breadth nodes are added in front of the queue
206 						queue.addFirst(neigh);
207 					}
208 				}
209 				else
210 				{
211 					// If we do not want to traverse this neighbor, we paint it black
212 					setColor(neigh, BLACK);
213 				}
214 			}
215 
216 
217 			labelEquivRecursive(neigh, UPWARD, getLabel(neigh), further, !neigh.isBreadthNode());
218 			labelEquivRecursive(neigh, DOWNWARD, getLabel(neigh), further, !neigh.isBreadthNode());
219 		}
220 	}
221 
222 	/**
223 	 * Labels equivalent nodes recursively.
224 	 * @param node Node to label equivalents
225 	 * @param up Traversing direction. Up means towards parents, if false then towards children
226 	 * @param dist The label
227 	 * @param enqueue Whether to enqueue equivalents
228 	 * @param head Where to enqueue. Head or tail.
229 	 */
230 	protected void labelEquivRecursive(Node node, boolean up, int dist,
231 		boolean enqueue, boolean head)
232 	{
233 		if(node == null) {
234 			LOG.error("labelEquivRecursive: null (Node)");
235 			return;
236 		}
237 		
238 		for (Node equiv : up ? node.getUpperEquivalent() : node.getLowerEquivalent())
239 		{
240 			if (getColor(equiv) == WHITE)
241 			{
242 				setLabel(equiv, dist);
243 
244 				if (enqueue)
245 				{
246 					setColor(equiv, GRAY);
247 
248 					if (head) queue.addFirst(equiv);
249 					else queue.add(equiv);
250 				}
251 				else
252 				{
253 					setColor(equiv, BLACK);
254 				}
255 			}
256 
257 			labelEquivRecursive(equiv, up, dist, enqueue, head);
258 		}
259 	}

261 	/**
262 	 * Checks if an equivalent of the given node is in the set.
263 	 * @param node Node to check equivalents
264 	 * @param set Node set
265 	 * @return true if an equivalent is in the set
266 	 */
267 	protected boolean isEquivalentInTheSet(Node node, Set<Node> set)
268 	{
269 		return set.contains(node) ||
270 			isEquivalentInTheSet(node, UPWARD, set) || isEquivalentInTheSet(node, DOWNWARD, set); 
271 	}
272 
273 	/**
274 	 * Checks if an equivalent of the given node is in the set.
275 	 * @param node Node to check equivalents
276 	 * @param direction Direction to go to get equivalents
277 	 * @param set Node set
278 	 * @return true if an equivalent is in the set
279 	 */
280 	protected boolean isEquivalentInTheSet(Node node, boolean direction, Set<Node> set)
281 	{
282 		for (Node eq : direction == UPWARD ? node.getUpperEquivalent() : node.getLowerEquivalent())
283 		{
284 			if (set.contains(eq)) return true;
285 			boolean isIn = isEquivalentInTheSet(eq, direction, set);
286 			if (isIn) return true;
287 		}
288 		return false;
289 	}

291 	/**
292 	 * Gets color tag of the node
293 	 * @param node Node to get color tag
294 	 * @return color tag
295 	 */
296 	protected int getColor(Node node)
297 	{
298 		if (!colors.containsKey(node))
299 		{
300 			// Absence of color is interpreted as white
301 			return WHITE;
302 		}
303 		else
304 		{
305 			return colors.get(node);
306 		}
307 	}
308 
309 	/**
310 	 * Sets color tag
311 	 * @param node node to set color tag
312 	 * @param color the color tag
313 	 */
314 	protected void setColor(Node node, int color)
315 	{
316 		colors.put(node, color);
317 	}

319 	/**
320 	 * Gets the distance label of the object.
321 	 * @param go object to get the distance
322 	 * @return the distance label
323 	 */
324 	public int getLabel(GraphObject go)
325 	{
326 		if (!dist.containsKey(go))
327 		{
328 			// Absence of label is interpreted as infinite
329 			return Integer.MAX_VALUE-(limit*2);
330 		}
331 		else
332 		{
333 			return dist.get(go);
334 		}
335 	}

337 	/**
338 	 * Sets the distance label.
339 	 * @param go object to set the distance label
340 	 * @param label the distance label
341 	 */
342 	protected void setLabel(GraphObject go, int label)
343 	{
344 //		System.out.println("Labeling(" + label + "): " + go);
345 		dist.put(go, label);
346 	}
