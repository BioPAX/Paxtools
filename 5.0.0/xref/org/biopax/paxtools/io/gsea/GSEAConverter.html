20  /**
21   * Converts a BioPAX model to the GMT format (used by GSEA software).
22   * 
23   * It creates GSEA entries from sequence entity reference xrefs
24   * in the BioPAX model as follows:
25   *
26   *   Each entry (row) consists of three columns (tab separated):
27   * name (we use pathway URI),
28   * description (e.g. "name: Apoptosis; datasource: reactome; organism: 9606 idtype: uniprot"),
29   * and the list of identifiers (of the same type). For participants not associated with any pathway,
30   * "other" is used for the pathway name and uri.
31   *
32   *   The list may have one or more IDs of the same type per entity reference,
33   * e.g., UniProt IDs or HGNC Symbols; entity references that do not have any xref of
34   * given db/id type are ignored. Optionally, if there are less than three protein
35   * references per entry, it will not be printed.
36   * 
37   * Note, to effectively enforce cross-species violation, 
38   * 'organism' property and pathways must be set
39   * to a BioSource object that has a valid unification xref: 
40   * db="Taxonomy" and id= some valid taxonomy id.
41   *
42   * Note, this code assumes that the model has successfully been validated
43   * and perhaps normalized (using the BioPAX Validator, Paxtools Normalizer).
44   * A BioPAX L1 or L2 model is first converted to the L3.
45   */
46  public class GSEAConverter

164 	/**
165 	 * Converts model to GSEA (GMT) and writes to out.
166 	 * See class declaration for more information.
167 	 *
168 	 * @param model Model
169 	 * @param out output stream to write the result to
170 	 * @throws IOException when there's an output stream error
171 	 */
172 	public void writeToGSEA(final Model model, OutputStream out) throws IOException
173 	{
174 		Collection<GMTEntry> entries = convert(model);
175 		if (entries.size() > 0)
176 		{
177 			Writer writer = new OutputStreamWriter(out);
178 			for (GMTEntry entry : entries) {
179 				if ((minNumOfGenesPerEntry <= 1 && !entry.identifiers().isEmpty())
180 						|| entry.identifiers().size() >= minNumOfGenesPerEntry)
181 				{
182 					writer.write(entry.toString() + "\n");
183 				}
184 			}
185 			writer.flush();
186 		}
187 	}

189 	/**
190 	 * Creates GSEA entries from the pathways contained in the model.
191 	 * @param model Model
192 	 * @return a set of GSEA entries
193 	 */
194 	public Collection<GMTEntry> convert(final Model model)
195 	{
196 		final Collection<GMTEntry> toReturn = new TreeSet<GMTEntry>(new Comparator<GMTEntry>() {
197 			@Override
198 			public int compare(GMTEntry o1, GMTEntry o2) {
199 				return o1.toString().compareTo(o2.toString());
200 			}
201 		});
202 
203 		Model l3Model;
204 		// convert to level 3 in necessary
205 		if (model.getLevel() == BioPAXLevel.L2)
206 			l3Model = (new LevelUpgrader()).filter(model);
207 		else
208 			l3Model = model;
209 		
210 		//a modifiable copy of the set of all PRs in the model - 
211 		//after all, it has all the PRs that do not belong to any pathway
212 		final Set<SequenceEntityReference> sequenceEntityReferences =
213 				new HashSet<SequenceEntityReference>(l3Model.getObjects(SequenceEntityReference.class));
214 
215 		final Set<Pathway> pathways = l3Model.getObjects(Pathway.class);
216 		for (Pathway pathway : pathways) 
217 		{
218 			String name = (pathway.getDisplayName() == null) ? pathway.getStandardName() : pathway.getDisplayName();
219 			if(name == null || name.isEmpty()) 
220 				name = pathway.getUri();
221 
222 			final Pathway currentPathway = pathway;
223 			final String currentPathwayName = name;
224 
225 			final boolean ignoreSubPathways =
226 				(!skipSubPathwaysOf.isEmpty() && shareSomeObjects(currentPathway.getDataSource(), skipSubPathwaysOf))
227 					|| skipSubPathways;
228 
229 			LOG.debug("Begin converting " + currentPathwayName + " pathway, uri=" + currentPathway.getUri());
230 			final Set<SequenceEntityReference> pathwaySers = new HashSet<SequenceEntityReference>();
231 			final Traverser traverser = new AbstractTraverser(SimpleEditorMap.L3,
232 					Fetcher.nextStepFilter, Fetcher.objectPropertiesOnlyFilter) {
233 				@Override
234 				protected void visit(Object range, BioPAXElement domain, Model model, PropertyEditor editor)
235 				{
236 					//by design (objectPropertiesOnlyFilter is used), it'll visit only object properties.
237 					BioPAXElement bpe = (BioPAXElement) range;
238 					if(bpe instanceof SequenceEntityReference) {
239 						pathwaySers.add((SequenceEntityReference) bpe);
240 					}
241 					if(bpe instanceof Pathway) {
242 						if(ignoreSubPathways)
243 						{	//do not traverse into the sub-pathway; log
244 							LOG.debug("Skipping sub-pathway: " + bpe.getUri());
245 						} else {
246 							traverse(bpe, model);
247 						}
248 					} else {
249 						traverse(bpe, model);
250 					}
251 				}
252 			};
253 			//run it - collect all PRs from the pathway
254 			traverser.traverse(currentPathway, null);
255 
256 			if(!pathwaySers.isEmpty()) {
257 				LOG.debug("For pathway: " + currentPathwayName + " (" + currentPathway.getUri()
258 						+ "), got " + pathwaySers.size() + " sERs; now - grouping by organism...");
259 				Map<String,Set<SequenceEntityReference>> orgToPrsMap = organismToProteinRefsMap(pathwaySers);
260 				// create GSEA/GMT entries - one entry per organism (null organism also makes one)
261 				String dataSource = getDataSource(currentPathway.getDataSource());
262 				Collection<GMTEntry> entries = createGseaEntries(currentPathway.getUri(),
263 						currentPathwayName, dataSource, orgToPrsMap);
264 				if(!entries.isEmpty())
265 					toReturn.addAll(entries);
266 				sequenceEntityReferences.removeAll(pathwaySers);//keep not processed PRs (a PR can be processed multiple times)
267 				LOG.debug("- collected " + entries.size() + "entries.");
268 			}
269 		}
270 		
271 		//when there're no pathways, only empty pathays, pathways w/o PRs, then use all/rest of PRs -
272 		//organize PRs by species (GSEA s/w can handle only same species identifiers in a data row)
273 		if(!sequenceEntityReferences.isEmpty() && !skipOutsidePathways) {
274 			LOG.info("Creating entries for the rest of PRs (outside any pathway)...");
275 			Map<String,Set<SequenceEntityReference>> orgToPrsMap = organismToProteinRefsMap(sequenceEntityReferences);
276 			if(!orgToPrsMap.isEmpty()) {
277 				// create GSEA/GMT entries - one entry per organism (null organism also makes one) 
278 				toReturn.addAll(createGseaEntries("other",
279 						"other", getDataSource(l3Model.getObjects(Provenance.class)), orgToPrsMap));
280 			}
281 		}
282 					
283 		return toReturn;
284 	}

286 	private Collection<GMTEntry> createGseaEntries(String uri, final String name, final String dataSource,
287 												   final Map<String, Set<SequenceEntityReference>> orgToPrsMap)
288 	{
289 		// generate GSEA entries for each taxId
290 		final Collection<GMTEntry> toReturn = new ArrayList<GMTEntry>();
291 		for (final String org : orgToPrsMap.keySet()) {
292 			if(orgToPrsMap.get(org).size() > 0) {
293 				LOG.debug("adding " + idType + " IDs of " + org +
294 						" proteins (PRs) from '" + name + "', " + dataSource + " pathway...");
295 				GMTEntry GMTEntry = new GMTEntry(uri, org, idType,
296 						String.format("name: %s; datasource: %s",name, dataSource));
297 				processEntityReferences(orgToPrsMap.get(org), GMTEntry);
298 				toReturn.add(GMTEntry);
299 			}
300 		}
301 		return toReturn;
302 	}

305 	//warn: there can be many equivalent BioSource objects (same taxonomy id, different URIs)
306 	private Map<String, Set<SequenceEntityReference>> organismToProteinRefsMap(Set<SequenceEntityReference> seqErs)
307 	{
308 		Map<String,Set<SequenceEntityReference>> map = new HashMap<String, Set<SequenceEntityReference>>();
309 
310 		if(seqErs.isEmpty())
311 			throw new IllegalArgumentException("Empty set");
312 		
313 		if (crossSpeciesCheckEnabled) {
314 			for (SequenceEntityReference r : seqErs) {
315 				String key = getOrganismKey(r.getOrganism()); // null org. is ok (key == "")
316 				//collect PRs only from allowed organisms
317 				if(allowedOrganisms==null || allowedOrganisms.isEmpty() || allowedOrganisms.contains(key)) {
318 					Set<SequenceEntityReference> sers = map.get(key);
319 					if (sers == null) {
320 						sers = new HashSet<SequenceEntityReference>();
321 						map.put(key, sers);
322 					}
323 					sers.add(r);
324 				}
325 			}
326 		} else {
327 			final Set<SequenceEntityReference> sers = new HashSet<SequenceEntityReference>();
328 			for (SequenceEntityReference r : seqErs) {
329 				String key = getOrganismKey(r.getOrganism());
330 				//collect PRs only from allowed organisms
331 				if(allowedOrganisms==null || allowedOrganisms.isEmpty() || allowedOrganisms.contains(key)) {
332 					sers.add(r);
333 				}
334 			}
335 			map.put("", sers);
336 		}
337 				
338 		return map;
339 	}

342 	void processEntityReferences(Set<SequenceEntityReference> sers, GMTEntry targetEntry)
343 	{
344 		prs_loop: for (SequenceEntityReference ser : sers)
345 		{
346 			// process PRs that belong to the same species (as targetEntry's) if crossSpeciesCheckEnabled==true
347 			if (crossSpeciesCheckEnabled && !targetEntry.taxID().equals(getOrganismKey(ser.getOrganism())))
348 				continue;
349 				
350 			if (idType != null && !idType.isEmpty())
351 			{
352 				final String db = idType.toLowerCase();
353 				// a shortcut if we are converting validated normalized BioPAX model:
354 				// get the primary ID from the URI of the ProteinReference
355 				final String uri = ser.getUri();
356 				if (uri.startsWith("http://identifiers.org/") && uri.contains(db))
357 				{
358 					String accession = ser.getUri();
359 					accession = accession.substring(accession.lastIndexOf("/") + 1);
360 					targetEntry.identifiers().add(accession);
361 				}
362 				else {
363 					int added = 0;
364 					for (Xref aXref : ser.getXref())
365 					{
366 						if(aXref instanceof UnificationXref
367 								&& aXref.getId() != null && aXref.getDb() != null
368 								&& aXref.getDb().toLowerCase().startsWith(db))
369 						{
370 							targetEntry.identifiers().add(aXref.getId());
371 							++added;
372 						}
373 					}
374 					if(added == 0) for (Xref aXref : ser.getXref())
375 					{
376 						if(aXref instanceof RelationshipXref
377 								&& aXref.getId() != null && aXref.getDb() != null
378 								&& aXref.getDb().toLowerCase().startsWith(db))
379 						{
380 							targetEntry.identifiers().add(aXref.getId());
381 							++added;
382 						}
383 					}
384 
385 					if(added > 12)
386 						LOG.info("In GSEA entry: " + targetEntry.taxID() + " " + targetEntry.name() +
387 								", sER " + ser.getUri() + " got " + added + " '" + db + "' identifiers...");
388 				}
389 			} else {
390 				// fallback - use URI (not really useful for the GSEA software)
391 				targetEntry.identifiers().add(ser.getUri());
392 			}
393 		}
394 	}

396 	/*
397 	 * Gets datasource names, if any, in a consistent way/order, excl. duplicates
398 	 */
399 	private String getDataSource(Set<Provenance> provenances)
400 	{
401 		if(provenances.isEmpty()) return "N/A";
402 		
403 		Set<String> dsNames = new TreeSet<String>();
404 		for (Provenance provenance : provenances)
405 		{
406 			String name = provenance.getDisplayName();
407 			if(name == null) 
408 				name = provenance.getStandardName();
409 			if(name == null && !provenance.getName().isEmpty()) 
410 				name = provenance.getName().iterator().next();
411 			if (name != null && name.length() > 0)
412 				dsNames.add(name.toLowerCase());
413 		}
414 		
415 		return StringUtils.join(dsNames, ";");
416 	}

419 	private String getOrganismKey(BioSource org) {
420 		String key = ""; //default value: unspecified/all species
421 
422 		if (org != null) {
423 			Set<Xref> xrefs = org.getXref();
424 
425 			if(!xrefs.isEmpty()) {
426 				for (Xref xref : xrefs) {
427 					if (xref instanceof UnificationXref
428 						&& xref.getDb().equalsIgnoreCase("taxonomy")) {
429 							if(key.isEmpty())
430 								key = xref.getId();
431 							else
432 								LOG.warn("BioSource " + org + " has multiple taxonomy unification xrefs; " +
433 										"I will use " + key);
434 					}
435 				}
436 			}
437 
438 			//when there're no Taxonomy xrefs - use a name
439 			if(key.isEmpty()) {
440 				if (org.getStandardName()!=null) {
441 					key = org.getStandardName().toLowerCase();
442 				} else if(org.getDisplayName()!=null) {
443 					key = org.getDisplayName().toLowerCase();
444 				} else if(!org.getName().isEmpty()) {
445 					key = org.getName().iterator().next().toLowerCase();
446 				}
447 			}
448 		}
449 
450 		return key;
451 	}

453 	private boolean shareSomeObjects(Set<?> setA, Set<?> setB) {
454 		return (!setA.isEmpty() && !setB.isEmpty())	? !CollectionUtils.intersection(setA, setB).isEmpty() : false;
455 	}
