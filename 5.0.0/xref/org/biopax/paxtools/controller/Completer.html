1   package org.biopax.paxtools.controller;
2   
3   import org.biopax.paxtools.model.BioPAXElement;
4   import org.biopax.paxtools.model.Model;
5   import org.biopax.paxtools.model.level2.pathway;
6   import org.biopax.paxtools.model.level3.Pathway;
7   
8   import java.util.Collection;
9   import java.util.HashSet;
10  import java.util.Set;
11  
12  /**
13   * This class is used for getting a valid subgraph from a set of BioPAX elements. For instance, if a
14   * BioPAX graph contains a complex with some members, but the subgraph contains the same complex
15   * without the members, then it is not valid. Similarly, an interaction is only valid with its
16   * participants; however, including its controls is optional.
17   *
18   * @author Ozgun Babur
19   */
20  public class Completer implements Visitor
21  {
22  	protected TraverserBilinked traverser;
23  
24  	private final Set<BioPAXElement> completed;
25  
26  	private boolean skipSubPathways;
27  
28  	public Completer(EditorMap map)
29  	{
30  		this.traverser = new TraverserBilinked(map, this, new PropertyFilterBilinked()
31  		{
32  			public boolean filter(PropertyEditor editor)
33  			{
34  				return editor instanceof ObjectPropertyEditor &&
35  					((ObjectPropertyEditor) editor).isCompleteForward();
36  			}
37  
38  			public boolean filterInverse(PropertyEditor editor)
39  			{
40  				return editor instanceof ObjectPropertyEditor &&
41  					((ObjectPropertyEditor) editor).isCompleteBackward();
42  			}
43  		});
44  
45  		completed = new HashSet<BioPAXElement>();
46  		skipSubPathways = true; //makes default - skip sub-pathways
47  	}
48  
49  	/**
50  	 * @deprecated use {@link #complete(Collection)} instead (- model is never explicitly used there)
51  	 * @param elements
52  	 * @param model
53  	 * @return
54  	 */
55  	public Set<BioPAXElement> complete(Collection<BioPAXElement> elements, Model model) {
56  		return complete(elements);
57  	}
58  
59  	public Set<BioPAXElement> complete(Collection<BioPAXElement> elements)
60  	{
61  		completed.clear();
62  
63  		for (BioPAXElement element : elements)
64  		{
65  			if (completed.add(element))
66  			{
67  				traverser.traverse(element, null); //model is not required here because of 'visit' impl. below
68  			}
69  		}
70  
71  		return completed;
72  	}
73  
74  	public void visit(BioPAXElement domain, Object range, Model model, PropertyEditor editor)
75  	{
76  		if (range instanceof BioPAXElement) //!=null works too (because of prop. filters)
77  		{
78  			BioPAXElement element = (BioPAXElement) range;
79  			if (!completed.contains(element))
80  			{
81  				// Ignore sub-pathways if skipSubPathways is true
82  				//(the effect must be the same as if @AutoComplete(forward=false) annotation
83  				// would be set on properties pathwayComponent and controlled, which unfortunately we cannot do
84  				// for values of type Pathway only via the property getter annotations,
85  				// because these properties' range is Process - Interaction or Pathway...)
86  				if( !(skipSubPathways && (element instanceof Pathway || element instanceof pathway)) ) {
87  					completed.add(element);
88  					traverser.traverse(element, null);
89  				}
90  			}
91  		}
92  	}
93  
94  	/**
95  	 * Use this property to optionally
96  	 * skip (if true) traversing into sub-pathways;
97  	 * i.e., when the value of BioPAX property 'pathwayComponent' or 'controlled' is a pathway.
98  	 *
99  	 * @param skipSubPathways true/false (default is 'true' - do skip)
100 	 */
101 	public void setSkipSubPathways(boolean skipSubPathways) {
102 		this.skipSubPathways = skipSubPathways;
103 	}
104 
105 	public boolean isSkipSubPathways() {
106 		return skipSubPathways;
107 	}
108 }
