1   package org.biopax.paxtools.query.wrapperL3;
2   
3   import org.biopax.paxtools.model.BioPAXElement;
4   import org.biopax.paxtools.model.level3.*;
5   import org.biopax.paxtools.query.model.AbstractNode;
6   import org.biopax.paxtools.query.model.Edge;
7   import org.biopax.paxtools.query.model.Graph;
8   
9   /**
10   * Wrapper for TemplateReaction class.
11   *
12   * @author Ozgun Babur
13   */
14  public class TemplateReactionWrapper extends EventWrapper
15  {
16  	/**
17  	 * Wrapped TemplateReaction.
18  	 */
19  	private TemplateReaction tempReac;
20  
21  	/**
22  	 * Constructor with the TemplateReaction and the owner graph.
23  	 * @param tempReac TemplateReaction to wrap
24  	 * @param graph Owner graph
25  	 */
26  	protected TemplateReactionWrapper(TemplateReaction tempReac, GraphL3 graph)
27  	{
28  		super(graph);
29  		this.tempReac = tempReac;
30  	}
31  
32  	/**
33  	 * Binds to template and controllers.
34  	 */
35  	@Override
36  	public void initUpstream()
37  	{
38  		NucleicAcid nuc = tempReac.getTemplate();
39  		if (nuc != null)
40  		addToUpstream(nuc, getGraph());
41  
42  		for (Control cont : tempReac.getControlledOf())
43  		{
44  			addToUpstream(cont, graph);
45  		}
46  	}
47  
48  	/**
49  	 * Binds to products.
50  	 */
51  	@Override
52  	public void initDownstream()
53  	{
54  		for (PhysicalEntity pe : tempReac.getProduct())
55  		{
56  			addToDownstream(pe, getGraph());
57  		}
58  	}
59  
60  	/**
61  	 * This is transcription.
62  	 * @return True
63  	 */
64  	@Override
65  	public boolean isTranscription()
66  	{
67  		return true;
68  	}
69  
70  	/**
71  	 * Binds the given PhysicalEntity to the downstream.
72  	 * @param pe PhysicalEntity to bind
73  	 * @param graph Owner graph
74  	 */
75  	protected void addToDownstream(PhysicalEntity pe, Graph graph)
76  	{
77  		AbstractNode node = (AbstractNode) graph.getGraphObject(pe);
78  
79  		if (node != null)
80  		{
81  			EdgeL3 edge = new EdgeL3(this, node, graph);
82  			edge.setTranscription(true);
83  
84  			node.getUpstreamNoInit().add(edge);
85  			this.getDownstreamNoInit().add(edge);
86  		}
87  	}
88  
89  	/**
90  	 * Binds the given element to the upstream.
91  	 * @param ele Element to bind
92  	 * @param graph Owner graph
93  	 */
94  	protected void addToUpstream(BioPAXElement ele, org.biopax.paxtools.query.model.Graph graph)
95  	{
96  		AbstractNode node = (AbstractNode) graph.getGraphObject(ele);
97  		if (node == null) return;
98  
99  		Edge edge = new EdgeL3(node, this, graph);
100 
101 		if (isTranscription())
102 		{
103 			if (node instanceof ControlWrapper)
104 			{
105 				((ControlWrapper) node).setTranscription(true);
106 			}
107 		}
108 
109 		node.getDownstreamNoInit().add(edge);
110 		this.getUpstreamNoInit().add(edge);
111 	}
112 
113 	/**
114 	 * Uses RDF ID of TemplateReaction as key.
115 	 * @return RDF ID of TemplateReaction
116 	 */
117 	public String getKey()
118 	{
119 		return tempReac.getUri();
120 	}
121 
122 	/**
123 	 * Gets the wrapped TemplateReaction
124 	 * @return The wrapped TemplateReaction
125 	 */
126 	public TemplateReaction getTempReac()
127 	{
128 		return tempReac;
129 	}
130 }
