3   import org.biopax.paxtools.model.BioPAXElement;
4   import org.biopax.paxtools.pattern.Constraint;
5   import org.biopax.paxtools.pattern.MappedConst;
6   import org.biopax.paxtools.pattern.Match;

8   import java.util.Collection;
9   import java.util.HashSet;
10  import java.util.Set;
11  
12  /**
13   * Used for using several constraints as a chain. The generated elements from a constraint is used
14   * as input by the next constraint in the chain. All constraints except the first one has to be
15   * size 2.
16   *
17   * @author Ozgun Babur
18   */
19  public class ConstraintChain extends ConstraintAdapter
20  {
21  	/**
22  	 * Constraint to be chained.
23  	 */
24  	Constraint[] con;
25  
26  	/**
27  	 * Constructor with the chained constraints. All constraints has to be generative, and all
28  	 * constraints, except the first one, has to be size 2. The input to these constraints is the
29  	 * element that is generated by the previous constraint in the chain.
30  	 * @param con mapped constraints
31  	 */
32  	public ConstraintChain(Constraint... con)
33  	{
34  		if (con.length < 2) throw new IllegalArgumentException(
35  			"Parameter constraint has to be more than one.");
36  
37  		this.con = con;
38  
39  		for (int i = 1; i < con.length; i++)
40  		{
41  			if (con[i].getVariableSize() > 2) throw new IllegalArgumentException("All chained " +
42  				"constraints (except first) has to have variable size 2. Unlike con[" + i + "]: "
43  				+ con[i]);
44  		}
45  
46  		setSize(con[0].getVariableSize());
47  	}
48  
49  	/**
50  	 * Chains the member constraints and returns the output of the last constraint.
51  	 * @param match match to process
52  	 * @param ind mapped indices
53  	 * @return satisfying elements
54  	 */
55  	@Override
56  	public Collection<BioPAXElement> generate(Match match, int... ind)
57  	{
58  
59  		Collection<BioPAXElement> gen = new HashSet<BioPAXElement> (
60  			con[0].generate(match, ind));
61  
62  		Set<BioPAXElement> input = new HashSet<BioPAXElement>();
63  		Set<BioPAXElement> output = new HashSet<BioPAXElement>(gen);
64  		int[] tempInd = {0, 1};
65  
66  		for (int i = 1; i < con.length; i++)
67  		{
68  			if (output.isEmpty()) break;
69  			input.clear();
70  			input.addAll(output);
71  			output.clear();
72  
73  			for (BioPAXElement ele : input)
74  			{
75  				Match m = new Match(2);
76  				m.set(ele, 0);
77  				output.addAll(con[i].generate(m, tempInd));
78  			}
79  		}
80  
81  		return output;
82  	}
83  }
