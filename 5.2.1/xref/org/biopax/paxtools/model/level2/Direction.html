6   public enum Direction

12  	IRREVERSIBLE_RIGHT_TO_LEFT;
13  
14  	public String toString()
15  	{
16  		if (this == REVERSIBLE) return "REVERSIBLE";
17  		else if (this == PHYSIOL_LEFT_TO_RIGHT) return "PHYSIOL_LEFT_TO_RIGHT";
18  		else if (this == PHYSIOL_RIGHT_TO_LEFT) return "PHYSIOL_RIGHT_TO_LEFT";
19  		else if (this == IRREVERSIBLE_LEFT_TO_RIGHT) return "IRREVERSIBLE_LEFT_TO_RIGHT";
20  		else if (this == IRREVERSIBLE_RIGHT_TO_LEFT) return "IRREVERSIBLE_RIGHT_TO_LEFT";
21  		else return null;
22  	}
23  }
