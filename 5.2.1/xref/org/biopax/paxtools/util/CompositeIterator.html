3   import java.util.Collection;
4   import java.util.Iterator;
5   import java.util.NoSuchElementException;
6   
7   /**
8    * A composite iterator that iterates over multiple iterators.
9    * @param <T> Base class type which all subiterator's type must extend from.
10   */
11  public class CompositeIterator<T> implements Iterator<T>
12  {
13      Iterator<? extends Collection<? extends T>> collectionIterator;
14      Iterator<? extends T> currentIterator;
15  
16  	/**
17  	 * This constructor creates an iterator instance from a set of collections
18  	 * @param collections to be iterated over.
19  	 */
20  	public CompositeIterator(Collection<? extends Collection<? extends T>> collections)
21      {
22          collectionIterator = collections.iterator();
23          currentIterator = getNextSet();
24  
25      }
26  
27      @Override
28      public boolean hasNext() {
29          if (currentIterator == null) {
30              return false;
31          } else if (!currentIterator.hasNext())
32          {
33              currentIterator = getNextSet();
34              return this.hasNext();
35          }
36          else return true;
37      }
38  
39      @Override
40      public T next() {
41          if (this.hasNext()) {
42              return currentIterator.next();
43          } else {
44              throw new NoSuchElementException();
45          }
46      }
47  
48      @Override
49      public void remove() {
50          throw new UnsupportedOperationException();
51      }
52  
53      private Iterator<? extends T> getNextSet() {
54          return collectionIterator.hasNext() ? collectionIterator.next().iterator() : null;
55      }
56  
57  }
