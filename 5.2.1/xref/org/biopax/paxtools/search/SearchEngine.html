173 	public SearchResult search(String query, int page,
174 			Class<? extends BioPAXElement> filterByType, String[] datasources,
175 			String[] organisms) 
176 	{
177 		SearchResult response = null;
178 		
179 		LOG.debug("search: " + query + ", page: " + page 
180 			+ ", filterBy: " + filterByType
181 			+ "; extra filters: ds in (" + Arrays.toString(datasources)
182 			+ "), org. in (" + Arrays.toString(organisms) + ")");
183 		
184 		IndexSearcher searcher = null;
185 	
186 		try {	
187 			QueryParser queryParser = new MultiFieldQueryParser(DEFAULT_FIELDS, analyzer);
188 			queryParser.setAllowLeadingWildcard(true);//TODO do we really want leading wildcards (e.g. *sulin)?
189 			
190 			searcher = searcherManager.acquire();	
191 			
192 			//find and transform top docs to search hits (beans), considering pagination...
193 			if(!query.trim().equals("*")) { //if not "*" query, which is not supported out-of-the-box, then
194 				//create the lucene query
195 				Query luceneQuery = queryParser.parse(query);
196 //do NOT (Lucene 4.1), or scoring/highlighting won't work for wildcard queries...				
197 //luceneQuery = searcher.rewrite(luceneQuery); 
198 				LOG.debug("parsed lucene query is " + luceneQuery.getClass().getSimpleName());
199 				
200 				//create filter: type AND (d OR d...) AND (o OR o...)
201 				Filter filter = createFilter(filterByType, datasources, organisms);
202 				
203 				//get the first page of top hits
204 				TopDocs topDocs = searcher.search(luceneQuery, filter, maxHitsPerPage);
205 				//get the required hits page if page>0
206 				if(page>0) {
207 					TopScoreDocCollector collector = TopScoreDocCollector.create(maxHitsPerPage*(page+1), true);  
208 					searcher.search(luceneQuery, filter, collector);
209 					topDocs = collector.topDocs(page * maxHitsPerPage, maxHitsPerPage);
210 				}
211 				
212 				//transform docs to hits, use a highlighter to get excerpts
213 				response = transform(luceneQuery, searcher, true, topDocs);
214 	
215 			} else { //find ALL objects of a particular BioPAX class (+ filters by organism, datasource)
216 				if(filterByType==null) 
217 					filterByType = Level3Element.class;
218 
219 				//replace q="*" with a search for the class or its sub-class name in the TYPE field
220 				BooleanQuery luceneQuery = new BooleanQuery();
221 				for(Class<? extends BioPAXElement> subType : SimpleEditorMap.L3.getKnownSubClassesOf(filterByType)) {
222 					luceneQuery.add(new TermQuery(new Term(FIELD_TYPE, subType.getSimpleName().toLowerCase())), Occur.SHOULD);
223 				}
224 				Filter filter = createFilter(null, datasources, organisms);
225 				
226 				//get the first page of top hits
227 				TopDocs topDocs = searcher.search(luceneQuery, filter, maxHitsPerPage);
228 				//get the required hits page if page>0
229 				if(page>0) {
230 					TopScoreDocCollector collector = TopScoreDocCollector.create(maxHitsPerPage*(page+1), true);  
231 					searcher.search(luceneQuery, filter, collector);
232 					topDocs = collector.topDocs(page * maxHitsPerPage, maxHitsPerPage);	
233 				}
234 				
235 				//convert
236 				response = transform(luceneQuery, searcher, false, topDocs);			
237 			}	
238 			
239 		} catch (ParseException e) {
240 			throw new RuntimeException("getTopDocs: failed to parse the query string.", e);
241 		} catch (IOException e) {
242 			throw new RuntimeException("getTopDocs: failed.", e);
243 		} finally {
244 			try {
245 				if(searcher!=null) {
246 					searcherManager.release(searcher);
247 					searcher = null;
248 				}
249 			} catch (IOException e) {}	
250 		}
251 		
252 		response.setPage(page);
253 		
254 		return response;
255 	}

258 	/**
259 	 * Returns a SearchResult
260 	 * that contains a List<BioPAXElement>,
261 	 * some parameters, totals, etc.
262 	 */
263 	private SearchResult transform(Query query, IndexSearcher searcher, boolean highlight, TopDocs topDocs) 
264 			throws CorruptIndexException, IOException 
265 	{	
266 		final SearchResultult.html#SearchResult">SearchResult response = new SearchResult();
267 		final List<BioPAXElement> hits = new ArrayList<BioPAXElement>();
268 		
269 		response.setMaxHitsPerPage(maxHitsPerPage);
270 		response.setHits(hits);
271 		
272 		for(ScoreDoc scoreDoc : topDocs.scoreDocs) {			
273 			Document doc = searcher.doc(scoreDoc.doc);
274 			String uri = doc.get(FIELD_URI);
275 			BioPAXElement bpe = model.getByID(uri);			
276 			LOG.debug("transform: doc:" + scoreDoc.doc + ", uri:" + uri);
277 			
278 			// use the highlighter (get matching fragments)
279 			// for this to work, all keywords were stored in the index field
280 			if (highlight && doc.get(FIELD_KEYWORD) != null) {				
281 				// use a Highlighter (store.YES must be enabled for 'keyword' field)
282 				QueryScorer scorer = new QueryScorer(query, FIELD_KEYWORD); 
283 				//this fixes scoring/highlighting for all-field wildcard queries like q=insulin* 
284 				//but not for term/prefix queries, i.e, q=name:insulin*, q=pathway:brca2. TODO
285 				scorer.setExpandMultiTermQuery(true);	
286 				
287 				//TODO try PostingsHighlighter once it's stable...
288 				// (see http://lucene.apache.org/core/4_10_0/highlighter/org/apache/lucene/search/postingshighlight/PostingsHighlighter.html)
289 				SimpleHTMLFormatter formatter = new SimpleHTMLFormatter("<span class='hitHL'>", "</span>");
290 				Highlighter highlighter = new Highlighter(formatter, scorer);
291 				highlighter.setTextFragmenter(new SimpleSpanFragmenter(scorer, 80));
292 										
293 				final String text = StringUtils.join(doc.getValues(FIELD_KEYWORD), " ");
294 				try {
295 					TokenStream tokenStream = analyzer.tokenStream("", new StringReader(text));
296 					String res = highlighter.getBestFragments(tokenStream, text, 7, "...");
297 
298 					if(res != null && !res.isEmpty()) {
299 						bpe.getAnnotations().put(HitAnnotation.HIT_EXCERPT.name(), res);
300 					}
301 
302 				} catch (Exception e) {throw new RuntimeException(e);}
303 
304 			} else if(highlight) {
305 				LOG.warn("Highlighter skipped, because KEYWORD field was null; hit: " 
306 						+ uri + ", " + bpe.getModelInterface().getSimpleName());
307 			}
308 
309 			// extract organisms (URI only) if not done before
310 			if(doc.get(FIELD_ORGANISM) != null && !bpe.getAnnotations().containsKey(HitAnnotation.HIT_ORGANISM.name())) {
311 				Set<String> uniqueVals = new TreeSet<String>();
312 				for(String o : doc.getValues(FIELD_ORGANISM)) {
313 					//note: only URIS are stored in the index					
314 					uniqueVals.add(o);
315 				}
316 				bpe.getAnnotations().put(HitAnnotation.HIT_ORGANISM.name(), uniqueVals);
317 			}
318 			
319 			// extract values form the index if not previously done
320 			if(doc.get(FIELD_DATASOURCE) != null && !bpe.getAnnotations().containsKey(HitAnnotation.HIT_DATASOURCE.name())) {
321 				Set<String> uniqueVals = new TreeSet<String>();
322 				for(String d : doc.getValues(FIELD_DATASOURCE)) {
323 					//note: only URIS are stored in the index
324 					uniqueVals.add(d);
325 				}
326 				bpe.getAnnotations().put(HitAnnotation.HIT_DATASOURCE.name(), uniqueVals);
327 			}	
328 			
329 			// extract only pathway URIs if not previously done
330 			//(because names and IDs used to be stored in the index field as well)
331 			if(doc.get(FIELD_PATHWAY) != null && !bpe.getAnnotations().containsKey(HitAnnotation.HIT_PATHWAY.name())) {
332 				Set<String> uniqueVals = new TreeSet<String>();
333 				for(String d : doc.getValues(FIELD_PATHWAY)) {
334 					//only URIs were stored (though all names/ids were indexed/analyzed)
335 					if(!d.equals(uri)) //exclude itself
336 						uniqueVals.add(d);
337 				}
338 				bpe.getAnnotations().put(HitAnnotation.HIT_PATHWAY.name(), uniqueVals);
339 			}
340 			
341 			//store the no. processes in the sub-network if not previously done
342 			if(doc.get(FIELD_N_PARTICIPANTS)!=null && !bpe.getAnnotations().containsKey(HitAnnotation.HIT_PARTICIPANTS.name()))
343 				bpe.getAnnotations().put(HitAnnotation.HIT_PARTICIPANTS.name(), Integer.valueOf(doc.get(FIELD_N_PARTICIPANTS)));
344 			if(doc.get(FIELD_N_PROCESSES)!=null && !bpe.getAnnotations().containsKey(HitAnnotation.HIT_PROCESSES.name()))
345 				bpe.getAnnotations().put(HitAnnotation.HIT_PROCESSES.name(), Integer.valueOf(doc.get(FIELD_N_PROCESSES)));
346 
347 			//store the Lucene's score and explanation.
348 			String excerpt = (String) bpe.getAnnotations().get(HitAnnotation.HIT_EXCERPT.name());
349 			if(excerpt == null) excerpt = "";
350 			excerpt += " -SCORE- " + scoreDoc.score + " -EXPLANATION- " + searcher.explain(query, scoreDoc.doc);
351 			bpe.getAnnotations().put(HitAnnotation.HIT_EXCERPT.name(), excerpt);
352 
353 			hits.add(bpe);
354 		}
355 						
356 		//set total no. hits	
357 		response.setTotalHits(topDocs.totalHits);
358 		
359 		return response;
360 	}

363 	public void index() {
364 		final int numObjects =  model.getObjects().size();
365 		LOG.info("index(), there are " + numObjects + " BioPAX objects to be (re-)indexed.");		
366 		IndexWriter iw;		
367 		try {
368 			//close the searcher manager if the old index exists
369 			if(searcherManager != null) {
370 				searcherManager.close();
371 				searcherManager = null;
372 			}
373 			IndexWriterConfig conf = new IndexWriterConfig(Version.LATEST, analyzer);
374 			iw = new IndexWriter(FSDirectory.open(indexFile), conf);
375 			//cleanup
376 			iw.deleteAll();
377 			iw.commit();
378 		} catch (IOException e) {
379 			throw new RuntimeException("Failed to create a new IndexWriter.", e);
380 		}		
381 		final IndexWriter indexWriter = iw;
382 
383 		ExecutorService exec = Executors.newFixedThreadPool(30);
384 		
385 		final AtomicInteger numLeft = new AtomicInteger(numObjects);
386 
387 		final org.biopax.paxtools.util.Filter<DataPropertyEditor> dataPropertiesToConsider
388 				= new org.biopax.paxtools.util.Filter<DataPropertyEditor>() {
389 			@Override
390 			public boolean filter(DataPropertyEditor editor) {
391 				final String prop = editor.getProperty();
392 				//to include in the index, as keywords, only the following properties
393 				// (basically, to exclude float type properties, embedded xml, db names, etc.):
394 				return (prop.equalsIgnoreCase("author") || prop.equalsIgnoreCase("availability")
395 						|| prop.equalsIgnoreCase("chemicalFormula") || prop.equalsIgnoreCase("comment")
396 						|| prop.equalsIgnoreCase("controlType") || prop.equalsIgnoreCase("conversionDirection")
397 						|| prop.equalsIgnoreCase("eCNumber") || prop.equalsIgnoreCase("id")
398 						|| prop.equalsIgnoreCase("name") || prop.equalsIgnoreCase("displayName")
399 						|| prop.equalsIgnoreCase("standardName") || prop.equalsIgnoreCase("sequence")
400 						|| prop.equalsIgnoreCase("source") || prop.equalsIgnoreCase("year")
401 						|| prop.equalsIgnoreCase("term") || prop.equalsIgnoreCase("stepDirection")
402 						|| prop.equalsIgnoreCase("structureData") || prop.equalsIgnoreCase("templateDirection")
403 						|| prop.equalsIgnoreCase("title") || prop.equalsIgnoreCase("url")
404 				);
405 			}
406 		};
407 
408 		final Fetcher/Fetcher.html#Fetcher">Fetcher fetcher = new Fetcher(SimpleEditorMap.L3, Fetcher.nextStepFilter);
409 		fetcher.setSkipSubPathways(true);
410 
411 		for(final BioPAXElement bpe : model.getObjects()) {	
412 			// prepare & index each element in a separate thread
413 			exec.execute(new Runnable() {
414 				public void run() {					
415 					// get or infer some important values if possible from this, child or parent objects:
416 					Set<String> keywords = ModelUtils.getKeywords(bpe, 3, dataPropertiesToConsider);
417 					// a hack to remove special (debugging) biopax comments
418 					for(String s : new HashSet<String>(keywords)) {
419 						//exclude additional comments generated by normalizer, merger, etc.
420 						if(s.startsWith("REPLACED ") || s.contains("ADDED"))
421 							keywords.remove(s);
422 					}
423 					
424 					bpe.getAnnotations().put(FIELD_KEYWORD, keywords);
425 					bpe.getAnnotations().put(FIELD_DATASOURCE, ModelUtils.getDatasources(bpe));
426 					bpe.getAnnotations().put(FIELD_ORGANISM, ModelUtils.getOrganisms(bpe));
427 					bpe.getAnnotations().put(FIELD_PATHWAY, ModelUtils.getParentPathways(bpe));
428 					//- includes itself if bpe is a pathway
429 
430 					//set <numparticipants> (PEs/Genes), <numprocesses> (interactions/pathways) index fields:
431 					if(bpe instanceof org.biopax.paxtools.model.level3.Process) {
432 						int numProc = fetcher.fetch(bpe, Process.class).size(); //except itself
433 						int numPeAndG = fetcher.fetch(bpe, PhysicalEntity.class).size()
434 								+ fetcher.fetch(bpe, Gene.class).size();
435 						bpe.getAnnotations().put(FIELD_N_PARTICIPANTS, Integer.toString(numPeAndG));
436 						bpe.getAnnotations().put(FIELD_N_PROCESSES, Integer.toString(numProc));
437 					} else if(bpe instanceof Complex) {
438 						int numPEs = fetcher.fetch(bpe, PhysicalEntity.class).size();
439 						bpe.getAnnotations().put(FIELD_N_PARTICIPANTS, Integer.toString(numPEs));
440 					}
441 
442 					//add IDs of uni. and rel. xrefs of this element and its children
443 					if(bpe instanceof Entity || bpe instanceof EntityReference)
444 					{
445 						final Set<String> ids = new HashSet<String>();
446 
447 						//fetch all children of (implicit) type XReferrable, which means - either
448 						//BioSource or ControlledVocabulary or Evidence or Provenance or Entity or EntityReference
449 						//(we actually want only the latter two types and their sub-types; will skip the rest later on):
450 						Set<XReferrable> children = fetcher.fetch(bpe, XReferrable.class);
451 						//include itself (- for fetcher only gets child elements)
452 						children.add((XReferrable) bpe);
453 						for(XReferrable child : children) {
454 							//skip IDs of Evidence,CV,Provenance (included into 'keyword' field anyway)
455 							if(!(child instanceof Entity || child instanceof EntityReference))
456 								continue;
457 							// collect standard bio IDs (skip publications); try/use id-mapping to associate more IDs:
458 							for(Xref x : child.getXref()) {
459 								if (!(x instanceof PublicationXref) && x.getId()!=null && x.getDb()!=null) {
460 									ids.add(x.getId());
461 								}
462 							}
463 						}
464 						if(!ids.isEmpty()) {
465 							bpe.getAnnotations().put(SearchEngine.FIELD_XREFID, ids);
466 						}
467 					}
468 
469 					index(bpe, indexWriter);
470 					
471 					//count, log a progress message
472 					int left = numLeft.decrementAndGet();
473 					if(left % 10000 == 0)
474 						LOG.info("index(), biopax objects left to index: " + left);
475 				}
476 			});
477 		}
478 		
479 		exec.shutdown(); //stop accepting new tasks	
480 		try { //wait
481 			exec.awaitTermination(Long.MAX_VALUE, TimeUnit.SECONDS);
482 		} catch (InterruptedException e) {
483 			throw new RuntimeException("Interrupted!", e);
484 		}
485 		
486 		try {
487 			indexWriter.close(); //wait for pending op., auto-commit, close.
488 		} catch (IOException e) {
489 			throw new RuntimeException("Failed to close IndexWriter.", e);
490 		} 
491 		
492 		//finally, create a new searcher manager
493 		initSearcherManager();
494 	}

499 	/**
500 	 * Creates a new Lucene Document that corresponds to a BioPAX object.
501 	 * It does not check whether the document exists (should not be there,
502 	 * because the {@link #index()} method cleans up the index)
503 	 * 
504 	 * Some fields also include biopax data type property values not only from 
505 	 * the biopax object but also from its child elements, up to some depth 
506 	 * (using key-value pairs in the pre-computed bpe.annotations map):
507 	 * 
508 	 *  'uri' - biopax object's absolute URI, analyze=no, store=yes;
509 	 * 
510 	 *  'name' - names, analyze=yes, store=yes; boosted;
511 	 * 
512 	 *  'keyword' - infer from this bpe and its child objects' data properties,
513 	 *            such as Score.value, structureData, structureFormat, chemicalFormula, 
514 	 *            availability, term, comment, patoData, author, source, title, url, published, 
515 	 *            up to given depth/level; and also all 'pathway' field values are included here; 
516 	 *            analyze=yes, store=yes;
517 	 *  
518 	 *  'datasource', 'organism' and 'pathway' - infer from this bpe and its child objects 
519 	 *  									  up to given depth/level, analyze=no, store=yes;
520 	 *  
521 	 *  'size' - number of child processes, an integer as string; analyze=no, store=yes
522 	 * 
523 	 * @param bpe BioPAX object
524 	 * @param indexWriter index writer
525 	*/
526 	void index(BioPAXElement bpe, IndexWriter indexWriter) {		
527 		// create a new document
528 		final Document doc = new Document();
529 		
530 		// save URI (not indexed field)
531 		Field field = new StoredField(FIELD_URI, bpe.getUri());
532 		doc.add(field);
533 		
534 		// index and store but not analyze/tokenize the biopax class name:
535 		field = new StringField(FIELD_TYPE, bpe.getModelInterface().getSimpleName().toLowerCase(), Field.Store.YES);
536 		doc.add(field);
537 		
538 		// make index fields from the annotations map (of pre-calculated/inferred values)
539 		if(!bpe.getAnnotations().isEmpty()) {
540 			if(bpe.getAnnotations().containsKey(FIELD_PATHWAY)) {
541 				addPathways((Set<Pathway>)bpe.getAnnotations().get(FIELD_PATHWAY), doc);
542 			}
543 			if(bpe.getAnnotations().containsKey(FIELD_ORGANISM)) {
544 				addOrganisms((Set<BioSource>)bpe.getAnnotations().get(FIELD_ORGANISM), doc);
545 			}
546 			if(bpe.getAnnotations().containsKey(FIELD_DATASOURCE)) {
547 				addDatasources((Set<Provenance>)bpe.getAnnotations().get(FIELD_DATASOURCE), doc);
548 			}
549 			if(bpe.getAnnotations().containsKey(FIELD_KEYWORD)) {
550 				addKeywords((Set<String>)bpe.getAnnotations().get(FIELD_KEYWORD), doc);
551 			}
552 			if(bpe.getAnnotations().containsKey(FIELD_N_PARTICIPANTS)) {
553 				field = new StoredField(FIELD_N_PARTICIPANTS,
554 						Integer.parseInt((String)bpe.getAnnotations().get(FIELD_N_PARTICIPANTS)));
555 				doc.add(field);
556 			}
557 			if(bpe.getAnnotations().containsKey(FIELD_N_PROCESSES)) {
558 				field = new IntField(FIELD_N,
559 						Integer.parseInt((String)bpe.getAnnotations().get(FIELD_N_PROCESSES)), Field.Store.NO);
560 				doc.add(field);
561 				field = new StoredField(FIELD_N_PROCESSES,
562 						Integer.parseInt((String)bpe.getAnnotations().get(FIELD_N_PROCESSES)));
563 				doc.add(field);
564 			}
565 			if(bpe.getAnnotations().containsKey(FIELD_XREFID)) {
566 				//index biological IDs as keywords
567 				addKeywords((Set<String>)bpe.getAnnotations().get(FIELD_XREFID), doc);
568 
569 				//index all IDs using "xrefid" fields
570 				for (String id : (Set<String>)bpe.getAnnotations().get(FIELD_XREFID)) {
571 					Field f = new StringField(FIELD_XREFID, id.toLowerCase(), Field.Store.NO);
572 					doc.add(f);
573 				}
574 			}
575 		}
576 		bpe.getAnnotations().remove(FIELD_KEYWORD);
577 		bpe.getAnnotations().remove(FIELD_DATASOURCE);
578 		bpe.getAnnotations().remove(FIELD_ORGANISM);
579 		bpe.getAnnotations().remove(FIELD_PATHWAY);
580 		bpe.getAnnotations().remove(FIELD_N_PARTICIPANTS);
581 		bpe.getAnnotations().remove(FIELD_N_PROCESSES);
582 		bpe.getAnnotations().remove(FIELD_XREFID);
583 
584 		// name
585 		if(bpe instanceof Named) {
586 			Named named = (Named) bpe;
587 			if(named.getStandardName() != null) {
588 				field = new TextField(FIELD_NAME, named.getStandardName(), Field.Store.NO);
589 				field.setBoost(3.5f);
590 				doc.add(field);
591 			}
592 			if(named.getDisplayName() != null && !named.getDisplayName().equalsIgnoreCase(named.getStandardName())) {
593 				field = new TextField(FIELD_NAME, named.getDisplayName(), Field.Store.NO);
594 				field.setBoost(3.0f);
595 				doc.add(field);
596 			}
597 			for(String name : named.getName()) {
598 				if(name.equalsIgnoreCase(named.getDisplayName()) || name.equalsIgnoreCase(named.getStandardName()))
599 					continue;
600 				field = new TextField(FIELD_NAME, name.toLowerCase(), Field.Store.NO);
601 				field.setBoost(2.5f);
602 				doc.add(field);
603 			}
604 		}
605 
606 		// write
607 		try {
608 			indexWriter.addDocument(doc);
609 		} catch (IOException e) {
610 			throw new RuntimeException("Failed to index; " + bpe.getUri(), e);
611 		}
612 	}

614 	private void addKeywords(Set<String> keywords, Document doc) {
615 		for (String keyword : keywords) {
616 			Field f = new TextField(FIELD_KEYWORD, keyword.toLowerCase(), Field.Store.YES);
617 			doc.add(f);
618 		}
619 	}

621 	private void addDatasources(Set<Provenance> set, Document doc) {
622 		for (Provenance p : set) {
623 			// Index and store URI (untokinized) - 
624 			// required to accurately calculate no. entities or to filter by data source (diff. datasources may share same names)
625 			doc.add(new StringField(FIELD_DATASOURCE, p.getUri(), Field.Store.YES));
626 			// index names as well
627 			for (String s : p.getName())
628 				doc.add(new TextField(FIELD_DATASOURCE, s.toLowerCase(), Field.Store.NO));
629 		}
630 	}

632 	private void addOrganisms(Set<BioSource> set, Document doc) {	
633 		for(BioSource bs : set) {
634 			// store URI as is (not indexed, untokinized)
635 			doc.add(new StoredField(FIELD_ORGANISM, bs.getUri()));
636 				
637 			// add organism names
638 			for(String s : bs.getName()) {
639 				doc.add(new TextField(FIELD_ORGANISM, s.toLowerCase(), Field.Store.NO));
640 			}
641 			// add taxonomy
642 			for(UnificationXref x : 
643 				new ClassFilterSet<Xref,UnificationXref>(bs.getXref(), UnificationXref.class)) {
644 				if(x.getId() != null)
645 					doc.add(new TextField(FIELD_ORGANISM, x.getId().toLowerCase(), Field.Store.NO));
646 			}
647 			// include tissue type terms
648 			if (bs.getTissue() != null) {
649 				for (String s : bs.getTissue().getTerm())
650 					doc.add(new TextField(FIELD_ORGANISM, s.toLowerCase(), Field.Store.NO));
651 			}
652 			// include cell type terms
653 			if (bs.getCellType() != null) {
654 				for (String s : bs.getCellType().getTerm()) {
655 					doc.add(new TextField(FIELD_ORGANISM, s.toLowerCase(), Field.Store.NO));
656 				}
657 			}
658 		}
659 	}

661 	private void addPathways(Set<Pathway> set, Document doc) {
662 		for(Pathway pw : set) {
663 			//add URI as is (do not lowercase; do not index; store=yes - required to report hits, e.g., as xml)
664 			doc.add(new StoredField(FIELD_PATHWAY, pw.getUri()));
665 			
666 			// add names to the 'pathway' (don't store) and 'keywords' (store, don't index) fields
667 			for (String s : pw.getName()) {
668 				doc.add(new TextField(FIELD_PATHWAY, s.toLowerCase(), Field.Store.NO));
669 				doc.add(new StoredField(FIELD_KEYWORD, s.toLowerCase()));//for highlighting only, not indexed
670 			}
671 			
672 			// add unification xref IDs too
673 			for (UnificationXref x : new ClassFilterSet<Xref, UnificationXref>(
674 					pw.getXref(), UnificationXref.class)) {
675 				if (x.getId() != null) {
676 					// index in both 'pathway' (don't store) and 'keywords' (store, don't index)
677 					doc.add(new TextField(FIELD_PATHWAY, x.getId().toLowerCase(), Field.Store.NO));
678 					doc.add(new StoredField(FIELD_KEYWORD, x.getId().toLowerCase()));//for highlighting only, not indexed
679 				}
680 			}
681 		}
682 	}

684 	/** 
685 	 * Creates a search filter like 
686 	 * type AND (datasource OR datasource...) 
687 	 *      AND (organism OR organism OR...)
688 	 * 
689 	 * Both names (partial or full) and URIs should work as filter values. 
690 	 * 
691 	 * @param type
692 	 * @param datasources
693 	 * @param organisms
694 	 */	
695 	private Filter createFilter(Class<? extends BioPAXElement> type, 
696 			String[] datasources, String[] organisms) {
697 		
698 		BooleanQuery filterQuery = new BooleanQuery();
699 		
700 		//AND datasources	
701 		if (datasources != null && datasources.length > 0) {
702 			filterQuery.add(subQuery(datasources, FIELD_DATASOURCE), Occur.MUST);
703 		}
704 		//AND organisms
705 		if (organisms != null && organisms.length > 0) {
706 			filterQuery.add(subQuery(organisms, FIELD_ORGANISM), Occur.MUST);
707 		}		
708 		//AND type	
709 		if(type != null) { //add biopax class filter
710 			BooleanQuery query = new BooleanQuery();
711 			query.add(new TermQuery(new Term(FIELD_TYPE, type.getSimpleName().toLowerCase())), Occur.SHOULD);//OR
712 			//for each biopax subclass (interface), add the name to the filter query
713 			for(Class<? extends BioPAXElement> subType : SimpleEditorMap.L3.getKnownSubClassesOf(type)) {
714 				query.add(new TermQuery(new Term(FIELD_TYPE, subType.getSimpleName().toLowerCase())), Occur.SHOULD);//OR
715 			}		
716 			filterQuery.add(query, Occur.MUST);
717 		}
718 		
719 		if(!filterQuery.clauses().isEmpty()) {
720 			LOG.debug("filterQuery: " + filterQuery.toString());
721 			return new CachingWrapperFilter( new QueryWrapperFilter(filterQuery) ); //TODO why CachingWrapperFilter, QueryWrapperFilter?
722 		} else 
723 			return null;
724 	}

726 	/**
727 	 * Filter values here are joint with 'OR' operator, 
728 	 * but if a value has internal whitespace symbols, this also makes a sub-query,
729 	 * in which terms are joint with 'AND'. This is to allow filtering
730 	 * by datasource/organism's full name, partial name, or uri 
731 	 * and allowing multiple datasources/organisms.
732 	 * 
733 	 * @param filterValues
734 	 * @param filterField
735 	 * @return
736 	 */
737 	private Query subQuery(String[] filterValues, String filterField) {
738 		BooleanQuery query = new BooleanQuery();	
739 		final Pattern pattern = Pattern.compile("\\s");		
740 		for(String v : filterValues) {
741 			//if v has whitespace chars (several words), make a "word1 AND word2 AND..." subquery
742 			if(pattern.matcher(v).find()) {
743 				BooleanQuery bq = new BooleanQuery();
744 //was bug: text with spaces and 'of', 'for', 'and', etc., did not match anything (we have to use the same analyzer as during indexing!)
745 //				for(String w : v.split("\\s+")) {
746 //					bq.add(new TermQuery(new Term(filterField, w.toLowerCase())), Occur.MUST);
747 //					LOG.debug("subQuery, add part: " + w.toLowerCase());
748 //				}
749 				try {
750 					//use the same analyser as when indexing
751 					TokenStream tokenStream = analyzer.tokenStream(filterField, new StringReader(v));
752 					CharTermAttribute chattr = tokenStream.addAttribute(CharTermAttribute.class);
753 					tokenStream.reset();
754 					while(tokenStream.incrementToken()) {
755 						//'of', 'and', 'for',.. never occur as tokens (this is how the std. analyzer works)
756 						String token = chattr.toString();
757 						bq.add(new TermQuery(new Term(filterField, token)), Occur.MUST);
758 					}
759 					tokenStream.end(); 
760 					tokenStream.close();
761 				} catch (IOException e) {
762 					//should never happen as we use StringReader
763 					throw new RuntimeException("Failed to open a token stream; "
764 							+ "field:" + filterField + ", value:" + v,e);
765 				}
766 				query.add(bq, Occur.SHOULD);
767 			} else {
768 				query.add(new TermQuery(new Term(filterField, v.toLowerCase())), Occur.SHOULD);
769 			}			
770 		}
771 		
772 		return query;
773 	}
