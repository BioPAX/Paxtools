3   import org.biopax.paxtools.model.BioPAXElement;
4   import org.biopax.paxtools.model.level3.EntityReference;
5   import org.biopax.paxtools.model.level3.PhysicalEntity;

7   import org.biopax.paxtools.util.BPCollections;
8   import org.slf4j.Logger;
9   import org.slf4j.LoggerFactory;
10  
11  import java.util.Set;
12  
13  
14  public abstract class SimplePhysicalEntityImpl extends PhysicalEntityImpl
15  		implements SimplePhysicalEntity
16  {
17  	private EntityReference entityReference;
18  
19    	private static final Logger log = LoggerFactory.getLogger(SimplePhysicalEntityImpl.class);
20  
21  	public SimplePhysicalEntityImpl() {
22  	}
23  
24  	public Set<EntityReference> getGenericEntityReferences()
25  	{
26  		Set<EntityReference> ger = BPCollections.I.createSet();
27  		EntityReference er = this.getEntityReference();
28  		if(er!=null)
29  		{
30  			ger.add(er);
31  			ger.addAll(er.getMemberEntityReference());
32  		}
33  		for (PhysicalEntity pe : this.getMemberPhysicalEntity())
34  		{
35  			if(pe instanceof SimplePhysicalEntity)
36  				ger.addAll(((SimplePhysicalEntity) pe).getGenericEntityReferences());
37  			else
38  				log.error("Member PE is of different class! Skipping..");
39  		}
40  		return ger;
41  	}
42  
43  	public EntityReference getEntityReference()
44  	{
45  		return entityReference;
46  	}
47  
48  	public void setEntityReference(EntityReference entityReference)
49  	{
50  		if (this.entityReference != null)
51  		{
52  			this.entityReference.getEntityReferenceOf().remove(this);
53  		}
54  		this.entityReference = entityReference;
55  		if (this.entityReference != null)
56  		{
57  			this.entityReference.getEntityReferenceOf().add(this);
58  		}
59  	}
60  
61  
62  	@Override
63  	public int equivalenceCode()
64  	{
65          return this.entityReference==null? hashCode():31 * super.locationAndFeatureCode() +
66  		       entityReference.equivalenceCode();
67  	}
68  
69  	@Override
70  	protected boolean semanticallyEquivalent(BioPAXElement element)
71  	{
72  		if(!(element instanceof SimplePhysicalEntity))
73  			return false;
74  		
75  		SimplePhysicalEntity that = (SimplePhysicalEntity) element;
76  		return ( (that.getEntityReference()!=null)
77  					? that.getEntityReference().isEquivalent(getEntityReference())
78  					: getEntityReference() == null
79  				) && super.semanticallyEquivalent(element);
80  	}
81  }
