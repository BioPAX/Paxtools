1   package org.biopax.paxtools.controller;
2   
3   import org.biopax.paxtools.model.BioPAXElement;
4   import org.biopax.paxtools.model.Model;
5   import org.biopax.paxtools.util.Filter;
6   import org.slf4j.Logger;
7   import org.slf4j.LoggerFactory;
8   
9   import java.util.HashSet;
10  import java.util.Set;
11  
12  /**
13   * This is a utility class for traversing over the dependent objects of a biopax element, based on
14   * property editors
15   */
16  public class Traverser
17  {
18  // ------------------------------ FIELDS ------------------------------
19  
20  	protected final EditorMap editorMap;
21  
22  	protected Visitor visitor;
23  
24  	protected Filter<PropertyEditor>[] filters;
25  
26  	private final static Logger log = LoggerFactory.getLogger(Traverser.class);
27  
28  
29  // --------------------------- CONSTRUCTORS ---------------------------
30  
31  	/**
32  	 * The full constructor.
33  	 * @param editorMap is used for looking up properties to traverse
34  	 * @param visitor is a visitor element that determines the action on reaching a biopax element during traversal
35  	 * @param filters limits the type of properties that are traversed.
36  	 */
37  	public Traverser(EditorMap editorMap, Visitor visitor, Filter<PropertyEditor>... filters)
38  	{
39  		this.editorMap = editorMap;
40  		this.visitor = visitor;
41  		this.filters = filters;
42  	}
43  
44  // -------------------------- SETTERS/GETTERS ------------------------
45  
46  
47  	public void setVisitor(Visitor visitor)
48  	{
49  		this.visitor = visitor;
50  	}
51  
52  	public Visitor getVisitor()
53  	{
54  		return visitor;
55  	}
56  
57  // -------------------------- OTHER METHODS --------------------------
58  
59  	/**
60  	 * Traverse and visit {@link Visitor} all properties of the element.
61  	 * This method does not traverse iteratively to the values.
62  	 * @param element BioPAX element to be traversed
63  	 * @param model to be traversed, but not necessarily (depends on the Visitor implementation).
64  	 * @param <D> actual BioPAX type which properties and inherited properties will be used
65  	 */
66  	public <D extends BioPAXElement> void traverse(D element, Model model)
67  	{
68  		Set<PropertyEditor> editors = editorMap.getEditorsOf(element);
69  
70  		if (editors == null)
71  		{
72  			log.warn("No editors for : " + element.getModelInterface());
73  			return;
74  		}
75  		for (PropertyEditor<? super D,?> editor : editors)
76  		{
77  			if (filter(editor))
78  			{
79  				Set<?> valueSet = editor.getValueFromBean(element);
80  				if (!valueSet.isEmpty())
81                      traverseElements(element, model, editor, valueSet);
82              }
83  		}
84  	}
85  
86      protected void traverseElements(BioPAXElement element, Model model, PropertyEditor<?,?> editor, Set<?> valueSet)
87      {
88          for (Object value : new HashSet<Object>(valueSet))
89          {
90          	visitor.visit(element, value, model, editor);
91          }
92      }
93  
94  
95      protected boolean filter(PropertyEditor<?, ?> editor)
96  	{
97  		for (Filter<PropertyEditor> filter : filters)
98  		{
99  			if (!filter.filter(editor))
100 			{
101 				return false;
102 			}
103 		}
104 		return true;
105 	}
106 }
