12  /**
13   * Specifically "Clones" the BioPAX elements set
14   * (traverses to obtain dependent elements), 
15   * puts them to the new model using the visitor and traverser framework;
16   * ignores elements that are not in the source list (compare to {@link Fetcher}).
17   *
18   * @see org.biopax.paxtools.controller.Visitor
19   * @see org.biopax.paxtools.controller.Traverser
20   */
21  public class Cloner implements Visitor
22  {
23  	private static final Logger LOG = LoggerFactory.getLogger(Cloner.class);
24  	
25  	private final Traverser traverser;
26  	private final BioPAXFactory factory;
27  
28  	public Cloner(EditorMap map, BioPAXFactory factory)
29  	{
30  		this.factory = factory;
31  		this.traverser = new Traverser(map, this);
32  	}
33  
34  	/**
35  	 * For each element from the 'toBeCloned' list,
36  	 * it creates a copy in the new model, setting all
37  	 * the data properties; however, object property values
38  	 * that refer to BioPAX elements not in 'toBeCloned' list
39  	 * are ignored.
40  	 *
41  	 * @deprecated use {@link #clone(Set)} instead ('source' model was not used in previous versions)
42  	 *
43  	 * @param source - actually, is never used... can be null all the same
44  	 * @param toBeCloned elements to clone
45  	 * @return a new model containing the cloned biopax objects
46  	 */
47  	public Modelref="../../../../org/biopax/paxtools/model/Model.html#Model">Model clone(Model source, Set<BioPAXElement> toBeCloned) {
48  		return clone(toBeCloned);
49  	}
50  
51  	/**
52  	 * For each element from the 'toBeCloned' list,
53  	 * it creates a copy in the new model, setting all 
54  	 * the data properties; however, object property values
55  	 * that refer to BioPAX elements not in 'toBeCloned' list
56  	 * are ignored. As the result, e.g., a Pathway can be incomplete,
57  	 * missing some or all its sub-processes, xrefs, pathway steps.
58  	 * 
59  	 * @param toBeCloned elements to clone
60  	 * @return a new model containing the cloned biopax objects
61  	 */
62  	public Model clone(Set<BioPAXElement> toBeCloned)
63  	{
64  		final Model targetModel = factory.createModel();
65  
66  		for (BioPAXElement bpe : new HashSet<BioPAXElement>(toBeCloned))
67  		{
68  			// create a new empty object using the URI (all the biopax properties are empty;
69  			// addNew fails when there are several objects having the same URI)
70  			targetModel.addNew(bpe.getModelInterface(), bpe.getUri());
71  		}
72  
73  		// a hack to avoid unnecessary checks for the valid sub-model being cloned, 
74  		// and warnings when the Cloner copies BPS.stepProcess values, 
75  		// and there is a Conversion among them (-always unless stepConversion is null).
76  		AbstractPropertyEditor.checkRestrictions.set(false);
77  		
78  		for (BioPAXElement bpe : toBeCloned)
79  		{
80  			traverser.traverse(bpe, targetModel);
81  		}
82  		
83  		AbstractPropertyEditor.checkRestrictions.set(true); //back to the default mode
84  
85  		return targetModel;
86  	}
87  
88  	//  Implement interface: Visitor
89  	public void visit(BioPAXElement domain, Object range, Model targetModel, PropertyEditor editor)
90  	{
91  		BioPAXElement targetDomain = targetModel.getByID(domain.getUri());
92  		
93  		if (range instanceof BioPAXElement)
94  		{
95  			BioPAXElement bpe = (BioPAXElement) range;
96  			BioPAXElement existing = targetModel.getByID(bpe.getUri());
97  			//set the property value if the value is already present in the target 
98  			if (existing != null) {
99  				editor.setValueToBean(existing, targetDomain);
100 			}
101 		}
102 		else
103 		{
104 			editor.setValueToBean(range, targetDomain);
105 		}
106 	}
107 }
