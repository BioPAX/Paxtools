4   import org.biopax.paxtools.model.Model;
5   import org.biopax.paxtools.util.Filter;
6   import org.slf4j.Logger;
7   import org.slf4j.LoggerFactory;
8   
9   import java.util.HashSet;
10  import java.util.Set;
11  
12  /**
13   * This is an all-in-one Traverser/Visitor combination 
14   * to use when deep recursive visiting is required. 
15   * Unlike {@link Traverser}, it keeps track of where current 
16   * object is in the model and whether it's been already visited, 
17   * which helps prevent infinite loops.
18   * Like it's for the {@link Traverser}, there is no any 
19   * particular order in which it processes properties.
20   * 
21   * @see Fetcher
22   * @see Traverser
23   * @see Visitor
24   * 
25   * @author rodch
26   */
27  public abstract class AbstractTraverser extends Traverser implements Visitor
28  {
29  	private final static Logger log = LoggerFactory.getLogger(AbstractTraverser.class);
30  	protected final Set<BioPAXElement> visited;
31  		
32  	public AbstractTraverser(EditorMap editorMap, 
33  		@SuppressWarnings("rawtypes") Filter<PropertyEditor>... filters)
34  	{
35  		super(editorMap, null, filters);
36  		setVisitor(this);
37  		visited = new HashSet<BioPAXElement>();
38  	}
39  
40  	/**
41  	 * This is to implement a real action here: 
42  	 * do something, return or even to continue (traverse)
43  	 * into the child (range) element's properties if it's a BioPAX object.
44  	 * 
45  	 * @param range property value
46  	 * @param domain parent/owner BioPAX element
47  	 * @param model the BioPAX model of interest
48  	 * @param editor the property editor
49  	 */
50  	protected abstract void visit(Object range, BioPAXElement domain, Model model, PropertyEditor<?,?> editor);
51  		
52  	/**
53  	 * Calls the protected abstract method visit that is to be
54  	 * implemented in subclasses of this abstract class.
55  	 * 
56  	 * @param domain BioPAX Element
57  	 * @param range property value (can be BioPAX element, primitive, enum, string)
58  	 * @param model the BioPAX model of interest
59  	 * @param editor parent's property PropertyEditor
60  	 */
61  	public void visit(BioPAXElement domain, Object range, Model model, PropertyEditor<?,?> editor) {
62  		// actions
63  		visit(range, domain, model, editor);
64  	}
65  
66  
67  	@Override
68  	public <D extends BioPAXElement> void traverse(D element, Model model) {
69  		if(visited.add(element)) {
70  			super.traverse(element, model);//calls visit method for each property value, taking prop. filters into acc.
71  		} else {
72  			log.debug("Escaped a loop: again " + element.getUri());
73  		}
74  	}
75  
76  
77  	/**
78  	 * Clears the internal set of traversed biopax objects.
79  	 * Apply if you're re-using the same traverser instance but
80  	 * start over from a different root biopax element.
81  	 */
82  	public void reset() {
83  		visited.clear();
84  	}
85  }
