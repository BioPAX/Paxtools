33  /**
34   * This class converts BioPAX L3 model into SBGN PD (SBGN-ML XML).
35   * It optionally applies a special COSE layout.
36   *
37   * Currently, this converter ignores several BioPAX types/properties:
38   * <ul>
39   * <li>Parent-child relationship between entity references</li>
40   * <li>Binding features and covalent binding features of physical entities</li>
41   * <li>Pathway, PathwayStep, Evidence, etc.</li>
42   * </ul>
43   *
44   * Also note that:
45   * <ul>
46   * <li>Compartment is just a controlled vocabulary in BioPAX, so nesting and neighborhood relations
47   * between compartments are not handled here.</li>
48   * <li>Control structures in BioPAX and in SBGN PD are a little different. We use logical
49   * glyphs and arcs to approximate controls in BioPAX. However, in fact,
50   * BioPAX does not imply a logical operator between controllers.</li>
51   * </ul>
52   *
53   * @author Ozgun Babur
54   */
55  public class L3ToSBGNPDConverter {

101 	/*
102 	 * SBGN process glyph can be used to show reversible reactions. In that case two ports of the
103 	 * process will only have product glyphs. However, this creates an incompatibility with BioPAX:
104 	 * reversible biochemical reactions can have catalysis with a direction. But if we use a single
105 	 * glyph for the process and direct that catalysis to it, then the direction of the catalysis
106 	 * will be lost. If we use two process nodes for the reversible reaction (one for left-to-right
107 	 * and another for right-to-left), then we can direct the directed catalysis to only the
108 	 * relevant process glyph.
109 	 *
110 	 * Also note that the layout do not support ports. If layout is run, the ports are removed. In
111 	 * that case it will be impossible to distinguish left and right components of the reversible
112 	 * process glyph because they will all be product edges.
113 	 */
114 	protected boolean useTwoGlyphsForReversibleConversion;

255 	/**
256 	 * Creates an SBGN object from the given BioPAX L3 model.
257 	 *
258 	 * Currently, it converts physical entities (interaction participants)
259 	 * and Conversion, Control, including their sub-classes, TemplateReaction,
260 	 * MolecularInteraction, GeneticInteraction, and base Interaction.class types.
261 	 * I does not convert other BioPAX entities and utility classes, such as:
262 	 * Pathway, Evidence, PathwayStep.
263 	 *
264 	 * @param model model to convert to SBGN
265 	 * @return SBGN representation of the BioPAX model
266 	 */
267 	public Sbgn createSBGN(Model model) {
268 		assert model.getLevel().equals(BioPAXLevel.L3) : "This method only supports L3 graphs";
269 
270 		glyphMap = new HashMap<String, Glyph>();
271 		compartmentMap = new HashMap<String, Glyph>();
272 		arcMap = new HashMap<String, Arc>();
273 		ubiqueSet = new HashSet<Glyph>();
274 
275 		int n = 0; //approximate number of SBGN nodes
276 
277 		// Create glyphs for Physical Entities
278 		for (Entity entity : model.getObjects(Entity.class)) {
279 			if (needsToBeCreatedInitially(entity)) {
280 				createGlyph(entity);
281 				++n;
282 			}
283 		}
284 
285 		// Create glyph for conversions and link with arcs
286 		for (Interaction interaction : model.getObjects(Interaction.class))
287 		{
288 			if(interaction.getParticipant().isEmpty())
289 				continue;
290 
291 			// For each conversion we check if we need to create a left-to-right and/or right-to-left process.
292 			if(interaction instanceof Conversion) {
293 				Conversion conv = (Conversion) interaction;
294 				if (conv.getConversionDirection() == null ||
295 						conv.getConversionDirection().equals(ConversionDirectionType.LEFT_TO_RIGHT) ||
296 						(conv.getConversionDirection().equals(ConversionDirectionType.REVERSIBLE) &&
297 								useTwoGlyphsForReversibleConversion)) {
298 					createProcessAndConnections(conv, ConversionDirectionType.LEFT_TO_RIGHT);
299 				} else if (conv.getConversionDirection() != null &&
300 						(conv.getConversionDirection().equals(ConversionDirectionType.RIGHT_TO_LEFT) ||
301 								(conv.getConversionDirection().equals(ConversionDirectionType.REVERSIBLE)) &&
302 										useTwoGlyphsForReversibleConversion)) {
303 					createProcessAndConnections(conv, ConversionDirectionType.RIGHT_TO_LEFT);
304 				} else if (conv.getConversionDirection() != null &&
305 						conv.getConversionDirection().equals(ConversionDirectionType.REVERSIBLE) &&
306 						!useTwoGlyphsForReversibleConversion) {
307 					createProcessAndConnections(conv, ConversionDirectionType.REVERSIBLE);
308 				}
309 			} else if(interaction instanceof TemplateReaction) {
310 				createProcessAndConnections((TemplateReaction) interaction);
311 			} else if(interaction instanceof MolecularInteraction) {
312 				createBasicProcess(interaction);
313 			} else if(interaction instanceof GeneticInteraction) {
314 				createGiProcess((GeneticInteraction) interaction);
315 			} else if(!(interaction instanceof Control)) {
316 				createBasicProcess(interaction);
317 			} else { //a Control special case, because these are normally added inside the methods in cases above
318 				Control control = (Control) interaction;
319 				//Control without any controlled process but with controller and controlledOf
320 				if(control.getControlled()==null || control.getControlled().isEmpty()) {
321 					Glyph g = createControlStructure(control);
322 					processControllers(control.getControlledOf(), g);
323 				}//else - do nothing - as it's converted anyway when the controlled interactions are processed
324 			}
325 
326 			++n;
327 		}
328 
329 		// Register created objects into sbgn construct
330 		final Sbgn sbgn = factory.createSbgn();
331 		org.sbgn.bindings.Map map = new org.sbgn.bindings.Map();
332 		sbgn.setMap(map);
333 		map.setLanguage(Language.PD.toString());
334 		map.getGlyph().addAll(getRootGlyphs(glyphMap.values()));
335 		map.getGlyph().addAll(getRootGlyphs(ubiqueSet));
336 		map.getGlyph().addAll(compartmentMap.values());
337 		map.getArc().addAll(arcMap.values());
338 
339 //    //Store some metadata within the standard SBGN-ML extension element: Notes
340 //		biopaxMetaDoc.setDocumentURI(model.getUri()); //can be null
341 //		SBGNBase.Notes modelNotes = new SBGNBase.Notes();
342 //		sbgn.setNotes(modelNotes);
343 //		Element elt = biopaxMetaDoc.createElementNS("","metadata");
344 //		elt.setTextContent(String.format("{name:\"%s\",uri:\"%s\"}", model.getName(), model.getUri())
345 //			.replaceAll("null",""));
346 //		modelNotes.getAny().add(elt);
347 
348 		final boolean layout = doLayout && n < this.maxNodes && !arcMap.isEmpty();
349 		try {
350 			//Must call this, although actual layout might never run;
351 			//in some real data tests, skipping createLayout method
352 			//led to malformed SBGN model, unfortunately...
353 			(new SBGNLayoutManager()).createLayout(sbgn, layout);
354 		} catch (Exception e) {
355 			throw new RuntimeException("SBGN Layout of " + model.getXmlBase()
356 					+ ((model.getName()==null) ? "" : model.getName()) + " failed.", e);
357 		}
358 		if(!layout) log.warn(String.format("No layout, for either " +
359 				"it's disabled: %s, or ~ no. nodes > %s: %s, or - no edges: %s",
360 				!doLayout, maxNodes, n>maxNodes, arcMap.isEmpty()));
361 
362 		return sbgn; //modified sbgn (even when no layout is run)
363 	}

365 	// Associate all the controllers of this process;
366 	private void processControllers(Set<Control> controls, Glyph process) {
367 		for (Control ctrl : controls) {
368 			Glyph g = createControlStructure(ctrl); //if ctrl has upstream controls (modulations), they're processed as well
369 			if (g != null) {
370 				createArc(g, process, getControlType(ctrl), null);
371 				processControllers(ctrl.getControlledOf(), g);
372 			}
373 		}
374 	}

377 	 * Initially, we don't want to represent every PhysicalEntity or Gene node.
378 	 * For example, if a Complex is nested under another Complex,
379 	 * and if it is not a participant of any interaction,
380 	 * then we don't want to draw it separately; also skip for dangling entities.
381 	 *
382 	 * @param ent physical entity or gene (it returns false for other entity types) to test
383 	 * @return true if we want to draw this entity in SBGN; false - otherwise, or - to be auto-created later
384 	 */
385 	private boolean needsToBeCreatedInitially(Entity ent) {
386 		boolean create = false;
387 
388 		if(ent instanceof PhysicalEntity || ent instanceof Gene) {
389 			if(ubiqueDet != null && ubiqueDet.isUbique(ent))
390 				create = false; // ubiques will be created where they are actually used.
391 			else if (!ent.getParticipantOf().isEmpty())
392 				create = true;
393 			else if(ent instanceof Complex./../../../org/biopax/paxtools/model/level3/Complex.html#Complex">Complex && ((Complex) ent).getComponentOf().isEmpty()
394 					&& ((Complex) ent).getMemberPhysicalEntityOf().isEmpty())
395 				create = true; //do make a root/top complex despite it's dangling
396 		}
397 
398 		return create;
399 	}

402 	 * Creates a glyph representing the given PhysicalEntity.
403 	 *
404 	 * @param e PhysicalEntity or Gene to represent
405 	 * @return the created glyph
406 	 */
407 	private Glyph createGlyph(Entity e)
408 	{
409 		String id = convertID(e.getUri());
410 		if (glyphMap.containsKey(id))
411 			return glyphMap.get(id);
412 
413 		// Create its glyph and register
414 		Glyph g = createGlyphBasics(e, true);
415 		glyphMap.put(g.getId(), g);
416 
417 		//TODO: export metadata (e.g., from bpe.getAnnotations() map) using the SBGN Extension feature
418 //		SBGNBase.Extension ext = new SBGNBase.Extension();
419 //		g.setExtension(ext);
420 //		Element el = biopaxMetaDoc.createElement(e.getModelInterface().getSimpleName());
421 //		el.setAttribute("uri", e.getUri());
422 //		ext.getAny().add(el);
423 
424 		if (g.getClone() != null)
425 			ubiqueSet.add(g);
426 
427 		if(e instanceof PhysicalEntity) {
428 			PhysicalEntity pe = (PhysicalEntity) e;
429 			assignLocation(pe, g);
430 			if("or".equalsIgnoreCase(g.getClazz())) {
431 				buildGeneric(pe, g, null);
432 			} else if (pe instanceof Complex) {
433 				createComplexContent((Complex) pe, g);
434 			}
435 		}
436 
437 		return g;
438 	}

453 	/*
454 	 * This method creates a glyph for the given PhysicalEntity, sets its title and state variables
455 	 * if applicable.
456 	 *
457 	 * @param e PhysicalEntity or Gene to represent
458 	 * @param idIsFinal if ID is final, then it is recorded for future reference
459 	 * @return the glyph
460 	 */
461 	private Glyph createGlyphBasics(Entity e, boolean idIsFinal)
462 	{
463 		Glyph g = factory.createGlyph();
464 		g.setId(convertID(e.getUri()));
465 
466 		String s = typeMatchMap.get(e.getModelInterface());
467 		if(( //use 'or' sbgn class for special generic physical entities
468 			e instanceof Complex/../../../org/biopax/paxtools/model/level3/Complex.html#Complex">Complex && !((Complex)e).getMemberPhysicalEntity().isEmpty()
469 				&& ((Complex) e).getComponent().isEmpty())
470 			||
471 			(e instanceof SimplePhysicalEntityg/biopax/paxtools/model/level3/SimplePhysicalEntity.html#SimplePhysicalEntity">SimplePhysicalEntity && ((SimplePhysicalEntity) e).getEntityReference()==null
472 				&& !((SimplePhysicalEntity) e).getMemberPhysicalEntity().isEmpty()))
473 		{
474 			s = GlyphClazz.OR.getClazz();
475 		}
476 		g.setClazz(s);
477 
478 		// Set the label
479 		Label label = factory.createLabel();
480 		label.setText(findLabelFor(e));
481 		g.setLabel(label);
482 
483 		// Detect if ubique
484 		if (ubiqueDet != null && ubiqueDet.isUbique(e))
485 		{
486 			g.setClone(factory.createGlyphClone());
487 		}
488 
489 		// Put on state variables
490 		if (!g.getClazz().equals(GlyphClazz.OR.getClazz())) {
491 			g.getGlyph().addAll(getInformation(e));
492 		}
493 
494 		// Record the mapping
495 		if (idIsFinal) {
496 			Set<String> uris = new HashSet<String>();
497 			uris.add(e.getUri());
498 			sbgn2BPMap.put(g.getId(), uris);
499 		}
500 
501 		return g;
502 	}

504 	/*
505 	 * Gets the representing glyph of the PhysicalEntity or Gene.
506 	 * @param e PhysicalEntity or Gene to get its glyph
507 	 * @param linkID Edge id, used if the Entity is ubique
508 	 * @return Representing glyph
509 	 */
510 	private Glyph getGlyphToLink(Entity e, String linkID)
511 	{
512 		if (ubiqueDet == null || !ubiqueDet.isUbique(e))
513 		{
514 			return glyphMap.get(convertID(e.getUri()));
515 		}
516 		else {
517 			// Create a new glyph for each use of ubique
518 			Glyph g = createGlyphBasics(e, false);
519 			g.setId(convertID(e.getUri()) + "_" + ModelUtils.md5hex(linkID));
520 			Set<String> uris = new HashSet<String>();
521 			uris.add(e.getUri());
522 			sbgn2BPMap.put(g.getId(), uris);
523 			if(e instanceof PhysicalEntity/../org/biopax/paxtools/model/level3/PhysicalEntity.html#PhysicalEntity">PhysicalEntity && ((PhysicalEntity)e).getCellularLocation() != null) {
524 				assignLocation((PhysicalEntity) e, g);
525 			}
526 			ubiqueSet.add(g);
527 			return g;
528 		}
529 	}

558 		}

561 	private void buildGeneric(PhysicalEntity generic, Glyph or, Glyph container) {
562 		assert "or".equalsIgnoreCase(or.getClazz()) : "must be 'or' glyph class";
563 
564 		for (PhysicalEntity m : generic.getMemberPhysicalEntity())
565 		{
566 			Glyph g = createGlyphBasics(m,false);
567 			if(container!=null)
568 				container.getGlyph().add(g);
569 			String gid = g.getId() + "_" + ModelUtils.md5hex("memberof_" + or.getId());
570 			g.setId(gid);
571 			glyphMap.put(gid, g);
572 			Set<String> uris =  new HashSet<String>();
573 			uris.add(m.getUri());
574 			sbgn2BPMap.put(gid, uris);
575 			assignLocation(m, g);
576 			createArc(g, or, ArcClazz.LOGIC_ARC.getClazz(), null);
577 			if(m instanceof Complex)
578 				createComplexContent((Complex) m, g);
579 		}
580 	}

584 	 * Recursive method for creating the content of a complex. A complex may contain other complexes
585 	 * (bad practice), but SBGN needs them flattened. If an inner complex is empty, then we
586 	 * represent it using a glyph. Otherwise we represent only the members of this inner complex,
587 	 * merging them with the most outer complex.
588 	 *
589 	 * @param cx inner complex to add as member

669 	/*
670 	 * Looks for the display name of this PhysicalEntity. If there is none, then it looks for the
671 	 * display name of its EntityReference. If still no name at hand, it tries the standard
672 	 * name, and then first element in name lists.
673 	 *
674 	 * A good BioPAX file will use a short and specific name (like HGNC symbols) as displayName.
675 	 *
676 	 * @param pe PhysicalEntity or Gene to find a name
677 	 * @return a name for labeling
678 	 */
679 	private String findLabelFor(Entity pe) {
680 		// Use gene symbol of PE
681 		for (Xref xref : pe.getXref()) {
682 			String sym = extractGeneSymbol(xref);
683 			if (sym != null) return sym;
684 		}
685 
686 		// Use gene symbol of ER
687 		EntityReference er = null;
688 
689 		if (pe instanceof SimplePhysicalEntity) {
690 			er = ((SimplePhysicalEntity) pe).getEntityReference();
691 		}
692 
693 		if (er != null) {
694 			for (Xref xref : er.getXref()) {
695 				String sym = extractGeneSymbol(xref);
696 				if (sym != null) return sym;
697 			}
698 		}
699 		
700 		// Use display name of entity
701 		String name = pe.getDisplayName();
702 
703 		if (name == null || name.trim().isEmpty())
704 		{
705 			if (er != null) {
706 				name = er.getDisplayName();
707 			}
708 
709 			if (name == null || name.trim().isEmpty()) {
710 				name = pe.getStandardName();
711 				if (name == null || name.trim().isEmpty()) {
712 					if (er != null) {
713 						name = er.getStandardName();
714 					}
715 					if (name == null || name.trim().isEmpty()) {
716 						if (!pe.getName().isEmpty()) {
717 							// Use first name of entity
718 							name = pe.getName().iterator().next();
719 						} else if (er != null && !er.getName().isEmpty()) {
720 							// Use first name of reference
721 							name = er.getName().iterator().next();
722 						}
723 					}
724 				}
725 			}
726 		}
727 
728 		// Search for the shortest name of chemicals
729 		if (pe instanceof SmallMolecule) {
730 			String shortName = getShortestName((SmallMolecule) pe);
731 			if (shortName != null) {
732 				if (name == null || (shortName.length() < name.length() && !shortName.isEmpty())) {
733 					name = shortName;
734 				}
735 			}
736 		}
737 		
738 		if (name == null || name.trim().isEmpty()) {
739 			// Don't leave it without a name
740 			name = "noname";
741 		}
742 
743 		return name;
744 	}

748 	 * @param spe entity to search in
749 	 * @return the shortest name

751 	private String getShortestName(SimplePhysicalEntity spe)

753 		String name = null;

770 	/**
771 	 * Searches for gene symbol in Xref.
772 	 * @param xref Xref to search
773 	 * @return gene symbol
774 	 */
775 	private String extractGeneSymbol(Xref xref)
776 	{
777 		if (xref.getDb() != null && (
778 			xref.getDb().equalsIgnoreCase("HGNC Symbol") ||
779 			xref.getDb().equalsIgnoreCase("Gene Symbol") ||
780 			xref.getDb().equalsIgnoreCase("HGNC")))
781 		{
782 			String ref = xref.getId();
783 
784 			if (ref != null) {
785 				ref = ref.trim();
786 				if (ref.contains(":")) ref = ref.substring(ref.indexOf(":") + 1);
787 				if (ref.contains("_")) ref = ref.substring(ref.indexOf("_") + 1);
788 				// if the reference is an HGNC ID, then convert it to a symbol
789 				if (!HGNC.containsSymbol(ref) && Character.isDigit(ref.charAt(0))) {
790 					ref = HGNC.getSymbol(ref);
791 				}
792 			}
793 
794 			return ref;
795 		}
796 
797 		return null;
798 	}

927 			cnv.getConversionDirection().equals(direction) ||
928 			cnv.getConversionDirection().equals(ConversionDirectionType.REVERSIBLE);

960 
961 		// Record mapping
962 		Set<String> uris = new HashSet<String>();

965 	}

1204 	 * @param ctrl Control to get its type
1205 	 * @return SBGN type of the Control
1206 	 */
1207 	private String getControlType(Control ctrl) {
1208 		if (ctrl instanceof Catalysis) {
1209 			// Catalysis has its own class
1210 			return ArcClazz.CATALYSIS.getClazz();
1211 		}
1212 
1213 		ControlType type = ctrl.getControlType();
1214 		if (type == null) {
1215 			// Use stimulation as the default control type
1216 			return ArcClazz.STIMULATION.getClazz();
1217 		}
1218 
1219 		// Map control type to stimulation or inhibition
1220 		switch (type) {
1221 			case ACTIVATION:
1222 			case ACTIVATION_ALLOSTERIC:
1223 			case ACTIVATION_NONALLOSTERIC:
1224 			case ACTIVATION_UNKMECH:
1225 				return ArcClazz.STIMULATION.getClazz();
1226 			case INHIBITION:
1227 			case INHIBITION_ALLOSTERIC:
1228 			case INHIBITION_OTHER:
1229 			case INHIBITION_UNKMECH:
1230 			case INHIBITION_COMPETITIVE:
1231 			case INHIBITION_IRREVERSIBLE:
1232 			case INHIBITION_UNCOMPETITIVE:
1233 			case INHIBITION_NONCOMPETITIVE:
1234 				return ArcClazz.INHIBITION.getClazz();
1235 			default:
1236 				throw new RuntimeException("Invalid control type: " + type);
1237 		}
1238 	}

1288 	 * Creates an arc from the source to the target, and sets its class to the specified clazz.
1289 	 * Puts the new arc in the sullied arcMap.
1290 	 *
1291 	 * @param source source of the arc -- either Glyph or Port
1292 	 * @param target target of the arc -- either Glyph or Port
1293 	 * @param clazz class of the arc
1294 	 */
1295 	private void createArc(Object source, Object target, String clazz, Stoichiometry stoic)
1296 	{
1297 		assert source instanceof Glyph || source instanceof Port : "source = " + source;
1298 		assert target instanceof Glyph || target instanceof Port : "target = " + target;
1299 
1300 		Arc arc = factory.createArc();
1301 		arc.setSource(source);
1302 		arc.setTarget(target);
1303 		arc.setClazz(clazz);
1304 		
1305 		String sourceID = source instanceof Glyph ?
1306 			((Glyph) source).getId() : ((Port) source).getId();
1307 		String targetID = target instanceof Glyph ?
1308 			((Glyph) target).getId() : ((Port) target).getId();
1309 
1310 		arc.setId(sourceID + "--TO--" + targetID);
1311 
1312 		if (stoic != null && stoic.getStoichiometricCoefficient() > 1) {
1313 			Glyph card = factory.createGlyph();
1314 			card.setClazz(GlyphClazz.CARDINALITY.getClazz());
1315 			Label label = factory.createLabel();
1316 			label.setText(new DecimalFormat("0.##").format(stoic.getStoichiometricCoefficient()));
1317 			card.setLabel(label);
1318 			arc.getGlyph().add(card);
1319 		}
1320 
1321 		Arc.Start start = new Arc.Start();
1322 		start.setX(0);
1323 		start.setY(0);
1324 		arc.setStart(start);
1325 
1326 		Arc.End end = new Arc.End();
1327 		end.setX(0);
1328 		end.setY(0);
1329 		arc.setEnd(end);
1330 
1331 		arcMap.put(arc.getId(), arc);
1332 	}

1335 	 * Collects root-level glyphs in the given glyph collection.
1336 	 *
1337 	 * @param glyphCol glyph collection to search
1338 	 * @return set of roots
1339 	 */
1340 	private Set<Glyph> getRootGlyphs(Collection<Glyph> glyphCol) {
1341 		Set<Glyph> root = new HashSet<Glyph>(glyphCol);
1342 		Set<Glyph> children = new HashSet<Glyph>();
1343 
1344 		for (Glyph glyph : glyphCol) {
1345 			addChildren(glyph, children);
1346 		}
1347 		root.removeAll(children);
1348

1350 	}
