54  	/**
55  	 * Normalizes BioPAX OWL data and returns
56  	 * the result as BioPAX OWL (string).
57  	 * 
58  	 * This public method is actually intended to use 
59  	 * outside the BioPAX Validator framework, with not too large models.
60  	 * 
61  	 * @param biopaxOwlData RDF/XML BioPAX content string
62  	 * @return normalized BioPAX RDF/XML
63  	 * @deprecated this method will fail if the data exceeds ~1Gb (max UTF8 java String length)
64  	 */
65  	public String normalize(String biopaxOwlData) {
66  		
67  		if(biopaxOwlData == null || biopaxOwlData.length() == 0) 
68  			throw new IllegalArgumentException("no data. " + description);
69  		
70  		// quick-fix for older BioPAX L3 version (v0.9x) property 'taxonXref' (range: BioSource)
71  		biopaxOwlData = biopaxOwlData.replaceAll("taxonXref","xref");
72  		
73  		// build the model
74  		Model model = null;
75  		try {
76  			model = biopaxReader.convertFromOWL(
77  				new ByteArrayInputStream(biopaxOwlData.getBytes("UTF-8")));
78  		} catch (UnsupportedEncodingException e) {
79  			throw new IllegalArgumentException("Failed! " + description, e);
80  		}
81  		
82  		if(model == null) {
83  			throw new IllegalArgumentException("Failed to create Model! " 
84  					+ description);
85  		}
86  		
87  		// auto-convert to Level3 model
88  		if (model.getLevel() != BioPAXLevel.L3) {
89  			log.info("Converting model to BioPAX Level3...");
90  			model = (new LevelUpgrader()).filter(model);
91  		}
92  		
93  		normalize(model); // L3 only!
94  		
95  		// return as BioPAX OWL
96  		return convertToOWL(model);
97  	}

100 	/**
101 	 * Normalizes all xrefs (not unification xrefs only) to help normalizing/merging other objects, 
102 	 * and also because some of the original xref URIs ("normalized")
103 	 * are in fact to be used for other biopax types (e.g., CV or ProteinReference); therefore
104 	 * this method will replace URI also for "bad" xrefs, i.e., those with empty/illegal 'db' or 'id' values.
105 	 * 
106 	 * @param model biopax model to update
107 	 */
108 	private void normalizeXrefs(Model model) {
109 		
110 		final NormalizerMap map = new NormalizerMap(model);
111 		final String xmlBase = getXmlBase(model); //current base, the default or model's one, if set.
112 		
113 		// use a copy of the xrefs set (to avoid concurrent modif. exception)
114 		Set<? extends Xref> xrefs = new HashSet<Xref>(model.getObjects(Xref.class));
115 		for(Xref ref : xrefs)
116 		{
117 			//skip not well-defined ones (incl. PublicationXrefs w/o db/id - won't normalize)
118 			if(ref.getDb() == null || ref.getId() == null)
119 				continue;
120 
121 			ref.setDb(ref.getDb().toLowerCase()); //set lowercase
122 			String idPart = ref.getId();
123 
124 			if(ref instanceof RelationshipXref) {
125 				// only normalize (replace URI of) RXs that could potentially clash with other RX, CVs, ERs;
126 				// skip, won't bother, for all other RXs...
127 				if(!ref.getUri().startsWith("http://identifiers.org/"))
128 					continue;
129 
130 				//RXs might have the same db, id but different rel. type.
131 				RelationshipTypeVocabulary cv = ((RelationshipXref) ref).getRelationshipType();
132 				if(ref.getIdVersion()!=null) {
133 					idPart += "_" + ref.getIdVersion();
134 				}
135 				if(cv != null && !cv.getTerm().isEmpty()) {
136 					idPart += "_" + StringUtils.join(cv.getTerm(), '_').toLowerCase();
137 				}
138 			}
139 			else if(ref instanceof UnificationXref) {
140 				// first, try to normalize the db name (using MIRIAM EBI registry)
141 				try {
142 					ref.setDb(MiriamLink.getName(ref.getDb()).toLowerCase());
143 				} catch (IllegalArgumentException e) {
144 					// - unknown/unmatched db name (normalize using defaults)
145 					if(ref.getIdVersion()!=null) {
146 						idPart += "_" + ref.getIdVersion();
147 					}
148 					map.put(ref, Normalizer.uri(xmlBase, ref.getDb(), idPart, ref.getModelInterface()));
149 					continue; //shortcut (non-standard db name)
150 				}
151 			
152 				// a hack for uniprot/isoform xrefs
153 				if (ref.getDb().startsWith("uniprot")) {
154 					//auto-fix (guess) for possibly incorrect db/id (can be 'uniprot isoform' with/no idVersion, etc..)
155 					if (isValidDbId("uniprot isoform", ref.getId())
156 							&& ref.getId().contains("-")) //the second condition is important
157 					{	//then it's certainly an isoform id; so - fix the db name
158 						ref.setDb("uniprot isoform"); //fix the db
159 					}
160 					else {
161 						//id does not end with "-\\d+", i.e., not a isoform id
162 						//(idVersion is a different thing, but id db was "uniprot isoform" they probably misused idVersion)
163 						if(ref.getDb().equals("uniprot isoform"))
164 						{
165 							if(ref.getIdVersion() != null && ref.getIdVersion().matches("^\\d+$"))
166 								idPart = ref.getId()+"-"+ref.getIdVersion(); //guess, by idVersion, they actually meant isoform
167 							if(isValidDbId(ref.getDb(), idPart)) {
168 								ref.setId(idPart); //moved the isoform # to the ID
169 								ref.setIdVersion(null);
170 							}
171 							else if(!isValidDbId(ref.getDb(), ref.getId())) {
172 								//certainly not isoform (might not even uniprot, but try...)
173 								ref.setDb("uniprot knowledgebase"); //guess, fix
174 							}
175 							idPart = ref.getId();
176 						}
177 					}
178 				}
179 			}
180 
181 			// shelve it for URI replace
182 			map.put(ref, Normalizer.uri(xmlBase, ref.getDb(), idPart, ref.getModelInterface()));
183 		}
184 		
185 		// execute replace xrefs
186 		map.doSubs();
187 	}

197 	/**
198 	 * Consistently generates a new BioPAX element URI 
199 	 * using given URI namespace (xml:base), BioPAX class, 
200 	 * and two different identifiers (at least one is required).
201 	 * Miriam registry is used to get the standard db name and 
202 	 * identifiers.org URI, if possible, only for relationship type vocabulary, 
203 	 * publication xref, and entity reference types.
204 	 * 
205 	 * @param xmlBase xml:base (common URI prefix for a BioPAX model), case-sensitive
206 	 * @param dbName a bio data collection name or synonym, case-insensitive
207 	 * @param idPart optional (can be null), e.g., xref.id, case-sensitive
208 	 * @param type BioPAX class
209 	 * @return URI
210 	 * @throws IllegalArgumentException if either type is null or both 'dbName' and 'idPart' are all nulls.
211 	 */
212 	public static String uri(final String xmlBase, 
213 		String dbName, String idPart, Class<? extends BioPAXElement> type)
214 	{
215 		if(type == null || (dbName == null && idPart == null))
216 			throw new IllegalArgumentException("'Either type' is null, or both dbName and idPart are nulls.");
217 
218 		if (idPart != null) idPart = idPart.trim();
219 		if (dbName != null) dbName = dbName.trim();
220 
221 		// try to find a standard URI, if exists, for a publication xref, or at least a standard name:
222 		if (dbName != null)
223 		{
224 			try {
225 				// try to get the preferred/standard name
226 				// for any type, for consistency
227 				dbName = MiriamLink.getName(dbName);
228 				// a shortcut: a standard and resolvable URI exists for some BioPAX types
229 				if ((type.equals(PublicationXref.class) && "pubmed".equalsIgnoreCase(dbName))
230 					|| type.equals(RelationshipTypeVocabulary.class)
231 					|| ProteinReference.class.isAssignableFrom(type)
232 					|| SmallMoleculeReference.class.isAssignableFrom(type)
233 					|| (type.equals(BioSource.class) && "taxonomy".equalsIgnoreCase(dbName)
234 						&& idPart!=null && idPart.matches("^\\d+$"))
235 				)
236 				{	//get the standard URI and quit (success), or fail and continue making a new URI below...
237 					return MiriamLink.getIdentifiersOrgURI(dbName, idPart);
238 				}
239 			} catch (IllegalArgumentException e) {
240 				log.info(String.format("uri(for a %s): db:%s, id:%s are not standard; %s)",
241 						type.getSimpleName(), dbName, idPart, e.getMessage()));
242 			}
243 		}
244 
245 		// If not returned above this point - no standard URI (Identifiers.org) was found -
246 		// then let's consistently build a new URI from args, anyway, the other way around:
247 		
248 		StringBuilder sb = new StringBuilder();		
249 		if (dbName != null) //lowercase for consistency
250 			sb.append(dbName.toLowerCase()); 	
251 		
252 		if (idPart != null) {
253 			if (dbName != null) sb.append("_");
254 			sb.append(idPart);
255 		}
256 		
257 		String localPart = sb.toString();
258 		String strategy = System.getProperty(PROPERTY_NORMALIZER_URI_STRATEGY, VALUE_NORMALIZER_URI_STRATEGY_MD5);
259 		if(VALUE_NORMALIZER_URI_STRATEGY_SIMPLE.equals(strategy) || Xref.class.isAssignableFrom(type))
260 		//i.e., for xrefs, always use the simple URI strategy (makes them human-readable)
261 		{
262 			//simply replace "unsafe" symbols with underscore (some uri clashes might be possible but rare...)
263 			localPart = localPart.replaceAll("[^-\\w]", "_");
264 		}
265 		else
266 		{
267 			//replace the local part with its md5 sum string (32-byte)
268 			localPart = ModelUtils.md5hex(localPart);
269 		}
270 		
271 		// create URI using the xml:base and digest of other values:
272 		return ((xmlBase != null) ? xmlBase : "") + type.getSimpleName() + "_" + localPart;
273 	}

294 	private void fixDisplayName(Model model) {
295 		log.info("Trying to auto-fix 'null' displayName...");
296 		// where it's null, set to the shortest name if possible
297 		for (Named e : model.getObjects(Named.class)) {
298 			if (e.getDisplayName() == null) {
299 				if (e.getStandardName() != null) {
300 					e.setDisplayName(e.getStandardName());
301 					log.info(e + " displayName auto-fix: "
302 							+ e.getDisplayName() + ". " + description);
303 				} else if (!e.getName().isEmpty()) {
304 					String dsp = e.getName().iterator().next();
305 					for (String name : e.getName()) {
306 						if (name.length() < dsp.length())
307 							dsp = name;
308 					}
309 					e.setDisplayName(dsp);
310 					log.info(e + " displayName auto-fix: " + dsp
311 						+ ". " + description);
312 				}
313 			}
314 		}
315 		// if required, set PE name to (already fixed) ER's name...
316 		for(EntityReference er : model.getObjects(EntityReference.class)) {
317 			for(SimplePhysicalEntity spe : er.getEntityReferenceOf()) {
318 				if(spe.getDisplayName() == null || spe.getDisplayName().trim().length() == 0) {
319 					if(er.getDisplayName() != null && er.getDisplayName().trim().length() > 0) {
320 						spe.setDisplayName(er.getDisplayName());
321 					}
322 				}
323 			}
324 		}
325 	}

335 	private Collection<UnificationXref> getUnificationXrefsSorted(XReferrable r) {
336 
337 		List<UnificationXref> urefs = new ArrayList<UnificationXref>();
338 		for(UnificationXref ux : new ClassFilterSet<Xref,UnificationXref>(r.getXref(), UnificationXref.class))
339 		{
340 			if(ux.getDb() != null && ux.getId() != null) {
341 				urefs.add(ux);
342 			} 
343 		}
344 
345 		Collections.sort(urefs, new Comparator<UnificationXref>() {
346 			public int compare(UnificationXref../../../org/biopax/paxtools/model/level3/UnificationXref.html#UnificationXref">UnificationXref o1, UnificationXref o2) {
347 				String s1 = o1.getDb() + o1.getId();
348 				String s2 = o2.getDb() + o2.getId();
349 				return s1.compareTo(s2);
350 			}
351 		});
352 
353 		return urefs;
354 	}

357 	/**
358 	 * Finds one preferred unification xref, if possible.
359 	 * Preferred db values are:
360 	 * "ncbi gene" - for NucleicAcidReference and the sub-classes;
361 	 * "uniprot" or "refseq" for ProteinReference;
362 	 * "chebi" - for SmallMoleculeReference;
363 	 * 
364 	 * @param bpe BioPAX object that can have xrefs
365 	 * @return the "best" first unification xref 
366 	 */
367 	private UnificationXref findPreferredUnificationXref(XReferrable bpe)
368 	{
369 		UnificationXref toReturn = null;
370 
371 		Collection<UnificationXref> orderedUrefs = getUnificationXrefsSorted(bpe);
372 
373 		//use preferred db prefix for different type of ER
374 		if(bpe instanceof ProteinReference) {
375 			// prefer xref having db starts with "uniprot" (preferred) or "refseq";
376 			toReturn = findSingleUnificationXref(orderedUrefs, "uniprot");
377 			// if null, next try refseq.
378 			if(toReturn==null)
379 				toReturn = findSingleUnificationXref(orderedUrefs, "refseq");
380 		} else if(bpe instanceof SmallMoleculeReference) {
381 			// - "chebi", then  "pubchem";
382 			toReturn = findSingleUnificationXref(orderedUrefs, "chebi");
383 			if(toReturn==null)
384 				toReturn = findSingleUnificationXref(orderedUrefs, "pubchem");
385 		} else if(bpe instanceof NucleicAcidReference) {
386 			//that includes NucleicAcidRegionReference, etc. sub-classes;
387 			toReturn = findSingleUnificationXref(orderedUrefs, "ncbi gene");
388 			if(toReturn==null)
389 				toReturn = findSingleUnificationXref(orderedUrefs, "entrez");
390 		} else {
391 			//for other XReferrable types (BioSource or ControlledVocabulary)
392 			//use if there's only one xref (return null if many)
393 			if(orderedUrefs.size()==1)
394 				toReturn = orderedUrefs.iterator().next();
395 		}
396 		
397 		return toReturn;
398 	}

400 	private UnificationXref findSingleUnificationXref(Collection<UnificationXref> uXrefs, String dbStartsWith) {
401 		UnificationXref ret = null;
402 
403 		for(Xref x : uXrefs) {
404 			if(x instanceof UnificationXref && x.getId() != null && x.getDb() != null
405 					&& x.getDb().toLowerCase().startsWith(dbStartsWith))
406 			{
407 				if(ret == null)
408 				{
409 					ret = (UnificationXref)x;
410 				}
411 				else if (ret.getDb().equalsIgnoreCase(x.getDb()) && !ret.getId().equals(x.getId())) {
412 					//exactly the same kind (same 'db') xref is found;
413 					//we can't test here if several IDs map to the same thing;
414 					//so, give up - for safety - and return null.
415 					ret = null;
416 					break;
417 				}
418 			}
419 		}
420 
421 		return ret;
422 	}

425 	/**
426 	 * BioPAX normalization 
427 	 * (modifies the original Model)
428 	 * 
429 	 * @param model BioPAX model to normalize
430 	 * @throws NullPointerException if model is null
431 	 * @throws IllegalArgumentException if model is not Level3 BioPAX
432 	 */
433 	public void normalize(Model model) {
434 		
435 		if(model.getLevel() != BioPAXLevel.L3)
436 			throw new IllegalArgumentException("Not Level3 model. " +
437 				"Consider converting it first (e.g., with the PaxTools).");
438 		
439 		//if set, update the xml:base
440 		if(xmlBase != null && !xmlBase.isEmpty())
441 			model.setXmlBase(xmlBase);
442 
443 		// Normalize/merge xrefs, first, and then CVs
444 		// (also because some of original xrefs might have "normalized" URIs 
445 		// that, in fact, must be used for other biopax types, such as CV or ProteinReference)
446 		log.info("Normalizing xrefs..." + description);
447 		normalizeXrefs(model);
448 		
449 		// fix displayName where possible
450 		if(fixDisplayName) {
451 			log.info("Normalizing display names..." + description);
452 			fixDisplayName(model);
453 		}
454 			
455 		log.info("Normalizing CVs..." + description);
456 		normalizeCVs(model);
457 		
458 		//normalize BioSource objects (better, as it is here, go after Xrefs and CVs)
459 		log.info("Normalizing organisms..." + description);
460 		normalizeBioSources(model);
461 
462 		// auto-generate missing entity references:
463 		for(SimplePhysicalEntity spe : new HashSet<SimplePhysicalEntity>(model.getObjects(SimplePhysicalEntity.class))) {
464 			//it skips if spe has entityReference or memberPE already
465 			ModelUtils.addMissingEntityReference(model, spe);
466 		}
467 
468 		log.info("Normalizing entity references..." + description);
469 		normalizeERs(model);
470 		
471 		// find/add lost (in replace) children
472 		log.info("Repairing..." + description);
473 		model.repair(); // it does not remove dangling utility class objects (can be done separately, later, if needed)
474 		
475 		log.info("Optional tasks (reasoning)..." + description);
476 	}

479 	private void normalizeCVs(Model model) {
480 		
481 		NormalizerMap map = new NormalizerMap(model);
482 		
483 		// process ControlledVocabulary objects (all sub-classes)
484 		for(ControlledVocabulary cv : model.getObjects(ControlledVocabulary.class))
485 		{
486 			//it does not check/fix the CV terms though (but a validation rule can do if run before the normalizer)...
487 			UnificationXref uref = findPreferredUnificationXref(cv); //usually, there's only one such xref
488 			if (uref != null) {
489 				// so let's generate a consistent URI
490 				map.put(cv, uri(xmlBase, uref.getDb(), uref.getId(), cv.getModelInterface()));
491 			} else if(!cv.getTerm().isEmpty()) {
492 				map.put(cv, uri(xmlBase, null, cv.getTerm().iterator().next(), cv.getModelInterface()));
493 			} else log.info("Cannot normalize " + cv.getModelInterface().getSimpleName() 
494 				+ " : no unification xrefs nor terms found in " + cv.getUri()
495 				+ ". " + description);
496 		} 
497 		
498 		// replace/update elements in the model
499 		map.doSubs();
500 	}

503 	private void normalizeBioSources(Model model) {
504 		
505 		NormalizerMap map = new NormalizerMap(model);
506 		
507 		for(BioSource bs : model.getObjects(BioSource.class))
508 		{
509 			UnificationXref uref = findPreferredUnificationXref(bs);
510 			//normally, the xref db is 'Taxonomy' (or a valid synonym)
511 			if (uref != null
512 				&& (uref.getDb().toLowerCase().contains("taxonomy") || uref.getDb().equalsIgnoreCase("newt")))
513 			{	
514 				String 	idPart = uref.getId();
515 
516 				//tissue/cellType terms can be added below:
517 				if(bs.getTissue()!=null && !bs.getTissue().getTerm().isEmpty()) 
518 					idPart += "_" + bs.getTissue().getTerm().iterator().next();
519 				if(bs.getCellType()!=null && !bs.getCellType().getTerm().isEmpty()) 
520 					idPart += "_" + bs.getCellType().getTerm().iterator().next();
521 				
522 				String uri = (idPart.equals(uref.getId()) //- no tissue or celltype were attached
523 						&& idPart.matches("^\\d+$")) //- is positive integer id
524 					? uri(xmlBase, uref.getDb(), idPart, BioSource.class)
525 						: "http://identifiers.org/taxonomy/" + idPart;
526 					// the latter is intentionally invalid identifiers.org/taxonomy URI - good (and important) for merging
527 				map.put(bs, uri);
528 
529 			} else 
530 				log.debug("Won't normalize BioSource" 
531 					+ " : no taxonomy unification xref found in " + bs.getUri()
532 					+ ". " + description);
533 		} 
534 		
535 		map.doSubs();
536 	}

538 	private void normalizeERs(Model model) {
539 		
540 		NormalizerMap map = new NormalizerMap(model);
541 		
542 		// process the rest of utility classes (selectively though)
543 		for (EntityReference bpe : model.getObjects(EntityReference.class)) {
544 			
545 			//skip those with already normalized URIs
546 			if(bpe.getUri().startsWith("http://identifiers.org/")) {
547 				log.info("Skip already normalized: " + bpe.getUri());
548 				continue;
549 			}			
550 			
551 			UnificationXref uref = findPreferredUnificationXref(bpe);
552 			if (uref != null) {
553 				// Create (with a new URI made from a unif. xref) 
554 				// and save the replacement object, if possible, 
555 				// but do not replace yet (will call doSubs later, for all).
556 				final String db = uref.getDb();
557 				final String id = uref.getId();
558 				// get the standard ID
559 				String uri = null;
560 				try { // make a new ID for the element
561 					uri = MiriamLink.getIdentifiersOrgURI(db, id);
562 				} catch (Exception e) {
563 					log.error("Cannot get a Miriam standard ID for " + bpe 
564 							+ " (" + bpe.getModelInterface().getSimpleName()
565 							+ ") " + ", using " + db + ":" + id 
566 							+ ". " + e.getMessage());
567 					return;
568 				}
569 
570 				if(uri != null) {
571 					map.put(bpe, uri);
572 				}	
573 			} else
574 				log.info("Cannot normalize EntityReference: "
575 					+ "no unification xrefs found in " + bpe.getUri()
576 					+ ". " + description);
577 		}
578 		
579 		// replace/update elements in the model
580 		map.doSubs();
581 	}

584 	/**
585 	 * Auto-generates standard and other names for the datasource
586 	 * from either its ID (if URN) or one of its existing names (preferably - standard name)
587 	 * 
588 	 * @param pro data source (BioPAX Provenance)
589 	 */
590 	public static void autoName(Provenance pro) {
591 		if(!(pro.getUri().startsWith("urn:miriam:") || pro.getUri().startsWith("http://identifiers.org/"))
592 				&& pro.getName().isEmpty()) {
593 			log.info("Skipping: cannot normalize Provenance: " + pro.getUri());
594 		}
595 		else { // i.e., 'name' is not empty or ID is the URN
596 			final SortedSet<String> names = new TreeSet<String>();
597 			
598 			String key = null;
599 			if(pro.getUri().startsWith("urn:miriam:") || pro.getUri().startsWith("http://identifiers.org/")) {
600 				key = pro.getUri();
601 			} else if (pro.getStandardName() != null) {
602 				key = pro.getStandardName();
603 			} else {
604 				key = pro.getDisplayName(); // can be null
605 			}
606 			
607 			if (key != null) {
608 				try {
609 					names.addAll(Arrays.asList(MiriamLink.getNames(key)));
610 					pro.setStandardName(MiriamLink.getName(key));
611 					// get the datasource description
612 					String description = MiriamLink.getDataTypeDef(pro.getStandardName());
613 					pro.addComment(description);
614 				} catch (IllegalArgumentException e) {
615 					// ignore (then, names is still empty...)
616 				}
617 			} 
618 			
619 			// when the above failed (no match in Miriam), or key was null -
620 			if(names.isEmpty()) {
621 				// finally, trying to find all valid names for each existing one
622 					for (String name : pro.getName()) {
623 						try {
624 							names.addAll(Arrays.asList(MiriamLink.getNames(name)));
625 						} catch (IllegalArgumentException e) {
626 							// ignore
627 						}
628 					}
629 					// pick up the first name, get the standard name
630 					if(!names.isEmpty())
631 						pro.setStandardName(MiriamLink
632 							.getName(names.iterator().next()));
633 			}
634 			
635 			// and add all the synonyms if any
636 			for(String name : names)
637 				pro.addName(name);
638 			
639 			//set display name if not set (standard name is set already)
640 			if(pro.getDisplayName() == null)
641 				pro.setDisplayName(pro.getStandardName());			
642 		}
643 	}

645 	/**
646 	 * Converts BioPAX L1 or L2 RDF/XML string data to BioPAX L3 string.
647 	 *
648 	 * WARN: this is not for huge (larger than 1GB) BioPAX RDF/XML data
649 	 * due to use of (UTF-8) String and Byte Array internally.
650 	 * This can be and is used by online web apps, such as the
651 	 * BioPAX Validator.
652 	 *
653 	 * @param biopaxData String
654 	 * @return BioPAX Level3 RDF/XML string
655 	 */
656 	public static String convertToLevel3(final String biopaxData) {
657 		String toReturn = "";
658 		
659 		try {
660 			ByteArrayOutputStream os = new ByteArrayOutputStream();
661 			InputStream is = new ByteArrayInputStream(biopaxData.getBytes());
662 			SimpleIOHandler io = new SimpleIOHandler();
663 			io.mergeDuplicates(true);
664 			Model model = io.convertFromOWL(is);
665 			if (model.getLevel() != BioPAXLevel.L3) {
666 				log.info("Converting to BioPAX Level3... " + model.getXmlBase());
667 				model = (new LevelUpgrader()).filter(model);
668 				if (model != null) {
669 					io.setFactory(model.getLevel().getDefaultFactory());
670 					io.convertToOWL(model, os);
671 					toReturn = os.toString();
672 				}
673 			} else {
674 				toReturn = biopaxData;
675 			}
676 		} catch(Exception e) {
677 			throw new RuntimeException(
678 				"Cannot convert to BioPAX Level3", e);
679 		}
680 
681 		return toReturn;
682 	}

714 	/**
715 	 * Helper class, to associate original 
716 	 * and new (either copy or existing) objects 
717 	 * within some biopax model and then execute 
718 	 * batch replace.
719 	 * 
720 	 * @author rodche
721 	 */
722 	private static class NormalizerMap {
723 
724 		//a model to modify by replacing biopax objects
725 		final Model model;
726 		
727 		// this is the biopax elements substitution map (old->new)
728 		final Map<BioPAXElement,BioPAXElement> subs;
729 		
730 		//the next map is to make sure the subs.values()  all have different URIs
731 		final Map<String,BioPAXElement> uriToSub;
732 		
733 		final ShallowCopy copier;
734 				
735 		NormalizerMap(Model model) {
736 			subs = BPCollections.I.createMap();
737 			uriToSub = BPCollections.I.createMap();
738 			this.model = model;
739 			copier = new ShallowCopy();
740 		}
741 
742 
743 		/**
744 		 * Creates (by URI) and saves the replacement object,
745 		 * but does not replace yet (call doSubs to replace).
746 		 * 
747 		 * @param bpe
748 		 * @param newUri
749 		 */
750 		void put(BioPAXElement bpe, String newUri)
751 		{
752 			if(model.containsID(newUri)) {
753 				// will use existing original (model) object that has the new Uri
754 				map(bpe, model.getByID(newUri));
755 			} else if(uriToSub.containsKey(newUri)) {
756 				// re-use the new object that's already added to replace another original
757 				map(bpe, uriToSub.get(newUri));
758 			} else {
759 				// will use the object's shallow copy that gets new Uri
760 				BioPAXElement copy = copier.copy(bpe, newUri);
761 				map(bpe, copy);
762 			}
763 		}		
764 		
765 		/**
766 		 * Executes the batch replace - migrating  
767 		 * to the normalized equivalent objects.
768 		 */
769 		void doSubs() {
770 			for(BioPAXElement e : subs.keySet()) {
771 				model.remove(e);
772 			}
773 			
774 			try {
775 				ModelUtils.replace(model, subs);
776 			} catch (Exception e) {
777 				log.error("Failed to replace BioPAX elements.", e);
778 				return;
779 			}
780 			
781 			for(BioPAXElement e : subs.values()) {
782 				if(!model.contains(e))
783 					model.add(e);
784 			}
785 		
786 			for(BioPAXElement e : model.getObjects()) {
787 				ModelUtils.fixDanglingInverseProperties(e, model);
788 			}
789 		}
790 
791 		private void map(BioPAXElement/../../../org/biopax/paxtools/model/BioPAXElement.html#BioPAXElement">BioPAXElement bpe, BioPAXElement newBpe) {
792 			subs.put(bpe, newBpe);
793 			uriToSub.put(newBpe.getUri(), newBpe);
794 		}
795 	}
