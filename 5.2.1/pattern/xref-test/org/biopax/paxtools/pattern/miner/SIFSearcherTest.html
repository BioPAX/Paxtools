212 	private static String getPrintable(long time)
213 	{
214 		int div = 1000 * 60 * 60;
215 		int hours = (int) (time / div);
216 		time %= div;
217 		div /= 60;
218 		int minutes = (int) (time / div);
219 		time %= div;
220 		div /= 60;
221 		int seconds = (int) (time / div);
222 
223 		String s = hours > 0 ? hours + "h, " : "";
224 		s += minutes > 0 ? minutes + "m, " : "";
225 		s += seconds + "s";
226 		return s;
227 	}

229 	private void confirmPresenceOfUbiques(Model model, Blacklist blacklist)
230 	{
231 		int present = 0;
232 		int absent = 0;
233 		for (String ubique : blacklist.getListed())
234 		{
235 			if (model.getByID(ubique) != null) present++;
236 			else
237 			{
238 				absent++;
239 			}
240 		}
241 		System.out.println("absent ubique  = " + absent);
242 		System.out.println("present ubique = " + present);
243 	}

347 	public static void generate(String inputModelFile, String ubiqueIDFile, String outputFile)
348 		throws IOException
349 	{
350 		SimpleIOHandler h = new SimpleIOHandler();
351 		Model model = h.convertFromOWL(new FileInputStream(inputModelFile));
352 
353 		List<SIFInteraction> sifs = new ArrayList<SIFInteraction>(generate(model,
354 			new Blacklist(ubiqueIDFile)));
355 
356 		Collections.sort(sifs);
357 
358 		BufferedWriter writer = new BufferedWriter(new FileWriter(outputFile));
359 
360 		for (SIFInteraction sif : sifs)
361 		{
362 			writer.write(sif + "\n");
363 		}
364 
365 		writer.close();
366 	}
