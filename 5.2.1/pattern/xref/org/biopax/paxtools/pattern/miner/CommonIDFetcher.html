13  /**
14   * Tries to get Gene Symbols or UniProt IDs for genes
15   * and - display names for small molecules;
16   *
17   * This id-fetcher is mainly to use with the BioPAX pathway data
18   * from Pathway Commons (PC2) db (normalized and enriched with xrefs).
19   *
20   * @author Ozgun Babur et al.
21   */
22  public class CommonIDFetcher implements IDFetcher
23  {
24  	boolean useUniprotIDs = false;
25  
26  	public Set<String> fetchID(BioPAXElement ele)
27  	{
28  		Set<String> set = new HashSet<String>();
29  
30  		if (ele instanceof SmallMoleculeReference || ele instanceof SmallMolecule)
31  		{
32  			Named e = (Named) ele;
33  			//avoid shortened/incomplete names -
34  			if (e.getDisplayName() != null && !e.getDisplayName().contains("..."))
35  				set.add(e.getDisplayName());
36  			else if (e.getStandardName() != null && !e.getStandardName().contains("..."))
37  				set.add(e.getStandardName());
38  			else if (!e.getName().isEmpty()) {
39  				Set<String> names = new TreeSet<String>();
40  				for(String name : e.getName()) {
41  					if(!name.contains("..."))
42  						names.add(name);
43  				}
44  				set.add(names.toString());
45  			}
46  		}
47  		else if (useUniprotIDs && ele.getUri().startsWith("http://identifiers.org/uniprot/"))
48  		{
49  			set.add(ele.getUri().substring(ele.getUri().lastIndexOf("/") + 1));
50  		}
51  		else if (ele instanceof XReferrable)
52  		{
53  			for (Xref xr : ((XReferrable) ele).getXref())
54  			{
55  				if(xr instanceof PublicationXref)
56  					continue;
57  
58  				String db = xr.getDb();
59  				String id = xr.getId();
60  				if (db != null && id != null && !id.isEmpty())
61  				{
62  					db = db.toLowerCase();
63  					if (!useUniprotIDs && db.startsWith("hgnc"))
64  					{
65  						//valid id is either a HGNC:1234 ID or Symbol (gene name)
66  						String symbol = HGNC.getSymbol(id);
67  						if (symbol != null && !symbol.isEmpty())
68  						{
69  							set.add(symbol);
70  						}
71  					}
72  					else if (useUniprotIDs && db.startsWith("uniprot"))
73  					{
74  						set.add(id);
75  					}
76  				}
77  			}
78  		}
79  
80  		if (set.isEmpty()) {
81  			// a second try - getting mirtarbase ids for a nucleic acid...
82  			if (ele instanceof NucleicAcidReference || ele instanceof NucleicAcid)
83  			{
84  				for (Xref xr : ((XReferrable) ele).getXref()) {
85  					if(xr instanceof PublicationXref) continue;
86  
87  					String db = xr.getDb();
88  					String id = xr.getId();
89  					if (db != null && id != null && !id.isEmpty()) {
90  						db = db.toLowerCase();
91  						if (db.equals("mirbase sequence")) {
92  							set.add(id);
93  						}
94  					}
95  				}
96  			}
97  		}
98  
99  		return set;
100 	}
101 
102 	public void setUseUniprotIDs(boolean useUniprotIDs)
103 	{
104 		this.useUniprotIDs = useUniprotIDs;
105 	}
106 }
