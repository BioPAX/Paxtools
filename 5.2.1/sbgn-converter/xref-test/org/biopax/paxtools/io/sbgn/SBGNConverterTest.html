52  	@Test
53  	public void testSBGNConversion() throws Exception
54  	{
55  		String input = "/AR-TP53";
56  		InputStream in = getClass().getResourceAsStream(input + ".owl");
57  		Model level3 = handler.convertFromOWL(in);
58  		String out = "target" + input + ".sbgn";
59  		L3ToSBGNPDConverter conv = new L3ToSBGNPDConverter(blacklist, null, true);
60  		conv.writeSBGN(level3, out);
61  		File outFile = new File(out);
62  		SbgnUtil.isValid(outFile); //ignore CName warning
63  
64  		// Now read from "f" and put the result in "sbgn"
65  		Sbgn result = (Sbgn)unmarshaller.unmarshal (outFile);
66  
67  		// Assert that the sbgn result contains glyphs
68  		assertTrue(!result.getMap().getGlyph().isEmpty());
69  
70  		// Assert that compartments do not contain members inside
71  		for (Glyph g : result.getMap().getGlyph())
72  		{
73  			if (g.getClazz().equals("compartment"))
74  			{
75  				assertTrue(g.getGlyph().isEmpty());
76  				String label = g.getLabel().getText();
77  				assertFalse(label.matches("go:"));
78  				assertTrue(StringUtils.isAllLowerCase(label.substring(0,1)));
79  				assertTrue(g.getId().matches(NOSPACE)); //refs issue #44
80  			}
81  		}
82  
83  		// Assert that the id mapping is not empty.
84  		assertFalse(conv.getSbgn2BPMap().isEmpty());
85  
86  		for (Set<String> set : conv.getSbgn2BPMap().values())
87  		{
88  			assertFalse(set.isEmpty());
89  		}
90  	}

93  	public void testNestedComplexes() throws Exception
94  	{
95  		String input = "/translation-initiation-complex-formation";
96  		InputStream in = getClass().getResourceAsStream(input + ".owl");
97  		Model level3 = handler.convertFromOWL(in);
98  
99  		String out = "target" + input + ".sbgn";
100 		L3ToSBGNPDConverter conv = new L3ToSBGNPDConverter(null, null, true);
101 		conv.setFlattenComplexContent(false);
102 		conv.writeSBGN(level3, out);
103 		File outFile = new File(out);
104 		SbgnUtil.isValid(outFile); //ignore CName warning
105 
106 		// Now read from "f" and put the result in "sbgn"
107 		Sbgn result = (Sbgn)unmarshaller.unmarshal (outFile);
108 
109 		boolean hasNestedComplex = false;
110 		for (Glyph g : result.getMap().getGlyph())
111 		{
112 			if (g.getClazz().equals("complex"))
113 			{
114 				for (Glyph mem : g.getGlyph())
115 				{
116 					if (mem.getClazz().equals("complex"))
117 					{
118 						hasNestedComplex = true;
119 						break;
120 					}
121 				}
122 				if (hasNestedComplex) break;
123 			}
124 		}
125 
126 		assertTrue(hasNestedComplex);
127 	}
