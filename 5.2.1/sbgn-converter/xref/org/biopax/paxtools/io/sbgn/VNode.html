79       * Default Constructor, sets the geometry of the bounds which are attributes of this class
80       * @param g glyph
81       * */
82      public VNode(Glyph g)
83      {

85          LOGICAL_OPERATOR_BOUND = new Bound(15,15);
86          PROCESS_NODES_BOUND = new Bound(15,15);

88          MACROMOLECULE_BOUND = new Bound(48,25);
89          NUCLEIC_ACID_FEATURE_BOUND = new Bound(50,25);
90  
91          SIMPLE_CHEMICAL_BOUND = new Bound(48,20);
92          UNSPECIFIED_ENTITY_BOUND = new Bound(40,40);
93          PHENOTYPE_BOUND = new Bound(50,20);
94          TAG_BOUND = new Bound(50,20);
95          PERTURBING_AGENT_BOUND = new Bound(50,20);
96          COMPLEX_BOUND = new Bound(48,20);
97  
98          INFO_BOUND = new Bound(MAX_STATE_AND_INFO_WIDTH,MAX_STATE_AND_INFO_HEIGHT);
99          STATE_BOUND = new Bound(MAX_STATE_AND_INFO_WIDTH,MAX_STATE_AND_INFO_HEIGHT);
100 
101         stateGlyphs = new ArrayList<Glyph>();
102         infoGlyphs = new  ArrayList<Glyph>();
103 
104         this.glyph = g;
105 
106         /*
107 		 * need to add bbox objects
108 		 */
109         Bbox b = new Bbox();
110         this.glyph.setBbox(b);
111 
112         if (this.glyph.getClazz() == null)
113             this.glyph.setClazz(NONE);
114 
115         this.setSizeAccordingToClass();
116     }

118     /**
119      * Function that will take place when VNode objects will update in layout process of ChiLay
120      *
121      * @param lGraphObj LGraphObject for whom the update will take place.
122      */
123     public void update(LGraphObject lGraphObj)
124     {
125         if (lGraphObj instanceof CoSEGraph)
126         {
127             return;
128         }
129 
130         LNode lNode = (LNode)lGraphObj;
131 
132         this.glyph.getBbox().setX((float) lNode.getLeft());
133         this.glyph.getBbox().setY((float) lNode.getTop());
134         this.placeStateAndInfoGlyphs();
135     }

148     /**
149      * Chooses a proper bound for this VNode according to its class.
150      */
151     public void setSizeAccordingToClass()
152     {
153         String glyphClass = this.glyph.getClazz();
154 
155         //If glyph class is not specified return here
156         if (glyphClass.equalsIgnoreCase(NONE))
157             return;
158 
159         if (glyphClass.equalsIgnoreCase(SOURCE_AND_SINK))
160         {
161             setBounds(SOURCE_AND_SINK_BOUND.getWidth(), SOURCE_AND_SINK_BOUND.getHeight());
162         }
163 
164         else if (glyphClass.equalsIgnoreCase(AND) || glyphClass.equalsIgnoreCase(OR) || glyphClass.equalsIgnoreCase(NOT))
165         {
166             setBounds(LOGICAL_OPERATOR_BOUND.getWidth(), LOGICAL_OPERATOR_BOUND.getHeight());
167         }
168 
169         else if (glyphClass.equalsIgnoreCase(ASSOCIATION) || glyphClass.equalsIgnoreCase(DISSOCIATION) || glyphClass.equalsIgnoreCase(OMITTED_PROCESS) ||
170                 glyphClass.equalsIgnoreCase(UNCERTAIN_PROCESS) || glyphClass.equalsIgnoreCase(PROCESS))
171         {
172             setBounds(PROCESS_NODES_BOUND.getWidth(), PROCESS_NODES_BOUND.getHeight());
173         }
174 
175         else if (glyphClass.equalsIgnoreCase(SIMPLE_CHEMICAL))
176         {
177             setBounds(SIMPLE_CHEMICAL_BOUND.getWidth(), SIMPLE_CHEMICAL_BOUND.getHeight());
178         }
179 
180         else if (glyphClass.equalsIgnoreCase(UNSPECIFIED_ENTITY))
181         {
182             setBounds(UNSPECIFIED_ENTITY_BOUND.getWidth(), UNSPECIFIED_ENTITY_BOUND.getHeight());
183         }
184 
185         else if (glyphClass.equalsIgnoreCase(MACROMOLECULE))
186         {
187             setBounds(MACROMOLECULE_BOUND.getWidth(), MACROMOLECULE_BOUND.getHeight());
188         }
189 
190         else if (glyphClass.equalsIgnoreCase(NUCLEIC_ACID_FEATURE))
191         {
192             setBounds(NUCLEIC_ACID_FEATURE_BOUND.getWidth(), NUCLEIC_ACID_FEATURE_BOUND.getHeight());
193         }
194 
195         else if (glyphClass.equalsIgnoreCase(STATE_VARIABLE))
196         {
197             setBounds(STATE_BOUND.getWidth(), STATE_BOUND.getHeight());
198         }
199 
200         else if (glyphClass.equalsIgnoreCase(UNIT_OF_INFORMATION))
201         {
202             setBounds(INFO_BOUND.getWidth(), INFO_BOUND.getHeight());
203         }
204 
205         else if (glyphClass.equalsIgnoreCase(PHENOTYPE))
206         {
207             setBounds(PHENOTYPE_BOUND.getWidth(), PHENOTYPE_BOUND.getHeight());
208         }
209 
210         else if (glyphClass.equalsIgnoreCase(PERTURBING_AGENT))
211         {
212             setBounds(PERTURBING_AGENT_BOUND.getWidth(), PERTURBING_AGENT_BOUND.getHeight());
213         }
214 
215         else if (glyphClass.equalsIgnoreCase(TAG))
216         {
217             setBounds(TAG_BOUND.getWidth(), TAG_BOUND.getHeight());
218         }
219 
220         else if (glyphClass.equalsIgnoreCase(COMPLEX))
221         {
222             setBounds(COMPLEX_BOUND.getWidth(), COMPLEX_BOUND.getHeight());
223         }
224 
225         if( this.glyph.getClone() != null )
226         {
227             Bbox glyphBbox = this.glyph.getBbox();
228             setBounds(3*glyphBbox.getW()/4, 3*glyphBbox.getH()/4);
229         }
230 
231         if (glyphClass.equalsIgnoreCase(MACROMOLECULE) || glyphClass.equalsIgnoreCase(NUCLEIC_ACID_FEATURE) || glyphClass.equalsIgnoreCase(SIMPLE_CHEMICAL) || glyphClass.equalsIgnoreCase(COMPLEX))
232         {
233             updateSizeForStateAndInfo();
234         }
235

238     /**
239      * Calculates required width according to the given list state and info glyphs of this VNode.
240      * This method also previously computes the width and height of state and info glyphs
241      * according to their label.
242      *
243      * @param  stateORinfoList list that keeps state or info glyphs of this VNode
244      * @return new width that is adjusted so that all glyphs in stateORinfoList are included.
245      * */
246     public int calcReqWidthByStateAndInfos(List<Glyph> stateORinfoList)
247     {
248         int wholeSize = 0;
249         int count = 0;
250 
251         for (Glyph tmpGlyph: stateORinfoList)
252         {
253             String text;
254 
255             if (tmpGlyph.getState() != null)
256             {
257                 text = tmpGlyph.getState().getValue();
258 
259                 if (tmpGlyph.getState().getVariable() != null &&
260                         tmpGlyph.getState().getVariable().length() > 0)
261                 {
262                     if(tmpGlyph.getState().getVariable() != null)
263                         text += "@" + tmpGlyph.getState().getVariable();
264                 }
265             }
266             else if (tmpGlyph.getLabel() != null)
267             {
268                 text = tmpGlyph.getLabel().getText();
269             }
270             else
271             {
272                 throw new RuntimeException("Encountered an information glyph with no state " +
273                         "variable (as modification boxes should have) and no label (as molecule type " +
274                         "boxed should have). glyph = " + tmpGlyph);
275             }
276 
277             int numOfUpper = 0;
278             int numOfLower = 0;
279 
280             for (int i = 0; i < text.length(); i++)
281             {
282                 if (Character.isLowerCase(text.charAt(i)))
283                 {
284                     numOfLower++;
285 
286                 } else
287                     numOfUpper++;
288             }
289 
290             Bbox b = new Bbox();
291             tmpGlyph.setBbox(b);
292 
293             //Set width
294             float requiredSize = numOfLower * LOWERCASE_LETTER_PIXEL_WIDTH + numOfUpper * UPPERCASE_LETTER_PIXEL_WIDTH;
295             if (requiredSize < MAX_STATE_AND_INFO_WIDTH)
296                 tmpGlyph.getBbox().setW(requiredSize);
297             else
298                 tmpGlyph.getBbox().setW(STATE_BOUND.width);
299 
300             //Set height
301             tmpGlyph.getBbox().setH(MAX_STATE_AND_INFO_HEIGHT);
302 
303             if (count < MAX_INFO_BOX_NUMBER / 2)
304                 wholeSize += tmpGlyph.getBbox().getW();
305 
306             count++;
307 
308         }
309 
310         return wholeSize;
311     }

313     /**
314      * 	If glyph attribute of this VNode object includes any "state of information" or "unit of information" glyphs, this method updates the
315      *  size of VNode accordingly.
316      * */
317     public void updateSizeForStateAndInfo()
318     {
319         // Find all state and info glyphs
320         for (Glyph glyph : this.glyph.getGlyph())
321         {
322             if (glyph.getClazz() == STATE_VARIABLE)
323             {
324                 stateGlyphs.add(glyph);
325             }
326             else if (glyph.getClazz() == UNIT_OF_INFORMATION)
327             {
328                 infoGlyphs.add(glyph);
329             }
330         }
331 
332         //Calculate "state of information" glyphs' sizes
333         int wholeWidthOfStates = calcReqWidthByStateAndInfos(stateGlyphs);
334         int wholeWidthOfInfos  = calcReqWidthByStateAndInfos(infoGlyphs);
335 
336         // Calculate  positions
337         int numOfStates = stateGlyphs.size();
338         int numOfInfos = infoGlyphs.size();
339 
340 
341 
342         //set a maximum width to glyph according to state and info boxes
343         int totNumStateInfo = numOfInfos + numOfStates;
344         int multiplier = totNumStateInfo <= 2 ? 2 : 3;
345         float requiredWidth = multiplier * OFFSET_BTW_INFO_GLYPHS + (multiplier-1) * MAX_STATE_AND_INFO_WIDTH;
346 
347         //Adjust heights so that info box offsets are taken into account in layout.
348         if(totNumStateInfo > 0 )
349             this.glyph.getBbox().setH(this.glyph.getBbox().getH()+MAX_STATE_AND_INFO_HEIGHT/2);
350 
351         if (this.glyph.getBbox().getW() < requiredWidth  )
352         {
353             this.glyph.getBbox().setW(requiredWidth);
354         }
355     }

357     /**
358      * Places state and info glyphs of this node
359      * */
360     public void placeStateAndInfoGlyphs() {
361         int numOfStates = stateGlyphs.size();
362         int numOfInfos = infoGlyphs.size();
363 
364 
365         //Adjust heights so that info box offsets are taken into account in layout.
366         if(numOfStates > 0 || numOfInfos > 0)
367             this.glyph.getBbox().setH(this.glyph.getBbox().getH()-MAX_STATE_AND_INFO_HEIGHT/2);
368 
369         float parent_y_up = this.glyph.getBbox().getY()-INFO_BOUND.height/2;
370         float parent_y_bot = this.glyph.getBbox().getY()+this.glyph.getBbox().getH()-INFO_BOUND.height/2;;
371         float parent_x_up = this.glyph.getBbox().getX();
372         float parentWidth = this.glyph.getBbox().getW();
373         String parentID = this.glyph.getId();
374 
375         int maxNumberOfStates =   stateGlyphs.size();
376         int maxNumberOfInfos =  infoGlyphs.size();
377         if (maxNumberOfInfos + maxNumberOfStates > 4)
378         {
379             if(maxNumberOfInfos < 2 || maxNumberOfStates < 2)
380             {
381                 int minimum = Math.min(maxNumberOfInfos,maxNumberOfStates);
382 
383                 if (maxNumberOfInfos - minimum == 0)
384                 {
385                     maxNumberOfInfos = minimum;
386                     maxNumberOfStates = maxNumberOfStates - maxNumberOfInfos;
387                 }
388                 else
389                 {
390                     maxNumberOfStates = minimum;
391                     maxNumberOfInfos = maxNumberOfInfos - maxNumberOfStates;
392                 }
393             }
394         }
395         //else normal placement 
396         
397         //Some variables for placement of state and info boxes
398         int totalNumberOfStateAndInfos = maxNumberOfInfos+maxNumberOfStates;
399         Glyph tmpglyph = null;
400         int infoIndex = 0;
401         int stateIndex = 0;
402         int topUsedWidth = 0;
403         int bottomUsedWidth = 0;
404         int additionalStateInfoOffset = 0;
405         boolean placeTopFlag = true;
406 
407         //Place state and info glyphs
408         for (int i = 0; i < totalNumberOfStateAndInfos; i++)
409         {
410             //Adjust offsets between state and info boxes
411             int offsetMultiplier = i % 4 <= 1 ? 1 : 2;
412 
413             if (maxNumberOfInfos - i > 0 && infoGlyphs.size() > 0)
414                 tmpglyph =  infoGlyphs.get(infoIndex++);
415             else if (stateGlyphs.size() > 0)
416                 tmpglyph = stateGlyphs.get(stateIndex++);
417 
418             //Top placement
419             if (placeTopFlag)
420             {
421                 if( totalNumberOfStateAndInfos <= 2 )
422                 {
423                     tmpglyph.getBbox().setX(parent_x_up+parentWidth/2-tmpglyph.getBbox().getW()/2);
424                     tmpglyph.getBbox().setY(parent_y_up - additionalStateInfoOffset);
425                     //set dummy id
426                     tmpglyph.setId(parentID + ".info." + (i+1) );
427                 }
428                 else
429                 {
430                     //set dummy id
431                     tmpglyph.setId(parentID + ".info." + (i+1) );
432 
433                     tmpglyph.getBbox().setX(parent_x_up + offsetMultiplier * OFFSET_BTW_INFO_GLYPHS + topUsedWidth);
434                     tmpglyph.getBbox().setY(parent_y_up - additionalStateInfoOffset);
435                 }
436 
437                 topUsedWidth += tmpglyph.getBbox().getW();
438             }
439             //bottom placement
440             else
441             {
442                 if(( i % 2 == 1 ) && ( totalNumberOfStateAndInfos == 2 || totalNumberOfStateAndInfos == 3 ))
443                 {
444                     tmpglyph.getBbox().setX(parent_x_up+parentWidth/2-tmpglyph.getBbox().getW()/2);
445                     tmpglyph.getBbox().setY(parent_y_bot + additionalStateInfoOffset);
446                     //set dummy id
447                     tmpglyph.setId(parentID + ".state." + (i+1) );
448                 }
449                 else
450                 {
451                     //set dummy id
452                     tmpglyph.setId(parentID + ".state." + (i+1) );
453                     tmpglyph.getBbox().setX(parent_x_up+ offsetMultiplier *OFFSET_BTW_INFO_GLYPHS + bottomUsedWidth);
454                     tmpglyph.getBbox().setY(parent_y_bot + additionalStateInfoOffset);
455                 }
456 
457                 bottomUsedWidth += tmpglyph.getBbox().getW();
458             }
459 
460             // Alternate between placing top and bottom
461             placeTopFlag = !placeTopFlag;
462 
463             //More than 4 state and info glyps are placed with an offset on top of previously placed state and info boxes
464             //This can be easily filtered out on the application side where the resultant graphs of this layout is used.
465             if ( ( i + 1 ) % 4 == 0 )
466                 additionalStateInfoOffset += NON_SUPPORTED_GLYPH_OFFSET;
467         }
