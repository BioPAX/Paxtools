27  	@Test
28  	public final void testFilter() throws IOException {
29  		SimpleIOHandler io = new SimpleIOHandler();
30  		Model model = io.convertFromOWL(
31  			getClass().getClassLoader()
32  				.getResourceAsStream("L2/biopax-example-short-pathway.owl"));
33  		
34  		physicalEntityParticipant pep31 = (physicalEntityParticipant) model.getByID(model.getXmlBase() + "physicalEntityParticipant31");
35  		assertTrue(pep31 instanceof physicalEntityParticipant);
36  		assertTrue(pep31.getPHYSICAL_ENTITY() instanceof protein);
37  		assertNotNull(pep31.getCELLULAR_LOCATION());
38  		
39  		model = (new LevelUpgrader()).filter(model);
40  		
41  		if (model != null) {
42  			io.convertToOWL(model, new FileOutputStream(
43  					getClass().getClassLoader().getResource("").getFile() + File.separator + "converted.owl"));
44  		}
45  		
46  		
47  		SmallMolecule p9 = (SmallMolecule) model.getByID(model.getXmlBase() + "physicalEntityParticipant9");
48  		Protein p31 = (Protein) model.getByID(model.getXmlBase() + "physicalEntityParticipant31");
49  		assertTrue(p9 instanceof SmallMolecule);
50  		assertTrue(p31 instanceof Protein);
51  		assertNotNull(p9.getCellularLocation());
52  		assertNotNull(p31.getCellularLocation());
53  		assertEquals(p9.getCellularLocation(), p31.getCellularLocation());
54  	}
