155 	@Test
156 	public final void testBioPAXDocument()
157 	{
158 
159 		String ID_COMPARTMENT_1 = "compartment_1";
160 		String ID_PROTEIN_1 = "PROTEIN_1";
161 		String ID_PROTEIN_2 = "PROTEIN_2";
162 		String ID_PROTEIN_3 = "PROTEIN_3";
163 
164 		String ID_PROTEIN_REFERENCE_1 = "PROTEIN_REFERENCE_1";
165 		String ID_PROTEIN_REFERENCE_2 = "PROTEIN_REFERENCE_2";
166 		String ID_PROTEIN_REFERENCE_3 = "PROTEIN_REFERENCE_3";
167 
168 		BioPAXFactory level3Factory = BioPAXLevel.L3.getDefaultFactory();
169 		Model biopaxModel = level3Factory.createModel();
170 
171 		// Create a compartment
172 		CellularLocationVocabulary clv =
173 				biopaxModel.addNew(CellularLocationVocabulary.class, ID_COMPARTMENT_1);
174 		String compartmentName = "golgi";
175 		clv.addTerm(compartmentName);
176 
177 
178 		// Create three proteins in golgi
179 		Protein p1 = biopaxModel.addNew(Protein.class, ID_PROTEIN_1);
180 		p1.setDisplayName("PE-Pro-BACE-1");
181 		Protein p2 = biopaxModel.addNew(Protein.class, ID_PROTEIN_2);
182 		p2.setDisplayName("PE-BACE-1");
183 		Protein p3 = biopaxModel.addNew(Protein.class, ID_PROTEIN_3);
184 		p3.setDisplayName("Furin");
185 		p1.setCellularLocation(clv);
186 		p2.setCellularLocation(clv);
187 		p3.setCellularLocation(clv);
188 
189 		// Create entity references for the proteins
190 		ProteinReference pr1 = biopaxModel.addNew(ProteinReference.class, ID_PROTEIN_REFERENCE_1);
191 		pr1.setStandardName("Pro-BACE-1");
192 		p1.setEntityReference(pr1);
193 		Stoichiometry stoichiometry1 = biopaxModel.addNew(Stoichiometry.class, "ST1");
194 		stoichiometry1.setPhysicalEntity(p1);
195 		stoichiometry1.setStoichiometricCoefficient(1);
196 
197 		ProteinReference pr2 = biopaxModel.addNew(ProteinReference.class, ID_PROTEIN_REFERENCE_2);
198 		pr2.setStandardName("BACE-1");
199 		p2.setEntityReference(pr2);
200 		Stoichiometry stoichiometry2 = biopaxModel.addNew(Stoichiometry.class, "ST2");
201 		stoichiometry2.setPhysicalEntity(p2);
202 		stoichiometry2.setStoichiometricCoefficient(1);
203 
204 		ProteinReference pr3 = biopaxModel.addNew(ProteinReference.class, ID_PROTEIN_REFERENCE_3);
205 		pr3.setStandardName("Furin");
206 		p3.setEntityReference(pr3);
207 		Stoichiometry stoichiometry3 = biopaxModel.addNew(Stoichiometry.class, "ST3");
208 		stoichiometry3.setPhysicalEntity(p3);
209 		stoichiometry3.setStoichiometricCoefficient(1);
210 
211 		// Create a reaction involving the three proteins
212 		BiochemicalReaction r = biopaxModel.addNew(BiochemicalReaction.class, "r1");
213 		r.addLeft(p1);
214 		r.addRight(p2);
215 
216 		Control c = biopaxModel.addNew(Catalysis.class, "cat1");
217 		c.setControlType(ControlType.ACTIVATION);
218 		c.addControlled(r);
219 
220 
221 		// Write out the owl file
222 		try
223 		{
224 			System.out.println("test");
225 			File f = new File(getClass().getClassLoader()
226 					.getResource("").getPath() + File.separator + "test.owl");
227 			FileOutputStream anOutputStream = new FileOutputStream(f);
228 			outputModel(biopaxModel, anOutputStream);
229 		}
230 		catch (Exception e)
231 		{
232 			e.printStackTrace();
233 		}
234 
235 	}

277     public void readWriteReadCheckModel(SimpleIOHandler io) throws IOException
278     {
279         // Read
280 		Model model = getL3Model(io);
281 
282 		//(since 5.1.0; no assertions here, for these are in-memory only)
283 		model.setName("A test Level3 model");
284 		model.setUri("http://biopax.org/test/io/l3model");
285 
286         // Write
287         ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
288         io.convertToOWL(model, outputStream);
289         outputStream.flush();
290 
291         // Read
292         ByteArrayInputStream inputStream = new ByteArrayInputStream(outputStream.toByteArray());
293         outputStream.close();
294         Model newModel = io.convertFromOWL(inputStream);
295         assertTrue(newModel != null);
296         inputStream.close();
297 
298         // Let's do a simple check to see if whether new model is OK
299         for(BioPAXElement bpe: model.getObjects())
300         {
301             assertTrue(newModel.containsID(bpe.getUri()));
302         }
303     }

321         BioPAXIOHandler io = new SimpleIOHandler(); //auto-detects level
322
