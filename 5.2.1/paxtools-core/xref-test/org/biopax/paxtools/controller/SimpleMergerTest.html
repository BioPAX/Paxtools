103 	@Test
104 	public final void testMergeToInconsistentModel() {
105 		// Let's merge an object having one xref into a model that
106 		// implicitly "contains" a different xref with the same URI;
107 		// merged model will explicitly contain only the xref
108 		// of the just merged object (not the original, implicit one;
109 		// and therefore the model is still not self-integral;
110 		// let's begin...
111 
112 		BioPAXFactory factory = BioPAXLevel.L3.getDefaultFactory();
113 		Model model = factory.createModel(); //target model
114 
115 		//create an xref outside the model,
116 		Xref mref =factory.create(UnificationXref.class, "10090");
117 		mref.setDb("Taxonomy");
118 		mref.setId("10090");
119 		//and a BioSource - in the model
120 		BioSource mouse = model.addNew(BioSource.class, "mouse");
121 		mouse.addXref(mref);
122 		//mref is owned by the mouse but is not in the model explicitly
123 		assertEquals(1, model.getObjects().size());
124 
125 		//Create two more objects outside the model, one owns the other -
126 		Xref href = factory.create(UnificationXref.class, "9606");
127 		href.setDb("Taxonomy");
128 		href.setId("9606");
129 		BioSource human = factory.create(BioSource.class, "human");
130 		human.addXref(href);
131 
132 		//do merge
133 		SimpleMerger merger = new SimpleMerger(SimpleEditorMap.L3);
134 		merger.merge(model, human);
135 
136 		assertEquals(3, model.getObjects().size()); //there's no mref - Xref 10090!
137 		assertNull(model.getByID("10090"));
138 
139 
140 		// now, if both xrefs had the same URI, such as 'ref',
141 		// then 'human' would become pointing to the mouse taxID,
142 		// or vice versa, depending on whether the initial model
143 		// was self-integral or not.
144 
145 		model = factory.createModel(); //fresh target model
146 
147 		//create an xref outside the model,
148 		mref = factory.create(UnificationXref.class, "ref");
149 		mref.setDb("Taxonomy");
150 		mref.setId("10090");
151 		mouse = model.addNew(BioSource.class, "mouse");
152 		mouse.addXref(mref);
153 
154 		//Create two more objects outside the model, one owns the other -
155 		href = factory.create(UnificationXref.class, "ref"); //on purpose - same URI, different xref!
156 		href.setDb("Taxonomy");
157 		href.setId("9606");
158 		human = factory.create(BioSource.class, "human");
159 		human.addXref(href);
160 
161 		merger.merge(model, human);
162 
163 		assertEquals(3, model.getObjects().size()); // - there's no Xref 10090
164 		Xref x = (UnificationXref)model.getByID("ref");
165 		assertNotNull(x);
166 		assertEquals("9606", x.getId());
167 		BioSource bs = (BioSource) model.getByID("mouse");
168 		assertNotNull(bs);
169 		assertEquals("10090", bs.getXref().iterator().next().getId()); //but that xref is not in the model
170 		assertEquals("9606", human.getXref().iterator().next().getId());
171 
172 		//But.
173 		//If the initial target model contained both the mouse and mref,
174 		model = factory.createModel(); //fresh target model
175 		mref = model.addNew(UnificationXref.class, "ref");
176 		mref.setDb("Taxonomy");
177 		mref.setId("10090");
178 		mouse = model.addNew(BioSource.class, "mouse");
179 		mouse.addXref(mref);
180 		//then
181 		merger.merge(model, human);
182 		//makes the human have mouse xref -
183 		assertEquals(3, model.getObjects().size());
184 		x = (UnificationXref)model.getByID("ref");
185 		assertNotNull(x);
186 		assertEquals("10090", x.getId()); //aha
187 		bs = (BioSource) model.getByID("mouse");
188 		assertNotNull(bs);
189 		assertEquals("10090", bs.getXref().iterator().next().getId()); //ok
190 		assertNotNull(model.getByID("human"));
191 		assertEquals("10090", human.getXref().iterator().next().getId());
192 		//- 'human' (as well as 'mouse') BioSource now points to mouse's taxID xref!
193 
194 		//PS:
195 		//in order to make a model self-integral, one can try merging it to itself,
196 		//which make all implicit objects become explicit members of the model, or fails,
197 		//if there are different objects using same URIs somewhere...
198 	}

200 	@Test
201 	public final void testMergeWithoutFilter() {
202 		BioPAXFactory factory = BioPAXLevel.L3.getDefaultFactory();
203 		Model model = factory.createModel();
204 		//create a PR, BS, etc., in the (target) model
205 		BioSource human = model.addNew(BioSource.class, "human");
206 		human.setStandardName("Homo sapiens"); //no displayName, no other names
207 		UnificationXref uxHuman = model.addNew(UnificationXref.class, "9606");
208 		human.addXref(uxHuman);
209 		ProteinReference pr = model.addNew(ProteinReference.class, "pr");
210 		pr.setDisplayName("one");
211 		pr.setOrganism(human);
212 
213 		//another PR with the SAME URI to be merged into the target model
214 		ProteinReference pr1 = factory.create(ProteinReference.class, "pr");
215 		Xref ux = factory.create(UnificationXref.class, "ux");
216 		ux.setDb("UniProt");
217 		ux.setId("P12345");
218 		pr1.addName("one");
219 		pr1.addName("two");
220 		pr1.addXref(ux);
221 		UnificationXref uxHuman1 = factory.create(UnificationXref.class, "9606"); //also, the same URI as for uxHuman above
222 		BioSource human1 = factory.create(BioSource.class, "human"); //same URI as for the other BioSource above
223 		human1.addXref(uxHuman1);
224 		human1.setDisplayName("human"); //but - no standardName nor other names are set
225 		pr1.setOrganism(human1);
226 		PublicationXref px = factory.create(PublicationXref.class, "publication");
227 		human1.addXref(px);
228 
229 		//merge NOT using any filters (in the Constructor)
230 		SimpleMerger merger = new SimpleMerger(SimpleEditorMap.L3);
231 		merger.merge(model, pr1);
232 
233 		assertEquals(5, model.getObjects().size());
234 		ProteinReference mergedPr = (ProteinReference) model.getByID("pr");
235 		assertTrue(mergedPr.getXref().isEmpty()); //xref wasn't copied
236 		assertEquals("one", mergedPr.getDisplayName());
237 		assertEquals(1, mergedPr.getName().size()); //name "two" wasn't copied (when no Filter used)
238 		assertEquals("Homo sapiens", mergedPr.getOrganism().getStandardName());
239 		assertNull(mergedPr.getOrganism().getDisplayName());
240 		assertEquals(1, mergedPr.getOrganism().getName().size()); //there's only displayName ('name' is a sub-prop. of that)
241 		assertNotNull(model.getByID("publication")); //the pub.xref was merged
242 		// but it's dangling, because
243 		assertEquals(1, mergedPr.getOrganism().getXref().size());
244 		assertEquals(uxHuman, mergedPr.getOrganism().getXref().iterator().next());
245 	}

248 	@Test
249 	public final void testMergeWithFilter() {
250 		BioPAXFactory factory = BioPAXLevel.L3.getDefaultFactory();
251 		Model model = factory.createModel();
252     	//create a PR, BS, etc., in the (target) model
253 		BioSource human = model.addNew(BioSource.class, "human");
254 		human.setStandardName("Homo sapiens"); //no displayName, no other names
255 		UnificationXref uxHuman = model.addNew(UnificationXref.class, "9606");
256 		uxHuman.setDb("taxonomy");
257 		uxHuman.setId("9606");
258 		human.addXref(uxHuman);
259     	ProteinReference pr = model.addNew(ProteinReference.class, "pr");
260     	pr.setDisplayName("one");
261 		pr.setOrganism(human);
262     	
263     	//another PR with the same URI to be merged into the target model
264     	ProteinReference pr1 = factory.create(ProteinReference.class, "pr");
265 		Xref ux = factory.create(UnificationXref.class, "ux");
266     	ux.setDb("UniProt"); 
267     	ux.setId("P12345"); 
268     	pr1.addName("one");
269     	pr1.addName("two");
270     	pr1.addXref(ux);
271 		UnificationXref uxHuman1 = factory.create(UnificationXref.class, "9606"); //use the same URI as for uxHuman above
272 		uxHuman1.setDb("Taxonomy"); //Capitalized, unlike the uxHuman above
273 		uxHuman1.setId("9606");
274     	BioSource human1 = factory.create(BioSource.class, "human"); //same URI as for the other BioSource above
275     	human1.addXref(uxHuman1);
276 		human1.setDisplayName("human"); //but - no standardName nor other names are set
277 		pr1.setOrganism(human1);
278 		PublicationXref px = factory.create(PublicationXref.class, "publication");
279 		px.setId("00000");
280 		px.setDb("PubMed");
281 		human1.addXref(px);
282 
283     	// Merge using a Filter, which
284     	// enables copying mul. card. property values of same-URI source EntityReference objects
285     	// (for all other types, if the target model has a same-type object with the same URI as in the sources
286     	// , incl. nested objects, then )
287     	SimpleMerger merger = new SimpleMerger(SimpleEditorMap.L3, new Filter<BioPAXElement>() {
288 			public boolean filter(BioPAXElement object) {
289 				return object instanceof EntityReference;
290 			}
291 		});
292     	merger.merge(model, pr1); 
293     	
294     	assertEquals(5, model.getObjects().size());
295     	ProteinReference mergedPr = (ProteinReference) model.getByID("pr");
296     	assertFalse(mergedPr.getXref().isEmpty());
297     	assertEquals("one", mergedPr.getDisplayName());
298     	assertEquals(2, mergedPr.getName().size());
299 
300 		assertEquals("Homo sapiens", mergedPr.getOrganism().getStandardName());
301 		assertNull(mergedPr.getOrganism().getDisplayName());
302 		//and, because BioSource is not considered in the Filter (see SimpleMerger's constructor parameter),
303 		// the 'human' name and publication xref px were not copied from source (human1) to the target (human) BioSource:
304 		assertEquals(1, mergedPr.getOrganism().getName().size());
305 		assertEquals(1, mergedPr.getOrganism().getXref().size());
306 		px = (PublicationXref)model.getByID("publication");
307 		assertNotNull(px); //the pub.xref was merged to the target model,
308 		// but it's dangling, because -
309 		for(XReferrable xReferrable : px.getXrefOf())
310 			assertFalse(model.contains(xReferrable));
311 
312 		// Well.
313 		// Next, merge using another Filter (adding BioSource type there)
314 		merger = new SimpleMerger(SimpleEditorMap.L3, new Filter<BioPAXElement>() {
315 			public boolean filter(BioPAXElement object) {
316 				return object instanceof EntityReference
317 						|| object instanceof BioSource; //i.e., enable copying BioSource's props too
318 			}
319 		});
320 		model = factory.createModel(); //a fresh empty model
321 		model.add(pr);
322 		model.add(human);
323 		model.add(uxHuman);
324 
325 		merger.merge(model, pr1);
326 
327 		assertEquals(5, model.getObjects().size());
328 		mergedPr = (ProteinReference) model.getByID("pr");
329 		assertFalse(mergedPr.getXref().isEmpty());
330 		assertEquals("one", mergedPr.getDisplayName());
331 		assertEquals(2, mergedPr.getName().size());
332 		assertEquals("Homo sapiens", mergedPr.getOrganism().getStandardName()); //it's the BioSource from orig. 'pr'
333 		assertNull(mergedPr.getOrganism().getDisplayName()); //single-cardinality prop. were not touched
334 		//and, because BioSource is now part of the Filter,
335 		//the 'human' (name) and publication xref px are copied from human1 to human BioSource:
336 		assertEquals(2, mergedPr.getOrganism().getName().size());
337 		assertTrue(mergedPr.getOrganism().getName().contains("human")); //originally, it wasn't there
338 		assertEquals(2, mergedPr.getOrganism().getXref().size()); //both xrefs
339 		px = (PublicationXref)model.getByID("publication");
340 		assertNotNull(px); //the pub.xref was copied; it's NOT dangling -
341 		assertTrue(mergedPr.getOrganism().getXref().contains(px));
342 		//the same URI uni.xref did actually not overwrite the original uxHuman (having lowercase "taxonomy" as its db)
343 		assertTrue(mergedPr.getOrganism().getXref().contains(uxHuman));
344 		assertFalse(mergedPr.getOrganism().getXref().contains(uxHuman1));
345 	}
