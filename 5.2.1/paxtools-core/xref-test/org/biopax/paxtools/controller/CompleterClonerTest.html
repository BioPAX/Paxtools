17  public class CompleterClonerTest
18  {
19  	private Model model;
20  	private String pwUri;
21  
22  	@Before
23  	public void init() {
24  		MockFactory mock= new MockFactory(BioPAXLevel.L3);
25  		model = mock.createModel();
26  
27  		Pathway pw = mock.create(model,Pathway.class,1)[0];
28  		pwUri = pw.getUri();
29  		BiochemicalReaction[] rx = mock.create(model, BiochemicalReaction.class,2);
30  		Control[] co = mock.create(model, Control.class,2);
31  		PathwayStep[] ps = mock.create(model,PathwayStep.class,2);
32  		Protein[] p = mock.create(model,Protein.class,2);
33  		SmallMolecule[] sm = mock.create(model,SmallMolecule.class,2);
34  
35  		pw.addPathwayComponent(co[0]);
36  		pw.addPathwayOrder(ps[0]);
37  		mock.bindArrays(mock.editor("controlled",Control.class),co,rx);
38  		mock.bindArrays(mock.editor("controller",Control.class),co,p);
39  		mock.bindArrays(mock.editor("right",BiochemicalReaction.class),rx,sm);
40  		ps[0].addNextStep(ps[1]);
41  		mock.bindArrays(mock.editor("stepProcess",PathwayStep.class),ps,rx);
42  		mock.bindArrays(mock.editor("stepProcess",PathwayStep.class),ps,co);
43  	}
44  
45  
46  	@Test
47  	public void testCompleteAndClone() throws Exception
48  	{
49  		Completer completer = new Completer(SimpleEditorMap.L3);
50  		Set<BioPAXElement> completed = completer.complete(Collections.singleton(model.getByID(pwUri)));
51  		assertTrue(completed.size()>1);
52  
53  		//PathwayStep1 should not be there if @AutoComplete(forward=false) has any effect on getNextStep()
54  		Set<PathwayStep> steps = new ClassFilterSet<BioPAXElement,PathwayStep>(completed,PathwayStep.class);
55  		assertEquals(1, steps.size());
56  		//There should be only one BiochemicalReaction that belongs to PathwayStep0
57  		Set<Conversion> rxs = new ClassFilterSet<BioPAXElement,Conversion>(completed,Conversion.class);
58  		assertEquals(1, rxs.size());
59  	}
