25  	@Test
26  	public final void testMergeAndReplace() {
27  		BioPAXFactory factory = BioPAXLevel.L3.getDefaultFactory();
28  		Model m = factory.createModel();
29  		
30  		Complex c = factory.create(Complex.class, "c1");
31  		Protein p1 = factory.create(Protein.class, "p1");
32  		Protein p2 = factory.create(Protein.class, "p2");
33  		
34  		ProteinReference pr1 = factory.create(ProteinReference.class, "pr1");
35  		UnificationXref x1 = factory.create(UnificationXref.class, "x1");		
36  		c.addComponent(p1);
37  		c.addComponent(p2);
38  		c.setDisplayName("complex");
39  		p1.setEntityReference(pr1);
40  		p1.setDisplayName("p1");
41  		pr1.addXref(x1);
42  		pr1.setDisplayName("pr1");
43  		x1.setId("1");
44  		x1.setDb("geneid");
45  		
46  		ProteinReference pr2 = factory.create(ProteinReference.class, "pr2");
47  		UnificationXref x2 = factory.create(UnificationXref.class, "x2");
48  		p2.setEntityReference(pr2);
49  		p2.setDisplayName("p2");
50  		pr2.addXref(x2);
51  		pr2.setDisplayName("pr2");
52  		x2.setId("2");
53  		x2.setDb("geneid");
54  		
55  		m.add(c);
56  		m.add(p1);
57  		m.add(p2);
58  		assertEquals(3, m.getObjects().size());
59  		assertEquals(2, c.getComponent().size());
60  
61  		
62  		// only 3 elements were added explicitly, but they have dependents -
63  		m.repair();
64  		
65  		assertEquals(7, m.getObjects().size());
66  		assertTrue(m.contains(x1));
67  		assertTrue(m.contains(x2));
68  		
69  		// create a new protein (the replacement)
70  		Protein p3 = factory.create(Protein.class, "p3");
71  		ProteinReference pr3 = factory.create(ProteinReference.class, "pr3");
72  		pr3.addXref(x2); // intentionally use same xref!
73  		pr3.setDisplayName("pr3");
74  		p3.setEntityReference(pr3);
75  		
76  		// do replace (replaces one element); 
77  		m.replace(p2, p3);	
78  		assertTrue(m.contains(p3)); // added!
79  		assertFalse(m.contains(p2)); // removed!		
80  
81  		assertEquals(7, m.getObjects().size()); // unchanged!
82  		assertEquals(2, c.getComponent().size());
83  		
84  		assertTrue(c.getComponent().contains(p1)); // untouched!
85  		assertTrue(c.getComponent().contains(p3)); // new!
86  		assertFalse(c.getComponent().contains(p2)); 
87  		
88  		//old children are still there
89  		assertTrue(m.contains(pr1));
90  		assertTrue(m.contains(pr2));
91  		assertTrue(m.contains(x1));
92  		assertTrue(m.contains(x2));
93  		
94  		// new children - not yet
95  		// (replace is 'shallow', does not affect children of replaced objects)
96  		assertFalse(m.contains(pr3));
97  		
98  		// now - repair will find all new children
99  		m.repair();
100 		
101 		assertEquals(8, m.getObjects().size()); // + pr3
102 		assertTrue(m.contains(pr3)); // added!
103 		
104 		assertTrue(m.contains(pr2)); // not deleted (may be dangling now)!
105 		assertTrue(m.contains(x2)); // not deleted (may be dangling now)!
106 		assertTrue(m.contains(pr1)); // not deleted (may be dangling now)!
107 		assertTrue(m.contains(x1)); // not deleted (may be dangling now)!
108 		
109 		// delete dangling
110 		ModelUtils.removeObjectsIfDangling(m, ProteinReference.class);
111 		
112 		// pr2 - removed
113 		assertEquals(7, m.getObjects().size());
114 		assertFalse(m.contains(pr2)); // deleted!
115 
116 		// ok to replace with null
117 		m.replace(p3, null);
118 		assertEquals(1, c.getComponent().size());
119 		ModelUtils.removeObjectsIfDangling(m, UtilityClass.class);
120 		assertFalse(m.contains(pr3)); // deleted!
121 		assertFalse(m.contains(x2)); // deleted!
122 		
123 		try {
124 			m.replace(pr1, x1); // fails, no effect on the model
125 			fail("Must fail!");
126 		} catch (IllegalBioPAXArgumentException e) {
127 			// fine!
128 		}
129 		// m is unchanged
130 		assertTrue(m.contains(pr1));
131 		assertEquals(pr1, p1.getEntityReference());
132 	}

153 	public final void testRemoveObjectsIfDangling() {
154 		Model model = (new SimpleIOHandler()).convertFromOWL(
155 			getClass().getClassLoader().getResourceAsStream("L3" + File.separator + "hcyc.owl"));
156 		assertEquals(6, model.getObjects().size());
157 		
158 		//there is no dangling util. class objects at the beginning
159 		ModelUtils.removeObjectsIfDangling(model, UtilityClass.class);
160 		assertEquals(6, model.getObjects().size());
161 		
162 		SmallMolecule sm1 = (SmallMolecule) model.getByID("SM1");
163 		SmallMoleculeReference smr1 = (SmallMoleculeReference) model.getByID("SMR1");
164 		SmallMoleculeReference a = (SmallMoleculeReference) model.getByID("A");
165 		
166 		assertNotNull(smr1);
167 		assertTrue(smr1.getMemberEntityReference().contains(a));
168 		
169 		// removing just SMR1 from SM1 and model makes both it and A dangling (though A is still member of SMR1)!
170 		sm1.setEntityReference(null); //unklinks SMR1
171 		model.remove(smr1);
172 		assertTrue(smr1.getEntityReferenceOf().isEmpty());
173 		
174 		// call to remove dangling things (but not B, which also belongs from SMR2, SM2)
175 		ModelUtils.removeObjectsIfDangling(model, UtilityClass.class);
176 		
177 		assertNull(model.getByID("SMR1"));		
178 		assertNull(model.getByID("A"));	
179 		SmallMoleculeReference b = (SmallMoleculeReference) model.getByID("B");
180 		assertNotNull(b);
181 		assertEquals(4, model.getObjects().size());
182 		
183 		// IMPORTANT to understand is that removed SMR1 and A still have B:
184 		assertEquals(3, b.getMemberEntityReferenceOf().size());
185 		assertTrue(a.getMemberEntityReference().contains(b));
186 		assertTrue(smr1.getMemberEntityReference().contains(b));
187 	}

258 	@Test
259 	public void testMergeEquivalentPhysicalEntities()
260 	{
261 		MockFactory mock = new MockFactory(BioPAXLevel.L3);
262 		Model model = mock.createModel();
263 
264 		ProteinReference[] pr = mock.create(model, ProteinReference.class, 2);
265 		Protein[] proteins = mock.create(model, Protein.class, 4);
266 		mock.bindInPairs("entityReference", proteins[0], pr[0], proteins[1], pr[0], proteins[2], pr[1], proteins[3], pr[1]);
267 
268 		//generate two semantically equivalent reactions
269 		BiochemicalReaction[] rxn = mock.create(model, BiochemicalReaction.class, 2);
270 		mock.bindInPairs("left", rxn[0], proteins[2], rxn[1], proteins[2]);
271 		mock.bindInPairs("right", rxn[0], proteins[3], rxn[1], proteins[3]);
272 
273 		Catalysis[] ctl = mock.create(model, Catalysis.class,2);
274 		mock.bindArrays("controlled", ctl, rxn);
275 		mock.bindInPairs("controller", ctl[0], proteins[0], ctl[1], proteins[1]);
276 
277 		ModelUtils.mergeEquivalentPhysicalEntities(model);
278 
279 		//either 0 or 1 kept by chance (and the other gets replaced)
280 		assertTrue(model.contains(proteins[0])^model.contains(proteins[1]));
281 		assertTrue(model.contains(proteins[2])^model.contains(proteins[3]));
282 		assertTrue(ctl[1].getController().contains(proteins[0])^ctl[1].getController().contains(proteins[1]));
283 		assertTrue(ctl[1].getController().contains(proteins[0])^ctl[1].getController().contains(proteins[1]));
284 		assertEquals(1, rxn[0].getRight().size());
285 		assertEquals(1, rxn[1].getRight().size());
286 		assertEquals(1, rxn[0].getLeft().size());
287 		assertEquals(1, rxn[1].getLeft().size());
288 		if(model.contains(proteins[2])) {
289 			assertTrue(rxn[0].getRight().contains(proteins[2]));
290 			assertTrue(rxn[1].getRight().contains(proteins[2]));
291 			assertTrue(rxn[0].getLeft().contains(proteins[2]));
292 			assertTrue(rxn[1].getLeft().contains(proteins[2]));
293 		} else if(model.contains(proteins[3])) {
294 			assertTrue(rxn[0].getRight().contains(proteins[3]));
295 			assertTrue(rxn[1].getRight().contains(proteins[3]));
296 			assertTrue(rxn[0].getLeft().contains(proteins[3]));
297 			assertTrue(rxn[1].getLeft().contains(proteins[3]));
298 		}
299 
300 		//merge interactions and test it
301 		ModelUtils.mergeEquivalentInteractions(model);
302 		assertTrue(model.contains(rxn[0])^model.contains(rxn[1]));
303 	}
