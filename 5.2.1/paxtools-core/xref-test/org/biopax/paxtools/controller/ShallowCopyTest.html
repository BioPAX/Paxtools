20  		Protein p1 = factory.create(Protein.class, "p1");
21  		ProteinReference pr1 = factory.create(ProteinReference.class, "pr1");
22  		UnificationXref x1 = factory.create(UnificationXref.class, "x1");
23  		UnificationXref x2 = factory.create(UnificationXref.class, "x2");
24  		p1.setEntityReference(pr1);
25  		p1.setDisplayName("p1");
26  		pr1.addXref(x1);
27  		pr1.addXref(x2);
28  		pr1.setDisplayName("pr1");
29  		x1.setId("1");
30  		x1.setDb("geneid");
31  		x2.setId("2");
32  		x2.setDb("geneid");
33  		
34  		assertEquals(pr1, p1.getEntityReference());
35  		assertEquals(2, p1.getEntityReference().getXref().size());
36  		
37  		Protein p2 = (new ShallowCopy()).copy(p1, "p1");
38  		
39  		assertEquals(p1.getUri(), p2.getUri());
40  		assertFalse(p1 == p2);
41  		assertTrue(p1.isEquivalent(p2));
42  		assertEquals(pr1, p2.getEntityReference());
43  		assertEquals(2, p2.getEntityReference().getXref().size());
44  	}
