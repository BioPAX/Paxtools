23  		BioPAXFactory factory = BioPAXLevel.L3.getDefaultFactory();
24  		Model model = factory.createModel();
25  		CommandManager manager = new CommandManager(model);
26  		Set<BioPAXElement> objects = new HashSet<BioPAXElement>();
27  
28  		Protein protein = factory.create(Protein.class, "1");
29  		objects.add(protein);
30  
31  		Protein protein2 = factory.create(Protein.class, "2");
32  		objects.add(protein2);
33  
34  		MolecularInteraction interaction = factory.create(MolecularInteraction.class, "3");
35  		objects.add(interaction);
36  
37  		manager.addObjects(objects);
38  		assertTrue(model.getObjects().size() == 3);
39  		manager.undo();
40  		assertTrue(model.getObjects().size()==0);
41  		manager.redo();
42  
43  		PropertyEditor propertyEditor =
44  				AbstractPropertyEditor.createPropertyEditor(MolecularInteraction.class, "participant");
45  		manager.addProperty(interaction, propertyEditor,protein);
46  		manager.addProperty(interaction, propertyEditor,protein2);
47  		assertTrue(model.getObjects().size()==3);
48  		manager.undo();
49  		manager.undo();
50  		assertTrue(model.getObjects().size()==3);
51  		manager.undo();
52  		assertTrue(model.getObjects().size()==0);
53  		manager.redo();
54  		manager.redo();
55  		manager.redo();
56  		assertTrue(model.getObjects().size()==3);
57  	}
