3   import java.util.Set;

5   /**
6    * Interface for all classes that can have names in BioPAX. All named BioPAX elements can have
7    * multiple names, and exactly one standardName and exactly one shortName. standardName and
8    * shortName are OWL subproperties of name in BioPAX, so these are automatically considered as a
9    * name if defined. Paxtools refle
10   * 
11   * Warning: There is a potential OWL-JAVA semantic mismatch when manipulating names. If a user
12   * decides to assign a different name to standardName or shortName, what happens to the old name is
13   * not well defined - they can be "demoted" to a regular name or removed from the names list
14   * altogether. Paxtools currently does the latter. If this is not the desired behaviour, the old
15   * name should be added manually back to the names list.
16   */
17  public interface Named extends XReferrable
18  {
19  	/**
20  	 * Names for this entity, including standardName and shortName if defined.
21  	 * 
22  	 * The contents of this set can be modified directly but semantic consistency is not guaranteed.
23  	 * Using {@link #addName} and {@link #removeName} is recommended.
24  	 *
25  	 * @return Names for this entity, including standardName and shortName if defined.
26  	 */
27  	public Set<String> getName();
28  
29  
30  	/**
31  	 * This method is reserved for batch operations and should not be used for normal use.
32  	 * Use add/remove name methods to manipulate the names instead.
33  	 *
34  	 * @param names new names to set (or replace all existing ones)
35  	 */
36  	public void setName(Set<String> names);
37  
38  	/**
39  	 * This method adds the given value to the name set.
40  	 *
41  	 * @param name a new name to be added
42  	 */
43  	public void addName(String name);
44  
45  	/**
46  	 * This method removes the given value from the name set.
47  	 *
48  	 * @param name a new name to be removed
49  	 */
50  	public void removeName(String name);
51  
52  
53  	/**
54  	 * An abbreviated name for this entity, preferably a name that is short enough to be used in a
55  	 * visualization application to label a graphical element that represents this entity. If no short
56  	 * name is available, an xref may be used for this purpose by the visualization application.
57  	 *
58  	 * @return An abbreviated name for this entity
59  	 */
60  	public String getDisplayName();
61  
62  
63  	/**
64  	 * An abbreviated name for this entity, preferably a name that is short enough to be used in a
65  	 * visualization application to label a graphical element that represents this entity. If no short
66  	 * name is available, an xref may be used for this purpose by the visualization application.
67  	 *
68  	 * @param displayName new display name (preferably a short one)
69  	 */
70  	public void setDisplayName(String displayName);
71  
72  
73  
74  	/**
75  	 * The preferred full name for this entity, if exists assigned by a standard nomenclature
76  	 * organization such as HUGO Gene Nomenclature Committee.
77  	 *
78  	 * @return standard name for this entity
79  	 */
80  	String getStandardName();
81  
82  
83  	/**
84  	 * The preferred full name for this entity, if exists assigned by a standard nomenclature
85  	 * organization such as HUGO Gene Nomenclature Committee.
86  	 *
87  	 * @param standardName standard name for this entity
88  	 */
89  	public void setStandardName(String standardName);
90  
91  }
