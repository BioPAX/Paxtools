1   package org.biopax.paxtools.model.level3;
2   
3   import java.util.Set;
4   
5   /**
6    * Definition: The form of a physical entity in a particular experiment, as it may be modified for purposes of
7    * experimental design.
8    *
9    * Examples: A His-tagged protein in a binding assay. A protein can be tagged by multiple tags,
10   * so can have more than 1 experimental form type terms
11   */
12  public interface ExperimentalForm extends UtilityClass
13  {
14  
15  
16  	/**
17  	 * Descriptor of this experimental form from a controlled vocabulary.
18  	 * The contents of this set should not be modified. Use add/remove methods instead.
19  	 * @return a CV term from PSI-MI participant identification methdod, experimental role, experimental preparation.
20  	 */
21  	Set<ExperimentalFormVocabulary> getExperimentalFormDescription();
22  
23  	/**
24  	 * Adds an experimental form description.
25  	 * @param experimentalFormType descriptor of this experimental form from a controlled vocabulary.
26  	 */
27  	void addExperimentalFormDescription(ExperimentalFormVocabulary experimentalFormType);
28  
29  	/**
30  	 * Removes an experimental form description.
31  	 * @param experimentalFormType descriptor of this experimental form from a controlled vocabulary.
32  	 */
33  	void removeExperimentalFormDescription(ExperimentalFormVocabulary experimentalFormType);
34  
35  
36  	/**
37  	 * @return The gene or physical entity that this experimental form describes.
38  	 */
39  	Entity getExperimentalFormEntity();
40  
41  	void setExperimentalFormEntity(Entity newPARTICIPANT);
42  
43  	/**
44  	 * Contents of this set should not be modified. Please use add/remove methods instead.
45  	 * @return A feature of the experimental form of the participant of the interaction,
46  	 *         such as a protein tag. It is not expected to occur in vivo or be necessary for the interaction.
47  	 */
48  	Set<EntityFeature> getExperimentalFeature();
49  
50  	/**
51  	 * This method adds an experimental feature
52  	 * @param experimentalFeature of the experimental form of the participant of the interaction,
53  	 *         such as a protein tag. It is not expected to occur in vivo or be necessary for the interaction.
54  	 */
55  	void addExperimentalFeature(EntityFeature experimentalFeature);
56  
57  	/**
58  	 * This method removes an experimental feature
59  	 * @param experimentalFeature of the experimental form of the participant of the interaction,
60  	 *         such as a protein tag. It is not expected to occur in vivo or be necessary for the interaction.
61  	 */
62  	void removeExperimentalFeature(EntityFeature experimentalFeature);
63  }
