12  /**
13   * Utility class for equivalence based comparison of a set of BioPAXElements.
14   *
15   * BioPAXElement by default uses equals and hash code methods based on the type and URI.
16   * On the other hand for many elements it is possible to determine semantic equivalence
17   * among elements. For example two entityFeatures with exactly the same type and location
18   * are equivalent. This logic is implemented in isEquivalent() and equivalenceCode()
19   * methods.
20   *
21   * For most Java collections that uses hashCode and equals there is no easy way to plug-in a
22   * comparator to switch to different comparison behavior. This is a simple Set implementation that uses
23   * equivalence codes when possible. It uses HashMap as the underlying implementation and will use a special bucket in
24   * the case of clashes.
25   *
26   */
27  public class EquivalenceGrouper<T extends BioPAXElement>
28  {
29  	HashSet<EquivalanceBucket<T>> buckets;
30  
31  	private static final Logger log = LoggerFactory.getLogger(EquivalenceGrouper.class);
32  
33  	public EquivalenceGrouper(Set<? extends T> bpes)
34  	{
35  		this();
36  		addAll(bpes);
37  	}
38  
39  	public HashSet<? extends List<T>> getBuckets()
40  	{
41  		return buckets;
42  	}
43  
44  	void addAll(Set<? extends T> bpes)
45  	{
46  		for (T bpe : bpes)
47  		{
48  			add(bpe);
49  		}
50  	}
51  
52  	public EquivalanceBucket<T> access(final T element)
53  	{
54  		EquivalanceBucket<T> value =null;
55  		if (element != null)
56  		{
57  			value = access(element, EquivalanceBucket.EQUALITY);
58  			if (value==null) //now try with equivalence
59  			{
60  				value=access(element,EquivalanceBucket.EQUIVALENCE);
61  			}
62  		}
63  		return value;
64  
65  	}
66  
67  	private EquivalanceBucket<T> access(final T element, final boolean parity)
68  	{
69  		final Object[] trap = new Object[]{null};
70  		if (this.buckets.contains(new Object()
71  		{
72  			public int hashCode()
73  			{
74  				return parity?element.equivalenceCode():element.hashCode();
75  			}
76  
77  			public boolean equals(Object other)
78  			{
79  				if(other != null && other.equals(element))
80  				{
81  					trap[0] = other;
82  					return true;
83  				}
84  				else return false;
85  			}
86  		}))
87  		{
88  			return (EquivalanceBucket<T>) trap[0];
89  		}
90  		return null;
91  	}
92  
93  
94  	public EquivalenceGrouper()
95  	{
96  		this.buckets = new HashSet<EquivalanceBucket<T>>();
97  	}
98  
99  	public void add(T bpe)
100 	{
101 
102 		// If this is the case then we will simply return false when
103 		// we have something that matches the evcode
104 		// AND if that something is a bucket contains something that is  equivalent to bpe
105 		// AND if that something is not a bucket and it is equivalent to bpe
106 		EquivalanceBucket<T> bucket = access(bpe);
107 		if (bucket == null)
108 		{
109 			bucket = new EquivalanceBucket(bpe);
110 			this.buckets.add(bucket);
111 		} else
112 		{
113 			for (T t : bucket)
114 			{
115 				if(t==bpe) return;
116 			}
117 			bucket.add(bpe);
118 		}
119 	}
120

122 	private class EquivalanceBucket<T extends BioPAXElement> extends LinkedList<T>
123 	{
124 
125 		static final boolean EQUALITY = false;
126 
127 		static final boolean EQUIVALENCE = true;
128 
129 		private final int code;
130 
131 		private boolean parity;
132 
133 		private EquivalanceBucket(T first)
134 		{
135 			this.add(first);
136 			if (first.equivalenceCode() == 0 || first.equivalenceCode() == first.hashCode())
137 			{
138 				parity = EQUALITY;
139 				this.code = first.hashCode();
140 			} else
141 			{
142 				parity = EQUIVALENCE;
143 				this.code = first.equivalenceCode();
144 			}
145 		}
146 
147 		@Override public int hashCode()
148 		{
149 			return code;
150 		}
151 
152 		@Override public boolean equals(Object o)
153 		{
154 			if (o instanceof EquivalanceBucket)
155 			{
156 				return super.equals(o);
157 			} else
158 			{
159 				T t = this.get(0);
160 				if (parity) //EQUIVALENCE
161 				{
162 					return t.isEquivalent((BioPAXElement) o);
163 				} else //EQUALITY
164 				{
165 					return t.equals(o);
166 				}
167 			}
168 		}
169 
170 	}
171 }
