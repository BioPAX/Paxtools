60  	public static void toGSEA(String[] argv) throws IOException {
61  		//argv[0] is the command name ('toGsea')
62  		boolean crossSpecies = false; //cross-check is enabled (i.e., no mixing different species IDs in one row)
63  		boolean subPathways = false; //no sub-pathways (i.e., going into sub-pathways is not enabled)
64  		boolean notPathways = false;
65  		Set<String> organisms = new HashSet<String>();
66  
67  		if (argv.length < 4)
68  			throw new IllegalArgumentException("Not enough arguments: " + argv);
69  
70  		if (argv.length > 4) {
71  			for (int i = 4; i < argv.length; i++) {
72  				if("-crossSpecies".equalsIgnoreCase(argv[i])) {
73  					crossSpecies = true;
74  				}
75  				else if("-subPathways".equalsIgnoreCase(argv[i])) {
76  					subPathways = true;
77  				}
78  				else if("-notPathway".equalsIgnoreCase(argv[i])) {
79  					notPathways = true;
80  				}
81  				else if(argv[i].startsWith("organisms=")) {
82  					for(String o : argv[i].substring(10).split(",")) {
83  						organisms.add(o.trim().toLowerCase());
84  					}
85  				}
86  			}
87  		}
88  
89  		// The Constructor args: GSEAConverter(idTypeNameOrPrefix, crossSpeciesCheckEnabled?, skipSubPathways?)
90  		GSEAConverter gseaConverter = new GSEAConverter(argv[3], !crossSpecies, !subPathways);
91  		gseaConverter.setSkipOutsidePathways(!notPathways);
92  		gseaConverter.setAllowedOrganisms(organisms);//if organisms is empty then all species are allowed (no filtering)
93  		gseaConverter.writeToGSEA(io.convertFromOWL(getInputStream(argv[1])), new FileOutputStream(argv[2]));
94  	}

103 	public static void getNeighbors(String[] argv) throws IOException
104 	{
105 		// set strings vars
106 		String in = argv[1];
107 		String[] ids = argv[2].split(",");
108 		String out = argv[3];
109 
110 		// read BioPAX from the file
111 		Model model = io.convertFromOWL(getInputStream(in));
112 
113 		// get elements (entities)
114 		Set<BioPAXElement> elements = new HashSet<BioPAXElement>();
115 		for (Object id : ids) {
116 			BioPAXElement e = model.getByID(id.toString());
117 			if (e != null && (e instanceof Entity || e instanceof entity)) {
118 				elements.add(e);
119 			} else {
120 				log.warn("Source element not found: " + id);
121 			}
122 		}
123 
124 		// execute the 'nearest neighborhood' query
125 		Set<BioPAXElement> result = QueryExecuter
126 			.runNeighborhood(elements, model, 1, Direction.BOTHSTREAM);
127 
128 		// auto-complete/clone the results in a new model
129 		// (this also cuts some less important edges, right?..)
130 		Completer c = new Completer(io.getEditorMap());
131 		result = c.complete(result, model);
132 		Cloner cln = new Cloner(io.getEditorMap(), io.getFactory());
133 		model = cln.clone(model, result); // new sub-model
134 
135 		if (model != null) {
136 			log.info("Elements in the result model: " + model.getObjects().size());
137 			// export to OWL
138 			io.convertToOWL(model, new FileOutputStream(out));
139 		} else {
140 			log.error("NULL model returned.");
141 		}
142 	}

150 		// set strings vars
151 		String in = argv[1];
152 		String out = argv[2];
153 		String[] uris = new String[0]; //empty means - all
154 		boolean absoluteUris = false;

156 			for(int i=3; i<argv.length; i++) {
157 				String param = argv[i];
158 				if (param.startsWith("uris=")) {
159 					uris = param.substring(5).split(",");
160 				}
161 				else if(param.startsWith("-absolute")) {
162 					absoluteUris = true;
163 				}
164 			}
165 		}
166 
167 		// import the model
168 		log.info("Loading the BioPAX model from " + in);
169 		Model model = io.convertFromOWL(getInputStream(in));
170 		log.info("Successfully loaded the BioPAX model. Writing to the output: " + out);
171 
172 		// extract and save the (sub-)model
173 		SimpleIOHandler biopaxWriter = new SimpleIOHandler(model.getLevel());
174 		biopaxWriter.absoluteUris(absoluteUris);
175 		if(uris.length > 0)
176 			biopaxWriter.convertToOWL(model, new FileOutputStream(out), uris);
177 		else
178 			biopaxWriter.convertToOWL(model, new FileOutputStream(out));
179 
180 		log.info("Done.");
181 	}

185 	 * Detects and converts a BioPAX Level 1, 2,
186 	 * or PSI-MI/MITAB model to BioPAX Level3..
187 	 *
188 	 * TODO: should not be public API, but is already used from ext. apps, such as PaxtoolsR...
189 	 */
190 	public static void toLevel3(String[] argv) throws IOException {
191 		final String input = argv[1];
192 		final String output = argv[2];
193 
194 		boolean forcePsiInteractionToComplex = false;
195 		if(argv.length > 3) {
196 			for(int i=3; i<argv.length; i++) {
197 				String param = argv[i];
198 				if (param.equalsIgnoreCase("-psimiToComplexes")) {
199 					forcePsiInteractionToComplex = true;
200 				}
201 			}
202 		}
203 
204 		Type type = detect(input);
205 
206 		InputStream is = getInputStream(input);
207 		FileOutputStream os = new FileOutputStream(output);
208 
209 		try {
210 			switch (type) {
211 				case BIOPAX:
212 					Model model = io.convertFromOWL(is);
213 					model = (new LevelUpgrader()).filter(model);
214 					if (model != null) {
215 						io.setFactory(model.getLevel().getDefaultFactory());
216 						io.convertToOWL(model, os); //os is closed already
217 					}
218 					break;
219 
220 				case PSIMI:
221 					PsiToBiopax3Converter psimiConverter = new PsiToBiopax3Converter();
222 					psimiConverter.convert(is, os, forcePsiInteractionToComplex);
223 					os.close();
224 					break;
225 
226 				default: //MITAB
227 					psimiConverter = new PsiToBiopax3Converter();
228 					psimiConverter.convertTab(is, os, forcePsiInteractionToComplex);
229 					os.close();
230 					break;
231 			}
232 
233 		} catch (Exception e) {
234 			throw new RuntimeException("Failed to convert " +
235 				input + "to BioPAX L3", e);
236 		}
237 	}

245 	//read a few lines to detect it's a BioPAX vs. PSI-MI vs. PSI-MITAB data.
246 	private static Type detect(String input) {
247 		StringBuilder sb = new StringBuilder();
248 		try {
249 			BufferedReader reader = new BufferedReader(new InputStreamReader(getInputStream(input)));
250 			int linesToCheck = 20;
251 			while (linesToCheck-- > 0) {
252 				sb.append(reader.readLine()).append('\n');
253 			}
254 			reader.close();
255 		} catch (IOException e) {
256 			throw new RuntimeException(e);
257 		}
258 
259 		String buf = sb.toString();
260 		if (buf.contains("<rdf:RDF") && buf.contains("http://www.biopax.org/release/biopax")) {
261 			return Type.BIOPAX;
262 		}
263 		else if (buf.contains("<entrySet") && buf.contains("/psidev")) {
264 			return Type.PSIMI;
265 		}
266 		else
267 			return Type.PSIMITAB; //default/guess
268 	}

307 	 * Checks files by the online BioPAX validator using the built-in BioPAX validator client.
308 	 *
309 	 * See about <a href="http://www.biopax.org/validator">BioPAX Validator Webservice</a>
310 	 *
311 	 * TODO: should not be public API, but is already used from ext. apps, such as PaxtoolsR...
312 	 */
313 	public static void validate(String[] argv) throws IOException
314 	{
315 		String input = argv[1];
316 		String output = argv[2];
317 		// default options
318 		RetFormat outf = RetFormat.HTML;
319 		boolean fix = false;
320 		Integer maxErrs = null;
321 		Behavior level = null; //will report both errors and warnings
322 		String profile = null;
323 
324 		// match optional args
325 		for (int i = 3; i < argv.length; i++) {
326 			if ("html".equalsIgnoreCase(argv[i])) {
327 				outf = RetFormat.HTML;
328 			} else if ("xml".equalsIgnoreCase(argv[i])) {
329 				outf = RetFormat.XML;
330 			} else if ("biopax".equalsIgnoreCase(argv[i])) {
331 				outf = RetFormat.OWL;
332 			} else if ("auto-fix".equalsIgnoreCase(argv[i])) {
333 				fix = true;
334 			} else if ("only-errors".equalsIgnoreCase(argv[i])) {
335 				level = Behavior.ERROR;
336 			} else if ((argv[i]).toLowerCase().startsWith("maxerrors=")) {
337 				String num = argv[i].substring(10);
338 				maxErrs = Integer.valueOf(num);
339 			} else if ("notstrict".equalsIgnoreCase(argv[i])) {
340 				profile = "notstrict";
341 			}
342 		}
343 
344 		Collection<File> files = new HashSet<File>();
345 		File fileOrDir = new File(input);
346 		if (!fileOrDir.canRead()) {
347 			System.out.println("Cannot read " + input);
348 		}
349 
350 		// collect files
351 		if (fileOrDir.isDirectory()) {
352 			// validate all the OWL files in the folder
353 			FilenameFilter filter = new FilenameFilter() {
354 				public boolean accept(File dir, String name) {
355 					return (name.endsWith(".owl"));
356 				}
357 			};
358 			for (String s : fileOrDir.list(filter)) {
359 				files.add(new File(fileOrDir.getCanonicalPath()
360 					+ File.separator + s));
361 			}
362 		} else {
363 			files.add(fileOrDir);
364 		}
365 
366 		// upload and validate using the default URL:
367 		// http://www.biopax.org/biopax-validator/check.html
368 		OutputStream os = new FileOutputStream(output);
369 		try {
370 			if (!files.isEmpty()) {
371 				BiopaxValidatorClient val = new BiopaxValidatorClient();
372 				val.validate(fix, profile, outf, level, maxErrs, null, files.toArray(new File[]{}), os);
373 			}
374 		} catch (Exception ex) {
375 			// fall-back: not using the remote validator; trying to read files
376 			String msg = "Unable to check with the biopax-validator web service: \n " +
377 				ex.toString() +
378 				"\n Fall-back: trying to parse the file(s) with paxtools " +
379 				"(up to the first syntax error in each file)...\n";
380 			log.error(msg, ex);
381 			os.write(msg.getBytes());
382 
383 			for (File f : files) {
384 				try {
385 					Model m = io.convertFromOWL(getInputStream(f.getPath()));
386 					msg = "Model that contains "
387 						+ m.getObjects().size()
388 						+ " elements is successfully created from "
389 						+ f.getPath()
390 						+ " (check the console output for warnings).\n";
391 					os.write(msg.getBytes());
392 				} catch (Exception e) {
393 					msg = "Error: " + e +
394 						" in building a BioPAX Model from: " +
395 						f.getPath() + "\n";
396 					os.write(msg.getBytes());
397 					e.printStackTrace();
398 					log.error(msg);
399 				}
400 				os.flush();
401 			}
402 		}
403 	}

406 	 * Exports a biopax model to SIF or/and customizable ExtendedSIF format
407 	 * (default classic SIF has only 3 data columns; ExtendedSIF and customized
408 	 * can have more colums plus nodes description at the bottom, after a blank line).
409 	 *
410 	 * TODO: should not be public API, but is already used from ext. apps, such as PaxtoolsR...
411 	 */
412 	public static void toSifnx(String[] argv) throws IOException {
413 		boolean extended = false; //if it stays 'false', then andSif==true will be in effect automatically
414 		boolean andSif = false; //if extended==false, SIF will be generated as it would be andSif==true
415 		boolean mergeInteractions = true;
416 		boolean useNameIfNoId = false;
417 		final Collection<SIFEnum> include = new HashSet<SIFEnum>();
418 		final Collection<SIFEnum> exclude = new HashSet<SIFEnum>();
419 		final ConfigurableIDFetcher idFetcher = new ConfigurableIDFetcher();
420 		final List<String> customFieldList = new ArrayList<String>(); //there may be custom field names (SIF mediators)
421 		//process arguments
422 		if(argv.length > 3) {
423 			for(int i=3; i<argv.length; i++) {
424 				String param = argv[i];
425 				if (param.startsWith("seqDb=")) {
426 					//remove the 'seqDb=' and split comma-sep. values (a single val. no comma is gonna be fine too)
427 					for (String db : param.substring(6).split(","))
428 						idFetcher.seqDbStartsWithOrEquals(db);
429 				} else if (param.startsWith("chemDb=")) {
430 					for (String db : param.substring(7).split(","))
431 						idFetcher.chemDbStartsWithOrEquals(db);
432 				} else if (param.equalsIgnoreCase("-andSif")) {
433 					andSif = true;
434 				} else if (param.equalsIgnoreCase("-extended")) {
435 					extended = true;
436 				} else if (param.equalsIgnoreCase("-dontMergeInteractions")) {
437 					mergeInteractions = false;
438 				} else if (param.equalsIgnoreCase("-useNameIfNoId")) {
439 					useNameIfNoId = true;
440 				} else if (param.startsWith("include=")) {
441 					for (String t : param.substring(8).split(","))
442 						include.add(SIFEnum.valueOf(t.toUpperCase()));
443 				} else if (param.startsWith("exclude=")) {
444 					for (String t : param.substring(8).split(","))
445 						exclude.add(SIFEnum.valueOf(t.toUpperCase())); //e.g. NEIGHBOR_OF
446 				} else {
447 					OutputColumn.Type type = OutputColumn.Type.getType(param.toUpperCase());
448 					if ((type != null && type != OutputColumn.Type.CUSTOM) || param.contains("/"))
449 					{
450 						if(!param.contains("/"))
451 							customFieldList.add(param.toUpperCase());
452 						else
453 							customFieldList.add(param);
454 					}
455 				}
456 			}
457 		}
458 
459 		//fall back to defaults when no ID types were provided
460 		if(idFetcher.getChemDbStartsWithOrEquals().isEmpty()) {
461 			idFetcher.chemDbStartsWithOrEquals("chebi");
462 		}
463 		if(idFetcher.getSeqDbStartsWithOrEquals().isEmpty()) {
464 			idFetcher.chemDbStartsWithOrEquals("hgnc");
465 		}
466 		idFetcher.useNameWhenNoDbMatch(useNameIfNoId);
467 
468 		Model model = getModel(io, argv[1]);
469 
470 		if(mergeInteractions)
471 			ModelUtils.mergeEquivalentInteractions(model);
472 
473 		//Create a new SIF searcher:
474 		//set SIF miners to use (default is to use all types, given no include/exclude args provided)
475 		final Collection<SIFEnum> sifTypes = (include.isEmpty())
476 			? new HashSet<SIFEnum>(Arrays.asList(SIFEnum.values())) : include;
477 		for(SIFType t : exclude) {
478 			sifTypes.remove(t); //remove if exists, otherwise - ignore
479 		}
480 		SIFSearcher searcher = new SIFSearcher(idFetcher, sifTypes.toArray(new SIFEnum[]{}));
481 		log.info("toSIF: using SIFTypes: " + sifTypes);
482 		//load and set blacklist.txt file if exists
483 		File blacklistFile = new File("blacklist.txt");
484 		if(blacklistFile.exists()) {
485 			log.info("toSIF: using blacklist.txt from current directory");
486 			searcher.setBlacklist(new Blacklist(new FileInputStream(blacklistFile)));
487 		} else {
488 			log.info("toSIF: not blacklisting ubiquitous molecules (no blacklist.txt found)");
489 		}
490 
491 		File outputFile = new File(argv[2]);
492 		OutputStream outputStream = new FileOutputStream(outputFile);
493 
494 		if (extended) {
495 			//using built-in PC EXTENDED_BINARY_SIF format (customFieldList parameter is ignored)
496 			Set<SIFInteraction> binaryInts = searcher.searchSIF(model);
497 			ExtendedSIFWriter.write(binaryInts, outputStream);
498 		} else if (customFieldList.isEmpty()) {
499 			searcher.searchSIF(model, outputStream); //classic SIF
500 		} else {
501 			// not really SIF format (this is useful but extra columns sometimes cannot be parsed correctly, by e.g. Cytoscape)
502 			log.info("toSIF: using custom fields (extra columns): " + customFieldList);
503 			searcher.searchSIF(model, outputStream, new CustomFormat(customFieldList.toArray(new String[]{})));
504 		}
505 		//outputStream is closed at this point (inside searchSIF(..) or write(..) method)
506 
507 		if(extended && andSif) {
508 			//convert the file into classic thee-column SIF format
509 			sifnxToSif(outputFile.getPath(), outputFile.getPath()+".sif");
510 		}
511 
512 		log.info("toSIF: done.");
513 	}

515 	/*
516 	 * TODO: should not be public API, but is already used from ext. apps, such as PaxtoolsR...
517 	 */
518 	public static void sifnxToSif(String inputFile, String outputFile) throws IOException {
519 		BufferedReader reader = new BufferedReader(new InputStreamReader(getInputStream(inputFile)));
520 		OutputStreamWriter writer = new OutputStreamWriter(new FileOutputStream(outputFile));
521 		//skip the first line (headers)
522 		if(reader.ready()) reader.readLine();
523 		while(reader.ready()) {
524 			String line = reader.readLine();
525 			//stop at the first blank line (because next come nodes with attributes)
526 			if(line==null || line.isEmpty())
527 				break;
528 			//keep only the first three columns (otherwise, it's not gonna be SIF format)
529 			writer.write(StringUtils.join(Arrays.copyOfRange(line.split("\t", 4), 0, 3), '\t') + '\n');
530 		}
531 		reader.close();
532 		writer.close();
533 	}

624 	 * For each physical entity participant in the BioPAX model,
625 	 * output uri, type, names, standard identifiers (in JSON format).
626 	 */
627 	private static void mapUriToIds(Model model, PrintStream out) {
628 		Set<String> elements = new TreeSet<String>();
629 
630 		//write one by one to insert EOLs and make potentially a very large file human-readable -
631 		for(PhysicalEntity pe : model.getObjects(PhysicalEntity.class))
632 		{
633 			JSONObject jo = new JSONObject();
634 			jo.put("uri", pe.getUri());
635 			jo.put("type", pe.getModelInterface().getSimpleName());
636 			jo.put("generic", ModelUtils.isGeneric(pe));
637 			jo.put("label", pe.getDisplayName());
638 
639 			JSONArray ja = new JSONArray();
640 			ja.addAll(pe.getName());
641 			jo.put("name", ja);
642 
643 			if(!(pe instanceof SmallMolecule)) {
644 				ja = new JSONArray();
645 				ja.addAll(identifiers(pe, "hgnc symbol", false, false));
646 				jo.put("HGNC Symbol", ja);
647 
648 				ja = new JSONArray();
649 				ja.addAll(identifiers(pe, "uniprot", true, false));
650 				jo.put("UniProt", ja);
651 			}
652 
653 			if(pe instanceof SmallMolecule || PhysicalEntity.class.equals(pe.getModelInterface())) {
654 				ja = new JSONArray();
655 				ja.addAll(identifiers(pe, "chebi", false, false));
656 				jo.put("ChEBI", ja);
657 			}
658 
659 			ja = new JSONArray();
660 			for(BioSource bs : ModelUtils.getOrganisms(pe)) ja.add(bs.getDisplayName());
661 			jo.put("organism", ja);
662 
663 			ja = new JSONArray();
664 			for(Provenance ds : pe.getDataSource()) ja.add(ds.getDisplayName());
665 			jo.put("datasource", ja);
666 
667 			elements.add(jo.toJSONString());
668 		}
669 
670 		// Write as JSON array to the output
671 		out.print("[\n" + StringUtils.join(elements,"\n,") + "\n]");
672 	}

676 	 * Recursively collects bio identifiers of given type (xref.db name)
677 	 * associated with the physical entity or generic, complex entity.
678 	 *
679 	 * TODO (options): process Gene and Interaction; traverse into 'evidence' property.
680 	 * @param entity a process participant (simple or generic)
681 	 * @param xrefdb identifier type, such as 'HGNC Symbol' or 'ChEBI' (matches Xref.db values in the BioPAX model)
682 	 * @param isPrefix whether the xrefdb value is a prefix rather than complete name.
683 	 * @param includeEvidence whether to traverse into property:evidence to collect ids.
684 	 */
685 	private static Set<String> identifiers(final PhysicalEntity entity, final String xrefdb,
686 																				 boolean isPrefix, boolean includeEvidence)
687 	{
688 		final Set<String> ids = new HashSet<String>();
689 		final Fetcher fetcher = (includeEvidence)
690 			? new Fetcher(SimpleEditorMap.L3, Fetcher.nextStepFilter)
691 			: new Fetcher(SimpleEditorMap.L3, Fetcher.nextStepFilter, Fetcher.evidenceFilter);
692 		fetcher.setSkipSubPathways(true); //makes no difference now  but good to have/know...
693 		Set<XReferrable> children = fetcher.fetch(entity, XReferrable.class);
694 		children.add(entity); //include itself
695 		for(XReferrable child : children) {//ignore some classes, such as controlled vocabularies, interactions, etc.
696 			if (child instanceof PhysicalEntity || child instanceof EntityReference || child instanceof Gene)
697 				for (Xref x : child.getXref())
698 					if ((x.getId()!=null && x.getDb()!=null) && (isPrefix)
699 						? x.getDb().toLowerCase().startsWith(xrefdb.toLowerCase())
700 						: xrefdb.equalsIgnoreCase(x.getDb()))
701 					{
702 						ids.add(x.getId());
703 					}
704 		}
705 		return ids;
706 	}

710 		final PathAccessor directChildPathwaysAccessor = new PathAccessor("Pathway/pathwayComponent:Pathway");
711 		final PathAccessor pathwayComponentAccessor = new PathAccessor("Pathway/pathwayComponent");
712 		final PathAccessor pathwayOrderStepProcessAccessor = new PathAccessor("Pathway/pathwayOrder/stepProcess");

720 			sb.append(pathway.getUri()).append('\t').append(pathway.getDisplayName()).append('\t');
721 			//add direct sub-pathways
722 			for(Object o : directChildPathwaysAccessor.getValueFromBean(pathway)) {
723 				Pathway p = (Pathway) o;
724 				sb.append(p.getUri()).append(";");
725 			}
726 			sb.append("\t");
727 			//add all sub-pathways
728 			Fetcher fetcher = new Fetcher(SimpleEditorMap.L3, Fetcher.nextStepFilter);

730 				sb.append(p.getUri()).append(";");
731 			}
732 			out.println(sb.toString());
733 		}
734 		// print pathway names, etc. after a blank line and title line
735 		out.println("\nPATHWAY_URI\tDATASOURCE\tDISPLAY_NAME\tALL_NAMES" +
736 			"\tNUM_DIRECT_COMPONENT_OR_STEP_PROCESSES");
737 		for(Pathway pathway : pathways) {
738 			final int size = pathwayComponentAccessor.getValueFromBean(pathway).size()
739 				+ pathwayOrderStepProcessAccessor.getValueFromBean(pathway).size();
740 			//pathways in PC2 normally has only one dataSource (Provenance)

747 			for(String name : pathway.getName())
748 				sb.append('"').append(name).append('"').append(";");
749 			//"size"
750 			sb.append('\t').append(size);
751 			out.println(sb.toString());
752 		}

755 	private static void summarizeHgncIds(Model model, PrintStream out) {
756 		boolean verbose = true; //TODO use another parameter here
757 
758 		//Analyse SERs (Protein-, Dna* and Rna* references) - HGNC usage, coverage,..
759 		//Calc. the no. non-generic ERs having >1 different HGNC symbols and IDs, or none, etc.
760 		Set<SequenceEntityReference> haveMultipleHgnc = new HashSet<SequenceEntityReference>();
761 		Map<Provenance,MutableInt> numErs = new HashMap<Provenance,MutableInt>();
762 		Map<Provenance,MutableInt> numProblematicErs = new HashMap<Provenance,MutableInt>();
763 		PathAccessor pa = new PathAccessor("EntityReference/entityReferenceOf/dataSource", model.getLevel());
764 		Set<String> problemErs = new TreeSet<String>();
765 		for(EntityReference ser : model.getObjects(EntityReference.class)) {
766 			//skip if it's a SMR or generic
767 			if(ser instanceof SmallMoleculeReference || !ser.getMemberEntityReference().isEmpty())
768 				continue;
769 
770 			Set<String> hgncSymbols = new HashSet<String>();
771 			Set<String> hgncIds = new HashSet<String>();
772 
773 			if(ser.getUri().startsWith("http://identifiers.org/hgnc")) {
774 				String s = ser.getUri().substring(ser.getUri().lastIndexOf("/")+1);
775 				if(s.startsWith("HGNC:"))
776 					hgncIds.add(s);
777 				else
778 					hgncSymbols.add(s);
779 			}
780 
781 			for(Xref x : ser.getXref()) {
782 				if(x instanceof PublicationXref || x.getDb()==null || x.getId()==null)
783 					continue; //skip
784 
785 				if(x.getDb().toLowerCase().startsWith("hgnc") && !x.getId().toLowerCase().startsWith("hgnc:")) {
786 					hgncSymbols.add(x.getId().toLowerCase());
787 				}
788 				else if(x.getDb().toLowerCase().startsWith("hgnc") && x.getId().toLowerCase().startsWith("hgnc:")) {
789 					hgncIds.add(x.getId().toLowerCase());
790 				}
791 			}
792 
793 			if(hgncIds.size()>1 || hgncSymbols.size()>1)
794 				haveMultipleHgnc.add((SequenceEntityReference) ser);
795 
796 			//increment "no hgnc" and "total" counts by data source
797 			for(Object provenance : pa.getValueFromBean(ser)) {
798 				if (hgncSymbols.isEmpty() && hgncIds.isEmpty()) {
799 					if (verbose) {
800 						problemErs.add(String.format("%s\t%s\t%s",
801 							((Provenance) provenance).getDisplayName(), ser.getDisplayName(), ser.getUri()));
802 					}
803 
804 					MutableInt n = numProblematicErs.get(provenance);
805 					if (n == null)
806 						numProblematicErs.put((Provenance) provenance, new MutableInt(1));
807 					else
808 						n.increment();
809 				}
810 
811 				MutableInt tot = numErs.get(provenance);
812 				if (tot == null)
813 					numErs.put((Provenance) provenance, new MutableInt(1));
814 				else
815 					tot.increment();
816 			}
817 		}
818 		//print results
819 		if(verbose) {
820 			out.println("SequenceEntityReferences (not generics) without any HGNC Symbol:");
821 			for(String line : problemErs) out.println(line);
822 		}
823 		out.println("The number of SERs (not generic) having more than one HGNC Symbols: " + haveMultipleHgnc.size());
824 		out.println("\nNumber of SequenceEntityReferences (not generics) without any HGNC ID, by data source:");
825 		int totalPrs = 0;
826 		int numPrsNoHgnc = 0;
827 		for(Provenance ds : numProblematicErs.keySet()) {
828 			int n = numProblematicErs.get(ds).intValue();
829 			numPrsNoHgnc += n;
830 			int t = numErs.get(ds).intValue();
831 			totalPrs += t;
832 			out.println(String.format("%s\t\t%d\t(%3.1f%%)", ds.getUri(), n, ((float)n)/t*100));
833 		}
834 		out.println(String.format("Total\t\t%d\t(%3.1f%%)", numPrsNoHgnc, ((float)numPrsNoHgnc)/totalPrs*100));
835 	}

975 				Class<?> returnType = getMethod.getReturnType();

1174 		{
1175 			public int compare(Object key1, Object key2)
1176 			{

1178 				int cnt2 = map.get(key2);

1181 				else return cnt2 - cnt1;
